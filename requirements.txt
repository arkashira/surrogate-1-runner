name: Ingest

on:
  workflow_dispatch:
    inputs:
      SHARD_ID:
        description: Shard ID
        required: true
      ITERATION_TIMESTAMP:
        description: Iteration timestamp
        required: true

jobs:
  ingest:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Set up Hugging Face token
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: |

          # Set the snapshot directory
          SNAPSHOT_DIR="batches/snapshot"

          # Set the shard ID
          SHARD_ID=${SHARD_ID}

          # Set the iteration timestamp
          ITERATION_TIMESTAMP=${ITERATION_TIMESTAMP}

          # Set the output file path
          OUTPUT_FILE="$SNAPSHOT_DIR/public-merged/$ITERATION_TIMESTAMP-$SHARD_ID-$RANDOM.jsonl"

          # Copy the pre-flight snapshot to the output file
          cp "$SNAPSHOT_DIR/*.parquet" "$OUTPUT_FILE"

          # Dedup the pairs using the central md5 hash store
          dedup.py "$OUTPUT_FILE"
