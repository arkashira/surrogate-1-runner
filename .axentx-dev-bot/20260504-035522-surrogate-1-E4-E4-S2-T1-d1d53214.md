/**
 * Token‑ and USD‑level cost information for a single LLM invocation.
 *
 * - `promptTokens`   – tokens that belong to the request payload.
 * - `completionTokens` – tokens generated by the model.

import { LlmClient, LlmResponse } from '../../llm';
import { Logger } from '../../utils';
import { LlmCostInfo } from '../models/LlmCostInfo';

/**
 * Result shape for an LLM action – includes the generated text and the cost payload.
 */
export interface LlmActionResult {
  /** The raw text generated by the LLM */
  output: string;
  /** Cost metrics for this invocation */
  cost: LlmCostInfo;
}

 * LlmAction executes a request against a configured LLM client and returns both
 * the generated output and a cost payload (tokens + USD). The payload is used
 * by the workflow engine and UI panels to display budgeting information.
 */
export class LlmAction implements Action<LlmActionResult> {
  private readonly client: LlmClient;
  private readonly logger: Logger;

  constructor(client: LlmClient, logger: Logger) {
    this.client = client;
    this.logger = logger;
  }

  /**
   * Executes the LLM request.
   *
   * @param ctx    Execution context supplied by the workflow engine.
   * @param prompt Prompt to send to the LLM.
   * @param model  Optional model identifier (defaults to the client’s default).
   * @returns      An object containing the generated text and cost information.
   */
  async execute(
    ctx: ActionContext,
    prompt: string,
    model?: string,
  ): Promise<LlmActionResult> {
    this.logger.debug('LlmAction.execute', { prompt, model });

    const response: LlmResponse = await this.client.generate({
      model,
      prompt,
