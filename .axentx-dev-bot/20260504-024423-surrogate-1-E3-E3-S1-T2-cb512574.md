"""
Export service for post-mortem data.
Provides functionality to export correlated artifacts and alerts in multiple formats.
"""
import json
import csv
from datetime import datetime
from typing import List, Dict, Any, Optional
from io import StringIO, BytesIO


class ExportService:
    """Service for exporting post-mortem data in various formats."""
    
    SUPPORTED_FORMATS = ['json', 'csv', 'html']
    
    def __init__(self):
        self.data_store = {}
    
    # =========================================================================
    # CORE EXPORT METHODS
    # =========================================================================
    
    def export_timeline(
        self, 
        artifacts: List[Dict[str, Any]], 
        format: str = 'json'
    ) -> bytes:
        """
        Export timeline data in the specified format.
        
        Args:
            artifacts: List of artifact dictionaries with timestamp data
            format: Export format (json, csv, html)
            
        Returns:
            Exported data as bytes
        """
        if format not in self.SUPPORTED_FORMATS:
            raise ValueError(
                f"Unsupported format: {format}. "
                f"Supported: {self.SUPPORTED_FORMATS}"
            )
        
        # Sort by timestamp (newest first for timeline view)
        sorted_artifacts = sorted(
            artifacts, 
            key=lambda x: x.get('timestamp', ''),
            reverse=True
        )
        
        if format == 'json':
            return self._export_json(sorted_artifacts)
        elif format == 'csv':
            return self._export_csv(sorted_artifacts)
        elif format == 'html':
            return self._export_html(sorted_artifacts)
    
    def _export_json(self, artifacts: List[Dict[str, Any]]) -> bytes:
        """Export to JSON format with metadata."""
        export_data = {
            'exported_at': datetime.utcnow().isoformat(),
            'total_items': len(artifacts),
            'timeline': artifacts
        }
        return json.dumps(export_data, indent=2, default=str).encode('utf-8')
    
    def _export_csv(self, artifacts: List[Dict[str, Any]]) -> bytes:
        """Export to CSV format using in-memory buffer."""
        if not artifacts:
            return b''
        
        output = StringIO()
        fieldnames = ['timestamp', 'type', 'deployment', 'severity', 'message', 'source']
        
        writer = csv.DictWriter(output, fieldnames=fieldnames, extrasaction='ignore')
        writer.writeheader()
        
        for artifact in artifacts:
            row = {k: artifact.get(k, '') for k in fieldnames}
            writer.writerow(row)
        
        return output.getvalue().encode('utf-8')
    
    def _export_html(self, artifacts: List[Dict[str, Any]]) -> bytes:
        """Export to HTML format with timeline visualization."""
        html_template = """<!DOCTYPE html>
<html>
<head>
