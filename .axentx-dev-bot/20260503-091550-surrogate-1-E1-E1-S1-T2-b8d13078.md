    type: str
    title: str
    owner: str
    eta: str

    ALLOWED_TYPES = {"type1", "type2", "type3"}

    @field_validator("type")

    def type_must_be_valid(cls, v: str) -> str:
        if v not in cls.ALLOWED_TYPES:
            raise ValueError(f"Invalid type. Allowed: {sorted(cls.ALLOWED_TYPES)}")
        return v

    @field_validator("title", "owner", "team")
    @classmethod
    def must_not_be_empty(cls, v: str, info: ValidationInfo) -> str:
        if not v or not v.strip():
            field = info.field_name
            raise ValueError(f"{field} cannot be empty")
        return v.strip()

    @field_validator("eta")
    @classmethod
    def eta_must_be_valid(cls, v: str) -> str:
        try:
            parsed = datetime.strptime(v, "%Y-%m-%d")
        except ValueError:
            raise ValueError("Invalid ETA format. Expected YYYY-MM-DD")
        # Optional: reject past dates if business rule requires
