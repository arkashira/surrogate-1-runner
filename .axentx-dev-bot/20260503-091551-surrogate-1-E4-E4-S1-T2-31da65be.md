import sqlite3
import os
from datetime import datetime, timedelta
from typing import Dict, Any

DB_PATH = os.getenv("DATABASE_URL", "surrogate.db")
if DB_PATH.startswith("sqlite:///"):
    DB_PATH = DB_PATH.replace("sqlite:///", "")

def _get_db():
    return sqlite3.connect(DB_PATH)

def get_wau_counts() -> Dict[str, Any]:
    """
    Returns WAU request counts per team and total for the last 7 days.
    A request is counted if it has at least one status update or view event
    within the rolling 7-day window.
    """
    week_ago = datetime.utcnow() - timedelta(days=7)
    week_ago_iso = week_ago.isoformat()

    try:
        conn = _get_db()
        cur = conn.cursor()

        # Count distinct requests per team that had a qualifying event in last 7 days
        cur.execute(
            """
            SELECT r.team_id, COUNT(DISTINCT r.id) AS cnt
            FROM requests r
            WHERE EXISTS (
                SELECT 1
                FROM request_events ev
                WHERE ev.request_id = r.id
                  AND ev.event_type IN ('status_update', 'view')
                  AND ev.created_at >= ?
            )
            GROUP BY r.team_id
            """,
            (week_ago_iso,)
        )
        rows = cur.fetchall()
        conn.close()

        per_team = {str(row[0]): row[1] for row in rows}
        total = sum(per_team.values())

        return {
            "wau_requests_last_7_days": {
                "per_team": per_team,
                "total": total,
                "window": {
                    "start": week_ago_iso,
                    "end": datetime.utcnow().isoformat(),
                    "unit": "days",
                    "length": 7
                },
                "definition": "requests with at least one status_update or view event in rolling 7-day window"
            }
        }
    except Exception:
        # Graceful fallback for runner environments without DB/schema
        return {
            "wau_requests_last_7_days": {
                "per_team": {},
                "total": 0,
                "window": {
                    "start": week_ago_iso,
                    "end": datetime.utcnow().isoformat(),
                    "unit": "days",
                    "length": 7
                },
                "definition": "requests with at least one status_update or view event in rolling 7-day window",
                "note": "no database available or schema missing"
            }
        }
```
