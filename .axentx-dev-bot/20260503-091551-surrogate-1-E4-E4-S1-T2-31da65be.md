### Implementation

**metrics.py**
```python
from datetime import datetime, timedelta
from typing import Dict, List

def calculate_wau(requests: List[Dict]) -> Dict:
    """
    Calculate WAU counts based on requests with at least one status update or view in the rolling 7-day window.

    Args:
        requests (List[Dict]): List of requests with their timestamps and status updates/views.

    Returns:
        Dict: WAU counts per team and total for the last 7 days.
    """
    wau_counts = {}
    current_date = datetime.now().date()
    seven_days_ago = current_date - timedelta(days=7)

    for request in requests:
        request_date = request['timestamp'].date()
        if seven_days_ago <= request_date <= current_date:
            team = request['team']
            if team not in wau_counts:
                wau_counts[team] = 0
            wau_counts[team] += 1

    return wau_counts

def get_wau_counts(requests: List[Dict]) -> Dict:
    """
    Get WAU counts for the last 7 days.

    Args:
        requests (List[Dict]): List of requests with their timestamps and status updates/views.

    Returns:
        Dict: WAU counts per team and total for the last 7 days.
    """
    wau_counts = calculate_wau(requests)
    total_wau = sum(wau_counts.values())
    return {'wau_counts': wau_counts, 'total_wau': total_wau}
```
