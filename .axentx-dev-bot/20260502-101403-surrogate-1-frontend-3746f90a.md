// cdnManifest.ts
// Deterministic CDN-only file loader for surrogate-1 datasets.
// Expects manifest at /static/file-list.json (or injected at build time).
// Each item: { path: string, size?: number, etag?: string }

const DATASET_OWNER = 'axentx';
const DATASET_REPO = 'surrogate-1';
const CDN_BASE = `https://huggingface.co/datasets/${DATASET_OWNER}/${DATASET_REPO}/resolve/main`;

export interface ManifestEntry {
  path: string;
  size?: number;
  etag?: string;
}

export interface CdnFile {
  url: string;
  path: string;
  size?: number;
  etag?: string;
}

export async function loadCdnManifest(manifestPath = '/static/file-list.json'): Promise<CdnFile[]> {
  const res = await fetch(manifestPath, { cache: 'no-store' });
  if (!res.ok) {
    throw new Error(`Failed to load manifest: ${res.status} ${res.statusText}`);
  }
  const list: ManifestEntry[] = await res.json();
  return list.map((entry) => ({
    ...entry,
    url: `${CDN_BASE}/${encodeURIComponent(entry.path).replace(/%2F/g, '/')}`,
  }));
}

// Lightweight streaming text fetch for a single file via CDN (no auth).
export async function fetchCdnTextFile(cdnFile: CdnFile, signal?: AbortSignal): Promise<string> {
  const res = await fetch(cdnFile.url, { signal, cache: 'no-store' });
  if (!res.ok) {
    throw new Error(`CDN fetch failed: ${res.status} ${res.statusText} for ${cdnFile.path}`);
  }
  return await res.text();
}

// Optional: stream JSON lines or NDJSON from a CDN file (memory-efficient).
export async function* streamCdnNdjson(
  cdnFile: CdnFile,
  signal?: AbortSignal
): AsyncGenerator<Record<string, unknown>, void, unknown> {
  const res = await fetch(cdnFile.url, { signal, cache: 'no-store' });
  if (!res.ok) {
    throw new Error(`CDN stream failed: ${res.status} ${res.statusText} for ${cdnFile.path}`);
  }
  if (!res.body) {
    throw new Error('ReadableStream not available');
  }

  const reader = res.body.getReader();
  const decoder = new TextDecoder();
  let buffer = '';

  while (true) {
    if (signal?.aborted) {
      reader.cancel();
      break;
    }
    const { done, value } = await reader.read();
    if (done) break;
    buffer += decoder.decode(value, { stream: true });

    let idx;
    while ((idx = buffer.indexOf('\n')) >= 0) {
      const line = buffer.slice(0, idx).trim();
      buffer = buffer.slice(idx + 1);
      if (!line) continue;
      try {
        yield JSON.parse(line);
      } catch {
        // skip malformed lines; caller can handle if needed
        continue;
      }
    }
  }

  // flush remainder
  if (buffer.trim()) {
    try {
      yield JSON.parse(buffer.trim());
    } catch {
      // ignore
    }
  }
}
