    for attempt in range(1, MAX_RETRIES + 1):
        try:
            resp = requests.get(url, headers=headers, timeout=TIMEOUT, **kwargs)
            if resp.status_code == 429:
                wait = RETRY_WAIT

                continue

            return resp
