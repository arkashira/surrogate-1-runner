const { sendEmailAlert } = require('./email');
const { sendSMSAlert } = require('./sms');

class AlertDelivery {

  }
}

```javascript
// /opt/axentx/surrogate-1/alerts/email.js
const nodemailer = require('nodemailer');

async function sendEmailAlert(alertData) {
  const transporter = nodemailer.createTransport({
    service: 'gmail',
    auth: {
      user: process.env.EMAIL_USER,
      pass: process.env.EMAIL_PASS
    }
  });

  const mailOptions = {
    from: 'cost.optimization@axentx.com',
    to: alertData.recipientEmail,
    subject: 'Cost Optimization Opportunity Identified',
    text: alertData.message
  };

  return transporter.sendMail(mailOptions);
}

module.exports = { sendEmailAlert };
```

```javascript
// /opt/axentx/surrogate-1/alerts/sms.js
const twilio = require('twilio');

async function sendSMSAlert(alertData) {
  const client = twilio(process.env.TWILIO_ACCOUNT_SID, process.env.TWILIO_AUTH_TOKEN);

  const message = await client.messages.create({
    body: alertData.message,
    from: process.env.TWILIO_PHONE_NUMBER,
    to: alertData.recipientPhone
  });

  return message.sid;
}

module.exports = { sendSMSAlert };
```
