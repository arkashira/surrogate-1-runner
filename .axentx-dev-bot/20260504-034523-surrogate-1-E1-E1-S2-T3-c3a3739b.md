class NotificationPreferencesServiceTest {

    @Mock
    NotificationPreferencesRepository repo;

    @InjectMocks
    NotificationPreferencesService service;

    @Test
    void getPreferences_returnsStoredValues() {
        when(repo.findByUserId(1L))
            .thenReturn(new NotificationPreferences(1L, true, false));

        NotificationPreferences prefs = service.getPreferences(1L);

        assertTrue(prefs.isEmailEnabled());
        assertFalse(prefs.isSlackEnabled());
    }

    @Test
    void updatePreferences_savesNewValues() {
        NotificationPreferences newPrefs = new NotificationPreferences(1L, false, true);
        service.updatePreferences(1L, newPrefs);

        verify(repo).save(argThat(p -> p.getUserId() == 1L
                                         && !p.isEmailEnabled()
                                         && p.isSlackEnabled()));
    }

    @Test
    void notifyNonCompliantModel_sendsEmailWhenEnabled() {
        NotificationPreferences prefs = new NotificationPreferences(1L, true, false);
        when(repo.findByUserId(1L)).thenReturn(prefs);

        service.notifyNonCompliantModel(1L, "model-123");

        verify(emailService).send(eq("admin@example.com"),
                                 eq("Non‑compliant model detected"),
                                 contains("model-123"));
        verify(slackClient, never()).postMessage(anyString(), anyString());
    }

    @Test
    void notifyNonCompliantModel_suppressesAllWhenDisabled() {
        NotificationPreferences prefs = new NotificationPreferences(1L, false, false);
        when(repo.findByUserId(1L)).thenReturn(prefs);

        service.notifyNonCompliantModel(1L, "model-123");

        verify(emailService, never()).send(anyString(), anyString(), anyString());
        verify(slackClient, never()).postMessage(anyString(), anyString());
    }
}
