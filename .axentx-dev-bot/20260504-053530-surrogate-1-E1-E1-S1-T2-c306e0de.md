from your_project.src.auth.claude_auth import ClaudeAuth

        self.config = {
            'claude': {
                'api_key': 'your_api_key_here',
                'api_secret': 'your_api_secret_here',
                'base_url': 'https://api.claude.ai'
            }
        }
        self.auth = ClaudeAuth(self.config)

    def test_init(self):
        self.assertEqual(self.auth.api_key, self.config['claude']['api_key'])
        self.assertEqual(self.auth.api_secret, self.config['claude']['api_secret'])
        self.assertEqual(self.auth.base_url, self.config['claude']['base_url'])

    @patch('requests.post')
    def test_authenticate_success(self, mock_post):
        mock_response = MagicMock(status_code=200)
        mock_post.return_value = mock_response
        result = self.auth.authenticate()
        self.assertTrue(result)
        mock_post.assert_called_once_with(self.auth.base_url + '/authenticate', headers={'Authorization': f'Bearer {self.auth.api_key}'})

    @patch('requests.post')
    def test_authenticate_failure(self, mock_post):
        mock_response = MagicMock(status_code=401)
        mock_post.return_value = mock_response
        result = self.auth.authenticate()
        self.assertFalse(result)
        mock_post.assert_called_once_with(self.auth.base_url + '/authenticate', headers={'Authorization': f'Bearer {self.auth.api_key}'})

    def test_log_request(self):
        self.auth.log_request('GET', '/test')
        self.assertEqual(self.auth.audit_log, ['GET /test'])

    def test_log_failure(self):
        self.auth.log_failure('Authentication failed')
        self.assertEqual(self.auth.audit_log, ['Authentication failed'])

if __name__ == '__main__':
    unittest.main()
