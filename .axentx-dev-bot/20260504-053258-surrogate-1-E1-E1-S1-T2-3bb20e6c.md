"""

import json
import logging
import os
import time
from dataclasses import dataclass, asdict
from datetime import datetime
from typing import Any, Dict, List, Optional

import confluent_kafka as kafka
from confluent_kafka import Producer, KafkaError

logger = logging.getLogger(__name__)


@dataclass
class CostRecord:
    """Represents a single GCP billing cost record."""
    record_id: str
    billing_account_id: str
    project_id: str
    cost_center: str
    cost_type: str
    currency: str
    amount: float
    timestamp: str
    resource_name: str
    labels: Dict[str, str]


class KafkaProducer:
    """
    High-throughput Kafka producer for cost data buffering.
    
    Features:
    - Batched message production for throughput optimization
    - Exponential backoff retry on failures
    - Dead letter queue for failed records
    - Metrics tracking for monitoring
    """

    def __init__(
        self,
        bootstrap_servers: str = "localhost:9092",
        topic: str = "axentx-cost-data",
        batch_size: int = 1000,
        linger_ms: int = 10,
        max_in_flight_requests: int = 5,
        retry_max_attempts: int = 5,
        retry_base_backoff_ms: int = 100,
        retry_max_backoff_ms: int = 30000,
        dead_letter_topic: Optional[str] = None,
    ):
        """
        Initialize the Kafka producer.

        Args:
            bootstrap_servers: Kafka broker addresses
            topic: Primary topic for cost data
            batch_size: Number of messages to batch before sending
            linger_ms: Time to wait for batch to fill
            max_in_flight_requests: Max concurrent requests per broker
            retry_max_attempts: Max retry attempts on failure
            retry_base_backoff_ms: Base backoff time in milliseconds
            retry_max_backoff_ms: Maximum backoff time in milliseconds
            dead_letter_topic: Topic for failed records
        """
        self.bootstrap_servers = bootstrap_servers
        self.topic = topic
        self.dead_letter_topic = dead_letter_topic or f"{topic}-dlq"
        self.batch_size = batch_size
        self.linger_ms = linger_ms
        self.max_in_flight_requests = max_in_flight_requests
        self.retry_max_attempts = retry_max_attempts
        self.retry_base_backoff_ms = retry_base_backoff_ms
        self.retry_max_backoff_ms = retry_max_backoff_ms

        # Producer configuration
        self.producer_config = {

            "message.max.bytes": 10485760,  # 10MB max message
