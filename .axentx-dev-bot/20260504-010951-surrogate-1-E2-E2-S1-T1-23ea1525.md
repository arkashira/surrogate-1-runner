## 2. Unit Tests (Jest + React Testing Library)

```js
// NavBar.test.jsx
import { render, screen, waitFor, fireEvent } from '@testing-library/react';
import NavBar from './NavBar';
import { BrowserRouter } from 'react-router-dom';
import fetchMock from 'jest-fetch-mock';

beforeEach(() => fetchMock.resetMocks());

const mockSiteMap = {
  items: [
    { id: 1, title: 'Home', path: '/' },
    { id: 2, title: 'Courses', path: '/courses' },
    { id: 3, title: 'Profile', path: '/profile' },
  ],
};

const renderNavBar = () =>
  render(
    <BrowserRouter>
      <NavBar />
    </BrowserRouter>
  );

test('renders NavBar on every page', () => {
  renderNavBar();
  expect(screen.getByRole('navigation')).toBeInTheDocument();
});

test('fetches site map and displays links', async () => {
  fetchMock.mockResponseOnce(JSON.stringify(mockSiteMap));

  renderNavBar();

  await waitFor(() => {
    expect(screen.getByText('Home')).toBeInTheDocument();
    expect(screen.getByText('Courses')).toBeInTheDocument();
    expect(screen.getByText('Profile')).toBeInTheDocument();
  });
});

test('links use client-side navigation', async () => {
  fetchMock.mockResponseOnce(JSON.stringify(mockSiteMap));

  renderNavBar();

  await waitFor(() => screen.getByText('Courses'));

  const coursesLink = screen.getByText('Courses').closest('a');
  expect(coursesLink).toHaveAttribute('href', '/courses');
});

test('shows fallback on API error', async () => {
  fetchMock.mockRejectOnce(new Error('Network error'));

  renderNavBar();

  await waitFor(() =>
    expect(screen.getByText('Navigation unavailable')).toBeInTheDocument()
  );
});

test('hamburger menu toggles on small screens', async () => {
  fetchMock.mockResponseOnce(JSON.stringify(mockSiteMap));

  // Mock window width
  global.innerWidth = 500;
  global.dispatchEvent(new Event('resize'));

  renderNavBar();

  const hamburger = await screen.findByLabelText('Toggle navigation');
  fireEvent.click(hamburger);

  expect(screen.getByText('Home')).toBeVisible();
});
```
