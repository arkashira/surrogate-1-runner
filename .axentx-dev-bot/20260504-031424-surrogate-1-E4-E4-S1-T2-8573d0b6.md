# src/pages/JobBoard.jsx
```jsx
import React, { useEffect, useState } from 'react';
import axios from 'axios';
import CredentialCard from '../components/CredentialCard';

const JobBoard = () => {
  const [credentials, setCredentials] = useState([]);
  const [filters, setFilters] = useState({ skillArea: '', location: '', experience: '' });

  useEffect(() => {
    const fetchCredentials = async () => {
      const response = await axios.get('/api/credentials', { params: filters });
      setCredentials(response.data);
    };

    fetchCredentials();
  }, [filters]);

  const handleFilterChange = (e) => {
    setFilters({ ...filters, [e.target.name]: e.target.value });
  };

  return (
    <div>
      <div>
        <input type="text" name="skillArea" placeholder="Skill Area" onChange={handleFilterChange} />
        <input type="text" name="location" placeholder="Location" onChange={handleFilterChange} />
        <input type="text" name="experience" placeholder="Experience" onChange={handleFilterChange} />
      </div>
      <div>
        {credentials.map((credential) => (
          <CredentialCard key={credential.id} credential={credential} />
        ))}
      </div>
    </div>
  );
};

export default JobBoard;
```
