import { v4 as uuidv4 } from 'uuid';

export type RequestType = 'incident' | 'change' | 'question' | 'task';
export type RequestStatus = 'open' | 'in_progress' | 'resolved' | 'closed';

export interface CreateRequestPayload {
  type: RequestType;
  title: string;
  owner: string;
  sla_target: string; // ISO-8601 datetime string (future)
  team?: string;
}

export interface Request extends CreateRequestPayload {
  id: string;
  created_at: string; // ISO-8601
  current_status: RequestStatus;
  public_url: string;
}

const VALID_TYPES = new Set<RequestType>(['incident', 'change', 'question', 'task']);

function isISO8601Future(value: string): boolean {
  const dt = new Date(value);
  return !isNaN(dt.getTime()) && dt > new Date();
}

export function validateCreateRequest(payload: any): { valid: true; value: CreateRequestPayload } | { valid: false; errors: string[] } {
  const errors: string[] = [];

  if (!payload || typeof payload !== 'object' || Array.isArray(payload)) {
    return { valid: false, errors: ['Invalid payload'] };
  }

  const { type, title, owner, sla_target, team } = payload;

  if (!type || typeof type !== 'string' || !VALID_TYPES.has(type as RequestType)) {
    errors.push('Invalid or missing "type". Must be one of: incident, change, question, task.');
  }

  if (!title || typeof title !== 'string' || title.trim().length === 0) {
    errors.push('Missing or invalid "title". Must be a non-empty string.');
  }

  if (!owner || typeof owner !== 'string' || owner.trim().length === 0) {
    errors.push('Missing or invalid "owner". Must be a non-empty string.');
  }

  if (!sla_target || typeof sla_target !== 'string' || !isISO8601Future(sla_target)) {
    errors.push('Invalid or missing "sla_target". Must be a future ISO-8601 datetime string.');
  }

  if (team !== undefined && (typeof team !== 'string' || team.trim().length === 0)) {
    errors.push('Invalid "team". Must be a non-empty string if provided.');
  }

  if (errors.length > 0) {
    return { valid: false, errors };
  }

  return {
    valid: true,
    value: {
      type: type as RequestType,
      title: title.trim(),
      owner: owner.trim(),
      sla_target: sla_target.trim(),
      team: team ? team.trim() : undefined,
    },
  };
}

export function createRequest(input: CreateRequestPayload): Request {
  const now = new Date().toISOString();
  const id = uuidv4();
  return {
    id,
    type: input.type,
    title: input.title,
    owner: input.owner,
    sla_target: input.sla_target,
    team: input.team,
    created_at: now,
    current_status: 'open',
    public_url: `/requests/${id}`,
  };
}
