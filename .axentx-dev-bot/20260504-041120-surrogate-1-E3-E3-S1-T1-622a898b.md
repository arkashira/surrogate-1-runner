import React, { useEffect, useState } from "react";
import PropTypes from "prop-types";
import "./CreditCard.css";

/**
 * CreditCard component displays monthly and bulk credit usage.
 *
 * Props
 * -----
 * endpoint: string
 *   API endpoint that returns a JSON object shaped like:
 *   {
 *     monthly: { total: number, used: number },
 *     bulk:    { total: number, used: number }
 *   }
 *
 * The component fetches the data on mount and then every 60 seconds.
 * It shows progress bars, percentages and a warning banner when bulk
 * credits have less than 20 % remaining.
 */
export default function CreditCard({ endpoint }) {
  const [data, setData] = useState({
    monthly: { total: 0, used: 0 },
    bulk: { total: 0, used: 0 },
  });
  const [error, setError] = useState(null);

  // -----------------------------------------------------------------
  // Data fetching
  // -----------------------------------------------------------------
  const fetchData = async () => {
    try {
      const resp = await fetch(endpoint, { cache: "no-store" });
      if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
      const json = await resp.json();
      setData({
        monthly: json.monthly ?? { total: 0, used: 0 },
        bulk: json.bulk ?? { total: 0, used: 0 },
      });
      setError(null);
    } catch (e) {
      setError(e.message);
    }
  };

  // -----------------------------------------------------------------
  // Effect: initial load + 60‑second polling
  // -----------------------------------------------------------------
  useEffect(() => {
    fetchData(); // initial
    const id = setInterval(fetchData, 60_000);
    return () => clearInterval(id);
  }, [endpoint]);

  };

  const bulk = calc(data.bulk.total, data.bulk.used);
