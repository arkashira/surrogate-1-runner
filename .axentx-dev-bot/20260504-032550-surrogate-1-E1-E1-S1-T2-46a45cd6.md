    p.write_text(json.dumps(data))
    return p

def test_load_templates_success(templates_path):
    # Temporarily patch the path used by load_templates
    os.environ["DATADOG_TEMPLATE_PATH"] = str(templates_path)
    templates = load_templates()
    assert "metrics_response" in templates
    assert templates["metrics_response"] == {"status": "ok"}

def test_load_templates_missing_file(tmp_path):
    os.environ["DATADOG_TEMPLATE_PATH"] = str(tmp_path / "missing.json")
    with pytest.raises(RuntimeError, match="synthetic_data.json not found"):
        load_templates()

    os.environ["DATADOG_TEMPLATE_PATH"] = str(p)
