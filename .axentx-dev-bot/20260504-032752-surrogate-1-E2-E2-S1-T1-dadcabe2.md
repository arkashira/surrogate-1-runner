#!/usr/bin/env python3
"""
Charge threshold controller for laptops.

Features
--------
* Read/write the kernel sysfs value
  /sys/class/power_supply/BAT0/charge_stop_threshold
* Persist the chosen threshold in /etc/axentx/charge_control.conf
* Apply the threshold automatically when the charger is plugged in
* Run as a short‑lived CLI tool or as a long‑running daemon
* Unit‑testable – all I/O is wrapped in small helper functions
"""

from __future__ import annotations

import argparse
import configparser
import os
import sys
import time
from pathlib import Path
from typing import Optional

# --------------------------------------------------------------------------- #
#  Configuration & constants
# --------------------------------------------------------------------------- #

CONFIG_PATH = Path("/etc/axentx/charge_control.conf")
BAT_PATH = Path("/sys/class/power_supply/BAT0")
CHARGE_STOP_PATH = BAT_PATH / "charge_stop_threshold"
STATUS_PATH = BAT_PATH / "status"

DEFAULT_THRESHOLD = 80
MIN_THRESHOLD = 50
MAX_THRESHOLD = 100

# --------------------------------------------------------------------------- #
#  Helper functions – these are the only places that touch the filesystem
# --------------------------------------------------------------------------- #

def read_config() -> int:
    """Return the persisted threshold, or DEFAULT_THRESHOLD if missing."""
    if not CONFIG_PATH.exists():
        return DEFAULT_THRESHOLD

    cfg = configparser.ConfigParser()
    cfg.read(CONFIG_PATH)
    try:
        return cfg.getint("settings", "threshold")
    except (configparser.Error, ValueError):
        return DEFAULT_THRESHOLD


def write_config(threshold: int) -> None:
    """Persist the threshold to disk."""
    CONFIG_PATH.parent.mkdir(parents=True, exist_ok=True)
    cfg = configparser.ConfigParser()
    cfg["settings"] = {"threshold": str(threshold)}
    with CONFIG_PATH.open("w") as f:
        cfg.write(f)


def read_sysfs(path: Path) -> str:
    """Read a single line from a sysfs file."""
    with path.open("r") as f:
        return f.read().strip()


def write_sysfs(path: Path, value: str) -> None:
    """Write a string to a sysfs file."""
    with path.open("w") as f:
        f.write(value)
