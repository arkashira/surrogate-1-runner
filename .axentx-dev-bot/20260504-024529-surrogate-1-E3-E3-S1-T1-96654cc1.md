import json
import logging
import os
import smtplib
import traceback
from email.message import EmailMessage
from typing import Any, Dict, List, Optional
from datetime import datetime

# --------------------------------------------------------------------------- #
# Configuration loading
# --------------------------------------------------------------------------- #
CONFIG_PATH = os.path.join(
    os.path.dirname(__file__), "..", "config", "error_notification_config.json"
)

def _load_config() -> Dict[str, Any]:
    """Load the JSON configuration for error notifications."""
    try:
        with open(CONFIG_PATH, "r", encoding="utf-8") as f:
            return json.load(f)
    except FileNotFoundError:
        # Default configuration – notifications disabled
        return {
            "enabled": False,
            "method": "email",
            "recipients": [],
            "smtp": {
                "host": "localhost",
                "port": 25,
                "use_tls": False,
                "username": None,
                "password": None,
                "sender": "surrogate-1@axentx.com"
            },
            "log_path": os.path.join(
                os.path.dirname(__file__), "..", "logs", "error.log"
            )
        }

_config = _load_config()

# --------------------------------------------------------------------------- #
# Logger setup
# --------------------------------------------------------------------------- #
log_path = _config.get("log_path", "error.log")
os.makedirs(os.path.dirname(log_path), exist_ok=True)

_logger = logging.getLogger("surrogate1.error")
_logger.setLevel(logging.ERROR)

# Add file handler
_handler = logging.FileHandler(log_path, encoding="utf-8")
_formatter = logging.Formatter(
    "%(asctime)s - %(levelname)s - %(name)s - %(message)s"
)
_handler.setFormatter(_formatter)
_logger.addHandler(_handler)

# Add console handler for development
console_handler = logging.StreamHandler()
console_handler.setFormatter(_formatter)
_logger.addHandler(console_handler)

# --------------------------------------------------------------------------- #
# Email helper with retry logic
# --------------------------------------------------------------------------- #
def _send_email(subject: str, body: str, recipients: List[str], retries: int = 3) -> bool:
    """Send an email using the SMTP configuration with retry logic."""
    smtp_cfg = _config.get("smtp", {})
    msg = EmailMessage()
    msg["Subject"] = subject
    msg["From"] = smtp_cfg.get("sender", "surrogate-1@axentx.com")
    msg["To"] = ", ".join(recipients)
    msg.set_content(body)
    
    last_error = None
    for attempt in range(retries):
        try:
            server = smtplib.SMTP(
                smtp_cfg.get("host", "localhost"), 
                smtp_cfg.get("port", 25)
            )
            try:
                if smtp_cfg.get("use_tls"):
                    server.starttls()
                
                if smtp_cfg.get("username"):
                    server.login(smtp_cfg["username"], smtp_cfg["password"] or "")
                
                server.send_message(msg)
                _logger.info(f"Email sent successfully to {recipients}")
                return True
            finally:
                server.quit()
        except Exception as e:
            last_error = e
            _logger.warning(f"Email send attempt {attempt + 1} failed: {e}")
            if attempt < retries - 1:
                import time
                time.sleep(2 ** attempt)  # Exponential backoff
    
    _logger.error(f"Failed to send email after {retries} attempts: {last_error}")
    return False
