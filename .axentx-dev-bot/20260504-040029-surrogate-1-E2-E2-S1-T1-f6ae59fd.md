## 1. Acceptance Criteria  
- **CLI Invocation** – Running `node src/cli.js` without arguments prints a usage message and exits with code `0`.  
- **Help Flag** – `node src/cli.js --help` displays the same usage message and exits with code `0`.  
- **Version Flag** – `node src/cli.js --version` prints the current package version and exits with code `0`.  
- **Valid Path** – Providing a valid directory path (`node src/cli.js ./src`) runs the import‑verification engine and outputs results in a human‑readable table.  
- **Invalid Path** – Supplying a non‑existent path prints an error message to stderr and exits with code `1`.  
- **Result Format** – The output table contains at least the columns `File`, `Missing Imports`, and `Total Imports`.  
- **Exit Codes** – Successful run exits with `0`; any internal error exits with `1`.

// tests/cli.unit.test.js
const { runCli } = require('../src/cli');
const { spawnSync } = require('child_process');
const path = require('path');

describe('CLI', () => {
  const cliPath = path.resolve(__dirname, '../src/cli.js');

  test('shows usage when no args', () => {
    const result = spawnSync('node', [cliPath], { encoding: 'utf8' });
    expect(result.stdout).toMatch(/Usage:/);
    expect(result.status).toBe(0);
  });

  test('--help shows usage', () => {
    const result = spawnSync('node', [cliPath, '--help'], { encoding: 'utf8' });
    expect(result.stdout).toMatch(/Usage:/);
    expect(result.status).toBe(0);
  });

  test('--version prints package version', () => {
    const pkg = require('../package.json');
    const result = spawnSync('node', [cliPath, '--version'], { encoding: 'utf8' });
    expect(result.stdout.trim()).toBe(pkg.version);
    expect(result.status).toBe(0);
  });

  test('valid path runs engine', () => {
    const result = spawnSync('node', [cliPath, './src'], { encoding: 'utf8' });
    expect(result.status).toBe(0);
    expect(result.stdout).toMatch(/File\s+Missing Imports\s+Total Imports/);
  });

  test('invalid path errors', () => {
    const result = spawnSync('node', [cliPath, './nonexistent'], { encoding: 'utf8' });
    expect(result.status).toBe(1);
    expect(result.stderr).toMatch(/not found/i);
  });
});
