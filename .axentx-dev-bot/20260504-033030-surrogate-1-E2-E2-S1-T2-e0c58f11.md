 * Validate service provider data
 */
function validateServiceProvider(data, isUpdate = false) {
  const errors = [];
  const required = isUpdate 
    ? serviceProviderSchema.required.filter(f => f !== 'email') // email not required for update
    : serviceProviderSchema.required;
  
  for (const field of required) {
    if (!data[field] || (typeof data[field] === 'string' && !data[field].trim())) {
      errors.push(`Missing required field: ${field}`);
    }
  }
  
  if (data.email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(data.email)) {
    errors.push('Invalid email format');
  }
  
  if (data.serviceType && !SERVICE_TYPES.includes(data.serviceType)) {
    errors.push(`Invalid serviceType. Must be one of: ${SERVICE_TYPES.join(', ')}`);
  }
  
  if (data.website && !/^https?:\/\/.+/.test(data.website)) {
    errors.push('Invalid website URL');
  }
  
  if (data.hourlyRate !== undefined && data.hourlyRate !== null) {
    const rate = parseFloat(data.hourlyRate);
    if (isNaN(rate) || rate < 0) {
      errors.push('hourlyRate must be a positive number');
    }
  }
  
  return { valid: errors.length === 0, errors };
}
