# test_lab_template_structure.py
import json
import os
import pytest

TEMPLATE_DIR = "/opt/axentx/surrogate-1/lab-templates"

def load_template(name):
    with open(os.path.join(TEMPLATE_DIR, name), "r") as f:
        return json.load(f)

@pytest.mark.parametrize("filename", [
    "dvac02-iam.json",
    "dvac02-vpc.json",
    "dvac02-s3.json",
    "dvac02-rds.json",
    "dvac02-lambda.json",
])
def test_template_exists_and_valid_json(filename):
    assert os.path.exists(os.path.join(TEMPLATE_DIR, filename))
    data = load_template(filename)
    assert isinstance(data, dict)

def test_iam_template_has_role_and_policy():
    data = load_template("dvac02-iam.json")
    resources = data.get("Resources", {})
    assert "DVAC02_IAM_TestRole" in resources
    role = resources["DVAC02_IAM_TestRole"]
    assert role["Type"] == "AWS::IAM::Role"
    assert "AssumeRolePolicyDocument" in role["Properties"]

def test_vpc_template_has_cidr_and_subnet():
    data = load_template("dvac02-vpc.json")
    resources = data.get("Resources", {})
    vpc = resources.get("DVAC02_VPC")
    assert vpc["Type"] == "AWS::EC2::VPC"
    assert vpc["Properties"]["CidrBlock"] == "10.0.0.0/16"
    subnets = [r for r in resources.values() if r["Type"] == "AWS::EC2::Subnet"]
    assert len(subnets) >= 1
