import { calculateCompletion, mapApiResponseToFeedback } from '@/progress/tracker';
import { fetchProgress, fetchFeedback } from '@/api';

describe('Progress Tracker Unit Tests', () => {
  test('calculateCompletion returns 0% when no items are completed', () => {
    const items = [{id:1, completed:false}, {id:2, completed:false}];
    expect(calculateCompletion(items)).toBe(0);
  });

  test('calculateCompletion returns correct percentage', () => {
    const items = [{id:1, completed:true}, {id:2, completed:false}, {id:3, completed:true}];
    expect(calculateCompletion(items)).toBeCloseTo(66.67, 2);
  });

  test('mapApiResponseToFeedback extracts recommendations and suggestions', () => {
    const apiResp = {
      recommendations: ['Read more articles on grammar'],
      suggestions: ['Practice speaking with a partner']
    };
    const feedback = mapApiResponseToFeedback(apiResp);
    expect(feedback.recommendations).toContain('Read more articles on grammar');
    expect(feedback.suggestions).toContain('Practice speaking with a partner');
  });

  test('fetchProgress resolves with stored progress data', async () => {
    jest.spyOn(global, 'fetch').mockResolvedValue({
      ok: true,
      json: async () => ({ videos: 3, articles: 2 })
    });
    const data = await fetchProgress('user123');
    expect(data).toEqual({ videos: 3, articles: 2 });
  });

  test('fetchFeedback returns empty payload on 204 No Content', async () => {
    jest.spyOn(global, 'fetch').mockResolvedValue({ ok: true, status: 204 });
    const feedback = await fetchFeedback('user123');
    expect(feedback).toEqual({ recommendations: [], suggestions: [] });
  });

  test('fetchProgress throws on network error', async () => {
    jest.spyOn(global, 'fetch').mockRejectedValue(new Error('Network error'));
    await expect(fetchProgress('user123')).rejects.toThrow('Network error');
  });
});
