```javascript
# /opt/axentx/surrogate-1/web/src/components/TaskItem.js
import React from 'react';
import { Button, StatusIndicator } from '@axentx/ui-components';

const TaskItem = ({ task, updateTaskStatus }) => {
  const handleStatusUpdate = (newStatus) => {
    updateTaskStatus(task.id, newStatus);
  };

  return (
    <div>
      <h2>{task.title}</h2>
      <p>{task.description}</p>
      <StatusIndicator status={task.status} />
      {task.status === 'pending' && (
        <Button onClick={() => handleStatusUpdate('in-progress')}>Start</Button>
      )}
      {task.status === 'in-progress' && (
        <Button onClick={() => handleStatusUpdate('done')}>Finish</Button>
      )}
    </div>
  );
};

export default TaskItem;
```

```javascript
class Task {
  constructor(id, title, description, status) {
    this.id = id;
    this.title = title;
    this.description = description;
    this.status = status;
  }

  updateStatus(newStatus) {
    this.status = newStatus;
  }
}
```

```javascript
# /opt/axentx/surrogate-1/web/src/api/taskApi.js
import axios from 'axios';

const updateTaskStatus = async (taskId, newStatus) => {
  try {
    const response = await axios.patch(`/api/tasks/${taskId}`, { status: newStatus });
    return response.data;
  } catch (error) {
    console.error(error);
  }
};

export default updateTaskStatus;
```

```javascript
// TaskItem.test.js
import React from 'react';
import { render, fireEvent } from '@testing-library/react';
import TaskItem from './TaskItem';

describe('TaskItem', () => {
  it('should render the status update button', () => {
    const { getByText } = render(<TaskItem task={{ status: 'pending' }} />);
    expect(getByText('Update Status')).toBeInTheDocument();
  });

  it('should update the task status to "in-progress" when the button is clicked', () => {
    const mockUpdateStatus = jest.fn();
    const { getByText } = render(<TaskItem task={{ status: 'pending' }} updateStatus={mockUpdateStatus} />);
    fireEvent.click(getByText('Update Status'));
    expect(mockUpdateStatus).toHaveBeenCalledWith('in-progress');
  });

  it('should update the task status to "done" when the button is clicked twice', () => {
    const mockUpdateStatus = jest.fn();
    const { getByText } = render(<TaskItem task={{ status: 'pending' }} updateStatus={mockUpdateStatus} />);
    fireEvent.click(getByText('Update Status'));
    fireEvent.click(getByText('Update Status'));
    expect(mockUpdateStatus).toHaveBeenCalledWith('done');
  });

  it('should disable the status update button for tasks that are already "done"', () => {
    const { getByText } = render(<TaskItem task={{ status: 'done' }} />);
    expect(getByText('Update Status')).toBeDisabled();
  });
});
```
