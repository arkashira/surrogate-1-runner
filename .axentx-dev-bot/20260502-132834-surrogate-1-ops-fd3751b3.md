### Phase 1: Generate Manifest Once (20 min)
Run this on the Mac orchestrator after any rate-limit window clears. It lists files non-recursively per date folder and produces a JSON manifest used by all shards.

```bash
#!/usr/bin/env bash
# bin/generate-manifest.sh
# Usage: ./generate-manifest.sh <date> [output.json]

set -euo pipefail

DATE="${1:-$(date +%Y-%m-%d)}"
OUTPUT="${2:-manifest-${DATE}.json}"
REPO="axentx/surrogate-1-training-pairs"

python3 -c "
import json, sys
from huggingface_hub import HfApi

api = HfApi()
repo_id = '${REPO}'
base_path = 'public-merged/${DATE}'

try:
    # Non-recursive list to find subfolders
    top = api.list_repo_tree(repo_id, path=base_path, recursive=False)
    files = []
    for item in top:
        if item.type == 'directory':
            sub = api.list_repo_tree(repo_id, path=item.path, recursive=True)
            for f in sub:
                if f.type == 'file' and f.path.endswith(('.parquet', '.jsonl', '.json')):
                    files.append({
                        'path': f.path,
                        'size': f.size,
                        'lfs': bool(getattr(f, 'lfs', None))
                    })
    manifest = {
        'date': '${DATE}',
        'repo': repo_id,
        'total_files': len(files),
        'files': files
    }
    with open('${OUTPUT}', 'w') as f:
        json.dump(manifest, f, indent=2)
    print(f'✅ {len(files)} files -> ${OUTPUT}')
except Exception as e:
    print(f'❌ {e}')
    sys.exit(1)
"
```

HF_REPO="axentx/surrogate-1-training-pairs"

[[ -f "${MANIFEST}" ]] || { echo "❌ Manifest not found: ${MANIFEST}"; exit 1; }

download_via_cdn() {
    local file_path="$1"
    local output_file="$2"
    local url="https://huggingface.co/datasets/${HF_REPO}/resolve/main/${file_path}"
    local max_retries=5 retry=0 wait_time=1

    while (( retry < max_retries )); do
        if curl -fsSL --retry 3 --retry-delay 2 \
             -H "User-Agent: axentx-surrogate-1/1.0" \
             -o "${output_file}" "${url}"; then
            return 0
        fi
        local code=$(curl -s -o /dev/null -w "%{http_code}" "${url}" || echo "000")
        if [[ "${code}" == "429" ]]; then
            echo "⚠️  429 rate-limited. Waiting ${wait_time}s..."
            sleep $wait_time
            ((wait_time *= 2))
            ((wait_time > 360)) && wait_time=360
            ((retry++))
        else
            sleep $wait_time
            ((retry++))
        fi
    done
    echo "❌ Failed ${file_path} after ${max_retries} retries"
    return 1
}
