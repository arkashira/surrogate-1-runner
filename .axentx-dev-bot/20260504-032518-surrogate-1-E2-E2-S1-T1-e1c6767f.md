<dependencies>

    </dependency>

    <!-- Testcontainers core -->
    <dependency>
        <groupId>org.testcontainers</groupId>

        <version>${testcontainers.version}</version>
        <scope>test</scope>
    </dependency>

    <!-- WireMock image for Testcontainers -->
    <dependency>
        <groupId>org.testcontainers</groupId>
        <artifactId>wiremock</artifactId>
        <version>${testcontainers.version}</version>
        <scope>test</scope>
    </dependency>

    <!-- WireMock Java API (used to build stubs) -->
    <dependency>
        <groupId>com.github.tomakehurst</groupId>
        <artifactId>wiremock-jre8</artifactId>
        <version>${wiremock.version}</version>
        <scope>test</scope>
    </dependency>

    <!-- JUnit 5 -->
    <dependency>
        <groupId>org.junit.jupiter</groupId>
        <artifactId>junit-jupiter</artifactId>
        <version>5.10.0</version>
        <scope>test</scope>
    </dependency>
</dependencies>
```

> **Tip** – If you’re using Spring Boot’s test starter (`spring-boot-starter-test`), it already pulls in JUnit 5 and Testcontainers, so you can drop the explicit JUnit dependency.
