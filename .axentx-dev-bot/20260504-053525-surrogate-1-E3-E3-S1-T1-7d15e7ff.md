package admin

import (
	"context"
	"encoding/json"
	"net/http"

	"github.com/axentx/surrogate-1/pkg/api"
	"github.com/axentx/surrogate-1/pkg/metrics"
	"github.com/axentx/surrogate-1/pkg/sso"
)

// MetricsResponse represents the response from the metrics API endpoint
type MetricsResponse struct {
	TotalRequests int `json:"total_requests"`
	SuccessRate   float64 `json:"success_rate"`
	TunnelUptime  float64 `json:"tunnel_uptime"`
}

// GetMetrics returns the metrics for the admin dashboard
func GetMetrics(ctx context.Context, r *http.Request) (*MetricsResponse, error) {
	// Get the SSO token from the request
	ssoToken, err := sso.GetSSOToken(ctx, r)
	if err != nil {
		return nil, err
	}

	// Get the metrics from the metrics package
	metricsData, err := metrics.GetMetrics()
	if err != nil {
		return nil, err
	}

	// Calculate the success rate and tunnel uptime
	successRate := metricsData.SuccessRate()
	tunnelUptime := metricsData.TunnelUptime()

	// Create the response
	response := &MetricsResponse{
		TotalRequests: metricsData.TotalRequests(),
		SuccessRate:   successRate,
		TunnelUptime:  tunnelUptime,
	}

	return response, nil
}

// MetricsHandler handles the GET request to the metrics API endpoint
func MetricsHandler(ctx context.Context, r *http.Request) (int, interface{}, error) {
	response, err := GetMetrics(ctx, r)
	if err != nil {
		return http.StatusInternalServerError, nil, err
	}

	return http.StatusOK, response, nil
}
