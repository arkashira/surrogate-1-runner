from __future__ import annotations

from dataclasses import dataclass
from decimal import Decimal
from typing import Dict, Optional, Tuple
import time


@dataclass(frozen=True)
class TradeContext:
    trade_id: str
    pipeline_id: str
    symbol: str
    side: str          # "buy" | "sell"
    price: Decimal
    size: Decimal
    timestamp: int     # epoch ms


@dataclass(frozen=True)
class RiskPolicy:
    max_size: Optional[Decimal] = None
    price_band_pct: Optional[Decimal] = None   # e.g. 0.02 for +/-2%
    limit_price: Optional[Decimal] = None      # hard cap for buys / floor for sells
    reference_price: Optional[Decimal] = None  # mark for band checks

    @staticmethod
    def from_dict(d: Dict) -> "RiskPolicy":
        def opt(key: str) -> Optional[Decimal]:
            v = d.get(key)
            return Decimal(str(v)) if v is not None else None

        return RiskPolicy(
            max_size=opt("max_size"),
            price_band_pct=opt("price_band_pct"),
            limit_price=opt("limit_price"),
            reference_price=opt("reference_price"),
        )


def check(trade_context: TradeContext, policy: RiskPolicy) -> Tuple[bool, Optional[str]]:
    """
    Deterministic risk checks.
    Returns (pass, reason) where reason is None on pass.
    Designed to complete well under 50 ms.
    """
    t0 = time.perf_counter()

    # 1) Max size
    if policy.max_size is not None and trade_context.size > policy.max_size:
        return False, f"size_violation: {trade_context.size} > {policy.max_size}"

    # 2) Price band
    if policy.price_band_pct is not None and policy.reference_price is not None:
        band = policy.reference_price * policy.price_band_pct
        lower = policy.reference_price - band
        upper = policy.reference_price + band
        if not (lower <= trade_context.price <= upper):
            return (
                False,
                f"price_band_violation: {trade_context.price} not in [{lower}, {upper}]",
            )

    # 3) Limit price
    if policy.limit_price is not None:
        if trade_context.side == "buy" and trade_context.price > policy.limit_price:
            return False, f"limit_violation: buy price {trade_context.price} > limit {policy.limit_price}"
        if trade_context.side == "sell" and trade_context.price < policy.limit_price:
            return False, f"limit_violation: sell price {trade_context.price} < limit {policy.limit_price}"

    # Defensive SLA monitoring (non-blocking)
    elapsed_ms = (time.perf_counter() - t0) * 1000
    if elapsed_ms >= 50:

        pass

    return True, None
