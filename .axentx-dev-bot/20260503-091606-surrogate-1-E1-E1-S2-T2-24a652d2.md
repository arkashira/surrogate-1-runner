from fastapi import APIRouter, HTTPException, status
from sqlalchemy.orm import Session
from datetime import datetime, timezone
from src.schemas.request import RequestStatusUpdate
from src.models import Request as RequestModel  # adjust import path as needed

router = APIRouter()

# Valid transitions map: current -> allowed next states
VALID_TRANSITIONS = {
    "open": {"in_progress", "closed"},
    "in_progress": {"resolved", "closed"},
    "resolved": {"closed"},
    "closed": set(),  # no outgoing transitions allowed
}

def recompute_sla_breached(request: RequestModel) -> bool:
    """
    Recompute SLA breached flag based on status history and timestamps.
    Example heuristic: if in_progress -> resolved took > 48h, breached = True.
    Adjust to match your SLA rules.
    """
    history = request.status_history or []
    in_progress_ts = None
    resolved_ts = None

    for entry in history:
        if entry.get("status") == "in_progress" and entry.get("transitioned_at"):
            in_progress_ts = entry["transitioned_at"]
        if entry.get("status") == "resolved" and entry.get("transitioned_at"):
            resolved_ts = entry["transitioned_at"]

    if in_progress_ts and resolved_ts:
        duration_hours = (resolved_ts - in_progress_ts).total_seconds() / 3600
        return duration_hours > 48  # example SLA: 48h to resolve
    return False

@router.patch("/requests/{request_id}", response_model=dict)
def update_request_status(request_id: str, update: RequestStatusUpdate, db: Session):
    req = db.query(RequestModel).filter(RequestModel.id == request_id).first()
    if not req:
        raise HTTPException(status_code=404, detail="Request not found")

    current = req.status
    next_status = update.status

    # Validate transition
    allowed = VALID_TRANSITIONS.get(current, set())
    if next_status not in allowed:
        raise HTTPException(
            status_code=409,
            detail=f"Invalid status transition: {current} -> {next_status}"
        )

    # Build history entry
    transitioned_at = datetime.now(timezone.utc)
    history_entry = {
        "status": next_status,
        "transitioned_at": transitioned_at.isoformat(),
    }

    # Append to history and update status
    history = req.status_history or []
    history.append(history_entry)
