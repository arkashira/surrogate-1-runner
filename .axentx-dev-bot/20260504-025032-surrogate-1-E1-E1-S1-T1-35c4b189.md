"""Enhanced GitHub API client for repository operations with robust error handling and configuration support."""

import os
import json
import base64
from typing import Dict, Any, Optional
import requests
import yaml

class GitHubAPIError(Exception):
    """Raised when the GitHub API returns an error status code."""
    def __init__(self, status_code: int, message: str):
        self.status_code = status_code
        self.message = message
        super().__init__(f"GitHub API error {status_code}: {message}")

class GitHubAPI:
    """Robust GitHub API client with authentication and request handling.

    Parameters:
        token: Personal access token for authentication. If None, uses GITHUB_TOKEN env var.
        base_url: Base URL for the GitHub API (defaults to public API).
        timeout: Request timeout in seconds (default: 10).
    """
    def __init__(
        self,
        token: Optional[str] = None,
        base_url: str = "https://api.github.com",
        timeout: int = 10
    ):
        self.token = token or os.getenv("GITHUB_TOKEN")
        if not self.token:
            raise ValueError("GitHub token must be provided via argument or GITHUB_TOKEN env var")

        self.base_url = base_url.rstrip("/")
        self.timeout = timeout
        self.session = requests.Session()
        self.session.headers.update({
            "Authorization": f"token {self.token}",
            "Accept": "application/vnd.github.v3+json",
            "User-Agent": "axentx-surrogate-1",
        })

    def _request(self, method: str, endpoint: str, **kwargs) -> Dict[str, Any]:
        """Internal method to make authenticated requests to GitHub API."""
        url = f"{self.base_url}{endpoint}"
        try:
            response = self.session.request(
                method,
                url,
                timeout=self.timeout,
                **kwargs
            )
            response.raise_for_status()
            return response.json() if response.content else {}
        except requests.exceptions.RequestException as e:
            raise GitHubAPIError(
                getattr(e.response, 'status_code', 500),
                str(e)
            )

    def get_repo(self, owner: str, repo: str) -> Dict[str, Any]:
        """Retrieve repository metadata."""
        return self._request("GET", f"/repos/{owner}/{repo}")

    def get_file_contents(
        self,
        owner: str,
        repo: str,
        path: str,
        ref: str = "main"
    ) -> str:
        """Retrieve raw file contents from repository.

        Args:
            owner: Repository owner
            repo: Repository name
            path: File path within repository
            ref: Branch/tag/commit reference (default: 'main')

        Returns:
            Decoded file contents as string
        """
        endpoint = f"/repos/{owner}/{repo}/contents/{path}"
        params = {"ref": ref}
        data = self._request("GET", endpoint, params=params)

        if data.get("encoding") == "base64":
            return base64.b64decode(data["content"]).decode('utf-8')
        return data.get("content", "")

    def get_formatter_config(
        self,
        owner: str,
        repo: str,
        ref: str = "main"
    ) -> Dict[str, Any]:
        """Retrieve formatter configuration from .github/formatter.yml.

        Returns empty dict if file doesn't exist or is invalid.
        """
        try:
            content = self.get_file_contents(
                owner,
                repo,
                ".github/formatter.yml",
                ref=ref
            )
            return yaml.safe_load(content) or {}
        except GitHubAPIError as e:
            if e.status_code == 404:
                return {}
            raise
