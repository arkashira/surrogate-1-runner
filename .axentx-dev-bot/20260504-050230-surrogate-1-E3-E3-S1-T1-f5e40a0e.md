"""
Retry decorator with exponential backoff, optional jitter, logging,
Prometheus metrics and an idempotency guard.

The module exposes two decorators:

* @retry(...) – retries a function on exception.
* @idempotent(key_func=None) – ensures a function is executed only once
  for a given key (by default the function name + arguments).

Metrics are only available if ``prometheus_client`` is installed.
"""

from __future__ import annotations

import functools
import logging
import random
import sys
import threading
import time
from typing import Any, Callable, Dict, Optional, Set

# Optional Prometheus metrics
try:
    from prometheus_client import Counter

    _PROMETHEUS_AVAILABLE = True
    ingestion_retries_total = Counter(
        "ingestion_retries_total",
        "Total number of retry attempts performed across all ingestion functions.",
    )
    ingestion_failures_total = Counter(
        "ingestion_failures_total",
        "Total number of ingestion attempts that ultimately failed after retries.",
    )
except ImportError:
    _PROMETHEUS_AVAILABLE = False
    ingestion_retries_total = None  # type: ignore
    ingestion_failures_total = None  # type: ignore


# ----------------------------------------------------------------------
# Idempotency guard
# ----------------------------------------------------------------------
_idempotent_keys: Set[str] = set()
_idempotent_lock = threading.Lock()


def idempotent(
    key_func: Optional[Callable[..., str]] = None,
) -> Callable[[Callable[..., Any]], Callable[..., Any]]:
    """
    Decorator that ensures the wrapped function runs at most once for a given key.

    Parameters
    ----------
    key_func : callable, optional
        A function that receives the same arguments as the decorated function
        and returns a unique string key. If omitted, the key is generated from
        the function's qualified name plus a JSON‑like representation of the
        arguments (using ``str``).

    The decorator is thread‑safe.
    """

    def decorator(func: Callable[..., Any]) -> Callable[..., Any]:
        @functools.wraps(func)
        def wrapper(*args: Any, **kwargs: Any) -> Any:
            # Build the key
            if key_func is not None:
                key = key_func(*args, **kwargs)
            else:
                # Simple default key: func name + repr of args/kwargs
                key = f"{func.__module__}.{func.__qualname__}::{args!r}::{kwargs!r}"

            with _idempotent_lock:
                if key in _idempotent_keys:
                    logging.getLogger("ingestion").info(
                        f"Skipping {func.__name__} (key={key}) – already processed"
                    )
                    return None
