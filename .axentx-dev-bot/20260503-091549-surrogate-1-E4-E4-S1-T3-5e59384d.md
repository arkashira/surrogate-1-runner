// lib/email.ts
import * as nodemailer from 'nodemailer';
import { EmailTemplate } from './email-template';

interface EmailConfig {
  host: string;
  port: number;
  auth: {
    user: string;
    pass: string;
  };
}

interface EmailTemplate {
  subject: string;
  render(requestId: string): string;
}

class EmailSender {
  private transporter: nodemailer.Transporter;

  constructor(private config: EmailConfig, private template: EmailTemplate) {}

  async sendEmail(to: string, requestId: string, channel: string): Promise<void> {
    const mailOptions = {
      from: 'axentx@axentx.com',
      to,
      subject: this.template.subject,
      text: this.template.render(requestId),
    };

    try {
      const result = await this.transporter.sendMail(mailOptions);
      console.log(`Email sent successfully: ${result.messageId}`);
    } catch (error) {
      console.error(`Error sending email: ${error.message}`);
      // retry logic will be implemented in notify.ts
    }
  }

  init(): void {
    this.transporter = nodemailer.createTransport({
      host: this.config.host,
      port: this.config.port,
      auth: this.config.auth,
    });
  }
}

export { EmailSender, EmailConfig, EmailTemplate };

// workers/notify.ts
import { EmailSender } from '../lib/email';
import { SlackWebhook } from '../lib/slack-webhook';
import { Queue } from 'bull';
import { Worker } from 'bull';

interface NotifyConfig {
  emailConfig: EmailConfig;
  template: EmailTemplate;
  queueName: string;
}

class NotifyWorker {
  private queue: Queue;
  private worker: Worker;

  constructor(private config: NotifyConfig) {
    this.queue = new Queue(this.config.queueName, {
      redis: {
        host: 'localhost',
        port: 6379,
      },
    });
    this.worker = new Worker(this.config.queueName, this.processJob.bind(this));
  }

  async processJob(job: any): Promise<void> {
    const { requestId, channel } = job.data;
    const emailSender = new EmailSender(this.config.emailConfig, this.config.template);
    await emailSender.init();
    await emailSender.sendEmail('stakeholders@example.com', requestId, channel);
    const slackWebhook = new SlackWebhook({
      webhookUrl: 'https://example.slack.com/webhook',
    });
    const slackMessage = {
      text: emailSender.template.render(requestId),
      channel,
    };
    await slackWebhook.postMessage(slackMessage);
  }

  async enqueueJob(requestId: string, channel: string): Promise<void> {
    await this.queue.add({ requestId, channel });
  }

  async retryFailedJobs(): Promise<void> {
    const failedJobs = await this.queue.getFailedJobs();
    for (const job of failedJobs) {
      await this.queue.add(job.data, { attempts: job.attempts, backoff: job.backoff });
    }
  }
}

export { NotifyWorker, NotifyConfig };
