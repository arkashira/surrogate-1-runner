+   url: string;
+   mimeType?: string;

+ const RECENT_KEY = 'axentx:recent-documents';
+ const ACTIVE_ENDPOINT = '/api/active';
+
+ export function pushRecentItem(id: string): void {
+   try {
+     const recent = getRecentItems();
+     const next = [id, ...recent.filter((x) => x !== id)].slice(0, 50);
