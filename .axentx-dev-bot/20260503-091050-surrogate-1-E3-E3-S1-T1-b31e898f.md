// After
import { open } from 'open';
import { analytics } from '../lib/analytics';
import { setRecentlyOpenedDocument } from '../lib/sessionStorage';

export const openDocument = (path: string) => {
  const startTime = Date.now();
  open(path).then(() => {
    const endTime = Date.now();
    const duration = endTime - startTime;
    if (duration <= 1000) {
      analytics.track('Document opened within 1s');
    } else {
      analytics.track('Document opened in more than 1s');
    }
    setRecentlyOpenedDocument(path);
  });
};
```
