export async function openDocument(doc: { id: string; url: string }): Promise<{ durationMs: number }> {
  const start = performance.now();
  try {
    // shell open or embedded viewer stub
    window.open(doc.url, '_blank');
    const durationMs = performance.now() - start;
    return { durationMs };
  } catch (err) {
    const durationMs = performance.now() - start;
    throw Object.assign(new Error('OPEN_FAILED'), { durationMs, cause: err });
  }
}
