import os
import logging
import time
from typing import List, Dict, Any, Optional
from urllib.parse import urljoin

import requests
from requests.adapters import HTTPAdapter
from requests.packages.urllib3.util.retry import Retry

# Configure logging
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


class GitHubAPIError(Exception):
    """Base exception for GitHub API errors."""
    pass


class GitHubRateLimitError(GitHubAPIError):
    """Raised when GitHub API rate limit is exceeded."""
    pass


class GitHubIntegration:
    """
    GitHub Integration for repository operations.
    
    Handles API authentication, rate limiting, retries, and 
    provides methods for fetching repository data and notifications.
    """
    
    BASE_URL = "https://api.github.com"
    
    def __init__(
        self, 
        token: Optional[str] = None, 
        repo: Optional[str] = None,
        max_retries: int = 3,
        timeout: int = 10
    ):
        """
        Initialize GitHub integration.
        
        Args:
            token: GitHub API token (defaults to config or env var)
            repo: Repository in format "owner/repo"
            max_retries: Number of retry attempts for failed requests
            timeout: Request timeout in seconds
        """
        self.token = token or os.getenv("GITHUB_API_TOKEN")
        self.repo = repo or os.getenv("GITHUB_REPO")
        
        if not self.token:
            raise ValueError("GitHub API token is required")
        if not self.repo:
            raise ValueError("GitHub repository is required")
            
        self.timeout = timeout
        self.session = self._create_session(max_retries)
        
    def _create_session(self, max_retries: int) -> requests.Session:
        """Create requests session with retry strategy."""
        session = requests.Session()
        
        retry_strategy = Retry(
            total=max_retries,
            backoff_factor=1,
            status_forcelist=[429, 500, 502, 503, 504],
            allowed_methods=["GET", "POST"]
        )
        
        adapter = HTTPAdapter(max_retries=retry_strategy)
        session.mount("https://", adapter)
        session.mount("http://", adapter)
        
        return session
    
    @property
    def headers(self) -> Dict[str, str]:
        """Return authentication headers for API requests."""
        return {
            "Authorization": f"token {self.token}",
            "Accept": "application/vnd.github.v3+json",
            "X-GitHub-Api-Version": "2022-11-28"
        }
    
    def get_repo_updates(
        self, 
        per_page: int = 30, 
        page: int = 1,
        branch: Optional[str] = None
    ) -> List[Dict[str, Any]]:
        """
        Fetch repository commits.
        
        Args:
            per_page: Number of results per page (max 100)
            page: Page number for pagination
            branch: Optional branch to filter commits
            
        Returns:
            List of commit dictionaries
            
        Raises:
            GitHubAPIError: If API request fails
            GitHubRateLimitError: If rate limit exceeded
        """
        url = f"{self.BASE_URL}/repos/{self.repo}/commits"
        params = {"per_page": per_page, "page": page}
        
        if branch:
            params["sha"] = branch
            
        try:
            response = self.session.get(
                url, 
                headers=self.headers, 
                params=params,
                timeout=self.timeout
            )
            
            self._handle_response(response)
            return response.json()
