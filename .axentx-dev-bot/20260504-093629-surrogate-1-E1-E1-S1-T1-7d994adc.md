import io

CHUNK_SIZE = 64 * 1024  # 64KB

def parse_stream(stream):
    chunks = []
    while True:
        chunk = stream.read(CHUNK_SIZE)
        if not chunk:
            break
        chunks.append(chunk)
    return b''.join(chunks)

def parse_file(file_path):
    with open(file_path, 'rb') as file:
        return parse_stream(file)
