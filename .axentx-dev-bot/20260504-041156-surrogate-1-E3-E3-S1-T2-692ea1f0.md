"""
Notification delivery system for CPC threshold breaches.

Features
--------
* Stores CPC history in SQLite (90‑day retention).
* Configurable per‑channel thresholds.
* Rate‑limits alerts per campaign.
* Thread‑safe, testable, and extensible.
"""

from __future__ import annotations

import sqlite3
import time
import threading
from datetime import datetime, timedelta
from typing import Dict, Tuple, List, Optional, Protocol

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #

DEFAULT_THRESHOLDS: Dict[str, float] = {
    "email": 0.20,  # 20 % above benchmark
    "sms": 0.20,
}

# --------------------------------------------------------------------------- #
# Persistence layer
# --------------------------------------------------------------------------- #

DB_PATH = ":memory:"  # In production, replace with a file path


class HistoricalStore:
    """SQLite wrapper for storing CPC benchmark history."""

    def __init__(self, db_path: str = DB_PATH) -> None:
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self._init_schema()

    def _init_schema(self) -> None:
        cur = self.conn.cursor()
        cur.execute(
            """
            CREATE TABLE IF NOT EXISTS cpc_history (
                campaign_id TEXT,
                timestamp INTEGER,
                benchmark REAL,
                cpc REAL,
                PRIMARY KEY (campaign_id, timestamp)
            )
            """
        )
        cur.execute(
            """
            CREATE INDEX IF NOT EXISTS idx_campaign
            ON cpc_history (campaign_id)
            """
        )
        self.conn.commit()

    def insert(self, campaign_id: str, benchmark: float, cpc: float) -> None:
        cur = self.conn.cursor()
        cur.execute(
            """
            INSERT OR REPLACE INTO cpc_history
            (campaign_id, timestamp, benchmark, cpc)
            VALUES (?, ?, ?, ?)
            """,
            (campaign_id, int(time.time()), benchmark, cpc),
        )
        self.conn.commit()

    def cleanup(self, days: int = 90) -> None:
        """Remove records older than `days`."""
        cutoff = int(time.time()) - days * 86400
        cur = self.conn.cursor()
        cur.execute("DELETE FROM cpc_history WHERE timestamp < ?", (cutoff,))
        self.conn.commit()

    def get_recent(
        self, campaign_id: str, limit: int = 10
    ) -> List[Tuple[int, float, float]]:
        cur = self.conn.cursor()
        cur.execute(
            """
            SELECT timestamp, benchmark, cpc
            FROM cpc_history
            WHERE campaign_id = ?
            ORDER BY timestamp DESC
            LIMIT ?
            """,
            (campaign_id, limit),
        )
        return cur.fetchall()
