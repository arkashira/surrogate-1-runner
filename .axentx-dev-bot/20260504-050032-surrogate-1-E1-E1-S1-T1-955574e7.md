"""
Cost savings prediction model using RandomForestRegressor.

This module provides functionality to train, predict, and manage
a machine learning model for cost savings recommendations.
"""

import joblib
import numpy as np
import pandas as pd
from pathlib import Path
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_absolute_error, r2_score
from sklearn.model_selection import train_test_split
from typing import Dict, Tuple, Optional

# Configuration
BASE_DIR = Path(__file__).resolve().parent.parent
MODEL_DIR = BASE_DIR / "models"
MODEL_DIR.mkdir(exist_ok=True)

DEFAULT_FEATURES = ['cpu_usage', 'memory_usage', 'storage_usage', 'network_usage']
TARGET = 'cost_savings'


class CostSavingsModel:
    """Machine learning model for predicting cost savings."""
    
    def __init__(
        self,
        n_estimators: int = 100,
        max_depth: Optional[int] = None,
        random_state: int = 42
    ):
        self.model = RandomForestRegressor(
            n_estimators=n_estimators,
            max_depth=max_depth,
            random_state=random_state
        )
        self.features = DEFAULT_FEATURES.copy()
        self.target = TARGET
        self.is_trained = False
        self.metrics: Dict[str, float] = {}
    
    def _validate_data(self, df: pd.DataFrame, require_target: bool = False) -> None:
        """Validate input data has required columns."""
        required_cols = self.features.copy()
        if require_target:
            required_cols.append(self.target
            
        missing = set(required_cols) - set(df.columns)
        if missing:
            raise ValueError(f"Missing required columns: {missing}")
        
        # Check for sufficient data
        if len(df) < 10:
            raise ValueError(f"Insufficient data: {len(df)} rows (minimum 10 required)")
    
    def train(self, data: pd.DataFrame, test_size: float = 0.2) -> Dict[str, float]:
        """
        Train the model on provided data.
        
        Args:
            data: DataFrame with features and target
            test_size: Fraction for test split (default 0.2)
            
        Returns:
            Dictionary with training metrics
        """
        self._validate_data(data, require_target=True)
        
        X = data[self.features]
        y = data[self.target]
        
        # Handle any NaN values
        if X.isnull().any().any():
            raise ValueError("Data contains NaN values. Please clean data before training.")
        
        X_train, X_test, y_train, y_test = train_test_split(
            X, y, test_size=test_size, random_state=42
        )
        
        self.model.fit(X_train, y_train)
        self.is_trained = True
        
        # Calculate metrics
        predictions = self.model.predict(X_test)
        self.metrics = {
            'mae': mean_absolute_error(y_test, predictions),
            'r2': r2_score(y_test, predictions),
            'train_size': len(X_train),
            'test_size': len(X_test)
        }
        
        return self.metrics
    
    def predict(self, data: pd.DataFrame) -> np.ndarray:
        """Generate predictions for input data."""
        if not self.is_trained:
            raise RuntimeError("Model must be trained before prediction")
