"""
Synthetic event generator for the Axentx surrogate pipeline.

Key features taken from the two candidate drafts:

* **Pattern loading** – read once at import time, with a clear error if the file is missing.
* **Service‑aware pattern selection** – pick a random pattern that matches the requested service,
  falling back to any pattern when none match (as in Candidate 1).
* **Volume control** – “low”, “normal”, “high” affect both the *emit* probability and the
  severity distribution (the weighting logic from Candidate 1, plus the explicit
  `emit_chance` mapping from Candidate 2).
* **Spike support** – optional “spike” configuration per‑service (the idea from Candidate 2).
* **Deterministic seeding** – callers can pass a `seed` to get reproducible streams, which is
  useful for tests.
* **Convenient CLI** – `python -m src.data.events` prints a small batch, making manual
  verification trivial.
"""

import json
import random
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any, Dict, List, Literal, Optional, Sequence, Tuple

# --------------------------------------------------------------------------- #
# 1️⃣  Load the JSON pattern file once at import time
# --------------------------------------------------------------------------- #
CONFIG_PATH = Path(__file__).resolve().parents[2] / "config" / "event-patterns.json"

if not CONFIG_PATH.is_file():
    raise FileNotFoundError(f"Event pattern config not found at {CONFIG_PATH}")

with CONFIG_PATH.open("r", encoding="utf-8") as f:
    _RAW_CONFIG: Dict[str, Any] = json.load(f)

_PATTERNS: List[Dict[str, Any]] = _RAW_CONFIG.get("patterns", [])
if not _PATTERNS:
    raise ValueError("No event patterns found in configuration.")
