import com.azure.core.credential.TokenCredential;
import com.azure.identity.DefaultAzureCredentialBuilder;
import com.azure.resourcemanager.costmanagement.CostManagementManager;
import com.azure.resourcemanager.costmanagement.models.QueryDefinition;
import com.azure.resourcemanager.costmanagement.models.QueryResult;
import com.azure.resourcemanager.costmanagement.models.QueryTimePeriod;
import com.fasterxml.jackson.databind.ObjectMapper;
import java.io.BufferedWriter;
import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Paths;
import java.time.Instant;
import java.time.LocalDateTime;
import java.time.format.DateTimeFormatter;
import java.util.Properties;
import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.TimeUnit;

public class AzureCostIngestor {

    private static final String CONFIG_FILE = "azure-cost-config.properties";
    private static final String OUTPUT_DIR = "output/azure-cost";
    private static final ObjectMapper MAPPER = new ObjectMapper();

    private final Properties config;
    private final ScheduledExecutorService scheduler = Executors.newSingleThreadScheduledExecutor();
    private final CostManagementManager costManagementManager;

    public AzureCostIngestor() throws IOException {
        config = new Properties();
        try (InputStream in = Files.newInputStream(Paths.get(CONFIG_FILE))) {
            config.load(in);
        }
        Files.createDirectories(Paths.get(OUTPUT_DIR));

        TokenCredential credential = new DefaultAzureCredentialBuilder().build();
        costManagementManager = new CostManagementManager(credential);
    }

    /**
     * Starts the ingestion job to run at the configured interval.
     */
    public void start() {
        long initialDelay = 0;
        long period = Long.parseLong(config.getProperty("ingest.interval.minutes", "30"));
        scheduler.scheduleAtFixedRate(this::runIngestion, initialDelay, period, TimeUnit.MINUTES);
    }

    /**
     * Stops the scheduled ingestion.
     */
    public void stop() {
        scheduler.shutdownNow();
    }

    /**
     * Main ingestion routine. Reads Azure credentials from config,
     * fetches cost data, and writes structured JSON to disk.
     */
    public void runIngestion() {
        try {
            System.out.println("AzureCostIngestor: Starting ingestion at " + Instant.now());

            // Create query definition
            QueryDefinition queryDefinition = new QueryDefinition()
                    .withType(QueryResultPropertiesQueryType.ACTUAL_COST)
                    .withTimePeriod(new QueryTimePeriod()
                            .withFrom(Instant.now().minusSeconds(3600))
                            .withTo(Instant.now()));

            // Fetch cost data
            QueryResult queryResult = costManagementManager.query().queryBySubscription(
                    config.getProperty("azure.subscription.id"),
                    queryDefinition);

            // Write to a timestamped file
            String fileName = String.format("azure-cost-%s.json", DateTimeFormatter.ofPattern("yyyyMMddHHmmss").format(LocalDateTime.now()));
            Path outFile = Paths.get(OUTPUT_DIR, fileName);
            try (BufferedWriter writer = Files.newBufferedWriter(outFile)) {
                MAPPER.writerWithDefaultPrettyPrinter().writeValue(writer, queryResult);
            }
            System.out.println("AzureCostIngestor: Written cost data to " + outFile);
        } catch (Exception e) {
            System.err.println("AzureCostIngestor: Error during ingestion - " + e.getMessage());
            e.printStackTrace();
        }
