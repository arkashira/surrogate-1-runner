package workflow

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"strings"
	"sync"
	"time"

	"github.com/google/uuid"
)

// Workflow represents a user‑defined workflow.
type Workflow struct {
	ID        string    `json:"id"`
	Name      string    `json:"name"`
	Steps     []string  `json:"steps,omitempty"`
	Owner     string    `json:"owner,omitempty"`
	CreatedAt time.Time `json:"created_at,omitempty"`
	UpdatedAt time.Time `json:"updated_at,omitempty"`
}

// ---------------------------------------------------------------------
// In‑memory store & concurrency primitives
// ---------------------------------------------------------------------

var (
	storeMu sync.RWMutex
	store   = make(map[string]*Workflow)

	subsMu      sync.RWMutex
	subscribers = make(map[string]map[chan string]struct{}) // workflowID → set of channels
)

// ---------------------------------------------------------------------
// Core CRUD operations
// ---------------------------------------------------------------------

// CreateWorkflow stores a new workflow and returns it.
// If w.ID is empty a UUID will be generated.
func CreateWorkflow(w *Workflow) (*Workflow, error) {
	if w == nil {
		return nil, fmt.Errorf("nil workflow")
	}
	storeMu.Lock()
	defer storeMu.Unlock()

	if w.ID == "" {
		w.ID = uuid.New().String()
	}
	now := time.Now()
	w.CreatedAt = now
	w.UpdatedAt = now
	store[w.ID] = w
	return w, nil
}

// GetWorkflow retrieves a workflow by its ID.
func GetWorkflow(id string) (*Workflow, error) {
	storeMu.RLock()
	defer storeMu.RUnlock()
	if w, ok := store[id]; ok {
		return w, nil
	}
	return nil, fmt.Errorf("workflow %s not found", id)
}

// ListWorkflows returns all stored workflows.
func ListWorkflows() ([]*Workflow, error) {
	storeMu.RLock()
	defer storeMu.RUnlock()
	ws := make([]*Workflow, 0, len(store))
	for _, w := range store {
		ws = append(ws, w)
	}
	return ws, nil
}

// UpdateWorkflow applies mutable fields from upd onto the stored workflow.
func UpdateWorkflow(id string, upd *Workflow) (*Workflow, error) {
	if upd == nil {
		return nil, fmt.Errorf("nil update payload")
	}
	storeMu.Lock()
	defer storeMu.Unlock()
	existing, ok := store[id]
	if !ok {
		return nil, fmt.Errorf("workflow %s not found", id)
	}
	if upd.Name != "" {
		existing.Name = upd.Name
	}
	if upd.Steps != nil {
		existing.Steps = upd.Steps
	}
	if upd.Owner != "" {
		existing.Owner = upd.Owner
	}
	existing.UpdatedAt = time.Now()

	// Broadcast change to real‑time listeners.
	broadcastWorkflowChange(existing)

	return existing, nil
}

// DeleteWorkflow removes a workflow from the store.
func DeleteWorkflow(id string) error {
	storeMu.Lock()
	defer storeMu.Unlock()
	if _, ok := store[id]; !ok {
		return fmt.Errorf("workflow %s not found", id)
	}
	delete(store, id)

	// Close any open subscriber channels.
	closeSubscribers(id)

	return nil
}

// ---------------------------------------------------------------------
// Real‑time collaboration (Server‑Sent Events)
// ---------------------------------------------------------------------

// broadcastWorkflowChange sends the updated workflow JSON to all subscribed clients.
func broadcastWorkflowChange(w *Workflow) {
	data, err := json.Marshal(w)
	if err != nil {
		return
	}
	subsMu.RLock()
	defer subsMu.RUnlock()
	if chans, ok := subscribers[w.ID]; ok {
		for ch := range chans {
			// Non‑blocking send; drop if client is slow.
			select {
			case ch <- string(data):
			default:
			}
		}
	}
}

// subscribe registers a channel to receive updates for a given workflow ID.
func subscribe(workflowID string) chan string {
	ch := make(chan string, 10) // buffered to avoid blocking
	subsMu.Lock()
	defer subsMu.Unlock()
	if _, ok := subscribers[workflowID]; !ok {
		subscribers[workflowID] = make(map[chan string]struct{})
	}
	subscribers[workflowID][ch] = struct{}{}
	return ch
}

// unsubscribe removes a channel from the subscription list.
