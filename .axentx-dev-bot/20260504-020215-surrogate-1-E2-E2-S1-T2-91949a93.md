name: Performance Benchmark Check

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  benchmark:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Docker
        uses: docker/setup-buildx-action@v2

      - name: Login to Docker Hub
        uses: docker/login-action@v2
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      - name: Build and run Docker container
        run: |
          docker build -t axentx/surrogate-1 .
          docker run -it --rm axentx/surrogate-1

      - name: Run performance benchmarks
        run: |
          docker run -it --rm axentx/surrogate-1 /bin/bash -c "python -m unittest discover -s tests -p 'test_*.py'"
          docker run -it --rm axentx/surrogate-1 /bin/bash -c "python -m unittest discover -s benchmarks -p 'bench_*.py'"

      - name: Compare results with baseline
        run: |
          python3 ./scripts/compare_benchmarks.py
        env:
          BASELINE_THRESHOLD: 0.05  # 5% regression threshold

      - name: Upload results to GitHub Actions
        uses: actions/upload-artifact@v3
        with:
          name: perf_results
          path: |
            perf_results.json
            benchmark_results.json

      - name: Report performance regression
        if: failure()
        run: |
          echo "::error::Performance regression detected! Results exceeded ${BASELINE_THRESHOLD}% threshold"
          exit 1
```
