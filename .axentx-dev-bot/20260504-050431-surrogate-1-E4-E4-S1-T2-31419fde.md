# ──────────────────────────────────────────────────────────────────────────────

"""
Real‑time ROI forecast API.

* The endpoint accepts a JSON payload describing a proposed workload.
* Forecasts are calculated deterministically (15 % savings, ±10 % CI) so the
  service can be exercised without external ML services.
* Results are persisted to a relational database via SQLAlchemy ORM.
* The module is intentionally lightweight – it can be dropped into any
  FastAPI project with minimal friction.
"""

from __future__ import annotations

import uuid
import datetime
from typing import Dict

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel, Field
from sqlalchemy import (
    Column,
    DateTime,
    Float,
    String,
    create_engine,
)
from sqlalchemy.orm import declarative_base, Session, sessionmaker

# --------------------------------------------------------------------------- #
# Configuration – keep this in a single place so it can be overridden in tests
# --------------------------------------------------------------------------- #
DATABASE_URL: str = "sqlite:///./forecast.db"  # change to Postgres, etc. in prod
engine = create_engine(DATABASE_URL, connect_args={"check_same_thread": False})
SessionLocal = sessionmaker(bind=engine, autoflush=False, autocommit=False)
Base = declarative_base()


def get_db() -> Session:
    """Yield a DB session; close it after the request."""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


# --------------------------------------------------------------------------- #
# ORM model – one row per forecast
# --------------------------------------------------------------------------- #
class Forecast(Base):
    __tablename__ = "forecasts"

    id = Column(String, primary_key=True, index=True, default=lambda: str(uuid.uuid4()))
    workload_id = Column(String, nullable=False, index=True)
    cpu_cores = Column(Integer, nullable=False)
    memory_gb = Column(Float, nullable=False)
    duration_days = Column(Integer, nullable=False)
    cost_per_core = Column(Float, nullable=False)
    cost_per_gb = Column(Float, nullable=False)

    savings = Column(Float, nullable=False)
    roi = Column(Float, nullable=False)
    confidence_low = Column(Float, nullable=False)
    confidence_high = Column(Float, nullable=False)

    created_at = Column(DateTime, default=datetime.datetime.utcnow, nullable=False)

    def to_response(self) -> "ForecastResponse":
        return ForecastResponse(
            forecast_id=self.id,
            workload_id=self.workload_id,
            savings=self.savings,
            roi=self.roi,
            confidence_interval={
                "low": self.confidence_low,
                "high": self.confidence_high,
            },
            created_at=self.created_at,
        )


# Create tables on first import – safe for SQLite demo; in prod use migrations
Base.metadata.create_all(bind=engine)
