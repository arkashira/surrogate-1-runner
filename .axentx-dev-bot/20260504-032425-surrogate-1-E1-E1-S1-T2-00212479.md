//! Clipboard handling utilities for AX “direct‑paste” attempts.
//!
//! The public API is deliberately small so that the same code can be used
//! both in production (real OS clipboard) and in unit‑tests (in‑memory mock).
//!
//! ```rust
//! use axentx_surrogate_1::clipboard::{attempt_direct_paste, Clipboard, SystemClipboard};
//!
//! let sys = SystemClipboard::new()?;               // real clipboard on macOS / Linux / Windows
//! let result = attempt_direct_paste(&sys, "my‑token")?;
//! println!("Direct paste result: {result:?}");
//! ```

use std::fmt;
use log::{info, warn};

/// Errors that can arise while working with the clipboard.
#[derive(Debug, thiserror::Error)]
pub enum ClipboardError {
    /// The OS‑specific clipboard API failed.
    #[error("clipboard backend error: {0}")]
    Backend(String),

    /// The direct‑paste logic decided that the supplied data is not a token
    /// that can be pasted directly.
    #[error("no direct‑paste performed")]
    NoOp,

    /// Any other unexpected condition.
    #[error("internal error: {0}")]
    Internal(String),
}

/// Result type used by the public functions.
pub type ClipboardResult<T> = Result<T, ClipboardError>;

/// The outcome of a direct‑paste attempt.
#[derive(Debug, PartialEq, Eq)]
pub enum DirectPasteResult {
    /// The token was recognised and written to the clipboard.
    Success,
    /// The token was *not* a direct‑paste token; nothing was changed.
    NoOp,
}

/// Minimal clipboard abstraction.  Production code implements it for the real
/// OS clipboard; tests implement a lightweight in‑memory version.
pub trait Clipboard: Send + Sync {
    /// Return the current clipboard contents as a `String`.
    fn get_contents(&self) -> ClipboardResult<String>;

    /// Replace the clipboard contents with `data`.
    fn set_contents(&self, data: &str) -> ClipboardResult<()>;
}
