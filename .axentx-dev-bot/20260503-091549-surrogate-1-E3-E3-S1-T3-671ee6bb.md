const db = require('../db');

const Audit = {
  create: async ({ requestId, actor, previousStatus, newStatus, comment, fieldUpdates = null }) => {
    if (!requestId || !actor || !newStatus) {
      throw new Error('Missing required audit fields: requestId, actor, newStatus');
    }
    const sql = `
      INSERT INTO audit_entries (
        request_id, actor, previous_status, new_status, comment, field_updates, created_at
      ) VALUES (?, ?, ?, ?, ?, ?, datetime('now'))
    `;
    const result = await db.run(sql, [
      requestId,
      actor,
      previousStatus,
      newStatus,
      comment || null,
      fieldUpdates ? JSON.stringify(fieldUpdates) : null
    ]);
    return {
      id: result.lastID,
      requestId,
      actor,
      previousStatus,
      newStatus,
      comment: comment || null,
      fieldUpdates,
      createdAt: new Date().toISOString()
    };
  },

  findByRequestId: async (requestId, page = 1, perPage = 50) => {
    const offset = (page - 1) * perPage;
    const sql = `
      SELECT id, request_id, actor, previous_status, new_status, comment, field_updates, created_at
      FROM audit_entries
      WHERE request_id = ?
      ORDER BY created_at DESC
      LIMIT ? OFFSET ?
    `;
    const rows = await db.all(sql, [requestId, perPage, offset]);
    return rows.map(r => ({
      id: r.id,
      requestId: r.request_id,
      actor: r.actor,
      previousStatus: r.previous_status,
      newStatus: r.new_status,
      comment: r.comment,
      fieldUpdates: r.field_updates ? JSON.parse(r.field_updates) : null,
      createdAt: r.created_at
    }));
  },

  countByRequestId: async (requestId) => {
    const sql = `SELECT COUNT(*) as count FROM audit_entries WHERE request_id = ?`;
    const row = await db.get(sql, [requestId]);
    return row.count;
  }
};

module.exports = Audit;
