def _public_payload(record: Dict[str, Any]) -> Dict[str, Any]:

@router.get("/public/requests/{public_id}")
async def get_public_request(
    request: Request,
    public_id: str,
    view: str | None = None,
) -> JSONResponse | HTMLResponse:
    record = _REQUESTS_DB.get(public_id)
    if not record:
        raise HTTPException(status_code=404, detail="Request not found")

    payload = _public_payload(record)

    # Prefer HTML if ?view=html or Accept header prefers HTML
    prefer_html = (view or "").lower() == "html" or "text/html" in (request.headers.get("accept") or "")
    if prefer_html:
        return templates.TemplateResponse(

                "request": request,
                "title": payload.get("title", "Status"),
                "current_status": payload.get("current_status", "unknown"),
                "owner": payload.get("owner", ""),
                "updated_at": payload.get("updated_at", ""),
                "timeline": payload.get("timeline", []),
                "public_id": public_id,
            },
        )
