import { v4 as uuidv4 } from 'uuid';

export type Status = string;

export interface TimelineEntry {
  id: string;
  requestId: string;
  fromStatus: Status | null;
  toStatus: Status;
  note: string;
  actor: string;
  timestamp: string; // ISO
}

export interface RequestRecord {
  id: string;
  createdAt: string; // ISO
  statusHistory: TimelineEntry[];
}

class SLAService {
  private requests = new Map<string, RequestRecord>();

  ensureRequest(requestId: string): RequestRecord {
    let req = this.requests.get(requestId);
    if (!req) {
      req = {
        id: requestId,
        createdAt: new Date().toISOString(),
        statusHistory: [],
      };
      this.requests.set(requestId, req);
    }
    return req;
  }

  recordStatusTransition(params: {
    requestId: string;
    fromStatus: Status | null;
    toStatus: Status;
    note: string;
    actor: string;
    timestamp?: string;
  }): TimelineEntry {
    const req = this.ensureRequest(params.requestId);
    const entry: TimelineEntry = {
      id: uuidv4(),
      requestId: params.requestId,
      fromStatus: params.fromStatus,
      toStatus: params.toStatus,
      note: params.note,
      actor: params.actor,
      timestamp: params.timestamp || new Date().toISOString(),
    };
    req.statusHistory.push(entry);
    // keep timeline ordered by timestamp ascending
    req.statusHistory.sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime());
    return entry;
  }

  getTimeline(requestId: string): TimelineEntry[] {
    const req = this.requests.get(requestId);
    if (!req) return [];
    // return ordered (already sorted)
    return [...req.statusHistory];
  }

  computeSLA(requestId: string): { createdAt: string; latestStatusAt: string | null; slaMs: number | null } {
    const req = this.requests.get(requestId);
    if (!req) {
      return { createdAt: new Date().toISOString(), latestStatusAt: null, slaMs: null };
    }
    const created = new Date(req.createdAt).getTime();
    const latest = req.statusHistory.length > 0
      ? new Date(req.statusHistory[req.statusHistory.length - 1].timestamp).getTime()
      : null;
    const slaMs = latest !== null ? latest - created : null;
    return {
      createdAt: req.createdAt,
      latestStatusAt: latest !== null ? new Date(latest).toISOString() : null,
      slaMs,
    };
  }
