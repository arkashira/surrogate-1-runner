"""
Utility definitions for the automated creator-tool setup.

This module provides a flexible and extensible catalog of common tools that creators often require, such as git, ffmpeg, ImageMagick, youtube-dl, node, npm, and httpie. Each tool is encapsulated within a `Tool` dataclass, which includes attributes for the tool's name, a command to check its presence, an installation command tailored for Debian-based systems, and a brief description. The automation logic remains decoupled from the tool definitions, allowing for easy extension by other parts of the codebase.

"""

from __future__ import annotations

import sys
from dataclasses import dataclass
from typing import List

@dataclass(frozen=True)
class Tool:
    """Immutable representation of a tool to be installed."""
    name: str
    check_cmd: List[str]
    install_cmd: List[str]
    description: str

def _sudo_prefix(cmd: List[str]) -> List[str]:
    """Prefixes the command with 'sudo' if the current user is not root on a Linux system."""
    if sys.platform.startswith("linux"):
        try:
            import os
            if os.geteuid() != 0:
                return ["sudo"] + cmd
        except Exception:
            return ["sudo"] + cmd
    return cmd

def _apt_install(pkg: str) -> List[str]:
    """Generates a sudo-prefixed apt-get install command for the specified package."""
    return _sudo_prefix(["apt-get", "update", "-y", "&&", "apt-get", "install", "-y", pkg])

def _pip_install(pkg: str) -> List[str]:
    """Generates a pip install command for the specified package at the user level."""
    return [sys.executable, "-m", "pip", "install", "--quiet", pkg]

def _npm_install(pkg: str) -> List[str]:
    """Generates an npm install command for the specified package globally, with sudo if necessary."""
    return _sudo_prefix(["npm", "install", "-g", pkg])

def get_common_tools() -> List[Tool]:
    """
    Provides a list of commonly used tools for creators.

    This list is intentionally concise to ensure the automated setup completes within five minutes on a typical CI runner. Tools are installed through the most suitable package manager based on the host operating system (apt, pip, npm).
    """

            description="Multimedia processing suite",

            check_cmd=["magick", "-version"],
            install_cmd=_apt_install("imagemagick"),
            description="Image manipulation utilities",
        ),
