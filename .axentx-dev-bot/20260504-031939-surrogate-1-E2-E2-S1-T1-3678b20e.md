## 1. Acceptance Criteria
- **Latency**: An alert must be emitted within **60 seconds** of an error or discrepancy being detected.
- **Content**: Each alert must contain:
  - A unique alert ID
  - Timestamp of detection
  - Severity level (`INFO`, `WARN`, `ERROR`)
  - A concise message (≤ 120 characters)
  - Suggested remediation steps (non‑empty string)
- **Dashboard**: The system must expose a method `get_active_alerts()` that returns a list of all currently active alerts in chronological order.
- **Deduplication**: Identical alerts (same message and severity) detected within a **5‑minute window** should be merged into a single alert entry.
- **Persistence**: Alerts must be stored in an in‑memory list; no external dependencies required for the test suite.
- **Extensibility**: The alerting logic should allow injection of custom detection functions and formatting functions.

describe('AlertingSystem', () => {
  let alerting;

  beforeEach(() => {
    alerting = new AlertingSystem();
  });

  test('should create an alert with required fields', () => {
    const alert = alerting.createAlert('ERROR', 'Database connection failed', 'Restart the DB service');
    expect(alert).toHaveProperty('id');
    expect(alert).toHaveProperty('timestamp');
    expect(alert.severity).toBe('ERROR');
    expect(alert.message).toBe('Database connection failed');
    expect(alert.recommendation).toBe('Restart the DB service');
  });

  test('should store alerts in chronological order', () => {
    const a1 = alerting.createAlert('WARN', 'Low disk space', 'Clean temp files');
    const a2 = alerting.createAlert('ERROR', 'Out of memory', 'Increase heap size');
    const active = alerting.get_active_alerts();
    expect(active[0].id).toBe(a1.id);
    expect(active[1].id).toBe(a2.id);
  });

  test('should deduplicate alerts within 5‑minute window', () => {
    const a1 = alerting.createAlert('WARN', 'High latency', 'Check network');
    // Simulate 2 minutes later
    advanceTimeBy(120000);
    const a2 = alerting.createAlert('WARN', 'High latency', 'Check network');
    const active = alerting.get_active_alerts();
    expect(active.length).toBe(1);
    expect(active[0].id).toBe(a1.id);
  });

  test('should not deduplicate alerts older than 5 minutes', () => {
    const a1 = alerting.createAlert('WARN', 'High latency', 'Check network');
    advanceTimeBy(301000); // 5 min + 1 sec
    const a2 = alerting.createAlert('WARN', 'High latency', 'Check network');
    const active = alerting.get_active_alerts();
    expect(active.length).toBe(2);
  });

  test('should allow custom detection function injection', () => {
    const customDetect = jest.fn().mockReturnValue(true);
    alerting.setDetectionFunction(customDetect);
    alerting.detectAndAlert();
    expect(customDetect).toHaveBeenCalled();
    expect(alerting.get_active_alerts().length).toBe(1);
  });

  test('should format alert messages correctly', () => {
    const formatter = jest.fn(msg => `ALERT: ${msg}`);
    alerting.setFormatter(formatter);
    const alert = alerting.createAlert('INFO', 'Routine check', 'All good');
    expect(alert.formattedMessage).toBe('ALERT: Routine check');
  });
});
