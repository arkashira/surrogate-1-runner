const fs = require('fs');
const axios = require('axios');

// Function to scan vulnerabilities
async function scanVulnerabilities() {
  try {
    const response = await axios.get('https://api.example.com/vulnerabilities');
    const vulnerabilities = response.data;
    return vulnerabilities;
  } catch (error) {
    console.error('Error scanning vulnerabilities:', error);
    return [];
  }
}

// Function to report vulnerabilities with severity levels
async function reportVulnerabilities() {
  const vulnerabilities = await scanVulnerabilities();
  const severityLevels = {
    low: [],
    medium: [],
    high: [],
    critical: []
  };

  vulnerabilities.forEach((vulnerability) => {
    switch (vulnerability.severity) {
      case 'low':
        severityLevels.low.push(vulnerability);
        break;
      case 'medium':
        severityLevels.medium.push(vulnerability);
        break;
      case 'high':
        severityLevels.high.push(vulnerability);
        break;
      case 'critical':
        severityLevels.critical.push(vulnerability);
        break;
      default:
        console.log(`Unknown severity level: ${vulnerability.severity}`);
    }
  });

  console.log('Vulnerabilities by severity level:');
  console.log(`Low: ${severityLevels.low.length}`);
  console.log(`Medium: ${severityLevels.medium.length}`);
  console.log(`High: ${severityLevels.high.length}`);
  console.log(`Critical: ${severityLevels.critical.length}`);

  // Save the report to a file
  const report = {
    low: severityLevels.low,
    medium: severityLevels.medium,
    high: severityLevels.high,
    critical: severityLevels.critical
  };
  fs.writeFileSync('vulnerability_report.json', JSON.stringify(report, null, 2));
}

// Run the report function
reportVulnerabilities();
