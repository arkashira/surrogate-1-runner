import fs from 'fs';
import path from 'path';
import { Mutex } from 'async-mutex';
import lockfile from 'proper-lockfile';

const DB_PATH = path.resolve(__dirname, '../db/requests.json');
const mutex = new Mutex();

// Ensure DB file exists
if (!fs.existsSync(DB_PATH)) {
  fs.writeFileSync(DB_PATH, JSON.stringify({ requests: {}, audit: {} }, null, 2), 'utf8');
}

interface AuditEntry {
  id: string;
  requestId: string;
  status: string;
  note?: string;
  actor: string;
  timestamp: string;
}

interface RequestRecord {
  id: string;
  status: string;
  createdAt: string;
  updatedAt: string;
}

interface Db {
  requests: Record<string, RequestRecord>;
  audit: Record<string, AuditEntry[]>;
}

function readDbSync(): Db {
  try {
    const raw = fs.readFileSync(DB_PATH, 'utf8');
    return JSON.parse(raw);
  } catch {
    return { requests: {}, audit: {} };
  }
}

async function writeDbAtomic(db: Db): Promise<void> {
  const tempPath = `${DB_PATH}.tmp.${Date.now()}`;
  fs.writeFileSync(tempPath, JSON.stringify(db, null, 2), 'utf8');

  // atomic replace + cross-process lock
  await lockfile.lock(DB_PATH, { retries: 3 });
  try {
    fs.renameSync(tempPath, DB_PATH);
  } finally {
    await lockfile.unlock(DB_PATH);
  }
}

function nowIso(): string {
  return new Date().toISOString();
}

function genId(): string {
  return `${Date.now()}-${Math.random().toString(36).slice(2, 9)}`;
}

export async function updateRequestStatus(
  id: string,
  status: string,
  note: string | undefined,
  actor: string
): Promise<{ request: RequestRecord; audit: AuditEntry }> {
  return mutex.runExclusive(async () => {
    const db = readDbSync();
    const request = db.requests[id];
    if (!request) {
      const err: any = new Error('Request not found');
      err.status = 404;
      throw err;
    }

    const updatedAt = nowIso();
    const updatedRequest: RequestRecord = {
      ...request,
      status,
      updatedAt
    };
    db.requests[id] = updatedRequest;

    const auditEntry: AuditEntry = {
      id: `audit-${genId()}`,
      requestId: id,
      status,
      note,
      actor,
      timestamp: updatedAt
    };

    const entries = db.audit[id] || [];
    entries.push(auditEntry);
    entries.sort((a, b) => a.timestamp.localeCompare(b.timestamp));
    db.audit[id] = entries;

    await writeDbAtomic(db);
    return { request: updatedRequest, audit: auditEntry };
  });
}
