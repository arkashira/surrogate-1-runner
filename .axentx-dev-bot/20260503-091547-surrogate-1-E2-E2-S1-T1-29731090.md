class RequestService {
  async get(id: string): Promise<RequestDocument> {
    // ...
  }

  async getStatus(id: string): Promise<RequestStatus> {
    // ...
  }

  async updateStatus(id: string, status: RequestStatus, note?: string): Promise<RequestDocument> {
    const request = await this.get(id);
    if (!request) {
      throw new Error(`Request not found`);
    }

    const version = request.version;
    const updatedRequest = await this.db.collection('requests').findOneAndUpdate(
      { _id: new ObjectId(id), version },
      { $set: { status, version: version + 1 }, $push: { audit: { status, note, timestamp: new Date() } } },
      { returnOriginal: false }
    );

    if (!updatedRequest.value) {
      throw new Error(`Request not found or concurrent update`);
    }

    return updatedRequest.value;
  }
}

export default RequestService;
