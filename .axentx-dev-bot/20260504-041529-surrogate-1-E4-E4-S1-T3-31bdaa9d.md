describe('Aggregator', () => {
  let aggregator;
  let mockStore;

  beforeEach(() => {
    mockStore = {
      query: jest.fn()
    };
    aggregator = new Aggregator(mockStore, { cacheTtl: 1000 }); // 1s TTL
  });

  test('get_account_stats aggregates correctly', async () => {
    mockStore.query.mockResolvedValue([
      { account_id: 'A1', severity: 'high', scan_time: '2026-05-01T10:00:00Z' },
      { account_id: 'A1', severity: 'low',  scan_time: '2026-05-01T10:00:00Z' },
      { account_id: 'A1', severity: 'high', scan_time: '2026-05-02T12:00:00Z' }
    ]);

    const stats = await aggregator.get_account_stats('A1');
    expect(stats.account_id).toBe('A1');
    expect(stats.total).toBe(3);
    expect(stats.high_severity).toBe(2);
    expect(stats.last_scan).toEqual(new Date('2026-05-02T12:00:00Z'));
  });

  test('get_all_account_stats returns sorted list', async () => {
    mockStore.query.mockResolvedValue([
      { account_id: 'A1', severity: 'low',  scan_time: '2026-05-01T10:00:00Z' },
      { account_id: 'B2', severity: 'high', scan_time: '2026-05-01T11:00:00Z' },
      { account_id: 'B2', severity: 'high', scan_time: '2026-05-01T11:00:00Z' }
    ]);

    const list = await aggregator.get_all_account_stats();
    expect(list[0].account_id).toBe('B2'); // 2 violations > 1
    expect(list[1].account_id).toBe('A1');
  });

  test('cache invalidation after TTL', async () => {
    mockStore.query.mockResolvedValueOnce([{ account_id: 'A1', severity: 'low', scan_time: '2026-05-01T10:00:00Z' }]);

    expect(first.total).toBe(1);

    // Advance time beyond TTL

    const second = await aggregator.get_account_stats('A1');
    expect(second.total).toBe(1);
    expect(second.high_severity).toBe(1);
  });
