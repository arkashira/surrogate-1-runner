os.makedirs(DATA_DIR, exist_ok=True)

class BusinessMetrics:
    """
    Typed container for the four mandatory fields.
    """
    mrr: float
    cac: float
    ltv: float
    audience_demographics: Dict[str, Any] = field(default_factory=dict)

    # --------------------------------------------------------------------- #
    # Validation helpers – used by the Django form and by any other caller
    # --------------------------------------------------------------------- #

        errors: List[str] = []

        # 1. Numeric fields must be present and non‑negative
        for field_name in ("mrr", "cac", "ltv"):

            if value is None:
                errors.append(f"{field_name.upper()} is required.")
            else:
                try:
                    fval = float(value)
                    if fval < 0:
                        errors.append(f"{field_name.upper()} must be non‑negative.")

                    errors.append(f"{field_name.upper()} must be a number.")

        # 2. Audience demographics must be a non‑empty dict

            errors.append("Audience demographics must be a non‑empty dictionary.")

        return errors

    def to_dict(self) -> Dict[str, Any]:
        return {
            "mrr": float(self.mrr),
