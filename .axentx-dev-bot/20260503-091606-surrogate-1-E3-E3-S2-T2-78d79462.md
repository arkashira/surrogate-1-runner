import os
from datetime import datetime
from typing import List, Dict, Any

try:
    import boto3
    from botocore.exceptions import ClientError
    SES_AVAILABLE = True
except Exception:
    SES_AVAILABLE = False


def _get_ses_client():
    if not SES_AVAILABLE:
        raise RuntimeError("SES not available; install boto3 or configure email backend")
    region = os.getenv("AWS_REGION", "us-east-1")
    return boto3.client("ses", region_name=region)


def send_digest(
    to_addresses: List[str],
    subject: str,
    body_text: str,
    sender: str | None = None,
    reply_to: List[str] | None = None,
) -> Dict[str, Any]:
    """
    Send a plain-text email via AWS SES (or print in dev).
    Returns a dict with message_id or error details.
    """
    sender = sender or os.getenv("DIGEST_SENDER", "no-reply@axentx.example")
    reply_to = reply_to or []

    # Dev mode: print instead of sending
    if os.getenv("ENV", "dev") == "dev":
        print(
            f"--- EMAIL DEV MODE ---\n"
            f"To: {to_addresses}\n"
            f"Subject: {subject}\n\n"
            f"{body_text}\n"
            f"--- END EMAIL ---\n"
        )
        return {"message_id": "dev-digest", "status": "printed"}

    client = _get_ses_client()
    try:
        kwargs = {
            "Source": sender,
            "Destination": {"ToAddresses": to_addresses},
            "Message": {
                "Subject": {"Data": subject},
                "Body": {"Text": {"Data": body_text}},
            },
        }
        if reply_to:
            kwargs["ReplyToAddresses"] = reply_to
        resp = client.send_email(**kwargs)
        return {"message_id": resp["MessageId"], "status": "sent"}
    except ClientError as e:
        return {"error": str(e), "status": "failed"}
```
