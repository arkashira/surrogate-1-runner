#!/usr/bin/env bash
set -euo pipefail

# Ensure conda is available
if ! command -v conda &>/dev/null; then
  echo "Conda is not installed. Please install Miniconda or Anaconda before running this script."
  exit 1
fi

# Create environments if they do not exist
for env_file in open_weight_llms.yml benchmark_suite.yml sdk.yml; do
  env_name=$(yq eval '.name' "$env_file")
  if conda env list | grep -q "^$env_name "; then
    echo "Environment '$env_name' already exists. Skipping creation."
  else
    echo "Creating environment '$env_name' from $env_file..."
    conda env create -f "$env_file"
  fi
done

echo "All environments are ready."
