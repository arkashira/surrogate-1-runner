// ...

import axios from 'axios';
import { WebClient } from '@slack/web-api';

const slackWebClient = new WebClient(process.env.SLACK_API_TOKEN);

const sendSlackNotification = async (stateChange: any) => {
  const formattedMessage = `**State Change**: ${stateChange.requestId} - ${stateChange.newState}\n`;
  formattedMessage += `**SLA Status**: ${stateChange.slaStatus}\n`;
  formattedMessage += `**Public Link**: ${stateChange.publicLink}`;

  const response = await slackWebClient.chat.postMessage({
    channel: process.env.SLACK_CHANNEL,
    text: formattedMessage,
  });

  console.log(response);
};

const sendDailyDigestEmail = async (requests: Request[]) => {
  const emailSubject = 'Daily Digest: Requests at Risk/Breached and New Updates';
  const emailBody = requests
    .map((request) => {
      const slaStatus = request.slaStatus;
      const state = request.state;
      return `Request ${request.id} has ${slaStatus} SLA status and is in state ${state}.`;
    })
    .join('\n');

  try {
    const response = await sendEmail({
      subject: emailSubject,
      body: emailBody,
      recipients: process.env.EMAIL_RECIPIENTS.split(','),
    });
    if (!response.ok) {
      console.error(`Error sending daily digest email: ${response.status}`);
    }
  } catch (error) {
    console.error(`Error sending daily digest email: ${error}`);
  }
};

// ...
