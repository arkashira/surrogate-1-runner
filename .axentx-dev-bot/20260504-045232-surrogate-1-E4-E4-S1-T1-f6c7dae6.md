import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { Table, TableBody, TableCell, TableContainer, TableHead, TableRow, Paper, TableSortLabel, TextField, Button } from '@mui/material';

interface Component {
  id: string;
  name: string;
  brand: string;
  price: number;
  benchmarkScore: number;
  cpu: string;
  gpu: string;
  ram: string;
  storage: string;
  psu: string;
  case: string;
}

const ComparisonTable: React.FC = () => {
  const [components, setComponents] = useState<Component[]>([]);
  const [order, setOrder] = useState<'asc' | 'desc'>('asc');
  const [orderBy, setOrderBy] = useState<keyof Component>('name');
  const [minPrice, setMinPrice] = useState<number>(0);
  const [maxPrice, setMaxPrice] = useState<number>(1000);
  const navigate = useNavigate();

  useEffect(() => {
    // Fetch components data from API
    fetch('/api/components')
      .then(response => response.json())
      .then(data => setComponents(data));
  }, []);

  const handleSort = (property: keyof Component) => {
    const isAsc = orderBy === property && order === 'asc';
    setOrder(isAsc ? 'desc' : 'asc');
    setOrderBy(property);
  };

  const handleFilter = () => {
    // Filter components based on price range
    fetch(`/api/components?minPrice=${minPrice}&maxPrice=${maxPrice}`)

      .then(data => setComponents(data));
  };

  const handleRowClick = (id: string) => {
    navigate(`/components/${id}`);

  const sortedComponents = [...components].sort((a, b) => {
    if (order === 'asc') {
      return a[orderBy] > b[orderBy] ? 1 : -1;
