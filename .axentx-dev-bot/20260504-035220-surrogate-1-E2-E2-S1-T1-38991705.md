    const term = new Terminal({
        rendererType: 'canvas',
        rows: 30,
        cols: 80,
        scrollback: 1000,
        cursorBlink: true

    const socket = new WebSocket('ws://your-websocket-server.com/shell');

    socket.onopen = () => {

    };

        term.write(event.data);
    };

    term.onData((data) => {
        socket.send(data);
    });
});

describe('TerminalManager', () => {
  let tm;

    expect(tm.terminal.rows).toBe(40);
  });

describe('WebSocketClient', () => {
  test('connects to WebSocket URL from config', () => {
    const ws = new WebSocketClient('ws://localhost:8080/shell');
    expect(ws.url).toBe('ws://localhost:8080/shell');
  });

  test('sends data via WebSocket', () => {
    const ws = new WebSocketClient('ws://localhost:8080/shell');
    ws.connect();
    ws.send('echo hello');
    expect(ws.lastSent).toBe('echo hello');
  });

  test('receives and forwards data to terminal', () => {
    const ws = new WebSocketClient('ws://localhost:8080/shell');
    const mockTerminal = { write: jest.fn() };
    ws.onMessage('response data', mockTerminal.write);
    expect(mockTerminal.write).toHaveBeenCalledWith('response data');
  });

  test('reconnects on connection loss (max 3 retries)', async () => {
    const ws = new WebSocketClient('ws://localhost:8080/shell');
    ws.connect();
    ws.simulateDisconnect();
    expect(ws.retryCount).toBe(1);
  });
