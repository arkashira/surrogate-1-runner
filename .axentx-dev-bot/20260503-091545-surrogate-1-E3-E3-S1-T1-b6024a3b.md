  id: string;

  status: RequestStatus;
  note?: string;
  actor: string;
  timestamp: string; // ISO 8601
}

export function isValidStatus(status: string): status is RequestStatus {
  return ALLOWED_STATUSES.has(status as RequestStatus);
}

// Persistence interface (in-memory implementation below; swap for DB later)
export interface UpdatesRepository {
  add(update: Update): void;
  findByRequestId(requestId: string): Update[];
}

// In-memory implementation
export class InMemoryUpdatesRepository implements UpdatesRepository {
  private store = new Map<string, Update[]>();

  add(update: Update): void {
    if (!this.store.has(update.requestId)) {
      this.store.set(update.requestId, []);
    }
    this.store.get(update.requestId)!.push(update);
  }

  findByRequestId(requestId: string): Update[] {
    return this.store.get(requestId) || [];
  }
}
