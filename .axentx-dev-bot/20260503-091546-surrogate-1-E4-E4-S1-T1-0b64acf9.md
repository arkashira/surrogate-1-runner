+
+ export const FollowerSchema = z.object({
+   id: z.string().cuid().optional(),
+   requestId: z.string().min(1, 'Request ID is required'),
+   userId: z.string().min(1, 'User ID is required'),
+   optOut: z.boolean().default(false),
+   createdAt: z.date().optional(),
+   updatedAt: z.date().optional(),
+ });
+
+ export type Follower = z.infer<typeof FollowerSchema>;
+
+ // Lightweight in-memory store shape (replace with DB adapter in real app)
+ export class FollowerModel {
+   private store: Map<string, Follower> = new Map();
+
+   create(data: Omit<Follower, 'id' | 'createdAt' | 'updatedAt'>): Follower {
+     const id = crypto.randomUUID();
+     const now = new Date();
+     const follower: Follower = {
+       id,
+       ...data,
+       optOut: data.optOut ?? false,
+       createdAt: now,
+       updatedAt: now,
+     };
+     this.store.set(id, follower);
+     return follower;
+   }
+
+   updateOptOut(requestId: string, userId: string, optOut: boolean): Follower | null {
+     const existing = Array.from(this.store.values()).find(
+       (f) => f.requestId === requestId && f.userId === userId
+     );
+     if (!existing) return null;
+     existing.optOut = optOut;
+     existing.updatedAt = new Date();
+     this.store.set(existing.id, existing);
+     return existing;
+   }
+
+   findByRequest(requestId: string): Follower[] {
+     return Array.from(this.store.values()).filter((f) => f.requestId === requestId);
+   }
+
+   findByUser(userId: string): Follower[] {
+     return Array.from(this.store.values()).filter((f) => f.userId === userId);
+   }
+ }
