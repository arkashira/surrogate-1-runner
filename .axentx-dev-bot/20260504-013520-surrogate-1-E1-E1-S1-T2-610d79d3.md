import pytest
from flask import Flask, render_template_string

# Assume the template is loaded from the actual file system
@pytest.fixture
def app():
    app = Flask(__name__, template_folder="/opt/axentx/surrogate-1/review/templates")
    @app.route("/report")
    def report():
        issues = [
            {"title": "Issue 1", "description": "Desc 1", "link": "/code/1"},
            {"title": "Issue 2", "description": "Desc 2", "link": "/code/2"},
        ]
        return render_template_string(open("report.html").read(), issues=issues)
    return app

def test_report_renders(app):
    client = app.test_client()
    resp = client.get("/report")
    assert resp.status_code == 200
    html = resp.data.decode()
    # AC2: list exists
    assert '<ul id="issue-list">' in html
    # AC3: each issue present
    assert html.count('<li class="issue-item">') == 2
    # AC3: link hrefs
    assert '<a href="/code/1">Issue 1</a>' in html
    assert '<a href="/code/2">Issue 2</a>' in html
    # AC3: descriptions
    assert '<p>Desc 1</p>' in html
    assert '<p>Desc 2</p>' in html

def test_css_link_present(app):
    client = app.test_client()
    html = client.get("/report").data.decode()
    assert '<link rel="stylesheet" href="/static/styles.css">' in html

def test_auto_escaping(app):
    client = app.test_client()
    # Inject a script tag in title
    issues = [{"title": "<script>alert(1)</script>", "description": "X", "link": "/x"}]
    with app.test_request_context():
        rendered = render_template_string(open("report.html").read(), issues=issues)
    # The script tag should be escaped
    assert '&lt;script&gt;alert(1)&lt;/script&gt;' in rendered
    assert '<script>' not in rendered
