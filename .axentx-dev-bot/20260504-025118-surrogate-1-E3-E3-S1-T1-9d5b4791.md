    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(message)s",
)
logger = logging.getLogger(__name__)

    """Main class handling deployment rollback operations."""

    def __init__(self):
        self.current_deployment_id = self._get_current_deployment_id()

        """Get the current deployment ID from environment or default."""
        return os.getenv("DEPLOYMENT_ID", "default-deployment")

    def load_last_deploy_state(self) -> Dict[str, Any]:
        """Load the last successful deployment state from disk."""
        if not DEPLOY_STATE_FILE.exists():

            return {}

        try:
            with DEPLOY_STATE_FILE.open("r") as f:
                state = json.load(f)

            return state

            logger.error(f"Failed to load deployment state: {e}")
            return {}
