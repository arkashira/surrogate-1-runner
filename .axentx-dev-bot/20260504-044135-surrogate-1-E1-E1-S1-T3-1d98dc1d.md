import subprocess, os, re

def test_freerouter_executable_exists():
    """Freerouter binary is present and executable."""
    path = "/opt/axentx/surrogate-1/bin/freerouter"
    assert os.path.isfile(path)
    assert os.access(path, os.X_OK)

def test_freerouter_runs_without_java():
    """Running freerouter must not spawn a Java process."""
    proc = subprocess.Popen(
        ["surrogate", "freerouter", "--dry-run"],
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        env=dict(os.environ, SURROGATE_DEBUG="1")
    )
    out, err = proc.communicate(timeout=10)
    # Ensure exit code 0
    assert proc.returncode == 0
    # Scan system process list for java while the command runs
    java_ps = subprocess.check_output(["pgrep", "-fl", "java"]).decode()
    assert not re.search(r'\bjava\b', java_ps), "Java process detected"

def test_kicad_version_detection():
    """Surrogate correctly detects KiCAD 9."""
    out = subprocess.check_output(["surrogate", "--detect-kicad"]).decode()
    assert "KiCAD 9" in out

def test_log_filter_removes_java_entries():
    """Log helper strips Java‑related lines."""
    from surrogate.logging import filter_java_errors
    sample = "INFO: Starting\nERROR: java.lang.OutOfMemoryError\nINFO: Done"
    cleaned = filter_java_errors(sample)
    assert "java.lang.OutOfMemoryError" not in cleaned
    assert "INFO: Starting" in cleaned
    assert "INFO: Done" in cleaned
