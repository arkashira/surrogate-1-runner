// /opt/axentx/surrogate-1/backend/services/thumbnails.ts
import crypto from 'crypto';

export interface ThumbnailOptions {
  title?: string;
  author?: string;
  width?: number;
  height?: number;
}

function colorFromSeed(seed: string): string {
  const hash = crypto.createHash('md5').update(seed).digest('hex');
  const r = parseInt(hash.slice(0, 2), 16);
  const g = parseInt(hash.slice(2, 4), 16);
  const b = parseInt(hash.slice(4, 6), 16);
  // desaturate slightly and normalize brightness for readability
  const gray = 0.2126 * r + 0.7152 * g + 0.0722 * b;
  const factor = gray < 100 ? 1.3 : 0.85;
  const nr = Math.min(255, Math.max(0, Math.floor(r * factor)));
  const ng = Math.min(255, Math.max(0, Math.floor(g * factor)));
  const nb = Math.min(255, Math.max(0, Math.floor(b * factor)));
  return `rgb(${nr},${ng},${nb})`;
}

function escapeHtml(s: string): string {
  return s.replace(/[&<>"']/g, (c) =>
    ({
      '&': '&amp;',
      '<': '&lt;',
      '>': '&gt;',
      '"': '&quot;',
      "'": '&#39;',
    }[c] || c)
  );
}

/**
 * Generate a data URL SVG thumbnail placeholder.
 */
export function generatePlaceholderThumbnail(
  seed: string,
  options: ThumbnailOptions = {}
): string {
  const { title = '', author = '', width = 300, height = 400 } = options;
  const bg = colorFromSeed(seed);
  const textColor = '#ffffff';
  const initials =
    (title ? title.slice(0, 1).toUpperCase() : '') +
    (author ? author.slice(0, 1).toUpperCase() : '');
  const shortTitle = title ? escapeHtml(title.slice(0, 20)) : 'Document';
  const shortAuthor = author ? escapeHtml(author.slice(0, 20)) : '';

  const svg = `
<svg xmlns="http://www.w3.org/2000/svg" width="${width}" height="${height}" viewBox="0 0 ${width} ${height}">
  <rect width="${width}" height="${height}" fill="${bg}"/>
  <text x="50%" y="42%" dominant-baseline="middle" text-anchor="middle" fill="${textColor}" font-family="system-ui, -apple-system, sans-serif" font-size="${Math.max(
    28,
    width * 0.14
  )}" font-weight="700">${initials || '?'}</text>
  <text x="50%" y="60%" dominant-baseline="middle" text-anchor="middle" fill="${textColor}" opacity="0.85" font-family="system-ui, -apple-system, sans-serif" font-size="${Math.max(
    12,
    width * 0.05
  )}" font-weight="600">${shortTitle}</text>
  <text x="50%" y="72%" dominant-baseline="middle" text-anchor="middle" fill="${textColor}" opacity="0.6" font-family="system-ui, -apple-system, sans-serif" font-size="${Math.max(
    10,
    width * 0.04
  )}">${shortAuthor}</text>
</svg>`.trim();

  return `data:image/svg+xml;base64,${Buffer.from(svg).toString('base64')}`;
}

/**
 * Get thumbnail URL for a document.
 * If a real cover exists (coverUrl), return it. Otherwise return generated placeholder.
 */
export function getThumbnailForDocument(
  doc: { id: string; title?: string; author?: string; coverUrl?: string },
  options?: ThumbnailOptions
): string {
  if (doc.coverUrl) {
    return doc.coverUrl;
  }
  return generatePlaceholderThumbnail(doc.id || doc.title || 'doc', {
    title: doc.title,
    author: doc.author,
    ...options,
  });
}
```
