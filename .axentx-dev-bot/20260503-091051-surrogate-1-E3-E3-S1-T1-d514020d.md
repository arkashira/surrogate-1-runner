// /opt/axentx/surrogate-1/backend/services/thumbnails.ts
import { createCanvas, loadImage } from 'canvas';
import fs from 'fs';
import path from 'path';

const THUMB_CACHE_DIR = path.resolve(process.env.THUMB_CACHE_DIR || '/tmp/axentx-thumbs');
if (!fs.existsSync(THUMB_CACHE_DIR)) fs.mkdirSync(THUMB_CACHE_DIR, { recursive: true });

function hashId(id: string): string {
  // simple deterministic hash for filenames
  let hash = 0;
  for (let i = 0; i < id.length; i++) {
    const char = id.charCodeAt(i);
    hash = (hash << 5) - hash + char;
    hash |= 0;
  }
  return Math.abs(hash).toString(36);
}

function stringToColor(str: string): string {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = str.charCodeAt(i) + ((hash << 5) - hash);
  }
  const c = (hash & 0x00FFFFFF).toString(16).toUpperCase();
  return '#' + '00000'.substring(0, 6 - c.length) + c;
}

function getInitials(title: string): string {
  const words = title.trim().split(/\s+/);
  if (words.length === 0) return '?';
  if (words.length === 1) return words[0].charAt(0).toUpperCase();
  return (words[0][0] + words[1][0]).toUpperCase();
}

export async function generateThumbnail(
  doc: { id: string; title: string; coverUrl?: string | null },
  size = 256
): Promise<{ path: string; mimeType: string }> {
  const idHash = hashId(doc.id);
  const outPath = path.join(THUMB_CACHE_DIR, `${idHash}.png`);

  // If cached, return
  if (fs.existsSync(outPath)) {
    return { path: outPath, mimeType: 'image/png' };
  }

  // If doc has coverUrl, try to fetch and resize
  if (doc.coverUrl) {
    try {
      const img = await loadImage(doc.coverUrl);
      const canvas = createCanvas(size, size);
      const ctx = canvas.getContext('2d');
      // cover and center crop
      const scale = Math.max(size / img.width, size / img.height);
      const w = img.width * scale;
      const h = img.height * scale;
      const x = (size - w) / 2;
      const y = (size - h) / 2;
      ctx.drawImage(img, x, y, w, h);
      const out = fs.createWriteStream(outPath);
      const stream = canvas.createPNGStream();
      await new Promise<void>((resolve, reject) => {
        stream.pipe(out);
        out.on('finish', () => resolve());
        out.on('error', reject);
      });
      return { path: outPath, mimeType: 'image/png' };
    } catch (err) {
      // fallback to generated cover
      console.warn('Failed to load cover for', doc.id, err);
    }
  }

  // Fallback: color
