from datetime import datetime, timezone
from typing import Any, Dict, List, Optional

from pydantic import BaseModel, Field, field_validator


class Quote(BaseModel):
    symbol: str = Field(..., description="Normalized symbol, e.g. AAPL.O")
    price: float = Field(..., ge=0)
    bid: Optional[float] = Field(None, ge=0)
    ask: Optional[float] = Field(None, ge=0)
    volume: Optional[int] = Field(None, ge=0)
    timestamp: datetime = Field(..., description="UTC timestamp of quote")

    @field_validator("timestamp")
    def ensure_utc(cls, v: datetime) -> datetime:
        if v.tzinfo is None:
            return v.replace(tzinfo=timezone.utc)
        return v.astimezone(timezone.utc)


class MarketSnapshotSchema(BaseModel):
    """
    Common schema for market snapshot research payloads.
    """

    snapshot_id: str = Field(..., description="Unique snapshot identifier")
    provider: str = Field(..., description="Raw data provider name")
    fetched_at: datetime = Field(..., description="UTC time connector fetched data")
    interval: str = Field("1m", description="Effective interval represented")
    quotes: List[Quote] = Field(default_factory=list)
    metadata: Dict[str, Any] = Field(default_factory=dict)

    @field_validator("fetched_at")
    def ensure_utc_fetched(cls, v: datetime) -> datetime:
        if v.tzinfo is None:
            return v.replace(tzinfo=timezone.utc)
