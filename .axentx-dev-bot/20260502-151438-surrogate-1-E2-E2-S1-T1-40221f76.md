from __future__ import annotations

from datetime import datetime, timezone
from typing import Any, Dict, Literal
from pydantic import BaseModel, Field


class MarketSnapshot(BaseModel):
    """Normalized market snapshot used across research connectors."""
    symbol: str
    timestamp: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))
    bid: float | None = None
    ask: float | None = None
    last: float | None = None
    open: float | None = None
    high: float | None = None
    low: float | None = None
    close: float | None = None
    volume: float | None = None
    vwap: float | None = None
    spread: float | None = None
    source: str = "market_snapshot"
    extra: Dict[str, Any] = Field(default_factory=dict)

    @classmethod
    def normalize(cls, raw: Dict[str, Any], symbol: str, source: str = "market_snapshot") -> "MarketSnapshot":
        bid, ask = raw.get("bid"), raw.get("ask")
        last = raw.get("last") or raw.get("close") or ask or bid
        return cls(
            symbol=symbol,
            timestamp=datetime.now(timezone.utc),
            bid=bid,
            ask=ask,
            last=last,
            open=raw.get("open"),
            high=raw.get("high"),
            low=raw.get("low"),
            close=raw.get("close") or last,
            volume=raw.get("volume"),
            vwap=raw.get("vwap"),
            spread=(ask - bid) if (ask is not None and bid is not None) else None,
            source=source,
            extra={k: v for k, v in raw.items() if k not in {"bid", "ask", "last", "open", "high", "low", "close", "volume", "vwap"}},
        )

    symbol: str
