def assign_shard(paths: list[str], shard_id: int, total: int) -> list[str]:
    """Deterministic shard assignment by slug hash."""
    assigned = []
    for p in paths:
        # stable hash across runs
        h = int(hashlib.md5(p.encode()).hexdigest(), 16)
        if h % total == shard_id:
            assigned.append(p)
    return assigned
