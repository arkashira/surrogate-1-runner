package com.axentx.surreal.dashboard.services;

import com.axentx.surreal.gpu.utils.GpuUtils;
import org.springframework.stereotype.Service;
import java.time.Instant;

@Service
public class FirmwareStatusService {

    private static final String DEFAULT_ERROR_MESSAGE = "Error fetching firmware information";

    public FirmwareInfo getFirmwareInfo() {
        try {
            String version = GpuUtils.getFirmwareVersion();
            String status = GpuUtils.getFirmwareStatus();
            Instant lastUpdated = Instant.now(); // Or fetch from GpuUtils if available

            return new FirmwareInfo(version, status, lastUpdated);
        } catch (Exception e) {
            return new FirmwareInfo(DEFAULT_ERROR_MESSAGE, "UNKNOWN", Instant.now());
        }
    }

    public record FirmwareInfo(
        String version,
        String status,
        Instant lastUpdated
    ) {}
}
```
