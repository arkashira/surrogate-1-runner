#### `dataset-enrich.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail

DATASET_REPO="axentx/surrogate-1-training-pairs"
DATE_FOLDER="${1:-$(date +"%Y-%m-%d")}"
FILE_LIST_JSON="file_list_${DATE_FOLDER}.json"
HF_TOKEN="${HF_TOKEN:-}"

# 1. List files (one API call)
if [ -z "$HF_TOKEN" ]; then
  echo "ERROR: HF_TOKEN is required to list repository tree."
  exit 1
fi

echo "Listing files for ${DATE_FOLDER}..."
curl -sS -H "Authorization: Bearer ${HF_TOKEN}" \
  "https://huggingface.co/api/v1/datasets/${DATASET_REPO}/tree/main/${DATE_FOLDER}" \
  -o "${FILE_LIST_JSON}"

# 2. Extract paths and download via CDN
echo "Downloading files via HF CDN..."
jq -r '.[].path' "${FILE_LIST_JSON}" | while read -r FILE; do
  CDN_URL="https://huggingface.co/datasets/${DATASET_REPO}/resolve/main/${FILE}"
  echo "Downloading ${FILE}..."
  curl -sS -L "${CDN_URL}" -o "${FILE}"

  # 3. Project to {prompt, response} (example for JSONL)
  # Adjust parser if format differs
  if [[ "$FILE" == *.jsonl ]]; then
    jq -c '{prompt: .prompt, response: .response}' "${FILE}" > "clean_${FILE}"
  elif [[ "$FILE" == *.tsv || "$FILE" == *.csv ]]; then
    awk -F'\t' '{print $1 "\t" $2}' "${FILE}" > "clean_${FILE}"
  else
    echo "Skipping projection for unsupported format: ${FILE}"
  fi
done

echo "Done. Cleaned files ready for training."
```
