// investor_interest.test.js
import request from 'supertest';
import app from '../../src/app';
import { ProviderStore } from '../../src/store/provider_store';
import { sendStatusEmail } from '../../src/services/email';

// Mock email service
jest.mock('../../src/services/email');

describe('Investor Interest - Unit', () => {
  beforeEach(() => {
    ProviderStore.clear(); // reset in‑memory store
    sendStatusEmail.mockClear();
  });

  test('listInterest returns only providers with signalInterest flag', async () => {
    ProviderStore.add({ id: 1, name: 'A', score: 85, signalInterest: true });
    ProviderStore.add({ id: 2, name: 'B', score: 70, signalInterest: false });

    const res = await request(app).get('/investor/interest');
    expect(res.status).toBe(200);
    expect(res.body).toHaveLength(1);
    expect(res.body[0].id).toBe(1);
  });

  test('listInterest sorts providers by descending score', async () => {
    ProviderStore.add({ id: 1, name: 'Low', score: 40, signalInterest: true });
    ProviderStore.add({ id: 2, name: 'High', score: 90, signalInterest: true });

    const res = await request(app).get('/investor/interest');
    expect(res.body[0].id).toBe(2);
    expect(res.body[1].id).toBe(1);
  });

  test('statusUpdate persists new status', async () => {
    ProviderStore.add({ id: 10, name: 'C', score: 77, status: 'IN_REVIEW', email: 'c@example.com' });

    const res = await request(app)
      .post('/investor/interest/10/status')
      .send({ status: 'DEAL_CLOSED' });

    expect(res.status).toBe(200);
    expect(res.body.status).toBe('DEAL_CLOSED');

    // Verify store mutated
    const stored = ProviderStore.get(10);
    expect(stored.status).toBe('DEAL_CLOSED');
  });

  test('invalid status yields 422', async () => {
    ProviderStore.add({ id: 5, name: 'D', score: 55, email: 'd@example.com' });

    const res = await request(app)
      .post('/investor/interest/5/status')
      .send({ status: 'UNKNOWN' });

    expect(res.status).toBe(422);
    expect(res.body.detail).toContain('value is not a valid enumeration member');
  });

  test('email service called once per status change', async () => {
    ProviderStore.add({ id: 7, name: 'E', score: 88, email: 'e@example.com' });

    await request(app)
      .post('/investor/interest/7/status')
      .send({ status: 'REJECTED' });

    expect(sendStatusEmail).toHaveBeenCalledTimes(1);
    expect(sendStatusEmail).toHaveBeenCalledWith(
      expect.objectContaining({ to: 'e@example.com', newStatus: 'REJECTED' })
    );
  });
});
