 import { Router } from 'express';
 import { db } from '../db';

 const router = Router();

+// PATCH /requests/:id/status - update status and record actor and timestamp
+router.patch('/requests/:id/status', async (req, res) => {
+  try {
+    const { id } = req.params;
+    const { status, note } = req.body;
+    const actor = req.user.username; // assuming req.user is the authenticated user
+    const updatedAt = new Date().toISOString();
+
+    // Update status in database
+    await db.requests.updateOne({ _id: id }, { $set: { status, note, updatedAt } });
+
+    // Record status change in audit trail
+    await db.audit.insertOne({ requestId: id, status, note, actor, updatedAt });
+
+    res.json({ ok: true, data: { status, note } });
+  } catch (err) {
+    console.error('Failed to update status:', err);
+    res.status(500).json({ ok: false, error: 'Failed to update status' });
+  }
+});

 // Existing GET /requests/:id/audit - chronological audit trail for a request
