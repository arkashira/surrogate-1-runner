import asyncio
import time
import uuid
from dataclasses import dataclass
from typing import Any, Callable, Dict, Optional

import httpx
from tenacity import (
    retry,
    stop_after_attempt,
    wait_exponential,
    retry_if_exception_type,
    before_sleep_log,
    retry_if_result,
)

from .base import BrokerAdapter, OrderFill, OrderRejected

import logging

logger = logging.getLogger(__name__)

@dataclass
class RESTLiveConfig:
    base_url: str
    api_key: str
    timeout: float = 5.0
    max_retries: int = 3
    idempotency_key_header: str = "Idempotency-Key"
    ack_deadline_ms: float = 200.0

    """
    Live broker REST adapter with retries and idempotency.

    Behavior:
    - submit_order sends order to broker REST endpoint.
    - Uses idempotency key to allow safe retries.
    - Retries on network errors, 5xx, and 429; does not retry 4xx (client errors).
    - Returns standardized ack with status accepted/rejected and latency_ms.
    - Emits OrderFill/OrderRejected events via provided callback for event store.
    """

    def __init__(
        self,
        config: RESTLiveConfig,
        event_callback: Optional[Callable[[Dict[str, Any]], None]] = None,
        http_client: Optional[httpx.AsyncClient] = None,
    ):
        self.config = config
        self.event_callback = event_callback
        self._client = http_client or httpx.AsyncClient(
            timeout=httpx.Timeout(config.timeout),
            headers={"Authorization": f"Bearer {config.api_key}", "Content-Type": "application/json"},
        )

        await self._client.aclose()
