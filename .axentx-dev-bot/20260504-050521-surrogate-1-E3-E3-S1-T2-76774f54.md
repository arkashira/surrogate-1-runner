];

const fetchBenchmarks = async (
  sizeFilter: string,
  domainFilter: string
): Promise<Benchmark[]> => {

  await new Promise((r) => setTimeout(r, 400));

  return ALL_BENCHMARKS.filter((b) => {
    const sizeMatch = sizeFilter ? b.size === sizeFilter : true;
    const domainMatch = domainFilter ? b.domain === domainFilter : true;
    return sizeMatch && domainMatch;
  });
};

/* ------------------------------------------------------------
 *  Component

const BenchmarkComparison: React.FC = () => {
