function costCalculatorMiddleware(req, res, next) {

  res.on('finish', () => {

    // Write cost metadata to API response
    res.locals.cost = {

      duration: duration.toFixed(2),
      cost: cost,

    // Write cost tracking to log file
    const logEntry = {

      timestamp: new Date().toISOString(),
      duration: duration.toFixed(2),
      cost: cost,

  });

  next();
}
