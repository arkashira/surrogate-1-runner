// Define the SLA calculator function
export function calculateSlaState(request: Request, latestStatusTimestamp: Date): SlaState {
    const slaTarget = request.slaTarget;
    const timeDiff = latestStatusTimestamp.getTime() - request.createdAt.getTime();

    const slaThreshold = slaTarget * 1000 * 60 * 60; // Convert hours to milliseconds
    const timeSinceLastStatus = timeDiff / 1000 / 60 / 60; // Convert milliseconds to hours

    if (timeSinceLastStatus <= slaThreshold) {
        return 'on_track';
    } else if (timeSinceLastStatus <= slaThreshold * 1.5) {
        return 'at_risk';
    } else {
        return 'breached';
    }
}

// Define the risk thresholds
export function getRiskThresholds(slaTarget: number): { breached: number; atRisk: number } {
    const slaThreshold = slaTarget * 1000 * 60 * 60; // Convert hours to milliseconds
    const riskThresholds = {
        breached: slaThreshold,
        atRisk: slaThreshold * 1.5,
    };
    return riskThresholds;
}

import { Router } from 'express';
import { RequestModel } from '../../models/request';
import { calculateSlaState, getRiskThresholds } from '../../lib/sla';

const router = Router();

/**
 * GET /reports/sla
 * Returns counts by SLA state for requests created in the last 7 days.
 * Accessible only to authenticated platform team members.
 */
router.get('/reports/sla', async (req, res) => {
    try {
        // Auth: platform team member required
        // Assumes auth middleware sets req.user with roles/groups
        const user = (req as any).user;
        if (!user || !user.roles?.includes('platform')) {
            return res.status(403).json({ error: 'Forbidden: platform access required' });
        }

        const sevenDaysAgo = new Date();
        sevenDaysAgo.setUTCDate(sevenDaysAgo.getUTCDate() - 7);

        const requests = await RequestModel.find({
            createdAt: { $gte: sevenDaysAgo },
        }).lean();

        const withState = requests.map((r) => ({
            state: calculateSlaState(r, r.latestStatusTimestamp),
        }));

        const riskThresholds = getRiskThresholds(72); // Assuming default slaTarget of 72 hours

        const summary = withState.reduce(
            (acc, r) => {
                acc[r.state] = (acc[r.state] || 0) + 1;
                return acc;
            },
            { on_track: 0, at_risk: 0, breached: 0 } as Record<SlaState, number>
        );

        return res.json({
            window: {
                from: sevenDaysAgo.toISOString(),
                to: new Date().toISOString(),
            },
            counts: summary,
            total: requests.length,
            riskThresholds,
        });
    } catch (err) {
        console.error('SLA report error:', err);
        return res.status(500).json({ error: 'Internal server error' });
    }
});

export default router;
```
