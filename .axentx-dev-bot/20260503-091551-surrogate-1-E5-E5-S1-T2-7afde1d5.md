from datetime import datetime, timezone
from sqlalchemy import Column, String, DateTime, ForeignKey, Enum as SQLEnum, Index
from sqlalchemy.orm import relationship, declared_attr
from .base import Base
import enum
import uuid


class ActivityType(str, enum.Enum):
    created = "created"
    status_transition = "status_transition"


class RequestActivity(Base):
    __tablename__ = "request_activities"

    id = Column(String, primary_key=True, default=lambda: str(uuid.uuid4()))
    request_id = Column(String, ForeignKey("requests.id", ondelete="CASCADE"), nullable=False)
    team_id = Column(String, nullable=False)
    event_type = Column(SQLEnum(ActivityType, create_constraint=True, values_callable=lambda x: [e.value for e in x]), nullable=False)
    from_status = Column(String, nullable=True)
    to_status = Column(String, nullable=True)
    timestamp = Column(DateTime(timezone=True), nullable=False, default=lambda: datetime.now(timezone.utc))

    # Optional bidirectional navigation; keep lightweight
    @declared_attr
    def request(self):
        return relationship("Request", back_populates="activities")

    __table_args__ = (
        Index("ix_request_activities_team_timestamp", "team_id", "timestamp"),
        Index("ix_request_activities_request", "request_id"),
    )

    def to_dict(self):
        return {
            "id": self.id,
            "request_id": self.request_id,
            "team_id": self.team_id,
            "event_type": self.event_type,
            "from_status": self.from_status,
            "to_status": self.to_status,
            "timestamp": self.timestamp.isoformat() if self.timestamp else None,
        }
