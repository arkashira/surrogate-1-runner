import csv
import io
from datetime import datetime
from typing import Any, Dict, Iterator, List
from sqlalchemy.orm import joinedload


def _isoformat(value: Any) -> str:
    if not value:
        return ""
    if isinstance(value, datetime):
        return value.isoformat()
    return str(value)


def _flatten_status_history(request: Dict[str, Any]) -> Iterator[Dict[str, Any]]:
    """
    Yield one row per status transition for a request.
    If no history, emit a single row with current state.
    """
    base = {
        "request_id": request.get("id"),
        "request_title": request.get("title", ""),
        "created_at": _isoformat(request.get("created_at")),
        "closed_at": _isoformat(request.get("closed_at")),
        "resolution_time_seconds": request.get("resolution_time_seconds"),
        "sla_status": request.get("sla_status", ""),
        "assigned_to": request.get("assigned_to", ""),
        "team_id": request.get("team_id", ""),
    }

    status_history: List[Dict[str, Any]] = request.get("status_history", [])
    if not status_history:
        yield {
            **base,
            "status": request.get("status", ""),
            "changed_at": _isoformat(request.get("updated_at", request.get("created_at"))),
            "previous_status": "",
        }
        return

    sorted_history = sorted(status_history, key=lambda h: h.get("changed_at", ""))
    previous_status = ""
    for entry in sorted_history:
        yield {
            **base,
            "status": entry.get("status", ""),
            "changed_at": _isoformat(entry.get("changed_at")),
            "previous_status": previous_status,
        }
        previous_status = entry.get("status", "")


def export_request_timelines_to_csv(requests: List[Dict[str, Any]]) -> Iterator[str]:
    """
    Stream CSV rows for request timelines.
    Field order is stable and includes all timeline + core request fields.
    """
    fieldnames = [
        "request_id",
        "request_title",
        "created_at",
        "closed_at",
        "resolution_time_seconds",
        "sla_status",
        "status",
        "changed_at",
        "previous_status",
        "assigned_to",
        "team_id",
    ]

    buffer = io.StringIO()
    writer = csv.DictWriter(buffer, fieldnames=fieldnames, extrasaction="ignore")
    writer.writeheader()
    yield buffer.getvalue()
    buffer.seek(0)
    buffer.truncate(0)

    for request in requests:
        for row in _flatten_status_history(request):
            writer.writerow(row)
            yield buffer.getvalue()
            buffer.seek(0)
            buffer.truncate(0)
