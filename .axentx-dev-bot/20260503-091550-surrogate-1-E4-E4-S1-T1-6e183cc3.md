from __future__ import annotations

import json
import logging
import multiprocessing
import os
import smtplib
import time
import urllib.request
from email.mime.text import MIMEText
from typing import Any, Dict, List, Optional

logger = logging.getLogger(__name__)

# ---- Queue (singleton, managed) ----
_manager = multiprocessing.Manager()
_queue = _manager.Queue()

# ---- Config (env) ----
SLACK_WEBHOOK_URL = os.getenv("SLACK_WEBHOOK_URL")
SMTP_HOST = os.getenv("SMTP_HOST")
SMTP_PORT = int(os.getenv("SMTP_PORT", "587"))
SMTP_USER = os.getenv("SMTP_USER")
SMTP_PASS = os.getenv("SMTP_PASS")
NOTIFY_FROM = os.getenv("NOTIFY_FROM", "notifications@axentx.local")
NOTIFY_TO = os.getenv("NOTIFY_TO", "")  # comma-separated default list
PUBLIC_URL_BASE = os.getenv("PUBLIC_URL_BASE", "https://requests.example.com")

# ---- Worker process ----
_worker_process: Optional[multiprocessing.Process] = None


# ---- Public API ----
def enqueue_notification(
    request_id: str,
    old_status: str,
    new_status: str,
    subscribers: Optional[List[str]] = None,
) -> None:
    """
    Non-blocking enqueue of a status-change notification.
    subscribers: explicit list; if omitted, NOTIFY_TO env is used as fallback.
    """
    if subscribers is None:
        subscribers = [a.strip() for a in NOTIFY_TO.split(",") if a.strip()]

    payload: Dict[str, Any] = {
        "request_id": request_id,
        "old_status": old_status,
        "new_status": new_status,
        "subscribers": subscribers,
        "public_url": f"{PUBLIC_URL_BASE.rstrip('/')}/r/{request_id}",
        "timestamp": time.time(),
    }

    try:
        _queue.put(payload, block=False)
        logger.info(
            "Enqueued notification request=%s %s -> %s", request_id, old_status, new_status
        )
    except Exception as exc:
        logger.warning("Failed to enqueue notification request=%s: %s", request_id, exc)
