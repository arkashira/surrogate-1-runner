import pandas as pd
from sklearn.ensemble import IsolationForest
from datetime import datetime, timedelta
import smtplib
from email.mime.text import MIMEText

class AnomalyDetection:
    def __init__(self, data_path):
        self.data_path = data_path
        self.model = IsolationForest(contamination=0.01)

    def load_data(self):
        return pd.read_csv(self.data_path)

    def preprocess_data(self, data):
        # Assuming 'cost' is the column we're monitoring for anomalies
        return data[['cost']]

    def train_model(self, preprocessed_data):
        self.model.fit(preprocessed_data)

    def detect_anomalies(self, data):
        preprocessed_data = self.preprocess_data(data)
        predictions = self.model.predict(preprocessed_data)
        anomalies = data[predictions == -1]
        return anomalies

    def send_alert(self, anomalies):
        if not anomalies.empty:
            msg = MIMEText(f"Anomalies detected:\n{anomalies.to_string()}")
            msg['Subject'] = 'Cloud Infrastructure Cost Anomalies Detected'
            msg['From'] = 'cloudoptimize@example.com'
            msg['To'] = 'platform-engineer@example.com'

            s = smtplib.SMTP('localhost')
            s.send_message(msg)
            s.quit()

def main():
    anomaly_detector = AnomalyDetection('/path/to/cloud_costs.csv')
    data = anomaly_detector.load_data()
    anomaly_detector.train_model(anomaly_detector.preprocess_data(data))
    
    while True:
        current_data = anomaly_detector.load_data()
        anomalies = anomaly_detector.detect_anomalies(current_data)
        anomaly_detector.send_alert(anomalies)
        
        # Check for anomalies every hour
        time.sleep(3600)

if __name__ == "__main__":
    main()

import { AnomalyDetection } from '../dashboard/anomaly_display';
import { EmailService } from '../services/email';
import mockData from './fixtures/cost_data.json';

describe('AnomalyDetection class unit tests', () => {
  let detector;
  let emailMock;

  beforeEach(() => {
    emailMock = jest.spyOn(EmailService, 'send').mockResolvedValue(true);
    detector = new AnomalyDetection({
      modelParams: { windowSize: 30, threshold: 2.5 },
      alertRecipients: ['user1@example.com', 'user2@example.com'],
    });
  });

  test('loadData parses CSV correctly', async () => {
    const data = await detector.loadData('test/fixtures/sample.csv');
    expect(data).toHaveLength(100);
    expect(data[0]).toMatchObject({ timestamp: expect.any(Date), cost: expect.any(Number) });
  });

  test('preprocessData fills missing dates and normalizes cost', () => {
    const raw = [{ timestamp: new Date('2023-01-01'), cost: 120 }, { timestamp: new Date('2023-01-03'), cost: 130 }];
    const processed = detector.preprocessData(raw);
    // Expect a missing 2023-01-02 entry filled with interpolated cost
    expect(processed).toHaveLength(3);
    expect(processed[1].cost).toBeCloseTo(125);
  });

  test('trainModel creates a model with expected properties', () => {
    detector.trainModel(mockData);
    expect(detector.model).toBeDefined();
    expect(detector.model.windowSize).toBe(30);
  });

  test('detectAnomalies returns only points beyond threshold', () => {
    detector.trainModel(mockData);
    const anomalies = detector.detectAnomalies(mockData);
    anomalies.forEach(a => {
      const z = (a.cost - detector.model.mean) / detector.model.std;
      expect(Math.abs(z)).toBeGreaterThanOrEqual(2.5);
    });
  });

  test('sendAlert composes correct email payload', async () => {
    const anomaly = { timestamp: new Date(), cost: 999 };
    await detector.sendAlert(anomaly);
    expect(emailMock).toHaveBeenCalledTimes(2); // two recipients
    const payload = emailMock.mock.calls[0][0];
    expect(payload.to).toContain('user1@example.com');
    expect(payload.subject).toMatch(/Cost Anomaly Detected/);
    expect(payload.body).toContain(anomaly.cost.toString());
    expect(payload.body).toContain('https://cloudoptimize.example.com/dashboard');
  });

  test('throttleAlert prevents duplicate alerts within same event', async () => {
    const anomaly = { id: 'a1', timestamp: new Date(), cost: 800 };
    await detector.sendAlert(anomaly);
    await detector.sendAlert(anomaly); // second call should be ignored
    expect(emailMock).toHaveBeenCalledTimes(2); // only first call sent
  });
});
