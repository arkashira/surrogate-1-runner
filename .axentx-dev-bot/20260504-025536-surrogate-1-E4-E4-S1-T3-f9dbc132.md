Dashboard customization module.

Provides data structures and helper functions to allow finance team members
to configure which widgets appear on the LLM usage and API cost dashboards.

from __future__ import annotations

import json
import os
from dataclasses import dataclass, field, asdict
from typing import Dict, List, Optional, Any, Union

# --------------------------------------------------------------------------- #
# Configuration data structures
# --------------------------------------------------------------------------- #

@dataclass
class WidgetConfig:
    """
    Configuration for a single dashboard widget.

    Attributes:
        id: Unique identifier for the widget.
        type: Type of widget (e.g., 'token_usage', 'api_cost', 'custom_chart').
        title: Human‑readable title.
        options: Widget‑specific options dictionary.
    """
    id: str
    type: str
    title: str
    options: Dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to a plain dict."""
        return asdict(self)

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "WidgetConfig":
        """Deserialize from a dict."""
        return cls(
            id=data["id"],
            type=data["type"],
            title=data["title"],
            options=data.get("options", {}),
        )


@dataclass
class DashboardConfig:
    """
    Configuration for an entire dashboard.

    Attributes:
        name: Dashboard name (e.g., 'LLM Usage', 'API Costs').
        widgets: Ordered list of widgets to display.
        layout: Optional layout specification (e.g., grid, tabs).
    """
    name: str
    widgets: List[WidgetConfig] = field(default_factory=list)
    layout: Optional[Dict[str, Any]] = None

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to a plain dict."""
        return {
            "name": self.name,
            "widgets": [w.to_dict() for w in self.widgets],
            "layout": self.layout,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "DashboardConfig":
        """Deserialize from a dict."""
        widgets = [WidgetConfig.from_dict(w) for w in data.get("widgets", [])]
        return cls(
            name=data["name"],
            widgets=widgets,
            layout=data.get("layout"),
        )

    def add_widget(self, widget: WidgetConfig) -> None:
        """Add a widget to the dashboard."""
        self.widgets.append(widget)

    def remove_widget(self, widget_id: str) -> bool:
        """Remove a widget by id. Returns True if removed."""
        for i, w in enumerate(self.widgets):
            if w.id == widget_id:
                del self.widgets[i]
                return True
        return False

    def find_widget(self, widget_id: str) -> Optional[WidgetConfig]:
        """Find a widget by id."""
        for w in self.widgets:
            if w.id == widget_id:
                return w
        return None
