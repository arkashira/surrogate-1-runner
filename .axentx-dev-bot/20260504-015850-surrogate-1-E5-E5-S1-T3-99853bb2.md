```go
func ProjectIdentifier(packageName, repoURL string) string {
    hasher := sha256.New()
    hasher.Write([]byte(packageName + "|" + repoURL))
    return hex.EncodeToString(hasher.Sum(nil))[:16]
}
```
- Uses SHA-256 for cryptographic security
- Combines package name and repo URL with pipe separator
- Truncates to 16 characters for brevity while maintaining uniqueness

### 3. Metrics Store
```go
type MetricsStore struct {
    mu    sync.RWMutex
    events []ScanEvent
}

func NewMetricsStore() *MetricsStore {
    return &MetricsStore{
        events: make([]ScanEvent, 0),
    }
}

func (m *MetricsStore) RecordScanEvent(event ScanEvent) {
    m.mu.Lock()
    defer m.mu.Unlock()
    m.events = append(m.events, event)
}

func (m *MetricsStore) GetWeeklyAggregation(weekStart time.Time) WeeklyAggregation {
    m.mu.RLock()
    defer m.mu.RUnlock()

    weekEnd := weekStart.AddDate(0, 0, 7)
    
    projectMap := make(map[string]*ProjectMetrics)
    var totalVulns int
    var scanCount int

    for _, event := range m.events {
        if event.Timestamp.After(weekStart) && event.Timestamp.Before(weekEnd) {
            scanCount++
            totalVulns += event.VulnerabilityCount

            if pm, ok := projectMap[event.ProjectID]; ok {
                pm.ScanCount++
                pm.TotalVulns += event.VulnerabilityCount
                if event.Timestamp.After(pm.LatestScan) {
                    pm.LatestScan = event.Timestamp
                }
            } else {
                projectMap[event.ProjectID] = &ProjectMetrics{
                    ProjectID:  event.ProjectID,
                    ScanCount:  1,
                    TotalVulns: event.VulnerabilityCount,
                    LatestScan: event.Timestamp,
                }
            }
        }
    }

    projects := make([]ProjectMetrics, 0, len(projectMap))
    for _, pm := range projectMap {
        projects = append(projects, *pm)
    }

    return WeeklyAggregation{
        WeekStart:            weekStart,
        WeekEnd:              weekEnd,
        TotalScans:           scanCount,
        UniqueProjects:       len(projectMap),
        TotalVulnerabilities: totalVulns,
        Projects:             projects,
    }
}
