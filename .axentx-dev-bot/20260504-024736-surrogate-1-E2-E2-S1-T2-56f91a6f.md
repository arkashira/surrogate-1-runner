    """
    Thin wrapper around the official Kubernetes Python client that:
    * auto‑detects the config source,
    * caches deployments for 30 s,
    * exposes a typed API for the most common operations.
    """

    def __init__(self, namespace: str = "default", use_incluster: Optional[bool] = None):
        """
