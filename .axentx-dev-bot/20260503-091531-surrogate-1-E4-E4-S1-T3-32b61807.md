import { Router, Request, Response } from 'express';
import crypto from 'crypto';
import { indexingQueue } from '../queues/indexingQueue';
import { handleGitHubPush } from '../webhooks/github';

const router = Router();

const WEBHOOK_SECRET = process.env.GITHUB_WEBHOOK_SECRET || '';

function verifySignature(payload: Buffer, signature: string): boolean {
  if (!WEBHOOK_SECRET) return false;
  const hmac = crypto.createHmac('sha256', WEBHOOK_SECRET);
  const digest = 'sha256=' + hmac.update(payload).digest('hex');
  return crypto.timingSafeEqual(Buffer.from(signature), Buffer.from(digest));
}

router.post('/github', async (req: Request, res: Response) => {
  const sig = req.headers['x-hub-signature-256'] as string;
  const payload = Buffer.from(JSON.stringify(req.body));

  if (!verifySignature(payload, sig)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const event = req.headers['x-github-event'] as string;
  if (event !== 'push') {
    return res.status(202).json({ message: 'Ignored non-push event' });
  }

  const body = req.body;
  const repo = body.repository?.full_name;
  const ref = body.ref;
  const commits = body.commits || [];

  // Collect changed files (adds + edits)
  const changedFiles = new Set<string>();
  for (const c of commits) {
    for (const f of c.added || []) changedFiles.add(f);
    for (const f of c.modified || []) changedFiles.add(f);
    // ignore removed
  }

  // Enqueue for async processing (fast 202 response)
  await indexingQueue.add(
    'github-push',
    { repo, ref, commits: commits.length, changedFiles: Array.from(changedFiles) },
    { attempts: 3, backoff: { type: 'exponential', delay: 5000 } }
  );

  res.status(202).json({ queued: true, files: changedFiles.size });
});

export default router;
```
