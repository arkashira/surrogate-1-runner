"""
Stripe Revenue Sync Job
~~~~~~~~~~~~~~~~~~~~~~~

Pulls the previous day’s balance transactions from Stripe and writes a
CSV to a Google Cloud Storage bucket.  Designed to run nightly via
GitHub Actions.

Environment variables
---------------------
STRIPE_SECRET_KEY          – Stripe secret key (required)
GOOGLE_APPLICATION_CREDENTIALS – GCP service‑account key (required)
GCS_BUCKET                 – GCS bucket name (required)
DATA_LAKE_PATH             – Optional prefix inside the bucket (default: data/stripe)

The script is intentionally small so that it can be unit‑tested easily.
"""

from __future__ import annotations

import csv
import datetime
import json
import logging
import os
import sys
from pathlib import Path
from typing import Iterable, List, Mapping, Optional

import pandas as pd
import stripe
from google.cloud import storage
from google.oauth2 import service_account

# --------------------------------------------------------------------------- #
# Logging
# --------------------------------------------------------------------------- #
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(name)s %(message)s",
    stream=sys.stdout,
)
log = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Configuration helpers
# --------------------------------------------------------------------------- #
def _require_env(name: str) -> str:
    """Return the value of an env var or raise a helpful error."""
    val = os.getenv(name)
    if not val:
        raise RuntimeError(f"Missing required environment variable: {name}")
    return val


# --------------------------------------------------------------------------- #
# Stripe client
# --------------------------------------------------------------------------- #
class StripeClient:
    """Thin wrapper around the official stripe library."""

    def __init__(self, api_key: str, account_id: Optional[str] = None) -> None:
        if not api_key:
            raise ValueError("API key must be provided")
        stripe.api_key = api_key
        self.account_id = account_id

    def list_balance_transactions(
        self,
        start: datetime.datetime,
        end: datetime.datetime,
        limit: int = 100,
    ) -> List[Mapping]:
        """
        Return all balance transactions between *start* (inclusive) and *end*
        (exclusive).  Pagination is handled automatically.
        """
        params = {
            "created": {"gte": int(start.timestamp()), "lt": int(end.timestamp())},
            "limit": limit,
        }
        if self.account_id:
            params["stripe_account"] = self.account_id

        log.debug("Fetching balance transactions: %s", params)
        return list(
            stripe.BalanceTransaction.list(**params).auto_paging_iter()
        )
