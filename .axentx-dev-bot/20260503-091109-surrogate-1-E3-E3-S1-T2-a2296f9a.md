const router = express.Router();

router.get('/', async (req, res) => {
  try {
    const start = Date.now();

    const documents = await prisma.document.findMany({
      select: {
        id: true,
        title: true,
        source: true,
        opened_at: true,
        open_count: true
      },
      orderBy: [
        { opened_at: 'desc' },
        { open_count: 'desc' }
      ],
      take: 1000
    });

    const elapsed = Date.now() - start;
    if (elapsed > 450) {
      console.warn(`Library index slow: ${elapsed}ms`);
    }

    res.json({
      documents: documents.map(d => ({
        ...d,
        last_opened_at: d.opened_at
      })),
      meta: { count: documents.length, elapsed_ms: elapsed }
    });
  } catch (err) {
    console.error('Library index error:', err);
    res.status(500).json({ error: 'Failed to load library index' });
  }
});

// POST /api/documents/:id/open — record open
router.post('/:id/open', async (req, res) => {
  try {
    const { id } = req.params;

    const updated = await prisma.document.update({
      where: { id },
      data: {
        open_count: { increment: 1 },
        opened_at: new Date()
      },
      select: {
        id: true,
        title: true,
        source: true,
        opened_at: true,
        open_count: true
      }
    });

    res.json({ document: { ...updated, last_opened_at: updated.opened_at } });
  } catch (err) {
    console.error('Record open error:', err);
    res.status(404).json({ error: 'Document not found' });
  }
});

module.exports = router;
