from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, Optional
import uuid


class StageName(str, Enum):
    RESEARCH = "research"
    RISK = "risk"
    EXECUTION = "execution"


class EventType(str, Enum):
    STAGE_START = "stage_start"
    STAGE_SUCCESS = "stage_success"
    STAGE_FAILURE = "stage_failure"


@dataclass(frozen=True)
class PipelineEvent:
    """Structured event emitted by pipeline stages with monotonic sequence IDs."""
    event_id: str
    pipeline_id: str
    stage: StageName
    event_type: EventType
    sequence: int
    timestamp: datetime
    payload: Dict[str, Any] = field(default_factory=dict)
    error: Optional[str] = None
    input_hash: Optional[str] = None  # deterministic input fingerprint for replay

    @classmethod
    def create(
        cls,
        pipeline_id: str,
        stage: StageName,
        event_type: EventType,
        sequence: int,
        payload: Dict[str, Any],
        *,
        error: Optional[str] = None,
        input_hash: Optional[str] = None,
    ) -> "PipelineEvent":
        return cls(
            event_id=str(uuid.uuid4()),
            pipeline_id=pipeline_id,
            stage=stage,
            event_type=event_type,
            sequence=sequence,
            timestamp=datetime.now(timezone.utc),
            payload=payload,
            error=error,
            input_hash=input_hash,
        )

    def to_dict(self) -> Dict[str, Any]:
        return {
            "event_id": self.event_id,
            "pipeline_id": self.pipeline_id,
            "stage": self.stage.value,
            "event_type": self.event_type.value,
            "sequence": self.sequence,
            "timestamp": self.timestamp.isoformat(),
            "payload": self.payload,
            "error": self.error,
            "input_hash": self.input_hash,
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "PipelineEvent":
        return cls(
            event_id=data["event_id"],
            pipeline_id=data["pipeline_id"],
            stage=StageName(data["stage"]),
            event_type=EventType(data["event_type"]),
            sequence=data["sequence"],
            timestamp=datetime.fromisoformat(data["timestamp"]).replace(tzinfo=timezone.utc),
            payload=data.get("payload", {}),
            error=data.get("error"),
            input_hash=data.get("input_hash"),
