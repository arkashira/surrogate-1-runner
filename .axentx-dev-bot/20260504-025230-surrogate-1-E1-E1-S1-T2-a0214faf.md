        const responseList = document.getElementById('response-list');

        const endpointInput = document.getElementById('endpoint');

        let responses = [];

        function addResponse(response) {
            responses.push(response);
            filterAndDisplayResponses();

        function filterAndDisplayResponses() {
            responseList.innerHTML = '';
            const filteredResponses = responses.filter(response => {
                const modelSelected = modelSelect.value === '' || response.model === modelSelect.value;

                const startTimeValid = !startTimeInput.value || new Date(response.timestamp) >= new Date(startTimeInput.value);
                const endTimeValid = !endTimeInput.value || new Date(response.timestamp) <= new Date(endTimeInput.value);

                return modelSelected && endpointMatch && startTimeValid && endTimeValid;
            });

            filteredResponses.forEach(response => {
                const item = document.createElement('div');
                item.className = 'response-item';
                item.textContent = `Model: ${response.model}, Endpoint: ${response.endpoint}, Timestamp: ${response.timestamp}, Response: ${JSON.stringify(response.data)}`;
                responseList.appendChild(item);
            });
        }

        function filterResponses() {
            filterAndDisplayResponses();

        // Simulate real-time data stream
        setInterval(() => {
            fetch('/api/responses')
                .then(response => response.json())
                .then(data => {
                    data.forEach(addResponse);
                });
        }, 1000);
    </script>
</body>
