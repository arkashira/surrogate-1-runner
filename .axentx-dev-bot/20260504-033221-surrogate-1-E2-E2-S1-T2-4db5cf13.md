describe('HighImpactFilter', () => {

  const owners = ['alice', 'bob'];

  let filter;

      ownerUsers: owners,
      minConfidenceThreshold: defaultThreshold,
    });

  });

  test('AC3: owner user with no keywords gives confidence 85', () => {

    expect(result[0].confidence).toBe(85);
