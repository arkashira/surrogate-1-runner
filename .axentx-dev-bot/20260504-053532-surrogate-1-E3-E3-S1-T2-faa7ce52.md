app = Dash(__name__)

    html.H2('Cloud Infrastructure Cost Forecast'),
    dcc.Graph(id='forecast-graph'),
    dcc.Dropdown(

        options=[

            {'label': 'Param 2', 'value': 'param2'},
        ],
        value='param1',
    ),
])
