### 1. `bin/lib/cdn_utils.py` (new)
```python
import hashlib
import json
from pathlib import Path
from typing import List, Dict

HF_DATASETS_ROOT = "https://huggingface.co/datasets"
SIBLING_REPOS = [
    "axentx/surrogate-1-training-pairs",
    "axentx/surrogate-1-training-pairs-sib1",
    "axentx/surrogate-1-training-pairs-sib2",
    "axentx/surrogate-1-training-pairs-sib3",
    "axentx/surrogate-1-training-pairs-sib4",
]

def cdn_url(repo: str, filepath: str) -> str:
    """Generate CDN URL that bypasses HF API auth/rate limits."""
    return f"{HF_DATASETS_ROOT}/{repo}/resolve/main/{filepath}"

def sibling_repo_for_slug(slug: str) -> str:
    """Deterministic sibling repo selection to spread commit load."""
    idx = int(hashlib.md5(slug.encode()).hexdigest(), 16) % len(SIBLING_REPOS)
    return SIBLING_REPOS[idx]

def shard_for_slug(slug: str, n_shards: int = 16) -> int:
    """Deterministic shard assignment (consistent with workflow matrix)."""
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % n_shards

def snapshot_date_files(repo: str, date: str, out_path: Path) -> List[Dict]:
    """Single API call: list top-level date folder (no recursion)."""
    from huggingface_hub import list_repo_tree
    items = list_repo_tree(repo, path=date, recursive=False)
    files = [
        {"path": f.rfilename, "cdn_url": cdn_url(repo, f.rfilename)}
        for f in items
        if f.rfilename.endswith(('.jsonl', '.parquet', '.json'))
    ]
    out_path.write_text(json.dumps(files, indent=2))
    return files
```
