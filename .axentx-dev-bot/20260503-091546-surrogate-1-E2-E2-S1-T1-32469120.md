```ts
import type { Request, Response, NextFunction } from 'express';

export type Role = 'owner' | 'admin' | 'member' | 'viewer';

export interface AuthnUser {
  sub: string;
  roles: Role[];
  requestRoles?: Record<string, Role[]>; // requestId -> roles
}

/**
 * Require an authenticated user with owner/admin role on the target request.
 * Rejects with 401 (no user), 403 (insufficient permissions).
 */
export function canUpdateRequestStatus(req: Request, res: Response, next: NextFunction) {
  const user = req.user as AuthnUser | undefined;
  const requestId = req.params?.id;

  if (!user) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  if (!requestId) {
    return res.status(400).json({ error: 'Request ID required' });
  }

  const roles = user.requestRoles?.[requestId] ?? [];
  const allowed = roles.includes('owner') || roles.includes('admin');

  if (!allowed) {
    return res.status(403).json({ error: 'Forbidden: only owners/admins can update status' });
  }

  next();
}
```
