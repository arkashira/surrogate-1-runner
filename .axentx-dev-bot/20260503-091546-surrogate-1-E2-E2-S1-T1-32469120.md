import { Request, Response, NextFunction } from 'express';

export interface AuthRequest extends Request {
  user?: {
    id: string;
    roles: string[];
  };
}

/**
 * Check if user can update a request's status.
 * Allowed: owner, or roles: request_admin, request_editor
 */
export function canUpdateRequestStatus(req: AuthRequest, res: Response, next: NextFunction) {
  const user = req.user;
  const request = (req as any).request; // set by prior middleware (e.g., loadRequest)

  if (!user) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  if (!request) {
    return res.status(404).json({ error: 'Request not found' });
  }

  const isOwner = request.ownerId === user.id;
  const hasRole = user.roles.some((r) => r === 'request_admin' || r === 'request_editor');

  if (!isOwner && !hasRole) {
    return res.status(403).json({ error: 'Forbidden: insufficient permissions' });
  }

  next();
}
