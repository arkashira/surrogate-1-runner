      status TEXT NOT NULL,
      createdAt TEXT NOT NULL,
      updatedAt TEXT NOT NULL
    );
    CREATE TABLE IF NOT EXISTS timeline (
      id TEXT PRIMARY KEY,

      status TEXT NOT NULL,
      occurredAt TEXT NOT NULL,
      actorId TEXT NOT NULL,
      notes TEXT,

    );
