### 1) Authz middleware (`src/middleware/authz.ts`)

```ts
// src/middleware/authz.ts
import type { Request, Response, NextFunction } from 'express';
import { db } from '../db';

const ALLOWED_ROLES = new Set(['admin', 'manager']);

declare global {
  namespace Express {
    interface Request {
      user?: {
        id: string;
        roles?: string[];
      };
    }
  }
}

export function requireCanUpdateRequestStatus(req: Request, res: Response, next: NextFunction) {
  const requestId = req.params.id;
  const user = req.user;

  if (!user?.id) {
    return res.status(401).json({ error: 'Unauthorized' });
  }

  const request = db.getRequest(requestId);
  if (!request) {
    return res.status(404).json({ error: 'Request not found' });
  }

  const isOwner = request.ownerId === user.id;
  const hasRole = user.roles?.some((r) => ALLOWED_ROLES.has(r)) ?? false;

  if (!isOwner && !hasRole) {
    return res.status(403).json({
      error: 'Forbidden: only owners or privileged roles can update status',
    });
  }

  // attach for downstream handlers
  (req as any).targetRequest = request;
  next();
}
```

### 2) Route: PATCH `/requests/:id/status` (`src/routes/requests.ts`)

```ts
// src/routes/requests.ts
import { Router } from 'express';
import { db } from '../db';
import { requireCanUpdateRequestStatus } from '../middleware/authz';

const router = Router();

/**
 * PATCH /requests/:id/status
 * Body: { status: string }
 * Response: { timeline: Array<{ status: string; occurredAt: string }> }
 */
router.patch(
  '/:id/status',
  requireCanUpdateRequestStatus,
  (req, res) => {
    const { status } = req.body;
    const requestId = req.params.id;

    if (!status || typeof status !== 'string') {
      return res.status(400).json({ error: 'Invalid or missing status' });
    }

    const request = (req as any).targetRequest;

    // Update current status
    db.updateRequestStatus(requestId, status);

    // Append timeline entry
    const timelineEntry = {
      status,
      occurredAt: new Date().toISOString(),
    };
    db.appendTimelineEntry(requestId, timelineEntry);

    // Fetch updated request and return sorted timeline
    const updated = db.getRequest(requestId);
    const timeline = [...updated.timeline].sort(
      (a, b) => new Date(a.occurredAt).getTime() - new Date(b.occurredAt).getTime()
    );

    return res.json({ timeline });
  }
);

export default router;
```
