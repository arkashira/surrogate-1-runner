  const { id, status } = ctx.request.body;
  const ownerId = await getUserId(id);
  if (ownerId === ctx.state.user.id || ctx.state.user.role === 'admin') {
    const updatedRequest = await updateRequest(id, status);
    const timelineEntry = await createTimelineEntry(id, status);
    ctx.body = { status: updatedRequest.status, timeline: timelineEntry };

    ctx.body = { error: 'Forbidden' };
  }
};
