from __future__ import annotations

from dataclasses import dataclass
from typing import List, Optional, Sequence
import numpy as np
from sentence_transformers import SentenceTransformer

MODEL_NAME = "sentence-transformers/all-MiniLM-L6-v2"


@dataclass
class Document:
    id: str
    title: str
    abstract: Optional[str]
    file_type: str  # "PDF" | "link"
    date_created: str  # ISO date "YYYY-MM-DD"
    content_snippet: Optional[str] = None  # short text for highlighting


class Vectorizer:
    """Encode titles + abstracts into 384-dim vectors."""

    def __init__(self, model_name: str = MODEL_NAME):
        self.model = SentenceTransformer(model_name)
        self._dim = self.model.get_sentence_embedding_dimension()

    @property
    def dim(self) -> int:
        return int(self._dim)

    def _build_text(self, doc: Document) -> str:
        title = (doc.title or "").strip()
        abstract = (doc.abstract or "").strip()
        if title and abstract:
            return f"{title}. {abstract}"
        return title or abstract or ""

    def encode_one(self, doc: Document) -> np.ndarray:
        return self.encode_batch([doc])[0]

    def encode_batch(self, docs: Sequence[Document]) -> np.ndarray:
        texts = [self._build_text(d) for d in docs]
        # normalize for cosine similarity in hnswlib
        vecs = self.model.encode(texts, normalize_embeddings=True, show_progress_bar=False)
        return vecs.astype(np.float32)
```
