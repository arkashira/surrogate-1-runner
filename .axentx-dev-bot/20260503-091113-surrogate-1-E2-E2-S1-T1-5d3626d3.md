import os
from typing import List, Sequence

import numpy as np
from sentence_transformers import SentenceTransformer

from src.models.document import Document

MODEL_NAME = os.getenv("EMBEDDING_MODEL", "all-MiniLM-L6-v2")

class Vectorizer:
    """
    Encode document title + abstract into a single vector.
    Uses sentence-transformers; supports batch encoding and caching.
    """

    def __init__(self, model_name: str = MODEL_NAME, device: str | None = None):
        self.model = SentenceTransformer(model_name, device=device)
        self._cache: dict[str, np.ndarray] = {}

    def _text(self, doc: Document) -> str:
        # Keep concatenation simple and deterministic
        title = (doc.title or "").strip()
        abstract = (doc.abstract or "").strip()
        if title and abstract:
            return f"{title}. {abstract}"
        return title or abstract or ""

    def encode_one(self, doc: Document) -> np.ndarray:
        key = doc.id
        if key in self._cache:
            return self._cache[key]
        vec = self.encode_batch([doc])[0]
        self._cache[key] = vec
        return vec

    def encode_batch(self, docs: Sequence[Document]) -> np.ndarray:
        texts = [self._text(d) for d in docs]
        # normalize to unit length for cosine-friendly HNSW
        vecs = self.model.encode(texts, normalize_embeddings=True, show_progress_bar=False)
        vecs = np.asarray(vecs, dtype=np.float32)
        for d, v in zip(docs, vecs):
            self._cache[d.id] = v
        return vecs

    def encode_query(self, query: str) -> np.ndarray:
        vec = self.model.encode([query], normalize_embeddings=True, show_progress_bar=False)
        return np.asarray(vec, dtype=np.float32)[0]
