// Status enum (single source of truth)
export const requestStatus = ['open', 'in_progress', 'resolved', 'closed'] as const;
export type RequestStatus = typeof requestStatus[number];

// Requests table (add/ensure these columns)
export const requests = pgTable(

    status: varchar('status', { enum: requestStatus }).notNull().default('open'),

    createdAt: timestamp('created_at').defaultNow().notNull(),
    updatedAt: timestamp('updated_at').defaultNow().notNull(),
    // other existing columns...
  },
  (table) => ({
    statusIdx: index('requests_status_idx').on(table.status),
  })
);

// Audit table for status transitions

  id: serial('id').primaryKey(),
  requestId: integer('request_id')

  newStatus: varchar('new_status', { enum: requestStatus }).notNull(),

### 2) Validators & SLA — `/opt/axentx/surrogate-1/src/lib/validators.ts`

```ts
import { requestStatus } from '../db/schema';

const statusIndex = (s: string) => requestStatus.indexOf(s as any);

export function isValidTransition(from: string, to: string): boolean {
  const fi = statusIndex(from);
  const ti = statusIndex(to);
  if (fi === -1 || ti === -1) return false;
  // Allow same status (idempotent) or move forward one step only
  return ti === fi || ti === fi + 1;
}

export function slaStatus(
  createdAt: Date,
  targetHours: number
): {
  elapsedHours: number;
  targetHours: number;
  status: 'on_track' | 'at_risk' | 'breached';
} {
  const now = new Date();
  const elapsed = Math.max(0, (now.getTime() - createdAt.getTime()) / (1000 * 60 * 60));
  const pct = targetHours > 0 ? elapsed / targetHours : 0;

  let status: 'on_track' | 'at_risk' | 'breached' = 'on_track';
  if (pct >= 1) status = 'breached';
  else if (pct >= 0.85) status = 'at_risk';

  return {
    elapsedHours: Number(elapsed.toFixed(2)),
    targetHours,
    status,
  };
}
```
