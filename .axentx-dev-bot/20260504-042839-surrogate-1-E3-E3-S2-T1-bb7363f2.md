from sdks.python_sdk import AxentxSDK, AxentxError

def sdk():

def test_add_rag_success(sdk):
    payload = {"question": "q", "answer": "a"}
    mock_resp = MagicMock(status_code=201, json=lambda: {"id": "123", **payload})
    with patch('sdks.python_sdk.requests.post', return_value=mock_resp) as mock_post:
        result = sdk.add_rag(payload)
        mock_post.assert_called_once_with(
            "http://mock/rag",
            json=payload,
            headers={"Authorization": "Bearer test-key"},
            timeout=10
        )
        assert result["id"] == "123"
        assert result["question"] == "q"

def test_get_rag_success(sdk):
    mock_resp = MagicMock(status_code=200, json=lambda: {"id": "123", "question": "q"})
    with patch('sdks.python_sdk.requests.get', return_value=mock_resp) as mock_get:
        result = sdk.get_rag("123")
        mock_get.assert_called_once_with(
            "http://mock/rag/123",
            headers={"Authorization": "Bearer test-key"},
            timeout=10
        )
        assert result["id"] == "123"

def test_add_rag_error(sdk):

        assert "bad request" in str(exc.value)
