compose2k8s –‑ Generate Kubernetes PersistentVolume / PersistentVolumeClaim
objects from a Docker‑Compose file.

Key features
-------------
* **Unified volume model** – works with both the short (``./host:/container``) and long
  (dictionary) syntaxes defined in the Compose specification.
* **Deterministic naming** – PV/PVC names are derived from the volume name (bind‑mounts
  get a predictable prefix).
* **Configurable capacity & access mode** – can be overridden per‑volume or globally.
* **Support for named‑volume “external” flag** – if a volume is declared ``external: true``
  we skip PV creation (the cluster is expected to have it already).
* **CLI & programmatic API** – import the functions or run ``compose2k8s -f docker-compose.yml``.
* **Extensible** – adding a new storage backend (NFS, CSI, etc.) is a matter of
  swapping the ``_host_path_for_volume`` helper or passing a custom ``volume_driver``.
"""

from __future__ import annotations

import os
import sys
import argparse
import yaml
from dataclasses import dataclass, field
from pathlib import Path
from typing import Dict, List, Optional, Any, Union

__all__ = [
    "VolumeInfo",
    "load_compose",
    "extract_volumes",
    "generate_pv_manifest",
    "generate_pvc_manifest",
    "generate_manifests",
    "main",
]

# ----------------------------------------------------------------------
# Data model
# ----------------------------------------------------------------------
@dataclass(frozen=True, order=True)
class VolumeInfo:
    """
    Normalised representation of a volume defined in a Docker‑Compose file.

    Attributes
    ----------
    name: str
        Identifier used in the generated PV/PVC (unique per volume).
    source: str
        Host path for bind mounts or the name of a named volume.
    target: str
        Container mount point (e.g. ``/data``).
    type: str
        ``"bind"`` or ``"named"``.
    read_only: bool
        Whether the mount was declared ``ro`` (read‑only).
    external: bool
        ``True`` if the compose file marks the volume as ``external: true`` –
        in that case we only generate a PVC (the PV is assumed to exist).
    """
    name: str
    source: str
    target: str
    type: str
    read_only: bool = False
    external: bool = False

    def __repr__(self) -> str:
        return (
            f"VolumeInfo(name={self.name!r}, source={self.source!r}, "
            f"target={self.target!r}, type={self.type!r}, "
            f"read_only={self.read_only}, external={self.external})"
        )


# ----------------------------------------------------------------------
# Helpers
# ----------------------------------------------------------------------
def _is_bind_like(source: str) -> bool:
    """
    Heuristic to decide if a short‑syntax source looks like a bind mount.
    Accepts absolute paths, ``./``, ``../`` or Windows drive letters.
    """
    s = source.strip()
    if os.path.isabs(s):
        return True
    if s.startswith(("./", "../", "~")):
        return True
    # simple Windows detection (e.g. "C:\...")
    if len(s) >= 2 and s[1] == ":":
        return True
    return False
