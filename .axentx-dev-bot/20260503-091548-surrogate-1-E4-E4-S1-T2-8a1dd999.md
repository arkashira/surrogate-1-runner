// src/models/team-settings.ts
import { db } from '../db';

export interface TeamSettingsAttributes {
  teamId: string;
  requestTypeId: string;
  defaultSlaHours: number;
  updatedAt: string;
}

export async function getDefaultSlaHours(
  teamId: string,
  requestTypeId: string
): Promise<number | null> {
  const setting = await db<TeamSettingsAttributes>('team_settings')
    .where({ team_id: teamId, request_type_id: requestTypeId })
    .first();
  return setting?.default_sla_hours ?? null;
}

export async function setDefaultSlaHours(
  teamId: string,
  requestTypeId: string,
  defaultSlaHours: number
): Promise<void> {
  await db<TeamSettingsAttributes>('team_settings')
    .insert({
      team_id: teamId,
      request_type_id: requestTypeId,
      default_sla_hours: defaultSlaHours,
      updated_at: new Date().toISOString(),
    })
    .onConflict(['team_id', 'request_type_id'])
    .merge();
}
