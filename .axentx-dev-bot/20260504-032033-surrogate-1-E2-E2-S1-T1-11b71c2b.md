const { Command } = require('commander');
const fs = require('fs');
const path = require('path');

const program = new Command();

program
  .name('surrogate-1')
  .description('CLI to generate synthetic metrics')
  .version('1.0.0');

program
  .command('generate-metrics')
  .description('Generate a predefined set of synthetic metrics')
  .option('--profile <profile>', 'Profile to use for generating metrics', 'basic')
  .action((options) => {
    const metrics = generateMetrics(options.profile);
    const output = JSON.stringify(metrics, null, 2);
    console.log(output);
    fs.writeFileSync(path.join(__dirname, 'metrics.json'), output);
  });

  if (profile === 'basic') {
    return [

      { name: 'metric8', value: 800 },
      { name: 'metric9', value: 900 },
      { name: 'metric10', value: 1000 }
    ];
  }
  throw new Error(`Unknown profile: ${profile}`);
}

// tests/generator.unit.test.js
const { generateMetrics } = require('../src/generator/cli');
const fs = require('fs');
const path = require('path');

describe('generateMetrics', () => {
  const profile = 'basic';
  const outputPath = path.resolve(__dirname, '..', 'src', 'generator', 'metrics.json');

  beforeEach(() => {
    if (fs.existsSync(outputPath)) fs.unlinkSync(outputPath);
  });

  test('returns an array of 10 metrics with correct names and values', () => {
    const metrics = generateMetrics(profile);
    expect(metrics).toHaveLength(10);
    metrics.forEach((m, idx) => {
      expect(m).toHaveProperty('name', `metric${idx + 1}`);
      expect(m).toHaveProperty('value', (idx + 1) * 100);
      expect(m).toHaveProperty('timestamp'); // schema requirement
    });
  });

  test('writes identical JSON to file', () => {
    const metrics = generateMetrics(profile);
    fs.writeFileSync(outputPath, JSON.stringify(metrics, null, 2));
    const fileContent = JSON.parse(fs.readFileSync(outputPath, 'utf8'));
    expect(fileContent).toEqual(metrics);
  });

  test('deterministic output across calls', () => {
    const first = generateMetrics(profile);
    const second = generateMetrics(profile);
    expect(first).toEqual(second);
  });
});
