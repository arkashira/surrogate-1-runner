- [ ] The system flags RDS instances idle for more than 5 days with CPU <10% and zero connections.

- [ ] Alerts are exposed via the `/api/v1/cost-radar/idle-resources` endpoint in the Cost Radar dashboard.
- [ ] Alerts include a recommendation to stop, terminate, or snapshot the resource.
- [ ] The system runs idle detection every 24 hours and updates alert state accordingly.

import { IdleResourceAlertSystem } from './idle-resource-alert-system';
import { CloudWatchClient } from '@aws-sdk/client-cloudwatch';
import { EC2Client } from '@aws-sdk/client-ec2';
import { RDSClient } from '@aws-sdk/client-rds';

describe('IdleResourceAlertSystem', () => {
  let system: IdleResourceAlertSystem;
  let mockEc2Client: jest.Mocked<EC2Client>;
  let mockRdsClient: jest.Mocked<RDSClient>;
  let mockCloudWatchClient: jest.Mocked<CloudWatchClient>;

  beforeEach(() => {
    mockEc2Client = {
      send: jest.fn(),
    } as unknown as jest.Mocked<EC2Client>;

    mockRdsClient = {
      send: jest.fn(),
    } as unknown as jest.Mocked<RDSClient>;

    mockCloudWatchClient = {
      send: jest.fn(),
    } as unknown as jest.Mocked<CloudWatchClient>;

    system = new IdleResourceAlertSystem(
      mockEc2Client,
      mockRdsClient,
      mockCloudWatchClient
    );
  });

  it('should detect EC2 instance idle for 7+ days with low CPU and no network', async () => {
    // Mock CloudWatch: CPU <5%, no network in/out
    mockCloudWatchClient.send.mockResolvedValue({
      Datapoints: Array(7).fill({ Average: 4 }),
    });

    const isIdle = await system.isEc2InstanceIdle('i-1234567890abcdef0', 7);
    expect(isIdle).toBe(true);
  });

  it('should not flag EC2 instance with CPU >5%', async () => {
    mockCloudWatchClient.send.mockResolvedValue({
      Datapoints: Array(7).fill({ Average: 6 }),
    });

    const isIdle = await system.isEc2InstanceIdle('i-1234567890abcdef0', 7);
    expect(isIdle).toBe(false);
  });

  it('should detect RDS instance idle for 5+ days with low CPU and zero connections', async () => {
    mockCloudWatchClient.send.mockResolvedValue({
      Datapoints: Array(5).fill({ Average: 8 }),
    });

    mockRdsClient.send.mockResolvedValue({
      DBInstances: [{ Status: 'available', Endpoint: { Address: 'db.example.com' } }],
    });

    const isIdle = await system.isRdsInstanceIdle('db-1234567890', 5);
    expect(isIdle).toBe(true);
  });

  it('should not flag RDS instance with active connections', async () => {
    mockCloudWatchClient.send.mockResolvedValue({
      Datapoints: Array(5).fill({ Average: 8 }),
    });

    mockRdsClient.send.mockResolvedValue({
      DBInstances: [{ Status: 'available', Endpoint: { Address: 'db.example.com' } }],
    });

    // Mock active connections via CloudWatch
    jest.spyOn(system as any, 'getRdsConnectionCount').mockResolvedValue(3);

    const isIdle = await system.isRdsInstanceIdle('db-1234567890', 5);
    expect(isIdle).toBe(false);
  });

  it('should generate alert with correct metadata and recommendation', async () => {
    const alert = system.generateAlert(
      'i-1234567890abcdef0',
      'EC2',
      8,
      150.5
    );

    expect(alert.resourceId).toBe('i-1234567890abcdef0');
    expect(alert.resourceType).toBe('EC2');
    expect(alert.idleDays).toBe(8);
    expect(alert.estimatedMonthlySavings).toBeCloseTo(150.5, 1);
    expect(alert.recommendation).toMatch(/stop|terminate/);
  });
});
