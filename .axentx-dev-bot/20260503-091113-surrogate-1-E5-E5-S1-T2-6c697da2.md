import pandas as pd
from datetime import datetime, timedelta
from anonymizer import Anonymizer
from pyCryptodome import AES

def compute_mau_wau(df):
    # Compute MAU and WAU
    mau = df['anon_id'].nunique()
    wau = df[df['timestamp'] >= (datetime.now() - timedelta(days=30))]['anon_id'].nunique()
    return mau, wau

def aggregate_metrics(df):
    # Aggregate metrics on a rolling basis
    mau = []
    wau = []
    for i in range(30, 0, -1):
        mau_i, wau_i = compute_mau_wau(df[df['timestamp'] >= (datetime.now() - timedelta(days=i))])
        mau.append(mau_i)
        wau.append(wau_i)
    return mau, wau

def expose_metrics(mau, wau):
    # Expose metrics via internal metrics endpoint
    metrics = {'mau': mau, 'wau': wau}
    return metrics

    # Expose metrics

- Daily job uses `BlockingScheduler` in the same module as the job definition and redefines `run_job`, which will cause import/runtime confusion; scheduling should be separate and idempotent.

- No retention or key-management strategy for anonymization secrets, which is required for GDPR-compliant pseudonymization.
