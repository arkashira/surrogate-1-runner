  TEMPLATE_FIRST_USE: 'template_first_use',
  SCHEDULING_FIRST_USE: 'scheduling_first_use',
  ANALYTICS_FIRST_USE: 'analytics_first_use'
};

const ALERT_THRESHOLD = 0.20; // 20%
const DEFAULT_COHORT_WINDOW_DAYS = 7;

class ActivationAlertSystem {
  constructor(config = {}) {
    this.threshold = config.threshold ?? ALERT_THRESHOLD;
    this.cohortWindowDays = config.cohortWindowDays ?? DEFAULT_COHORT_WINDOW_DAYS;
    this.alertCallbacks = [];
    this.cohortData = new Map();
    this.activeAlerts = new Map();

  }

  /**
   * Register an alert callback for notifications
   * @param {Function} callback - Called with alert object when triggered
   * @returns {Function} Unsubscribe function
   */
  onAlert(callback) {
    this.alertCallbacks.push(callback);
    return () => {
      const idx = this.alertCallbacks.indexOf(callback);
      if (idx > -1) this.alertCallbacks.splice(idx, 1);
    };
  }

   * Track a first-time feature usage event
