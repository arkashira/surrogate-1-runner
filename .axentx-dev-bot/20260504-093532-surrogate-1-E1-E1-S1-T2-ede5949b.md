import { Readable } from 'stream';
import { parseStream } from '../src/parser/parseStream';
import { Result } from '../src/types';

// Mock parseEntry to isolate stream logic
jest.mock('../src/parser/parseEntry', () => ({

describe('parseStream', () => {
  it('should resolve with Result when stream emits valid JSON lines', async () => {
    const input = ['{"id":1}', '{"id":2}'].join('\n');
    const stream = Readable.from(input.split('\n').map(line => line + '\n'));

    const result: Result = await parseStream(stream);

    expect(result.success).toBe(true);
    expect(result.entries.length).toBe(2);
    expect(result.errors).toHaveLength(0);
  });

  it('should accumulate errors for malformed lines but continue processing', async () => {
    const input = ['{"id":1}', '{"id":', '{"id":3}'].join('\n');
    const stream = Readable.from(input.split('\n').map(line => line + '\n'));

    const result: Result = await parseStream(stream);

    expect(result.success).toBe(false);
    expect(result.entries.length).toBe(2); // Only two valid
    expect(result.errors.length).toBe(1);
    expect(result.errors[0].line).toBe('{"id":');
  });

  it('should reject if stream emits an error', async () => {
    const stream = new Readable({
      read() {
        this.destroy(new Error('Simulated read error'));
      },
    });

    await expect(parseStream(stream)).rejects.toThrow('Simulated read error');
  });

  it('should handle empty stream gracefully', async () => {
    const stream = Readable.from([]);

    const result: Result = await parseStream(stream);

    expect(result.success).toBe(true);
    expect(result.entries).toHaveLength(0);
    expect(result.errors).toHaveLength(0);
  });

  it('should process large stream in chunks without memory spike (indirect test)', async () => {
    const largeInput = Array.from({ length: 100_000 }, (_, i) => `{"id":${i}}\n`);
    const stream = Readable.from(largeInput);

    const result: Result = await parseStream(stream);

    expect(result.entries.length).toBe(100_000);
    expect(result.errors).toHaveLength(0);
  });
});
