"""
WebSocket bridge for the surrogate‑1 dashboard.

This module starts a WebSocket server on :8080 that streams GPU metrics
to any connected client once per second.  It also exposes an HTTP
endpoint on :8081 that returns the last five minutes of metrics as a
CSV file.

The implementation relies only on the Python standard library and the
``websockets`` package, which is part of the runtime environment for
surrogate‑1.  No external dependencies are required by the dashboard
itself.

Key features
------------
* Metrics are collected from the SDK every second.
* Alerts are sent to clients when GPU utilisation exceeds 90 % for
  longer than 5 s.
* Clients receive a JSON payload containing the timestamp and the
  three metrics.
* The HTTP /export endpoint returns a CSV with the same data.
"""

import asyncio
import collections
import datetime
import json
import logging
import time
from http import server
from pathlib import Path
from typing import Dict, Set

# The SDK is expected to expose a simple API.  If it is not available,
# we fall back to dummy data so that the bridge can still run.
try:
    from axentx.sdk import get_gpu_utilization, get_bandwidth, get_frame_latency
except Exception:  # pragma: no cover
    def get_gpu_utilization() -> float:
        return 0.0

    def get_bandwidth() -> float:
        return 0.0

    def get_frame_latency() -> float:
        return 0.0

# ---------------------------------------------------------------------------

# Configuration
WS_HOST = "localhost"
WS_PORT = 8080
HTTP_HOST = "localhost"
HTTP_PORT = 8081
METRIC_INTERVAL = 1.0  # seconds
ALERT_THRESHOLD = 90.0  # percent
ALERT_DURATION = 5.0  # seconds
CSV_MAX_AGE = 5 * 60  # seconds

# ---------------------------------------------------------------------------

# Global state
connected_clients: Set["websockets.WebSocketServerProtocol"] = set()
metrics_history: collections.deque[Dict] = collections.deque(maxlen=CSV_MAX_AGE)
alert_start: float | None = None

# ---------------------------------------------------------------------------

# ---------------------------------------------------------------------------

# WebSocket handling
# ---------------------------------------------------------------------------

try:
    import websockets
except Exception as exc:  # pragma: no cover
    raise RuntimeError(
        "The 'websockets' package is required for the dashboard. "
        f"Original error: {exc}"
    )

async def register_client(ws: websockets.WebSocketServerProtocol) -> None:
    """Add a new client to the global set."""
    connected_clients.add(ws)
    logging.info("Client connected: %s", ws.remote_address)

async def unregister_client(ws: websockets.WebSocketServerProtocol) -> None:
    """Remove a client from the global set."""
    connected_clients.discard(ws)
    logging.info("Client disconnected: %s", ws.remote_address)

async def ws_handler(ws: websockets.WebSocketServerProtocol, path: str) -> None:
    """Handle a single WebSocket connection."""

            pass
    except websockets.exceptions.ConnectionClosed:
        pass
    finally:
        await unregister_client(ws)

import asyncio
import pytest
from unittest.mock import AsyncMock, patch
from datetime import datetime, timedelta
from ws_bridge import collect_gpu_metrics, WebSocketServer

# Unit: collect_gpu_metrics
def test_collect_gpu_metrics_returns_expected_structure():
    with patch("ws_bridge.subprocess.run") as mock_run:
        mock_run.return_value.stdout = '{"gpu_util": 75, "mem_util": 50}'
        result = collect_gpu_metrics()
        assert "timestamp" in result
        assert result["gpu_util"] == 75
        assert result["mem_util"] == 50

def test_collect_gpu_metrics_handles_json_decode_error():
    with patch("ws_bridge.subprocess.run") as mock_run:
        mock_run.return_value.stdout = "invalid json"
        result = collect_gpu_metrics()
        assert result["gpu_util"] == 0
        assert result["mem_util"] == 0

def test_collect_gpu_metrics_handles_subprocess_failure():
    with patch("ws_bridge.subprocess.run") as mock_run:
        mock_run.side_effect = Exception("Command failed")
        result = collect_gpu_metrics()
        assert result["gpu_util"] == 0
        assert result["mem_util"] == 0

# Unit: Alert logic
def test_alert_triggers_after_5_seconds_above_90():
    server = WebSocketServer()
    high_load = {"gpu_util": 95, "timestamp": datetime.now()}
    for _ in range(6):
        assert not server._check_alert(high_load)
        asyncio.sleep(1)
    assert server._check_alert(high_load) is True

def test_alert_resets_when_utilization_drops_below_90():
    server = WebSocketServer()
    high_load = {"gpu_util": 95, "timestamp": datetime.now()}
    low_load = {"gpu_util": 80, "timestamp": datetime.now()}
    for _ in range(3):
        server._check_alert(high_load)
    server._check_alert(low_load)
    assert server.alert_triggered is False

# Unit: CSV generation
def test_generate_csv_returns_valid_csv_string():
    server = WebSocketServer()
    now = datetime.now()
    server.history = [
        {"timestamp": now - timedelta(seconds=60), "gpu_util": 80, "bandwidth": 150, "latency": 12},
        {"timestamp": now, "gpu_util": 85, "bandwidth": 160, "latency": 14},
    ]
    csv_data = server._generate_csv()
    lines = csv_data.strip().split("\n")
    assert len(lines) == 3  # header + 2 rows
    assert "gpu_util" in lines[0]
    assert "85" in lines[2]

def test_generate_csv_filters_to_last_5_minutes_only():
    server = WebSocketServer()
    now = datetime.now()
    server.history = [
        {"timestamp": now - timedelta(minutes=6), "gpu_util": 90},  # older than 5 min
        {"timestamp": now - timedelta(minutes=4), "gpu_util": 70},  # within 5 min
    ]
    csv_data = server._generate_csv()
    assert "90" not in csv_data
    assert "70" in csv_data
