import re
from html import escape
from typing import List, Optional


def _find_best_window(text: str, terms: List[str], window: int) -> Optional[str]:
    """
    Return a substring window around the earliest, highest-density match.
    Falls back to start of text if no term found.
    """
    if not text or not terms:
        return text[:window] if text else ""

    # Build a case-insensitive regex that matches any term as a whole word-ish boundary
    escaped_terms = [re.escape(t) for t in terms if t.strip()]
    if not escaped_terms:
        return text[:window]

    pattern = re.compile("|".join(escaped_terms), re.IGNORECASE)
    matches = list(pattern.finditer(text))
    if not matches:
        return text[:window]

    # Pick first match and center window around it
    m = matches[0]
    start = max(0, m.start() - window // 2)
    end = min(len(text), start + window)
    # Adjust start if we hit the end
    if end - start < window and start > 0:
        start = max(0, end - window)

    snippet = text[start:end]
    # Prefix ellipsis if not at start
    if start > 0:
        snippet = "…" + snippet.lstrip()
    # Suffix ellipsis if not at end
    if end < len(text):
        snippet = snippet.rstrip() + "…"
    return snippet
