import re

def extract_snippet(text, query, max_length=200):
    """
    Extract a snippet from the text around the query match.
    """
    matches = re.finditer(query, text, re.IGNORECASE)
    snippets = []
    for match in matches:
        start = max(0, match.start() - max_length // 2)
        end = min(len(text), match.end() + max_length // 2)
        snippet = text[start:end]
        snippets.append(snippet)
    return snippets

def highlight_matched_terms(snippet, query):
    """
    Highlight the matched terms in the snippet.
    """
    highlighted_snippet = re.sub(query, lambda match: f"<strong>{match.group()}</strong>", snippet, flags=re.IGNORECASE)
    return highlighted_snippet
```
