# test_content_exporter.py

import pytest
from unittest.mock import patch, mock_open
from content_exporter import list_templates, export_template, _walk_templates, set_base_dir

def test_list_templates():
    # Test that list_templates filters templates correctly
    templates = [
        {"path": "blog_post.md", "tactic_type": "blog", "industry": "tech"},
        {"path": "social_media.md", "tactic_type": "social", "industry": "tech"},
    ]
    assert list_templates(templates, "blog", "tech") == [templates[0]]
    assert list_templates(templates, "social", "tech") == [templates[1]]
    assert list_templates(templates, "blog", "health") == []

def test_export_template_md():
    # Test that export_template handles .md format correctly
    template = {"path": "blog_post.md", "content": "# Blog Post"}
    with patch("builtins.open", mock_open(read_data="# Blog Post")) as mock_file:
        result = export_template(template, "md")
        assert result == "# Blog Post"
        mock_file.assert_called_once_with("blog_post.md", "r")

def test_export_template_docx():
    # Test that export_template handles .docx format correctly
    template = {"path": "blog_post.md", "content": "# Blog Post"}
    with patch("docx.Document") as mock_docx:
        result = export_template(template, "docx")
        assert result is not None
        mock_docx.assert_called_once()

def test_walk_templates():
    # Test that _walk_templates yields all Markdown files under BASE_DIR
    with patch("os.walk") as mock_walk:
        mock_walk.return_value = [
