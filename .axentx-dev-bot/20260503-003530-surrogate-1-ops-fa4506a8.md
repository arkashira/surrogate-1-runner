## 1) Mac orchestration tool (single API call)

`tools/snapshot_manifest.py`

- One call: `list_repo_tree(path=date_partition, recursive=False)`  
- Deterministic file list (sorted), parquet-only filter, CDN URLs, integrity metadata  
- Exits non-zero on 429/404 with clear message; retries/backoff for transient errors  
- Emits compact `file_manifest.json` usable by training on any host

```python
#!/usr/bin/env python3
"""
Create CDN-only manifest for one date-partition of a HF dataset repo.

Usage:
    python tools/snapshot_manifest.py \
      --repo axentx/surrogate-1-training-pairs \
      --date 2026-05-03 \
      --out manifest.json
"""
import argparse
import hashlib
import json
import sys
import time
from datetime import datetime, timezone

from huggingface_hub import HfApi, RateLimitError, RepositoryNotFoundError

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def build_manifest(repo: str, date: str, out_path: str) -> None:
    api = HfApi()
    prefix = f"{date}/"

    for attempt in range(1, 4):
        try:
            entries = api.list_repo_tree(repo=repo, path=prefix, recursive=False)
            break
        except RateLimitError:
            wait = 60 * (2 ** attempt)
            print(f"HF API 429. Retry {attempt}/3 after {wait}s.", file=sys.stderr)
            time.sleep(wait)
            continue
        except RepositoryNotFoundError:
            print(f"Repo not found: {repo}", file=sys.stderr)
            sys.exit(1)
    else:
        print("HF API rate limit retries exhausted.", file=sys.stderr)
        sys.exit(2)

    files = sorted(
        e.path for e in entries
        if e.type == "file" and e.path.lower().endswith((".parquet", ".jsonl"))
    )
    if not files:
        print(f"No parquet/jsonl files under prefix '{prefix}'", file=sys.stderr)
        sys.exit(1)

    items = [
        {
            "path": p,
            "url": CDN_TEMPLATE.format(repo=repo, path=p),
        }
        for p in files
    ]

    manifest = {
        "repo": repo,
        "date": date,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": items,
        "count": len(items),
    }
    # content hash for quick change detection
    manifest["sha256"] = hashlib.sha256(
        json.dumps(manifest, sort_keys=True).encode()
    ).hexdigest()

    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2, sort_keys=True)
    print(f"Wrote {len(items)} files -> {out_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Create CDN file manifest for HF dataset date partition.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--date", required=True, help="YYYY-MM-DD folder in dataset repo")
    parser.add_argument("--out", default="file_manifest.json")
    args = parser.parse_args()
    build_manifest(args.repo, args.date, args.out)
```
