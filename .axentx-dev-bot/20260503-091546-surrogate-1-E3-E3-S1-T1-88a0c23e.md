export type RiskLabel = 'ok' | 'at-risk' | 'breached' | 'unknown';

export interface Request {
  id: string;
  team: string;
  title: string;
  description?: string;
  status: string;
  statusChangedAt: string; // ISO 8601
  timeInCurrentStatusSeconds: number;
  slaTargetSeconds?: number;
  riskLabel: RiskLabel;
  createdAt: string;
  updatedAt: string;
}

/**
 * Compute risk label from current age and optional SLA.
 * Rules:
 * - missing/invalid slaTargetSeconds => 'unknown'
 * - time >= slaTargetSeconds          => 'breached'
 * - time >= 0.8 * slaTargetSeconds    => 'at-risk'
 * - otherwise                         => 'ok'
 */
export function computeRiskLabel(
  timeInCurrentStatusSeconds: number,
  slaTargetSeconds?: number
): RiskLabel {
  if (slaTargetSeconds == null || slaTargetSeconds <= 0 || Number.isNaN(slaTargetSeconds)) {
    return 'unknown';
  }

  if (timeInCurrentStatusSeconds >= slaTargetSeconds) {
    return 'breached';
  }

  if (timeInCurrentStatusSeconds >= 0.8 * slaTargetSeconds) {
    return 'at-risk';
  }

  return 'ok';
}

/**
 * Build a fully-enriched Request from persisted/partial data.
 * Computes timeInCurrentStatusSeconds from statusChangedAt vs now
 * and derives riskLabel.
 */
export function buildRequest(partial: {
  id: string;
  team: string;
  title: string;
  description?: string;
  status: string;
  statusChangedAt: string;
  createdAt: string;
  updatedAt: string;
  slaTargetSeconds?: number;
}): Request {
  const now = Date.now();
  const statusChangedAtMs = new Date(partial.statusChangedAt).getTime();

  // Guard against invalid dates; clamp to non-negative integer seconds.
  const timeInCurrentStatusSeconds =
    statusChangedAtMs > 0 && !Number.isNaN(statusChangedAtMs)
      ? Math.max(0, Math.floor((now - statusChangedAtMs) / 1000))
      : 0;

  const riskLabel = computeRiskLabel(timeInCurrentStatusSeconds, partial.slaTargetSeconds);

  return {
    ...partial,
    timeInCurrentStatusSeconds,
    riskLabel,
  };
}
