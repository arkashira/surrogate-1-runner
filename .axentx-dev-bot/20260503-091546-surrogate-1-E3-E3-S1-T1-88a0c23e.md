export type RiskLabel = 'ok' | 'at-risk' | 'breached' | 'unknown';

export interface Request {
  id: string;
  team: string;
  title: string;
  status: string;
  statusUpdatedAt: Date; // when it entered current status
  createdAt: Date;
  updatedAt: Date;
  slaTargetSeconds?: number; // optional SLA target for current status
  timeInCurrentStatusSeconds: number; // computed
  riskLabel: RiskLabel; // computed
  metadata?: Record<string, unknown>;
}

export function computeTimeInCurrentStatusSeconds(statusUpdatedAt: Date): number {
  return Math.floor((Date.now() - statusUpdatedAt.getTime()) / 1000);
}

export function computeRiskLabel(
  timeInCurrentStatusSeconds: number,
  slaTargetSeconds?: number
): RiskLabel {
  if (slaTargetSeconds == null || slaTargetSeconds <= 0) {
    return 'unknown';
  }
  if (timeInCurrentStatusSeconds >= slaTargetSeconds) {
    return 'breached';
  }
  if (timeInCurrentStatusSeconds >= slaTargetSeconds * 0.8) {
    return 'at-risk';
  }
  return 'ok';
}

export function enrichRequest(request: Omit<Request, 'timeInCurrentStatusSeconds' | 'riskLabel'>): Request {
  const timeInCurrentStatusSeconds = computeTimeInCurrentStatusSeconds(request.statusUpdatedAt);
  const riskLabel = computeRiskLabel(timeInCurrentStatusSeconds, request.slaTargetSeconds);
  return {
    ...request,
    timeInCurrentStatusSeconds,
    riskLabel,
  };
}
