# src/tracking.py
"""
Change tracking for pinned external actions.

The module watches a directory for file changes, records a SHA‑256 hash
of each file, persists a history of changes, and emits notifications
via the :mod:`notifications` module.

Public API
----------
* :func:`track_changes` – start watching a directory.
* :func:`record_action` – manually record a file’s current state.
* :func:`get_history` – retrieve the change history for a file.
"""

import hashlib
import json
import os
import time
from datetime import datetime
from pathlib import Path
from typing import Dict, List

from watchdog.events import FileSystemEventHandler
from watchdog.observers import Observer

# --------------------------------------------------------------------------- #
# Persistence helpers
# --------------------------------------------------------------------------- #

HISTORY_FILE = Path("data/action_change_history.json")

# Ensure the directory exists
HISTORY_FILE.parent.mkdir(parents=True, exist_ok=True)
if not HISTORY_FILE.exists():
    HISTORY_FILE.write_text("{}", encoding="utf-8")


def _load_history() -> Dict[str, List[Dict]]:
    try:
        return json.loads(HISTORY_FILE.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, FileNotFoundError):
        return {}


def _save_history(history: Dict[str, List[Dict]]) -> None:
    HISTORY_FILE.write_text(json.dumps(history, indent=2), encoding="utf-8")


# --------------------------------------------------------------------------- #
# In‑memory cache of the last known hash for each file
# --------------------------------------------------------------------------- #

_last_hash: Dict[str, str] = {}


def _hash_file(path: Path) -> str:
    """Return a SHA‑256 hex digest of the file contents."""
    with path.open("rb") as f:
        return hashlib.sha256(f.read()).hexdigest()


# --------------------------------------------------------------------------- #
# Core logic
# --------------------------------------------------------------------------- #

def _append_change_record(action_id: str, new_hash: str, preview: str) -> Dict:
    """Create a change record and persist it."""
    record = {
        "timestamp": datetime.utcnow().isoformat() + "Z",
        "hash": new_hash,
        "content_preview": preview,
    }

    history = _load_history()
    history.setdefault(action_id, []).append(record)
    _save_history(history)

    return record


def record_action(action_id: str, content: str) -> None:
    """
    Record the current state of an action.

    If the content has changed since the last record, a change event is
    appended to the history and a notification is sent.
    """
    global _last_hash

    new_hash = hashlib.sha256(content.encode("utf-8")).hexdigest()
    last_hash = _last_hash.get(action_id)

    if new_hash == last_hash:
        return  # No change

    # Persist change
    change_record = _append_change_record(
        action_id, new_hash, content[:200]
    )
