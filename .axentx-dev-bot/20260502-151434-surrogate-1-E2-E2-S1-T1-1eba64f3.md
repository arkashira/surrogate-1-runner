from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, Optional


class ConnectorError(Exception):
    """Base error for all connector failures."""

    def __init__(self, message: str, code: str, context: Optional[Dict[str, Any]] = None):
        super().__init__(message)
        self.code = code
        self.context = context or {}
        self.timestamp = datetime.now(timezone.utc).isoformat()


class TimeoutError(ConnectorError):
    def __init__(self, message: str = "Operation timed out", context: Optional[Dict[str, Any]] = None):
        super().__init__(message, "TIMEOUT", context)


class ValidationError(ConnectorError):
    def __init__(self, message: str, context: Optional[Dict[str, Any]] = None):
        super().__init__(message, "VALIDATION", context)


class RejectionError(ConnectorError):
    def __init__(self, message: str, context: Optional[Dict[str, Any]] = None):
        super().__init__(message, "REJECTED", context)


class OrderSide(str, Enum):
    BUY = "buy"
    SELL = "sell"


class OrderType(str, Enum):
    MARKET = "market"
    LIMIT = "limit"


@dataclass(frozen=True)
class Order:
    symbol: str
    side: OrderSide
    quantity: float
    order_type: OrderType = OrderType.MARKET
    limit_price: Optional[float] = None
    client_order_id: Optional[str] = None

from surrogate_1.connectors.models import (
    ConnectorError, ValidationError, OrderType, Order,
    MarketSnapshot, RiskCheckRequest, RiskCheckResult

def test_connector_error_serializable():
    exc = ConnectorError(code="TEST", context={"x": 1}, timestamp=datetime.now(timezone.utc).isoformat())
    payload = json.dumps({"code": exc.code, "context": exc.context, "timestamp": exc.timestamp})
    loaded = json.loads(payload)
    assert loaded["code"] == exc.code
    assert loaded["context"] == exc.context
    assert isinstance(loaded["timestamp"], str)

def test_validation_error_is_connector_error():
    exc = ValidationError(code="VALIDATION_ERROR", context={"field": "quantity"})
    assert isinstance(exc, ConnectorError)

def test_order_rejects_non_positive_quantity():
    with pytest.raises(ValidationError) as ei:
        Order(order_id="1", symbol="AAPL", order_type=OrderType.LIMIT, quantity=0, limit_price=100.0)
    assert ei.value.code == "VALIDATION_ERROR"

def test_order_rejects_missing_limit_for_limit_order():
    with pytest.raises(ValidationError) as ei:
