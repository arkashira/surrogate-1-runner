    - Test type (unit, integration, e2e)

    - Tags (custom metadata)
    - Regex patterns on test names
    """

        self.filters = []

        """
        Add a custom filter function.
        
        Args:
            filter_func: A function that takes a test dict and returns True if it should be included
        """
        self.filters.append(filter_func)

        """
