"""
Document ingestion and chunking module for surrogate-1.

Supports parallel ingestion, configurable chunk sizes, and lossless
chunking of large documents.
"""

import hashlib
import json
import logging
import os
import sys
from concurrent.futures import ProcessPoolExecutor, as_completed
from dataclasses import dataclass, field
from pathlib import Path
from typing import Any, Callable, Iterable, Optional
from uuid import uuid4

import click

logger = logging.getLogger(__name__)


@dataclass
class Chunk:
    """Represents a single chunk of a document."""
    chunk_id: str
    document_id: str
    content: str
    offset: int
    length: int
    metadata: dict = field(default_factory=dict)


@dataclass
class IngestionResult:
    """Result of a document ingestion operation."""
    document_id: str
    total_chunks: int
    chunks: list[Chunk]
    status: str  # "success", "partial", "failed"
    error: Optional[str] = None


class DocumentChunker:
    """Handles document chunking with configurable parameters."""

    def __init__(
        self,
        chunk_size: int = 1024,
        chunk_overlap: int = 0,
        encoding: str = "utf-8"
    ):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.encoding = encoding

    def chunk_document(self, content: str, document_id: str) -> list[Chunk]:
        """
        Split document content into overlapping chunks.
        
        Args:
            content: Raw document content
            document_id: Unique identifier for the document
            
        Returns:
            List of Chunk objects
        """
        if not content:
            return []
        
        chunks = []
        offset = 0
        chunk_id_counter = 0
        
        while offset < len(content):
            end = min(offset + self.chunk_size, len(content))
            chunk_content = content[offset:end]
            
            chunk = Chunk(
                chunk_id=f"{document_id}-{uuid4().hex[:8]}",
                document_id=document_id,
                content=chunk_content,
                offset=offset,
                length=len(chunk_content),
                metadata={
                    "original_length": len(content),
                    "chunk_index": chunk_id_counter
                }
            )
            chunks.append(chunk)
            
            offset += (self.chunk_size - self.chunk_overlap)
            chunk_id_counter += 1
        
        return chunks

    def verify_no_loss(self, original: str, chunks: list[Chunk]) -> bool:
        """Verify that chunking didn't lose any information."""
        reconstructed = "".join(chunk.content for chunk in chunks)
        return reconstructed == original
