import { Job } from '../types/job';
import { Request } from '../types/request';
import { SlackClient } from '../clients/slackClient';
import { EmailClient } from '../clients/emailClient';

export class NotificationJob implements Job {
  private slackClient: SlackClient;
  private emailClient: EmailClient;

  constructor(slackClient: SlackClient, emailClient: EmailClient) {
    this.slackClient = slackClient;
    this.emailClient = emailClient;
  }

  async execute(request: Request) {
    const formattedMessage = this.formatMessage(request);
    await this.slackClient.sendMessage(formattedMessage);
    await this.emailClient.sendEmail(formattedMessage);
  }

  private formatMessage(request: Request) {
    return `Request ${request.id} has been updated to ${request.status}. View it here: ${request.publicUrl}`;
  }
}
