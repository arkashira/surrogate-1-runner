### 2) Notification queue + worker (`src/queues/notificationQueue.ts`)
```ts
import { Queue, Worker, QueueScheduler } from 'bullmq';
import redis from './redis';
import { processNotification } from '../jobs/notificationJob';

const QUEUE_NAME = 'notifications';

export const notificationQueue = new Queue(QUEUE_NAME, {
  connection: redis,
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 5000 },
    removeOnComplete: 100,
    removeOnFail: 50,
  },
});

export const notificationScheduler = new QueueScheduler(QUEUE_NAME, {
  connection: redis,
});

export const notificationWorker = new Worker(
  QUEUE_NAME,
  async (job) => processNotification(job.data),
  {
    connection: redis,
    concurrency: 5,
    autorun: true,
  }
);

// Graceful shutdown helpers (call on app termination)
export async function shutdownNotifications() {
  await notificationWorker.close();
  await notificationQueue.close();
  await redis.quit();
}
```
