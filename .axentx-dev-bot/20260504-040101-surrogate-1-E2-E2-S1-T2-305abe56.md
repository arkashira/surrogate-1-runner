from functools import wraps
from flask import request, jsonify
from datetime import datetime, timedelta
from collections import defaultdict
import time

class RateLimiter:
    def __init__(self, max_requests=100, time_window=timedelta(minutes=1)):
        """
        Initialize rate limiter with configurable parameters.

        Args:
            max_requests (int): Maximum allowed requests within the time window
            time_window (timedelta): Time window for rate limiting
        """
        self.max_requests = max_requests
        self.time_window = time_window
        self.request_counts = defaultdict(list)

    def _is_allowed(self, client_id):
        """Check if client is allowed to make another request."""
        current_time = datetime.now()
        # Remove timestamps older than the time window
        self.request_counts[client_id] = [
            t for t in self.request_counts[client_id]
            if t > current_time - self.time_window
        ]

        if len(self.request_counts[client_id]) < self.max_requests:
            self.request_counts[client_id].append(current_time)
            return True
        return False

    def limit(self, func):
        """Decorator to apply rate limiting to a Flask route."""
        @wraps(func)
        def wrapper(*args, **kwargs):
            client_id = request.remote_addr
            if self._is_allowed(client_id):
                return func(*args, **kwargs)
            else:
                return jsonify({
                    "error": "Rate limit exceeded",
                    "message": f"Maximum {self.max_requests} requests per minute allowed"
                }), 429
        return wrapper
