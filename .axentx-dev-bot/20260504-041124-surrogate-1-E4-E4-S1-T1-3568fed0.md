    Returns:
        Processed DataFrame with last 30 days of data

    Raises:
        ForecastError: If validation fails
    """

    if not required_cols.issubset(df.columns):

    df = df.sort_values("date")

    # Keep only the last 30 unique dates
    unique_dates = df["date"].unique()
    if len(unique_dates) < 7:
        raise ForecastError("At least 7 days of history are required for forecasting")
