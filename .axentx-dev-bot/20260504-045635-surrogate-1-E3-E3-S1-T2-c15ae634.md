// NewPolicyEngine creates a new policy enforcement engine
func NewPolicyEngine(repoOwner, repoName, token string, autoRemediate, dryRun bool) *PolicyEngine {
	var client *github.Client
	if token != "" {
		ts := oauth2.StaticTokenSource(
			&oauth2.Token{AccessToken: token},
		)
		tc := oauth2.NewClient(context.Background(), ts)
		client = github.NewClient(tc)
	}

	webhookURL := os.Getenv("SECURITY_WEBHOOK_URL")

	return &PolicyEngine{
		githubClient:  client,
		repoOwner:     repoOwner,
		repoName:      repoName,
		webhookURL:    webhookURL,
		autoRemediate: autoRemediate,
		dryRun:        dryRun,
	}
}

// LoadPolicies loads policies from a JSON file
func (pe *PolicyEngine) LoadPolicies() error {
	data, err := ioutil.ReadFile("policies.json")
	if err != nil {
		return err
	}
	var policies []Policy
	if err := json.Unmarshal(data, &policies); err != nil {
		return err
	}
	pe.policies = policies
	return nil
}

// RunPolicies executes all security policy checks
func (pe *PolicyEngine) RunPolicies(ctx context.Context, commitSHA, branch string) (*PolicyResult, error) {
	result := &PolicyResult{
		Timestamp: time.Now().UTC().Format(time.RFC3339),
		CommitSHA: commitSHA,
		Branch:    branch,
		Passed:    true,
	}

	// Run policy checks
	for _, policy := range pe.policies {
		violations, err := pe.checkPolicy(policy)
		if err != nil {
			log.Printf("Policy check %s failed: %v", policy.Name, err)
			continue
		}
		result.Violations = append(result.Violations, violations...)
	}

	// Auto-remediate if enabled
	if pe.autoRemediate && len(result.Violations) > 0 {
		remediated, errs := pe.autoRemediateViolations(result.Violations)
		result.Remedied = remediated
		result.RemediationErr = errs
	}

	// Update passed status
	if len(result.Violations) > result.Remedied {
		result.Passed = false
	}

	return result, nil
}
