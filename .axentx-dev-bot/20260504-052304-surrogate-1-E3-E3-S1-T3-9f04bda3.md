// src/api/demoCall.js
/**
 * Demo API call handler for the in-app tutorial.
 *
 * This handler serves static demo responses that are used in the
 * interactive tutorial to illustrate Surrogate-1's capabilities.
 *
 * The handler expects a query parameter `action` that selects a
 * predefined demo response. If the action is not recognized,
 * a default response is returned.
 *
 * @param {import('express').Request} req
 * @param {import('express').Response} res
 */
const { demoResponses } = require('../utils/demoResponses');

function demoCall(req, res) {
  const { action } = req.query;
  const response = demoResponses[action] || demoResponses.default;
  res.json(response);
}

module.exports = { demoCall };

## 1. Acceptance Criteria
1. **First‑load trigger** – The tutorial is automatically displayed on the very first dashboard load for a user who has never completed or skipped it.  
2. **Three‑step flow** – The tutorial consists of exactly three sequential steps, each rendered with a distinct heading and description.  
3. **Live demo call** – Step 2 performs a real HTTP request to `/api/demoCall` and displays the returned JSON payload in the UI.  
4. **Progress indicator** – A visual progress bar or step counter updates correctly (0 → 1 → 2 → 3) as the user advances or skips.  
5. **Skip functionality** – Clicking “Skip” at any step immediately closes the tutorial and records a “skipped” flag in local storage.  
6. **Persistence** – Once the tutorial is completed or skipped, subsequent dashboard loads do **not** show the tutorial again.  
7. **Error handling** – If the demo API call fails (network error or non‑2xx response), an inline error message is shown and the user can retry the call without leaving the tutorial.
