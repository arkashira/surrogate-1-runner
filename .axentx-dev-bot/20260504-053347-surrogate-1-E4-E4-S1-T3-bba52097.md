## 1. Acceptance Criteria
- **AC1 – Randomized User‑Agent**: Every outbound HTTP request must include a `User-Agent` header selected randomly from a configurable list of at least 20 distinct strings; the same request must never reuse the exact same UA twice in a row.
- **AC2 – Proxy IP Rotation**: Outbound traffic must be routed through a proxy pool of ≥5 IPs, rotating on a per‑request basis; the selected proxy IP must differ from the previous request’s IP.
- **AC3 – TLS 1.3 Enforcement**: All outbound connections must negotiate TLS 1.3; any fallback to lower versions must cause the request to fail.
- **AC4 – Custom Stealth Header**: Every request must contain a header `X‑Stealth‑Token: <random‑hex‑16>`; the token must be newly generated for each request.
- **AC5 – HTTPS Endpoint**: Requests are sent to `https://api.surrogate.internal/v1/collect` and must succeed with HTTP 200 under normal conditions.
- **AC6 – Health‑Check Endpoint**: A GET to `/health/stealth` returns JSON `{ "status":"ok", "blocked":false }` within 2 s, confirming no firewall blockage.
- **AC7 – Logging**: The gateway logs the chosen User‑Agent, proxy IP, and generated stealth token for each request at INFO level.

func TestGetRandomUserAgent(t *testing.T) {
    uaList := []string{"UA1", "UA2", "UA3", "UA4", "UA5"}
    SetUserAgentPool(uaList)

    first := getRandomUserAgent()
    assert.Contains(t, uaList, first)

    // Run many times to ensure randomness and no immediate repeat
    prev := first
    for i := 0; i < 50; i++ {
        cur := getRandomUserAgent()
        assert.Contains(t, uaList, cur)
        assert.NotEqual(t, prev, cur, "User-Agent should not repeat consecutively")
        prev = cur
    }
}

func TestGenerateStealthToken(t *testing.T) {
    token1 := generateStealthToken()
    token2 := generateStealthToken()

    assert.NotEqual(t, token1, token2, "Each token must be unique")
}

    pool := []string{"1.1.1.1:8080", "2.2.2.2:8080", "3.3.3.3:8080"}

        prev = cur
    }
}
