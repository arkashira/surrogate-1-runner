# .github/actions/format/action.yml
name: 'Format Code'
description: 'Run surrogate-1 formatter and check for changes'
inputs:
  token:
    description: 'GitHub token for creating comments'
    required: true
runs:
  using: 'composite'
  steps:
    - name: Checkout repository
      uses: actions/checkout@v3

    - name: Set up surrogate-1
      run: |
        python -m pip install --upgrade pip
        pip install surrogate-1
      shell: bash

    - name: Run formatter
      id: format
      run: |
        surrogate format
        if [ -n "$(git status --porcelain)" ]; then
          echo "Formatted files:"
          git diff --name-only
          echo "::error::Code formatting failed. Please run 'surrogate format' locally and commit the changes."
          exit 1
        fi
      shell: bash

    - name: Create summary comment
      if: failure() && steps.format.outcome == 'failure'
      uses: actions/github-script@v6
      with:
        script: |
          const { data: comments } = await github.rest.issues.listComments({
            owner: context.repo.owner,
            repo: context.repo.repo,
            issue_number: context.issue.number,
          });

          const formattingComment = comments.find(comment =>
            comment.body.includes('Formatted files')
          );

          if (!formattingComment) {
            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.issue.number,
              body: `### Formatted files\n\n\`\`\`\n${{ steps.format.outputs.diff }}\n\`\`\``
            });
          }
```
