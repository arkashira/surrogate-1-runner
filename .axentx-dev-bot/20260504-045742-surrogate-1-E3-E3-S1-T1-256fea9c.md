// tests/ui/configForm.test.js
import { render, fireEvent } from '@testing-library/svelte';
import ConfigForm from '../../src/ui/components/ConfigForm.svelte';

describe('ConfigForm', () => {
  test('renders all configuration inputs with default values', () => {
    const { getByLabelText, getByDisplayValue } = render(ConfigForm);
    expect(getByLabelText(/sensitivity threshold/i)).toBeInTheDocument();
    expect(getByDisplayValue(0.1)).toBeInTheDocument(); // default
    expect(getByLabelText(/recheck interval/i)).toBeInTheDocument();
    expect(getByDisplayValue(60)).toBeInTheDocument();
    expect(getByLabelText(/alert destination/i)).toBeInTheDocument();
  });

  test('emits save event with updated config on form submit', () => {
    const { component, getByText } = render(ConfigForm);
    const mockSave = jest.fn();
    component.$on('save', mockSave);

    fireEvent.input(getByText(/sensitivity threshold/i), { target: { value: 0.5 } });
    fireEvent.click(getByText(/save configuration/i));

    expect(mockSave).toHaveBeenCalledWith(expect.objectContaining({
      threshold: 0.5,
      interval: 60,
      alertDestination: ''
    }));
  });

  test('persists config to localStorage on save', () => {
    const { getByText } = render(ConfigForm);
    fireEvent.input(getByText(/sensitivity threshold/i), { target: { value: 0.7 } });
    fireEvent.click(getByText(/save configuration/i));

    const saved = localStorage.getItem('driftDetectorConfig');
    expect(JSON.parse(saved).threshold).toBe(0.7);
  });
});

// tests/ui/MonitorDashboard.test.js
import { render, screen } from '@testing-library/svelte';
import MonitorDashboard from '../../src/ui/components/MonitorDashboard.svelte';

describe('MonitorDashboard', () => {
  test('displays current drift status and score', () => {
    render(MonitorDashboard, { driftScore: 0.12, status: 'Stable' });
    expect(screen.getByText(/stable/i)).toBeInTheDocument();
    expect(screen.getByText(/0.12/i)).toBeInTheDocument();
  });

  test('shows alert banner when drift exceeds threshold', () => {
    render(MonitorDashboard, { driftScore: 0.8, threshold: 0.5 });
    expect(screen.getByRole('alert')).toHaveTextContent(/drift detected/i);
  });

  test('toggles graph visibility when widget control is clicked', async () => {
    const { getByText, queryByTestId } = render(MonitorDashboard);
    const toggle = getByText(/show history/i);

    expect(queryByTestId('drift-graph')).not.toBeInTheDocument();
    await fireEvent.click(toggle);
    expect(queryByTestId('drift-graph')).toBeInTheDocument();
  });
});
