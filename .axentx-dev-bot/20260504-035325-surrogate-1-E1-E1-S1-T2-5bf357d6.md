  });

  const session = createSession({ user: 'carol', mfa: 'token456' });
  const plaintext = Buffer.from('Hello, world!');

  test('encrypts data', () => {
    const cipher = encryptData(session, plaintext);
    expect(cipher).not.toEqual(plaintext);
  });

  test('decrypts data back to original', () => {
    const cipher = encryptData(session, plaintext);
    const decrypted = decryptData(session, cipher);
    expect(decrypted).toEqual(plaintext);
  });

describe('Logging', () => {
  test('logs session start with correct fields', () => {
    const session = createSession({ user: 'dave', mfa: 'token789' });

    const log = readLastLogEntry(); // helper that reads last line
    expect(log.user).toBe('dave');
    expect(log.action).toBe('START');
    expect(log).toHaveProperty('timestamp');
