interface TooltipProps {
  /** Unique identifier for this tooltip */
  id: string;
  /** Tooltip content text or JSX */
  content: React.ReactNode;
  /** Element that triggers the tooltip */
  children: React.ReactNode;
  /** Tooltip placement relative to trigger */
  placement?: 'top' | 'right' | 'bottom' | 'left';
  /** Delay before auto-showing tooltip (ms), 0 to disable */
  autoShowDelay?: number;
  /** Additional CSS classes for tooltip */
  className?: string;
}

/**
 * Tooltip component with session-storage persistence.
 * 
 * Features:
 * - SessionStorage-based dismissal that persists across page navigation
 * - Supports mouse hover and keyboard focus
 * - Configurable placement (top, right, bottom, left)
 * - Optional auto-show delay
 * - Full accessibility support
 */
export default function Tooltip({
  id,
  content,
  children,
  placement = 'top',
  autoShowDelay = 0,
  className = '',
}: TooltipProps) {
  const [isDismissed, setIsDismissed] = useSessionStorage(`tooltipDismissed_${id}`, false);
  const [isVisible, setIsVisible] = useState(false);
  const triggerRef = useRef<HTMLSpanElement>(null);
  const autoShowTimerRef = useRef<NodeJS.Timeout | null>(null);

  // Clear timer on unmount
  useEffect(() => {
    return () => {
      if (autoShowTimerRef.current) {
        clearTimeout(autoShowTimerRef.current);
      }
    };
  }, []);

  // Auto-show after delay if not dismissed
  useEffect(() => {
    if (!isDismissed && autoShowDelay > 0) {
      autoShowTimerRef.current = setTimeout(() => {
        setIsVisible(true);
      }, autoShowDelay);
    }
    return () => {
      if (autoShowTimerRef.current) {
        clearTimeout(autoShowTimerRef.current);
      }
    };
  }, [isDismissed, autoShowDelay]);

  const showTooltip = useCallback(() => {
    if (!isDismissed) {
      setIsVisible(true);
    }
  }, [isDismissed]);

  const hideTooltip = useCallback(() => {
    setIsVisible(false);
  }, []);

    e.stopPropagation();
    setIsDismissed(true);
    setIsVisible(false);
  }, [setIsDismissed]);

  // Handle both mouse and keyboard events
  const handleFocus = useCallback(() => {
    showTooltip();
  }, [showTooltip]);
