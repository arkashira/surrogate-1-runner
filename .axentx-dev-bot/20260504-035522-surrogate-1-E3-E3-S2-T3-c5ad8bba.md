## 1. Acceptance Criteria
- **Toggle Availability** – The sync feature can be enabled/disabled in the Settings UI; the toggle state persists across app restarts.  
- **Encrypted Upload** – When enabled, the app uploads a single encrypted blob to the Surrogate‑1 S3 bucket; the blob size matches the size of the local workflow store.  
- **Status Display** – Settings shows the timestamp of the last successful backup and the size of the uploaded blob in human‑readable form.  
- **Restore Confirmation** – On restore, the app prompts the user for confirmation; only after confirmation does it download, decrypt, and overwrite the local store.  
- **Error Handling** – If upload/download fails, an error message is shown and the status remains unchanged.  
- **Passcode Consistency** – The same passcode used for encryption is required for decryption; mismatched passcodes result in a decryption error.

// S3Client.test.js
describe('S3Client', () => {
  let client;
  beforeEach(() => {
    client = new S3Client({ bucket: 'surrogate-1', region: 'us-east-1' });
  });

  test('uploadBlob encrypts data before sending', async () => {
    const data = Buffer.from('workflow data');
    const encryptSpy = jest.spyOn(crypto, 'encrypt').mockReturnValue('encrypted');
    const putSpy = jest.spyOn(client.s3, 'putObject').mockResolvedValue({});
    await client.uploadBlob(data, 'passcode');
    expect(encryptSpy).toHaveBeenCalledWith(data, 'passcode');
    expect(putSpy).toHaveBeenCalledWith(expect.objectContaining({
      Body: 'encrypted',
      Key: expect.any(String),
    }));
  });

  test('downloadBlob returns decrypted data', async () => {
    const encrypted = 'encrypted';
    const decryptSpy = jest.spyOn(crypto, 'decrypt').mockReturnValue('workflow data');
    const getSpy = jest.spyOn(client.s3, 'getObject').mockResolvedValue({ Body: encrypted });
    const result = await client.downloadBlob('passcode');
    expect(decryptSpy).toHaveBeenCalledWith(encrypted, 'passcode');
    expect(result).toBe('workflow data');
  });

  test('throws on decryption error', async () => {
    jest.spyOn(client.s3, 'getObject').mockResolvedValue({ Body: 'bad' });
    jest.spyOn(crypto, 'decrypt').mockImplementation(() => { throw new Error('bad key'); });
    await expect(client.downloadBlob('wrong')).rejects.toThrow('bad key');
  });
});
