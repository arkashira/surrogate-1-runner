        <h2>Credit Balances</h2>
        <div class="credit-card">
            <span class="credit-label">Bulk Credits:</span>
            <span id="bulk-balance" class="credit-value">Loading…</span>
        </div>
        <div class="credit-card">
            <span class="credit-label">Monthly Credits:</span>
            <span id="monthly-balance" class="credit-value">Loading…</span>
        </div>
    </section>

    <!-- -------------------------------------------------
         Existing dashboard content (if any)
         ------------------------------------------------- -->
    {{ dashboard_content|safe }}

    <footer>
        <p>&copy; 2026 Axentx</p>
    </footer>

    <!-- -------------------------------------------------
         JavaScript – fetches credit data and updates UI
         ------------------------------------------------- -->
    <script>
        /**
         * Pull the latest credit balances from the back‑end.
         * Uses `credentials: 'same-origin'` so Flask session cookies are sent.
         */
        async function refreshCredits() {
            try {
                const resp = await fetch('/api/credits', { credentials: 'same-origin' });
                if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
                const data = await resp.json();

                // Guard against malformed payloads
                const bulk   = typeof data.bulk   === 'number' ? data.bulk   : null;
                const monthly= typeof data.monthly=== 'number' ? data.monthly: null;

                document.getElementById('bulk-balance')
                        .textContent = bulk !== null ? bulk.toLocaleString() : '—';
                document.getElementById('monthly-balance')
                        .textContent = monthly !== null ? monthly.toLocaleString() : '—';
            } catch (err) {
                console.error('Failed to fetch credit balances:', err);
                // Show a non‑intrusive error state
                document.getElementById('bulk-balance').textContent   = 'Error';
                document.getElementById('monthly-balance').textContent = 'Error';
            }
        }

        // Initial load
        refreshCredits();

        // Real‑time updates – poll every 15 seconds (fast enough for UI responsiveness,
        // light on the server, and matches the more aggressive interval from Candidate 2)
        setInterval(refreshCredits, 15000);
    </script>
</body>
</html>
