def get_costs_last_n_days(n: int) -> List[Dict]:
    """
    Return a list of cost records for the last `n` days.
    Each record is a dict: {"date": datetime.date, "provider": str, "amount": float}
    The implementation is deterministic (seeded RNG) so unit‑tests are stable.
    """
    today = datetime.date.today()
    providers = ["aws", "gcp", "azure"]
    np.random.seed(0)                     # deterministic for tests / CI
    data: List[Dict] = []

    for i in range(n):
        day = today - datetime.timedelta(days=i)
        for p in providers:
            # sinusoidal seasonality + tiny random jitter
            amount = 100 + 20 * np.sin(i / 5.0) + np.random.rand() * 5
            data.append({"date": day, "provider": p, "amount": amount})
    return data


# ----------------------------------------------------------------------
# 2️⃣  Forecast core – linear regression + simple confidence band
# ----------------------------------------------------------------------
def linear_regression_forecast(
    dates: List[datetime.date], totals: List[float]
) -> Tuple[float, Tuple[float, float]]:
    """
    Fit a simple OLS line `y = a*x + b` on daily totals.
    Returns:
        * projected total spend for the *next* 30 days,
        * a symmetric confidence interval (currently ±10 % of the projection).
    """
    if len(dates) < 2:
        raise ValueError("At least two data points required for regression")

    # Convert dates → ordinal numbers (int) for a linear relationship
    x = np.array([d.toordinal() for d in dates])
    y = np.array(totals)

    # Least‑squares solution
    A = np.vstack([x, np.ones_like(x)]).T
    a, b = np.linalg.lstsq(A, y, rcond=None)[0]

    # Predict the next 30 days
    last_day = max(dates)
    future = [last_day + datetime.timedelta(days=i) for i in range(1, 31)]
    x_future = np.array([d.toordinal() for d in future])
    y_pred = a * x_future + b
    total_projection = float(y_pred.sum())

    # ±10 % confidence band (simple, but works as a placeholder)
    low = total_projection * 0.9
    high = total_projection * 1.1
    return total_projection, (low, high)
