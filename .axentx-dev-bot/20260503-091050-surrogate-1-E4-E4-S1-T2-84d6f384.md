// Synchronous, typed tour state + signup reporting
const STORAGE_KEY = 'axentx:tourState';

export interface TourStateData {
  hasSeenTour?: boolean;
  signupTimestamp?: string;
  tourCompletedAt?: string;
}

function readState(): TourStateData {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : {};
  } catch {
    return {};
  }
}

function writeState(patch: Partial<TourStateData>) {
  try {
    const next = { ...readState(), ...patch };
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
  } catch {
    // ignore storage errors (private mode, quota)
  }
}

function nowISO() {
  return new Date().toISOString();
}

function emitAnalytics(event: string, detail?: unknown) {
  try {
    // Prefer beacon; fallback to fetch keepalive
    const payload = JSON.stringify({ event, timestamp: nowISO(), detail });
    const url = '/api/analytics';

    if (navigator.sendBeacon?.(url, payload)) return;

    try {
      void fetch(url, {
        method: 'POST',
        body: payload,
        headers: { 'Content-Type': 'application/json' },
        keepalive: true,
      }).catch(() => {});
    } catch {
      // ignore
    }
  } catch {
    // ignore
  }
}

    writeState({ hasSeenTour: v });
