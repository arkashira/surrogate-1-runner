// Synchronous, typed tour state + signup reporting
const STORAGE_KEY = 'axentx:tourState';

export interface TourStateData {
  hasSeenTour?: boolean;
  signupTimestamp?: string;
  tourCompletedAt?: string;
}

function readState(): TourStateData {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : {};
  } catch {
    return {};
  }
}

function writeState(patch: Partial<TourStateData>) {
  try {
    const next = { ...readState(), ...patch };
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
  } catch {
    // ignore storage errors (private mode, quota)
  }
}

function nowISO() {
  return new Date().toISOString();
}

function emitAnalytics(event: string, detail?: unknown) {
  try {
    // Prefer beacon; fallback to fetch keepalive
    const payload = JSON.stringify({ event, timestamp: nowISO(), detail });
    const url = '/api/analytics';

    if (navigator.sendBeacon?.(url, payload)) return;

    try {
      void fetch(url, {
        method: 'POST',
        body: payload,
        headers: { 'Content-Type': 'application/json' },
        keepalive: true,
      }).catch(() => {});
    } catch {
      // ignore
    }
  } catch {
    // ignore
  }
}

  get hasSeenTour(): boolean {
    return !!readState().hasSeenTour;
  },

    writeState({ hasSeenTour: v });

    return readState().signupTimestamp;

import { tourState } from '../lib/tourState';

describe('tourState', () => {
  beforeEach(() => {
    localStorage.clear();
    jest.spyOn(window, 'dispatchEvent').mockImplementation(() => {});
    jest.spyOn(navigator, 'sendBeacon').mockImplementation(() => true);
    global.fetch = jest.fn(() => Promise.resolve({ ok: true } as any));
  });

  test('completeTour sets hasSeenTour=true and records tourCompletedAt', () => {
    tourState.completeTour();
    expect(localStorage.getItem('hasSeenTour')).toBe('true');
    expect(JSON.parse(localStorage.getItem('tourCompletedAt') || 'null')).toBeTruthy();
    expect(window.dispatchEvent).toHaveBeenCalledWith(expect.objectContaining({ type: 'axentx:tourCompleted' }));
    expect(global.fetch).toHaveBeenCalledWith(expect.stringContaining('tour_completed'), expect.objectContaining({ method: 'POST', keepalive: true }));
  });

  test('skipTour sets hasSeenTour=true and emits skip events', () => {
    tourState.skipTour();
    expect(localStorage.getItem('hasSeenTour')).toBe('true');
    expect(window.dispatchEvent).toHaveBeenCalledWith(expect.objectContaining({ type: 'axentx:tourSkipped' }));
    expect(global.fetch).toHaveBeenCalledWith(expect.stringContaining('tour_skipped'), expect.anything());
  });

  test('recordSignup records timestamp once unless overridden', () => {
    const t1 = tourState.recordSignup();
    const stored = JSON.parse(localStorage.getItem('signupTimestamp') || 'null');
    expect(stored).toBe(t1);

    const t2 = tourState.recordSignup();
    expect(t2).toBe(t1); // unchanged

    const forced = tourState.recordSignup(true);
    expect(forced).toBeGreaterThanOrEqual(t1);
  });

  test('graceful degradation when localStorage unavailable', () => {
    const stub = { getItem: jest.fn(() => { throw new Error('quota'); }), setItem: jest.fn(() => { throw new Error('quota'); }) };
    Object.defineProperty(window, 'localStorage', { value: stub as any });
    expect(() => tourState.completeTour()).not.toThrow();
    // analytics fallback
    expect(global.fetch).toHaveBeenCalled();
  });

  test('analytics fallback from sendBeacon to fetch with keepalive', () => {
    (navigator.sendBeacon as jest.Mock).mockReturnValue(false);
    tourState.completeTour();
    expect(global.fetch).toHaveBeenCalledWith(expect.anything(), expect.objectContaining({ keepalive: true }));
  });
});
