// Synchronous, SSR-safe tour + signup telemetry helpers
const TOUR_KEY = 'shelf:tourCompleted';
const SIGNUP_EVENT_KEY = 'shelf:signupEventSent';

export function isBrowser(): boolean {
  return typeof window !== 'undefined';
}

export function hasSeenTour(): boolean {
  return isBrowser() && localStorage.getItem(TOUR_KEY) === 'true';
}

export function markTourSeen(): void {
  if (isBrowser()) localStorage.setItem(TOUR_KEY, 'true');
}

export function shouldShowTour(): boolean {
  return isBrowser() && !hasSeenTour();
}

export function hasSignupEventBeenSent(): boolean {
  return isBrowser() && localStorage.getItem(SIGNUP_EVENT_KEY) === 'true';
}

function markSignupEventSent(): void {
  if (isBrowser()) localStorage.setItem(SIGNUP_EVENT_KEY, 'true');
}

export function recordSignupEvent(userId?: string): void {
  if (!isBrowser()) return;
  // Deduplicate: send once per user/device
  if (hasSignupEventBeenSent()) return;

  const timestamp = new Date().toISOString();
  const payload = JSON.stringify({
    event: 'signup_completed',
    userId,
    timestamp,
    source: 'welcome_tour'
  });

  const done = () => markSignupEventSent();

  // Prefer sendBeacon (reliable, non-blocking)
  if (navigator.sendBeacon) {
    try {
      navigator.sendBeacon('/api/telemetry', payload);
      done();
      return;
    } catch {
      // fallback
    }
  }

  // Fallback fetch with keepalive
  fetch('/api/telemetry', {
    method: 'POST',
    body: payload,
    headers: { 'Content-Type': 'application/json' },
    keepalive: true
  }).then(done).catch(() => {});
}
```
