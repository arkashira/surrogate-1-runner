```python
import os
import smtplib
import logging
from datetime import datetime, timezone
from email.mime.text import MIMEText
from email.utils import formatdate, make_msgid
from typing import List, Optional

logger = logging.getLogger(__name__)


class EmailService:
    """
    Send emails via SMTP/SES. Supports unsubscribe link placeholder.
    Logs sent timestamp for idempotency/tracking.
    """

    def __init__(
        self,
        smtp_host: Optional[str] = None,
        smtp_port: int = 587,
        smtp_user: Optional[str] = None,
        smtp_pass: Optional[str] = None,
        use_tls: bool = True,
        from_addr: Optional[str] = None,
        reply_to: Optional[str] = None,
    ):
        self.smtp_host = smtp_host or os.getenv("SMTP_HOST")
        self.smtp_port = int(os.getenv("SMTP_PORT", smtp_port))
        self.smtp_user = smtp_user or os.getenv("SMTP_USER")
        self.smtp_pass = smtp_pass or os.getenv("SMTP_PASS")
        self.use_tls = bool(os.getenv("SMTP_TLS", str(use_tls).lower() not in ("0", "false", "no")))
        self.from_addr = from_addr or os.getenv("EMAIL_FROM") or "no-reply@axentx.invalid"
        self.reply_to = reply_to or os.getenv("EMAIL_REPLY_TO")

    def send(
        self,
        to_addrs: List[str],
        subject: str,
        html_body: str,
        unsubscribe_url: Optional[str] = None,
        message_id: Optional[str] = None,
    ) -> dict:
        """
        Send email and return metadata including sent timestamp.
        """
        if not self.smtp_host:
            raise RuntimeError("SMTP_HOST is not configured")

        msg = MIMEText(html_body, "html", "utf-8")
        msg["Subject"] = subject
        msg["From"] = self.from_addr
        msg["To"] = ", ".join(to_addrs)
        msg["Date"] = formatdate(localtime=True)
        msg["Message-ID"] = message_id or make_msgid(domain=self.from_addr.split("@")[-1])
        if self.reply_to:
            msg["Reply-To"] = self.reply_to

        # Unsubscribe header (best-effort)
        if unsubscribe_url:
            msg["List-Unsubscribe"] = f"<{unsubscribe_url}>"

        sent_ts = datetime.now(timezone.utc).isoformat()
        logger.info("Sending email to=%s subject=%s message_id=%s", to_addrs, subject, msg["Message-ID"])

        try:
            with smtplib.SMTP(self.smtp_host, self.smtp_port, timeout=30) as server:
                if self.use_tls:
                    server.starttls()
                if self.smtp_user and self.smtp_pass:
                    server.login(self.smtp_user, self.smtp_pass)
