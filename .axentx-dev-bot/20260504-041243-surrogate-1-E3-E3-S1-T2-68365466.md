"""
Real‑time Cost Dashboard API for Surrogate‑1

Features
--------
* **Cost ingestion** – Add single or bulk entries.
* **Filtering** – By provider, region, instance type, service, and time window.
* **Aggregation** – Total, per‑provider, per‑region, per‑instance‑type, per‑service, daily breakdowns.
* **Alerts** – Configure threshold‑based alerts (daily, hourly, percentage increase) with severity levels.
* **History** – Persist alert triggers for auditability.

Design notes
------------
* Uses `dataclasses` for clear, immutable value objects.
* All dates are UTC ISO‑8601 strings – no timezone confusion.
* `Enum` types guard against typos in provider/alert severity.
* The module is intentionally lightweight – no external dependencies beyond the stdlib.
"""

from __future__ import annotations

from dataclasses import dataclass, field, asdict
from datetime import datetime, timedelta
from enum import Enum
from typing import Dict, List, Optional, Any, Iterable
from collections import defaultdict

# --------------------------------------------------------------------------- #
# 1. Domain types
# --------------------------------------------------------------------------- #

class CloudProvider(str, Enum):
    """Supported cloud providers."""
    AWS   = "aws"
    GCP   = "gcp"
    AZURE = "azure"
    ALL   = "all"  # used only for filtering


class AlertSeverity(str, Enum):
    """Severity of a cost alert."""
    INFO     = "info"
    WARNING  = "warning"
    CRITICAL = "critical"


@dataclass(frozen=True)
class CostEntry:
    """A single cost record."""
    timestamp: datetime
    provider: CloudProvider
    region: str
    instance_type: str
    service: str
    cost: float
    currency: str = "USD"
    usage_quantity: float = 1.0
    usage_unit: str = "hours"

    def to_dict(self) -> Dict[str, Any]:
        return {
            "timestamp": self.timestamp.isoformat(),
            "provider": self.provider.value,
            "region": self.region,
            "instance_type": self.instance_type,
            "service": self.service,
            "cost": self.cost,
            "currency": self.currency,
            "usage_quantity": self.usage_quantity,
            "usage_unit": self.usage_unit,
        }

    enabled: bool = True
    created_at: datetime = field(default_factory=datetime.utcnow)

def test_cost_dashboard_aggregation():
    dashboard = CostDashboard()
    entries = [
        CostEntry(CloudProvider.AWS, InstanceType.T2_MICRO, "us-east-1", 10.5, datetime.now()),
        CostEntry(CloudProvider.AWS, InstanceType.T2_SMALL, "us-east-1", 20.0, datetime.now()),
        CostEntry(CloudProvider.AZURE, InstanceType.BASIC_A1, "eastus", 15.0, datetime.now())
    ]
    for entry in entries:
        dashboard.add_entry(entry)
    summary = dashboard.aggregate()
    assert summary.total_cost == 45.5
    assert summary.provider_breakdown == {
        "AWS": 30.5,
        "AZURE": 15.0
    }
    assert summary.instance_type_breakdown == {
        "T2_MICRO": 10.5,
        "T2_SMALL": 20.0,
        "BASIC_A1": 15.0
    }
