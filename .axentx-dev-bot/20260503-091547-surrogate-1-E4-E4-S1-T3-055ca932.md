<html lang="en">
<head>

    .badge.breach { background: #dc2626; }

    .timer { font-weight: 700; font-family: ui-monospace, SFMono-Regular, Menlo, monospace; }
  </style>
</head>
<body>
  <h1>Active Requests — SLA Status</h1>

  <!-- Server should render one card per active request -->
  <div class="request"
       data-request-id="req-12345"
       data-sla-hours="24"
       data-created-at="2025-06-01T08:30:00.000Z"
       data-sla-breach-logged="false">
    <div class="meta">
      <span class="badge" id="sla-badge-req-12345">—</span>
      <span class="sla">
        SLA target: <strong id="sla-target-req-12345">24</strong>h |
        Elapsed: <span class="timer" id="sla-elapsed-req-12345">—</span>
      </span>
    </div>
    <div>Status: <span id="status-text-req-12345">open</span></div>
  </div>

  <script>
    // SLA thresholds: ok < 80%, warning 80–100%, breach >100%
    function getSlaState(elapsedHours, targetHours) {
      if (targetHours <= 0) return 'ok';
      const pct = elapsedHours / targetHours;
      if (pct > 1) return 'breach';
      if (pct >= 0.8) return 'warning';
      return 'ok';
    }

    function formatHms(ms) {
      const totalSec = Math.floor(ms / 1000);
      const h = Math.floor(totalSec / 3600);
      const m = Math.floor((totalSec % 3600) / 60);
      const s = totalSec % 60;
      return `${h}h ${m}m ${s}s`;
    }

    function updateRequestCard(el) {
      const id = el.dataset.requestId;
      const targetHours = Number(el.dataset.slaHours || 0);
      const createdAt = new Date(el.dataset.createdAt || Date.now());
      const breachLogged = el.dataset.slaBreachLogged === 'true';

      const now = new Date();
      const elapsedMs = now - createdAt;
      const elapsedHours = elapsedMs / (1000 * 3600);

      const state = getSlaState(elapsedHours, targetHours);
      const badge = document.getElementById(`sla-badge-req-${id}`);
      const elapsedEl = document.getElementById(`sla-elapsed-req-${id}`);
      const targetEl = document.getElementById(`sla-target-req-${id}`);

      badge.textContent = state.toUpperCase();
      badge.className = `badge ${state}`;
      elapsedEl.textContent = formatHms(elapsedMs);
      targetEl.textContent = String(targetHours);

      // Server-driven breach logging (single source of truth)
      if (state === 'breach' && !breachLogged) {
        fetch(`/api/requests/${id}/sla-breach`, { method: 'POST' })
          .then(() => { el.dataset.slaBreachLogged = 'true'; })
          .catch(() => {});
      }
    }

    function tick() {
      document.querySelectorAll('.request').forEach(updateRequestCard);
    }
    setInterval(tick, 10000);
    tick();
  </script>
</body>
</html>
```
