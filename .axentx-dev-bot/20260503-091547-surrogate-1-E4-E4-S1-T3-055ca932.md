// models/sla.ts
export interface SLA {
  id: string;
  request: Request;
  breached: boolean;
  breachTime: Date;
}

export function calculateSLAState(request: Request, sla: SLA) {
  const timeElapsed = Date.now() - request.createdAt.getTime();
  const slaTarget = request.slaTarget;
  const breachThreshold = slaTarget * 60 * 60 * 1000; // 1 hour in milliseconds
  if (timeElapsed > breachThreshold) {
    sla.breached = true;
    sla.breachTime = new Date(timeElapsed);
  }
  return sla;
}
```
