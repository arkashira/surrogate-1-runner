package com.axentx.surrogate1.llm;

/**
 * Base contract for any LLM provider (Claude, Minimax, OpenAI, etc.).
 *
 * <p>All providers must implement:
 * <ul>
 *   <li> {@link #getProviderName()} – a stable, human‑readable identifier.</li>
 *   <li> {@link #isAvailable()} – quick health‑check (e.g. env vars present).</li>
 *   <li> {@link #getMaxTokens()} – the hard token ceiling for the provider.</li>
 *   <li> {@link #generate(String)} – legacy single‑string prompt support.</li>
 *   <li> {@link #chat(LLMRequest)} – full chat‑completion support.</li>
 * </ul>
 */
public interface LLMProvider {

    /** @return the canonical provider name (e.g. "Claude", "Minimax") */
    String getProviderName();

    /** @return true if the provider is configured and reachable */
    boolean isAvailable();

    /** @return the maximum token count the provider can return in one call */
    int getMaxTokens();

    /**
     * Legacy single‑string prompt API – useful for quick tests or legacy code.
     *
     * @param prompt the prompt text
     * @return the generated text
     */
    String generate(String prompt);

    /**
     * Full chat‑completion API.
     *
     * @param request the request payload
     * @return a normalised response
     */
    LLMResponse chat(LLMRequest request);
}
