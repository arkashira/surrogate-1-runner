const WebSocket = require('ws');

const wss = new WebSocket.Server({ noServer: true });
const clients = new Set();

wss.on('connection', (ws, req) => {
  // Optional: attach request info to ws if you want per-socket auth/subscriptions
  ws._req = req;
  clients.add(ws);
  ws.on('close', () => clients.delete(ws));
  ws.on('error', () => clients.delete(ws));
});

function broadcast(payload) {
  const data = JSON.stringify(payload);
  for (const client of clients) {
    if (client.readyState === WebSocket.OPEN) {
      client.send(data);
    }
  }
}

/**
 * Broadcast a request status update.
 * Keep payload minimal and stable for UI.
 */
function broadcastRequestStatus(requestId, status, metadata = {}) {
  broadcast({
    type: 'request_status_update',
    requestId,
    status,
    timestamp: new Date().toISOString(),
    ...metadata,
  });
}

module.exports = {
  wss,
  broadcast,
  broadcastRequestStatus,
};
