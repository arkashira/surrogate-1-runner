Surrogate-1 Workflow – Java‑free KiCad 9 integration.

This module provides a minimal, self‑contained workflow that:
* Loads a KiCad project (.kicad_pro) using the pcbnew Python API.
* Generates Gerber files, drill files, and a PDF preview.
* Stores the generated files in a user‑specified output directory.
* Is fully compatible with KiCad 9 and does not invoke any Java processes.

The workflow is driven by a YAML configuration file located at
`/opt/axentx/surrogate-1/workflow/config.yaml`.  The configuration
parameters are documented in the docstring below and in the example
`config.yaml` file.

Author: axentx engineering team
"""

import sys
import logging
import yaml
import pcbnew

# --------------------------------------------------------------------------- #
# Configuration handling
# --------------------------------------------------------------------------- #

DEFAULT_CONFIG_PATH = "/opt/axentx/surrogate-1/workflow/config.yaml"

def load_config(config_path: str = DEFAULT_CONFIG_PATH) -> dict:
    """
    Load the YAML configuration file.

    Parameters
    ----------
    config_path : str
        Path to the YAML configuration file.

    Returns
    -------
    dict
        Dictionary containing configuration values.
    """
    if not os.path.exists(config_path):
        raise FileNotFoundError(f"Configuration file not found: {config_path}")

    with open(config_path, "r", encoding="utf-8") as fp:
        cfg = yaml.safe_load(fp)

    # Basic validation
    required_keys = ["project_path", "output_dir", "gerber_options"]
    missing = [k for k in required_keys if k not in cfg]
    if missing:
        raise KeyError(f"Missing required config keys: {missing}")

    return cfg

def export_gerbers(board: pcbnew.BOARD, output_dir: str, options: dict) -> None:
    """
    Export Gerber files for the given board.

    Parameters
    ----------
    board : pcbnew.BOARD
        The loaded KiCad board.
    output_dir : str
        Directory where Gerber files will be written.
    options : dict
        Export options (mirroring, drill file format, etc.).
    """
    # Ensure output directory exists
    os.makedirs(output_dir, exist_ok=True)

    # Prepare export settings
    gerber_exporter = pcbnew.gerber.ExportBoardToGerbers(board)
    gerber_exporter.SetOutputDirectory(output_dir)

    # Apply user‑defined options
    # Example: mirroring
    if options.get("mirror_x", False):
        gerber_exporter.SetMirrorX(True)
    if options.get("mirror_y", False):
        gerber_exporter.SetMirrorY(True)

    # Drill file format
    drill_format = options.get("drill_file_format", "Excellon")
    if drill_format.lower() == "excellon":
        gerber_exporter.SetDrillFileFormat(pcbnew.gerber.DRILL_EXCELLON)
    elif drill_format.lower() == "dxf":
        gerber_exporter.SetDrillFileFormat(pcbnew.gerber.DRILL_DXF)
    else:
        logging.warning(f"Unsupported drill file format '{drill_format}', defaulting to Excellon")

    # Run export
