const transporter = nodemailer.createTransport({
  host: SMTP_HOST,
  port: SMTP_PORT,
  secure: SMTP_PORT === 465,
  auth: SMTP_USER ? { user: SMTP_USER, pass: SMTP_PASS } : undefined,
});

function exponentialBackoff(attempt: number): number {
  // capped exponential backoff with jitter
  return Math.min(1000 * 2 ** attempt + Math.floor(Math.random() * 200), 10000);
}

async function sendWithRetry(
  to: string,
  subject: string,
  html: string,
  text: string,
  attempt = 0
): Promise<boolean> {
  try {
    await transporter.sendMail({ from: FROM, to, subject, html, text });
    logger.info({ to, subject }, 'Email delivered');
    return true;
  } catch (err) {
    if (attempt >= 3) {
      logger.error({ to, subject, err, attempt }, 'Email delivery failed after retries');
      return false;
    }
    const delay = exponentialBackoff(attempt);
    logger.warn({ to, subject, attempt, delay, err }, 'Email delivery failed, retrying');
    await new Promise((r) => setTimeout(r, delay));
    return sendWithRetry(to, subject, html, text, attempt + 1);
  }
}

  const params = new URLSearchParams({ watcher: watcherId });

}
