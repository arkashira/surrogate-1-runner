// Base types
export interface CreateRequestPayload {
  type: string;
  title: string;
  owner: string;
  slaTargetHours: number;
}

export interface RequestItem {
  id: string;
  type: string;
  title: string;
  owner: string;
  slaTargetHours: number;
  status: 'open' | 'closed' | 'pending';
  createdAt: string;
  slaDeadline: string; // ISO8601 (server authoritative)
}

// API errors carry structured details when available
export class ApiError extends Error {
  constructor(
    message: string,
    public status: number,
    public details?: unknown
  ) {
    super(message);
    name = 'ApiError';
  }
}

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(path, {
    headers: { 'Content-Type': 'application/json', ...(init?.headers || {}) },
    ...init,
  });

  if (!res.ok) {
    let details;
    try {
      details = await res.json();
    } catch {
      details = undefined;
    }
    throw new ApiError(details?.message || `Request failed (${res.status})`, res.status, details);
  }

  return res.json();
}

export async function createRequest(payload: CreateRequestPayload): Promise<RequestItem> {
  return request<RequestItem>('/requests', {
    method: 'POST',
    body: JSON.stringify(payload),
  });
}

export async function fetchOpenRequests(): Promise<RequestItem[]> {
  return request<RequestItem[]>('/requests?status=open');
}
```

export function formatSlaDeadline(iso: string): string {
  const d = new Date(iso);
  if (Number.isNaN(d.getTime())) return iso;
  return d.toLocaleString(undefined, {
