class OverloadLinkGenerator {
    fun generateLink(input: String?): String {
        if (input == null || input.isEmpty()) {
            return ""
        }

        val links = input.split(",").map { it.trim() }.filter { it.isNotBlank() }
        return links.joinToString(",")
    }

    fun generate(
        signature: String,
        returnType: String?,
        paramTypes: List<String>
    ): String {
        if (signature.isBlank() || returnType == null) {
            throw IllegalArgumentException("Invalid input")
        }

        val paramTypesStr = paramTypes.joinToString(",")
        val link = "$signature#$returnType:$paramTypesStr"
        return link.trim()
    }
}
