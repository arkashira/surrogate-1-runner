    branches: [main, develop]
    paths:
      - 'charts/surrogate-1/**'

  lint-chart:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Helm
        uses: azure/setup-helm@v3
        with:
          version: 'v3.13.3'

      - name: Run helm lint
        run: helm lint charts/surrogate-1
        id: helm-lint

      - name: Check linting results
        if: steps.helm-lint.outcome != 'success'
        run: |
          echo "Helm lint failed"
          exit 1

      - name: Run helm template
        run: helm template surrogate-1 charts/surrogate-1 --debug

      - name: Bump chart version
        if: steps.helm-lint.outcome == 'success'
        run: |
          helm version --app-version $(git rev-parse --short HEAD)

      - name: Commit and push bumped chart version
        if: steps.helm-lint.outcome == 'success'
        uses: actions/checkout@v4
        with:
          token: ${{ secrets.GITHUB_TOKEN }}
        run: |
          git config --global user.name 'github-actions[bot]'
          git config --global user.email 'github-actions[bot]@users.noreply.github.com'
          git add .
          git commit -m "Bump chart version to $(git rev-parse --short HEAD)"
          git push origin main
```
