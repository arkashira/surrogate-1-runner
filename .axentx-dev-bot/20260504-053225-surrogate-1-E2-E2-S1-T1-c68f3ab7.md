import logging
from datetime import datetime
from typing import Any, Dict

from flask import request, g

# This middleware assumes a Flask-like web app. Adjust imports if using FastAPI or Django.
# It logs: timestamp, user_id, endpoint, and request metadata.

logger = logging.getLogger("ai_audit_logger")


def _extract_user_id() -> str:
    # Try common headers or Flask g/user object; adapt to your auth system.
    # 1) If application already parsed an authenticated user into g.current_user
    user = getattr(g, "current_user", None)
    if user is not None:
        uid = getattr(user, "id", None)
        if uid:
            return str(uid)

    # 2) Try standard headers (e.g., X-User-Id)
    uid = request.headers.get("X-User-Id")
    if uid:
        return str(uid)

    # 3) Fallback to anonymous
    return "anonymous"


def _extract_endpoint() -> str:
    # Endpoint path or function name
    return request.path or request.endpoint or "/unknown"


def _extract_request_meta() -> Dict[str, Any]:
    try:
        # Read body if not too large; some frameworks may not allow reading twice.
        data = request.get_json(silent=True)
        if data is None:
            # Fall back to raw data if JSON not provided
            data = request.data.decode("utf-8", errors="replace")
        return {"payload": data}
    except Exception:
        return {"payload": "<unreadable>"}


def _build_log_record() -> Dict[str, Any]:
    ts = datetime.utcnow().isoformat(timespec="microseconds") + "Z"
    user_id = _extract_user_id()
    endpoint = _extract_endpoint()
    meta = _extract_request_meta()

    record = {
        "timestamp_utc": ts,
        "user_id": user_id,
        "endpoint": endpoint,
        "request_method": request.method,
        "headers": dict(request.headers),
        "meta": meta,
    }
    return record


def log_ai_request():
    try:
        rec = _build_log_record()
        logger.info("ai_request", extra={"audit_record": rec})
    except Exception:
        # Do not disrupt the main request flow if logging fails
        logger.exception("Failed to log AI request audit")


# Optional: expose as decorator for explicit use
def ai_audit_endpoint(func):
    def wrapper(*args, **kwargs):
        log_ai_request()
        return func(*args, **kwargs)
    wrapper.__name__ = func.__name__
    return wrapper
