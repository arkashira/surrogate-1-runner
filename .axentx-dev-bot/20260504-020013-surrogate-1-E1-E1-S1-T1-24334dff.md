## 1. Acceptance Criteria
1. **Token Generation** – An admin can create a recovery token for any user via UI or CLI; the returned token is a 64‑character base‑64 string (derived from 32 random bytes) and is stored hashed (SHA‑256) in the database.  
2. **Expiration** – Generated tokens automatically become invalid 30 days after `created_at`; the system must reject any use of an expired token.  
3. **Uniqueness** – No two active (non‑expired, non‑revoked) tokens for the same user may have the same plaintext value.  
4. **Usage Logging** – Each token record must capture `used_at`, `used_by_ip`, and `used_by_admin_id` when the token is consumed; this history must be queryable by admins.  
5. **Revocation** – Admins can manually revoke a token; a revoked token is marked `revoked_at` and cannot be used thereafter.  
6. **Audit Trail** – All token creation, usage, and revocation events are written to the audit log with timestamps and admin identifiers.  
7. **Performance** – Retrieval of a user’s active recovery tokens must execute in ≤ 50 ms on a dataset of 10 k tokens (validated by integration test).

package models_test

import (
    "testing"
    "time"
    "crypto/rand"

    "github.com/stretchr/testify/assert"

// Helper to generate a raw token for testing
func rawToken() string {
    b := make([]byte, 32)
    _, _ = rand.Read(b)
    return base64.RawURLEncoding.EncodeToString(b)
}

func TestGenerateRecoveryToken(t *testing.T) {
    token, err := models.GenerateRecoveryToken(123) // userID = 123
    assert.NoError(t, err)
    assert.NotEmpty(t, token.Plaintext)               // raw token returned
    assert.Len(t, token.Plaintext, 43)                // 32 bytes base64url -> 43 chars
    assert.NotEmpty(t, token.Hashed)                  // SHA‑256 hex string length 64
    assert.Equal(t, 123, token.UserID)
    assert.WithinDuration(t, time.Now(), token.CreatedAt, time.Second)
}

func TestHashingIsDeterministic(t *testing.T) {
    raw := rawToken()
    h1 := models.HashToken(raw)
    h2 := models.HashToken(raw)
    assert.Equal(t, h1, h2)
    assert.Len(t, h1, 64) // SHA‑256 hex
}

func TestTokenExpiration(t *testing.T) {
    token := models.NewRecoveryTokenForTest(1) // creates token with CreatedAt = now
    token.CreatedAt = time.Now().Add(-31 * 24 * time.Hour) // simulate old token
    assert.True(t, token.IsExpired())
    token.CreatedAt = time.Now().Add(-29 * 24 * time.Hour)
    assert.False(t, token.IsExpired())
}

func TestUsageLogging(t *testing.T) {
    token := models.NewRecoveryTokenForTest(1)
    err := token.MarkUsed(time.Now(), "192.0.2.10", 42) // adminID = 42
    assert.NoError(t, err)
    assert.NotZero(t, token.UsedAt)
    assert.Equal(t, "192.0.2.10", token.UsedByIP)
    assert.Equal(t, 42, token.UsedByAdminID)
    // second usage should fail
    err = token.MarkUsed(time.Now(), "192.0.2.11", 43)
    assert.Error(t, err) // already used
}

func TestRevocation(t *testing.T) {
    token := models.NewRecoveryTokenForTest(1)
    err := token.Revoke(time.Now())
    assert.NoError(t, err)
    assert.NotZero(t, token.RevokedAt)
    // revoked token cannot be marked used
    err = token.MarkUsed(time.Now(), "10.0.0.1", 99)
    assert.Error(t, err)
