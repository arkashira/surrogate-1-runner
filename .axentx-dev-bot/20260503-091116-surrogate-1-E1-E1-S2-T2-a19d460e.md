// /opt/axentx/surrogate-1/frontend/components/MetadataEditor.tsx

import React, { useState, useEffect } from 'react';
import { useForm } from 'react-hook-form';
import axios from 'axios';
import pdfjs from 'pdfjs';

interface Props {
  documentId: string;
}

const MetadataEditor: React.FC<Props> = ({ documentId }) => {
  const { register, handleSubmit, watch } = useForm();
  const [title, setTitle] = useState('');
  const [author, setAuthor] = useState('');
  const [tags, setTags] = useState<string[]>([]);
  const [saving, setSaving] = useState(false);

  useEffect(() => {
    const saveChanges = async () => {
      const { title: newTitle, author: newAuthor, tags: newTags } = watch();
      setSaving(true);
      try {
        const response = await axios.put(`/api/documents/${documentId}`, {
          title: newTitle,
          author: newAuthor,
          tags: newTags,
        });
        console.log(response.data);
      } catch (error) {
        console.error(error);
      } finally {
        setSaving(false);
      }
    };

    saveChanges();
  }, [watch]);

  const onSubmit = async (data: any) => {
    const { title: newTitle, author: newAuthor, tags: newTags } = data;
    setTitle(newTitle);
    setAuthor(newAuthor);
    setTags(newTags);
  };

  return (
    <div>
      <h2>Metadata Editor</h2>
      <form onSubmit={handleSubmit(onSubmit)}>
        <label>
          Title:
          <input
            type="text"
            defaultValue={title}
            {...register('title')}
          />
        </label>
        <br />
        <label>
          Author:
          <input
            type="text"
            defaultValue={author}
            {...register('author')}
          />
        </label>
        <br />
        <label>
          Tags:
          <input
            type="text"
            defaultValue={tags.join(', ')}
            {...register('tags')}
          />
        </label>
        <br />
        <button type="submit" disabled={saving}>
          Save
        </button>
      </form>
    </div>
  );
};

export default MetadataEditor;
```
