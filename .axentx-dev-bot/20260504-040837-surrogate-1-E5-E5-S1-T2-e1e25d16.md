import logging
from typing import Dict, List
import smtplib
from email.mime.text import MIMEText

class NotificationSystem:
    def __init__(self, config: Dict):
        self.config = config
        self.logger = logging.getLogger(__name__)

    def log_error(self, error: str, error_type: str):
        self.logger.error(f"Error Type: {error_type}, Error: {error}")

    def send_notification(self, error: str, error_type: str):
        if error_type in self.config.get('critical_errors', []):
            self._send_email_notification(error, error_type)

    def _send_email_notification(self, error: str, error_type: str):
        msg = MIMEText(f"Error Type: {error_type}\nError: {error}")
        msg['Subject'] = f"Critical Error Notification: {error_type}"
        msg['From'] = self.config.get('email_from')
        msg['To'] = self.config.get('email_to')

        try:
            with smtplib.SMTP(self.config.get('smtp_server'), self.config.get('smtp_port')) as server:
                server.starttls()
                server.login(self.config.get('smtp_username'), self.config.get('smtp_password'))
                server.send_message(msg)
            self.logger.info("Notification email sent successfully")
        except Exception as e:
            self.logger.error(f"Failed to send notification email: {e}")

from unittest.mock import patch, MagicMock
import pytest
from datetime import datetime

from surrogate_1.notifications.system import NotificationSystem, ErrorEvent, ErrorSeverity

def test_log_error_with_severity():
    ns = NotificationSystem(channels=["email"])
    error = ErrorEvent(
        error_type="WorkflowTimeout",
        message="Task took too long",
        severity=ErrorSeverity.CRITICAL,
        timestamp=datetime.now()
    )
    with patch("logging.error") as mock_log:
        ns.log_error(error)
        mock_log.assert_called_once_with(
            f"[{error.severity.name}] {error.error_type}: {error.message} at {error.timestamp}"
        )

def test_send_notification_on_critical_error():
    ns = NotificationSystem(channels=["email"])
    error = ErrorEvent(
        error_type="WorkflowFailure",
        message="Step failed: upload",
        severity=ErrorSeverity.CRITICAL,
        timestamp=datetime.now()
    )
    with patch.object(ns, "send_email_notification") as mock_notify:
        ns.handle_error(error)
        mock_notify.assert_called_once_with(error)

def test_no_notification_on_non_critical_error():
    ns = NotificationSystem(channels=["email"])
    error = ErrorEvent(
        error_type="RetryAttempt",
        message="Retrying step 1",
        severity=ErrorSeverity.WARNING,
        timestamp=datetime.now()
    )
    with patch.object(ns, "send_email_notification") as mock_notify:
        ns.handle_error(error)
        mock_notify.assert_not_called()

def test_email_notification_format():
    ns = NotificationSystem(channels=["email"])
    error = ErrorEvent(
        error_type="AuthFailure",
        message="Invalid credentials",
        severity=ErrorSeverity.CRITICAL,
        timestamp=datetime(2026, 5, 3, 10, 0, 0)
    )
    with patch("smtplib.SMTP") as mock_smtp:
        ns.send_email_notification(error)
        mock_smtp_instance = mock_smtp.return_value
        args, kwargs = mock_smtp_instance.send_message.call_args
        msg = args[0]
        assert "Subject: [CRITICAL] Workflow Error: AuthFailure" in msg.as_string()
        assert "Invalid credentials" in msg.as_string()
        assert "2026-05-03 10:00:00" in msg.as_string()
