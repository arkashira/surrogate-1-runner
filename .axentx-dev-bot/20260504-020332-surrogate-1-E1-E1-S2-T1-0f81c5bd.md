import threading
import time
import pytest
from surrogate_1.issue_store import IssueStore, Issue

def test_issue_store_thread_safety():
    store = IssueStore()
    issues = [Issue(id=i, title=f"Issue {i}", severity="high") for i in range(100)]

    def writer():
        for issue in issues:
            store.add_issue(issue)
            time.sleep(0.001)

    def reader():
        # Random reads while writes are happening
        for _ in range(200):
            _ = store.list_issues()
            time.sleep(0.001)

    threads = [threading.Thread(target=writer) for _ in range(3)] + \
              [threading.Thread(target=reader) for _ in range(3)]

    for t in threads:
        t.start()
    for t in threads:
        t.join()

    # After all threads complete, all issues must be present exactly once
    stored_ids = {i.id for i in store.list_issues()}
    assert stored_ids == {i.id for i in issues}
