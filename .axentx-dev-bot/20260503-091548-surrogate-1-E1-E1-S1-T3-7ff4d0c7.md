```python
from __future__ import annotations

import string
import time
import secrets
from typing import Optional, Callable

ALPHANUM = string.ascii_lowercase + string.digits
SLUG_LENGTH = 8

def _random_suffix(length: int = 4) -> str:
    return "".join(secrets.choice(ALPHANUM) for _ in range(length))

def generate_slug(
    title: str,
    exists_fn: Optional[Callable[[str], bool]] = None,
    max_attempts: int = 20,
) -> str:
    """
    Generate a short, URL-friendly unique slug.

    Strategy:
      - Normalize title to a short prefix (alphanum, lowercase, truncated)
      - Append timestamp base36 (seconds since epoch) for time ordering
      - Append random suffix for collision resistance
      - Retry with new random suffix if collision detected (exists_fn)

    Returns a string safe for use in URLs (lowercase, alnum).
    """
    # Normalize title prefix
    prefix = "".join(c for c in title.lower() if c in ALPHANUM)[:12]
    if not prefix:
        prefix = "req"

    base = int(time.time())
    ts_part = base_to36(base)

    for _ in range(max_attempts):
        rand = _random_suffix(SLUG_LENGTH - len(prefix) - len(ts_part) - 1)
        slug = f"{prefix}-{ts_part}{rand}"
        slug = slug[:SLUG_LENGTH + 12]  # keep reasonable length
        if exists_fn is None or not exists_fn(slug):
            return slug

    # Fallback: fully random
    for _ in range(max_attempts):
        slug = _random_suffix(SLUG_LENGTH)
        if exists_fn is None or not exists_fn(slug):
            return slug

    raise RuntimeError("Unable to generate unique slug after retries")

def base_to36(value: int) -> str:
    if value < 0:
        raise ValueError("base_to36 requires non-negative integer")
    if value == 0:
        return "0"
    digits = []
    while value:
        value, rem = divmod(value, 36)
        digits.append("0123456789abcdefghijklmnopqrstuvwxyz"[rem])
    return "".join(reversed(digits))
```
