import json
import os
import platform
import time
from pathlib import Path
from typing import Dict, Any
import requests

# Configuration
CONFIG_DIR = Path.home() / ".batteryguard"
CONFIG_PATH = CONFIG_DIR / "telemetry.json"
DEFAULT_ENDPOINT = "https://axentx-telemetry.example.com/api/v1/events"

# Helper functions
def _ensure_config_dir() -> None:
    CONFIG_DIR.mkdir(parents=True, exist_ok=True)

def _load_config() -> Dict[str, Any]:
    _ensure_config_dir()
    if not CONFIG_PATH.exists():
        return {"opt_out": False}
    try:
        with CONFIG_PATH.open("r", encoding="utf-8") as f:
            return json.load(f)
    except Exception:
        return {"opt_out": False}

def _save_config(cfg: Dict[str, Any]) -> None:
    _ensure_config_dir()
    with CONFIG_PATH.open("w", encoding="utf-8") as f:
        json.dump(cfg, f, indent=2)

# Public API
def disable() -> None:
    cfg = _load_config()
    cfg["opt_out"] = True
    _save_config(cfg)

def enable() -> None:
    cfg = _load_config()
    cfg["opt_out"] = False
    _save_config(cfg)

def is_opted_out() -> bool:
    cfg = _load_config()
    return cfg.get("opt_out", False)

def send_event(command_name: str, endpoint: str = DEFAULT_ENDPOINT) -> None:
    if is_opted_out():
        return

    event_data = {
        'command': command_name,
        'timestamp': int(time.time()),
        'os': platform.system(),
        'os_version': platform.release()
    }

    try:
        requests.post(endpoint, json=event_data, timeout=2)
    except Exception:
        pass
```
