package models

import (
	"sort"
	"sync"
	"time"
)

// DailyCost represents the aggregated cost for a single day
type DailyCost struct {
	Date             time.Time         `json:"date"`
	Total            float64           `json:"total"`
	ProviderBreakdown map[string]float64 `json:"providerBreakdown"`
}

// CostHistoryStore stores daily cost entries with thread-safe operations
type CostHistoryStore struct {
	mu   sync.RWMutex
	data map[string]*DailyCost // key: "YYYY-MM-DD"
}

// NewCostHistoryStore creates a new empty store
func NewCostHistoryStore() *CostHistoryStore {
	return &CostHistoryStore{
		data: make(map[string]*DailyCost),
	}
}

// addOrUpdate adds amount to the given provider for the supplied date
func (s *CostHistoryStore) addOrUpdate(date time.Time, provider string, amount float64) {
	key := date.Format("2006-01-02")
	s.mu.Lock()
	defer s.mu.Unlock()

	d, ok := s.data[key]
	if !ok {
		// Truncate to midnight to keep consistent dates
		day := time.Date(date.Year(), date.Month(), date.Day(), 0, 0, 0, 0, date.Location())
		d = &DailyCost{
			Date:             day,
			ProviderBreakdown: make(map[string]float64),
		}
		s.data[key] = d
	}

	d.Total += amount
	d.ProviderBreakdown[provider] += amount
	s.cleanup()
}

// RecordCost records a cost for today
func (s *CostHistoryStore) RecordCost(provider string, amount float64) {
	s.addOrUpdate(time.Now(), provider, amount)
}

// RecordCostAt records a cost for an arbitrary date (exported for testing/back-fill)
func (s *CostHistoryStore) RecordCostAt(date time.Time, provider string, amount float64) {
	s.addOrUpdate(date, provider, amount)
}

// GetLast30Days returns the daily cost entries for the most recent 30 days, ordered from oldest to newest
func (s *CostHistoryStore) GetLast30Days() []DailyCost {
	s.mu.RLock()
	defer s.mu.RUnlock()

	// Collect and sort dates
	keys := make([]string, 0, len(s.data))
	for k := range s.data {
		keys = append(keys, k)
	}
	sort.Strings(keys)

	cutoff := time.Now().AddDate(0, 0, -30)
	var result []DailyCost
	for _, k := range keys {
		entry := s.data[k]
		if entry.Date.Before(cutoff) {
			continue
		}
		result = append(result, *entry)
	}
	return result
}

// cleanup removes entries older than 30 days
func (s *CostHistoryStore) cleanup() {
	cutoff := time.Now().AddDate(0, 0, -30)
	for k, v := range s.data {
		if v.Date.Before(cutoff) {
			delete(s.data, k)
		}
	}
}
