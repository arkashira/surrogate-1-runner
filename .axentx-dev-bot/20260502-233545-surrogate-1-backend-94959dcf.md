Update the GitHub Actions workflow to run the `bin/snapshot.sh` script before running the `bin/dataset-enrich.sh` script:

```yml
name: Ingest Dataset

on:
  workflow_dispatch:
    inputs:
      date-folder:
        description: 'Date folder to process'
        required: true
      num-shards:
        description: 'Number of shards to process'
        required: true

jobs:
  ingest:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Run pre-flight snapshot generation script
        run: |
          bin/snapshot.sh
          export HF_TOKEN=${{ secrets.HF_TOKEN }}
          export DATE_FOLDER=${{ inputs.date-folder }}
          export NUM_SHARDS=${{ inputs.num-shards }}

      - name: Run dataset enrichment script
        run: |
          bin/dataset-enrich.sh
          export HF_TOKEN=${{ secrets.HF_TOKEN }}
          export DATE_FOLDER=${{ inputs.date-folder }}
          export NUM_SHARDS=${{ inputs.num-shards }}
```
