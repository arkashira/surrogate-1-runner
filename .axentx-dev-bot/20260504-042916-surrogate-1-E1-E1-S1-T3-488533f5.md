import sqlite3
from datetime import datetime
from typing import List, Dict, Optional, Tuple

class ConversationStore:
    """Persistent storage for conversation history using SQLite"""

    def __init__(self, db_path: str = 'conversation_history.db'):
        self.conn = sqlite3.connect(db_path)
        self.cursor = self.conn.cursor()
        self._initialize_db()

    def _initialize_db(self) -> None:
        """Create the database table if it doesn't exist"""
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS conversations (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id TEXT NOT NULL,
                user_id TEXT NOT NULL,
                message_type TEXT NOT NULL CHECK(message_type IN ('user', 'ai')),
                message TEXT NOT NULL,
                timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        self.conn.commit()

    def save_message(
        self,
        session_id: str,
        user_id: str,
        message_type: str,
        message: str
    ) -> None:
        """Store a message in the conversation history"""
        if message_type not in ('user', 'ai'):
            raise ValueError("message_type must be either 'user' or 'ai'")

        self.cursor.execute('''
            INSERT INTO conversations
            (session_id, user_id, message_type, message)
            VALUES (?, ?, ?, ?)
        ''', (session_id, user_id, message_type, message))
        self.conn.commit()

    def get_conversation_history(
        self,
        session_id: str,
        user_id: Optional[str] = None,
        limit: Optional[int] = None
    ) -> List[Dict]:
        """Retrieve conversation history for a session"""
        query = '''
            SELECT message_type, message, timestamp
            FROM conversations
            WHERE session_id = ?
        '''
        params = [session_id]

        if user_id:
            query += ' AND user_id = ?'
            params.append(user_id)

        query += ' ORDER BY timestamp ASC'

        if limit:
            query += ' LIMIT ?'
            params.append(limit)

        self.cursor.execute(query, params)

        return [
            {
                'message_type': row[0],
                'message': row[1],
                'timestamp': datetime.fromisoformat(row[2])
            }
            for row in self.cursor.fetchall()
        ]

    def close(self) -> None:
        """Close the database connection"""
        self.conn.close()
