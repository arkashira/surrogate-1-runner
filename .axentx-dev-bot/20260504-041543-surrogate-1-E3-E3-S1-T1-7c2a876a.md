import React, { useState, useEffect, useCallback } from 'react';

const CreditAllocator = ({ 
  totalCredits = 1000, 
  bulkProjectedUsage = 0, 
  monthlyProjectedUsage = 0,
  onAllocationChange 
}) => {
  const [bulkPercentage, setBulkPercentage] = useState(50);
  const [monthlyPercentage, setMonthlyPercentage] = useState(30);
  const [error, setError] = useState('');

  // Calculate actual credit amounts from percentages
  const bulkAllocation = Math.round(totalCredits * (bulkPercentage / 100));
  const monthlyAllocation = Math.round(totalCredits * (monthlyPercentage / 100));
  const totalAllocated = bulkAllocation + monthlyAllocation;
  const remainingCredits = totalCredits - totalAllocated;

  // Validate allocation doesn't exceed total
  useEffect(() => {
    const newError = totalAllocated > totalCredits 
      ? `Allocation exceeds available credits by ${totalAllocated - totalCredits}`
      : '';
    setError(newError);
  }, [totalAllocated, totalCredits]);

  // Notify parent of allocation changes
  useEffect(() => {
    if (onAllocationChange) {
      onAllocationChange({
        bulk: bulkAllocation,
        monthly: monthlyAllocation,
        remaining: remainingCredits,
        isValid: totalAllocated <= totalCredits
      });
    }
  }, [bulkAllocation, monthlyAllocation, remainingCredits, totalAllocated, onAllocationChange]);

  const handleBulkChange = (e) => {
    const value = Math.min(100, Math.max(0, parseInt(e.target.value) || 0));
    setBulkPercentage(value);
  };

  const handleMonthlyChange = (e) => {
    const value = Math.min(100, Math.max(0, parseInt(e.target.value) || 0));
    setMonthlyPercentage(value);
  };

  const getUsageStatus = (allocated, projected) => {
    if (allocated === 0) return projected > 0 ? 'over-budget' : 'healthy';
    const percentage = (projected / allocated) * 100;
    if (percentage > 100) return 'over-budget';
    if (percentage > 80) return 'warning';
    return 'healthy';
  };

  const bulkStatus = getUsageStatus(bulkAllocation, bulkProjectedUsage);
  const monthlyStatus = getUsageStatus(monthlyAllocation, monthlyProjectedUsage);

  const isOverBudget = totalAllocated > totalCredits;

  return (
    <div className="credit-allocator">
      <h3>Credit Budget Allocation</h3>
      
      <div className="total-credits">
        <span className="label">Total Available Credits:</span>
        <span className="value">{totalCredits.toLocaleString()}</span>
      </div>

      <div className="allocation-grid">
        {/* Bulk Tier Allocation */}
        <div className="allocation-card bulk">
          <h4>Bulk Tier</h4>
          <div className="percentage-input">
            <label>
              <input
                type="number"
                min="0"
                max="100"
                value={bulkPercentage}
                onChange={handleBulkChange}
                disabled={isOverBudget}
              />
              %
            </label>
          </div>
          <div className="allocation-amount">
            <span className="label">Allocated:</span>
            <span className="value">{bulkAllocation.toLocaleString()} credits</span>
          </div>
          <div className="projected-usage">
            <span className="label">Projected Usage:</span>

              {bulkProjectedUsage.toLocaleString()} credits
              {bulkProjectedUsage > bulkAllocation && ' (OVER)'}
