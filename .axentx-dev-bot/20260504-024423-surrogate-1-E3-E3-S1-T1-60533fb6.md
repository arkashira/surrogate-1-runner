## Unified Post‑Mortem Dashboard

Below is a single, production‑ready implementation that pulls together the strongest ideas from the two candidates:

* **Real‑world data fetching** (`fetch('/api/postmortem')`) with proper error handling  
* **Two‑way filtering** – by artifact type *and* a free‑text search on title/description  
* **Export** – both JSON (for downstream tooling) and CSV (for spreadsheets)  
* **Responsive, accessible UI** – semantic markup, ARIA labels, and a clean CSS module  
* **Graceful UX** – loading spinner, error banner, and “no results” placeholder  

> **Tip** – If you need to support older browsers, replace the `fetch` call with a polyfill or `axios`.

### 1. `src/ui/pages/PostMortemDashboard.jsx`

```jsx
import React, { useState, useEffect, useMemo } from 'react';
import './postmortem.css';

/**
 * PostMortemDashboard
 *
 * • Fetches correlated artifacts and alerts from `/api/postmortem`
 * • Allows filtering by type (alert / artifact / all) and a free‑text search
 * • Renders a chronological timeline
 * • Supports JSON & CSV export of the current view
 */
const PostMortemDashboard = () => {
  /* ---------- State ---------- */
  const [data, setData] = useState([]);
  const [filterType, setFilterType] = useState('all'); // 'all' | 'alert' | 'artifact'
  const [search, setSearch] = useState('');
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [exporting, setExporting] = useState(false);

  /* ---------- Data fetching ---------- */
  useEffect(() => {
    const fetchData = async () => {
      try {
        const resp = await fetch('/api/postmortem');
        if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
        const json = await resp.json();

        // Normalise & sort
        const items = json
          .map((i) => ({
            id: i.id,
            type: i.type,          // 'alert' | 'artifact'
            title: i.title || i.description || '',
            timestamp: i.timestamp,
            details: i.details || i.description || '',
          }))
          .sort((a, b) => new Date(a.timestamp) - new Date(b.timestamp));

        setData(items);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    };
    fetchData();
  }, []);

  /* ---------- Filtering ---------- */
  const filtered = useMemo(() => {
    return data.filter((item) => {
      const typeMatch = filterType === 'all' || item.type === filterType;
      const searchMatch =
        !search ||
        item.title.toLowerCase().includes(search.toLowerCase()) ||
        item.details.toLowerCase().includes(search.toLowerCase());
      return typeMatch && searchMatch;
    });
  }, [data, filterType, search]);

  /* ---------- Export helpers ---------- */
  const exportJSON = () => {
    const blob = new Blob([JSON.stringify(filtered, null, 2)], {
      type: 'application/json',
    });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `postmortem_${new Date().toISOString()}.json`;
    a.click();
    URL.revokeObjectURL(url);
  };

  const exportCSV = () => {
    setExporting(true);
    const header = ['ID', 'Type', 'Title', 'Timestamp', 'Details'];
    const rows = filtered.map((i) => [
      i.id,
      i.type,
      i.title,
      i.timestamp,
      i.details,
    ]);
    const csv = [header, ...rows]
      .map((row) => row.map((v) => `"${String(v).replace(/"/g, '""')}"`).join(','))
      .join('\n');

    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `postmortem_${new Date().toISOString()}.csv`;
    a.click();
    URL.revokeObjectURL(url);
    setExporting(false);
  };

  /* ---------- UI ---------- */
  if (loading) return <div className="pm-loading">Loading post‑mortem data…</div>;
