const nodemailer = require('nodemailer');
const smtpTransport = require('nodemailer-smtp-transport');

// Securely manage credentials using environment variables or a secrets manager
const transporter = nodemailer.createTransport(smtpTransport({
  host: process.env.SMTP_HOST || 'smtp.gmail.com',
  port: parseInt(process.env.SMTP_PORT) || 587,
  secure: false, // true for 465, false for other ports
  auth: {
    user: process.env.EMAIL_USER,
    pass: process.env.EMAIL_PASS
  }
}));

function sendAlert(email, message) {
  const mailOptions = {
    from: process.env.EMAIL_FROM || 'your-email@gmail.com',
    to: email,
    subject: 'Non-Compliance Alert',
    text: message
  };

  return new Promise((resolve, reject) => {
    transporter.sendMail(mailOptions, (error, info) => {
      if (error) {
        console.error('Error sending email:', error);
        reject(error);
      } else {
        console.log('Email sent: ' + info.response);
        resolve(info);
      }
    });
  });
}

module.exports = { sendAlert };

## 1. Acceptance Criteria

- **AC1** – *Alert configuration*  
  The system must allow a sysadmin to create an alert rule that specifies a non‑compliance condition (e.g., `cpu_usage > 90%`) and a list of recipient email addresses.

- **AC2** – *Real‑time delivery*  
  When a non‑compliance event occurs, the alert must be dispatched within **≤ 5 seconds** of event detection.

- **AC3** – *Escalation*  
  If the primary recipient does not acknowledge the alert within **10 minutes**, the alert must be automatically resent to the escalation list.

- **AC4** – *API validation*  
  The `/api/alerts` endpoint must reject malformed payloads with a **400 Bad Request** and a descriptive error message.

- **AC5** – *Error handling*  
  Failed email sends must be retried **3 times** with exponential back‑off; after the third failure the system must log the incident and return a **502 Bad Gateway** to the caller.
