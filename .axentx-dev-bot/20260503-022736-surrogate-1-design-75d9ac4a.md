Replace `bin/dataset-enrich.sh` with a **manifest-driven, CDN-bypass ingestion worker** (`bin/dataset-enrich.py`) that:

- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN`, `REPO_ID` (default: `axentx/surrogate-1-training-pairs`)
- Uses **one API call** (`list_repo_tree` on the date folder) → saves `manifest.json`
- Each shard deterministically hashes `file_path` → picks its slice; only processes its slice
- Downloads via **CDN bypass** (`https://huggingface.co/datasets/{repo}/resolve/main/{path}`) — no Authorization header, avoids 429 API limits
- Projects to `{prompt, response}` only at parse time (avoids pyarrow CastError on mixed schemas)
- Dedups via content-based md5 store (`lib/dedup.py`)
- Uploads to `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl`
- Reuses existing HF token and repo layout; keeps GitHub Actions matrix unchanged

def deterministic_shard(slug: str, total: int) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % total

def list_date_folder(repo: str, date: str, token: str) -> List[str]:
    """Single API call: list files under date/ (non-recursive)."""
    try:
        items = HF_API.list_repo_tree(
            repo=repo,
            path=date,
            recursive=False,
            token=token,
        )
    except Exception:
        # fallback: list root and filter by prefix
        items = HF_API.list_repo_tree(
            repo=repo,
            path="",
            recursive=False,
            token=token,
        )
        items = [i for i in items if i.startswith(f"{date}/")]
    # keep only leaf files (skip subfolders)
    return [f for f in items if not f.endswith("/")]

def build_manifest(repo: str, date: str, token: str, out_path: Path) -> List[str]:
    files = list_date_folder(repo, date, token)
    manifest = {
        "repo": repo,
        "date": date,
        "files": sorted(files),
        "generated_at": datetime.datetime.utcnow().isoformat() + "Z",
    }
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest["files"]

def cdn_download_url(repo: str, path: str) -> str:
    return f"{CDN_ROOT}/{repo}/resolve/main/{path}"

def safe_download(url: str, timeout: int = 30) -> bytes:
    resp = requests.get(url, timeout=timeout)
