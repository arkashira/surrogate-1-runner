"""
Workflow automation for KiCAD 9 design generation.

This script provides a command‑line interface to:
  * Load a KiCAD schematic (.kicad_sch)
  * Generate a PCB layout (.kicad_pcb) from the schematic
  * Run Design Rule Checks (DRC)
  * Export Gerber files
  * Generate a Bill of Materials (BOM)

It is intended to be invoked by a CI workflow or a local developer
to automate repetitive tasks and reduce manual effort.
"""

import argparse
import logging
import os
import sys
import tempfile
from pathlib import Path
from typing import List, Optional

# KiCAD 9 provides a Python API under the `pcbnew` module.
# Importing it may fail if KiCAD is not installed; we guard the import
# and provide clear error messages.
try:
    import pcbnew
except ImportError as exc:
    # Provide a helpful message for developers
    raise RuntimeError(
        "KiCAD 9 Python API (pcbnew) is not available. "
        "Ensure KiCAD 9 is installed and the environment variable "
        "PYTHONPATH includes the KiCAD Python modules."
    ) from exc


# --------------------------------------------------------------------------- #
# Helper functions
# --------------------------------------------------------------------------- #
def _setup_logging(verbose: bool) -> None:
    """Configure root logger."""
    level = logging.DEBUG if verbose else logging.INFO
    logging.basicConfig(
        level=level,
        format="%(asctime)s [%(levelname)s] %(message)s",
        datefmt="%H:%M:%S",
    )


def _load_schematic(schematic_path: Path) -> pcbnew.SCH:
    """Load a KiCAD schematic file."""
    logging.debug(f"Loading schematic: {schematic_path}")
    if not schematic_path.exists():
        raise FileNotFoundError(f"Schematic not found: {schematic_path}")
    sch = pcbnew.SCH()
    sch.Load(schematic_path.as_posix())
    return sch


def _generate_pcb_from_schematic(
    sch: pcbnew.SCH, pcb_path: Path
) -> pcbnew.PCBNEW:
    """Generate a PCB layout from a schematic."""
    logging.debug(f"Generating PCB: {pcb_path}")
    pcb = pcbnew.PCBNEW()
    pcb.Load(sch.GetFileName())  # Load the schematic into the PCB context
    pcb.Save(pcb_path.as_posix())
    return pcb


def _run_drc(pcb: pcbnew.PCBNEW) -> List[str]:
    """Run Design Rule Checks and return a list of error messages."""
    logging.debug("Running DRC")
    drc = pcbnew.DRC()
    drc.Run()
    errors = []
    for err in drc.GetErrors():
        errors.append(err.GetText())
    if errors:
        logging.warning(f"DRC found {len(errors)} errors")
    else:
        logging.info("DRC passed with no errors")
    return errors


def _export_gerbers(
    pcb: pcbnew.PCBNEW, output_dir: Path, gerber_options: Optional[dict] = None
) -> List[Path]:
    """Export Gerber files to the specified directory."""
    logging.debug(f"Exporting Gerbers to {output_dir}")
    output_dir.mkdir(parents=True, exist_ok=True)
    # Default Gerber options
    options = {
        "output_dir": output_dir.as_posix(),
        "gerber_format": pcbnew.GERBER_FORMAT_AUTO,
        "gerber_units": pcbnew.GERBER_UNITS_MM,
        "gerber_precision": 4,
        "gerber_omit_protection": False,
        "gerber_omit_footprint": False,
    }
    if gerber_options:
        options.update(gerber_options)

    exporter = pcbnew.GerberExporter()
    exporter.SetOptions(options)
    exporter.Export(pcb)

    # Collect exported files
    exported = list(output_dir.glob("*"))
    logging.info(f"Exported {len(exported)} Gerber files")
    return exported


def _generate_bom(pcb: pcbnew.PCBNEW, bom_path: Path) -> None:
    """Generate a BOM CSV file."""
    logging.debug(f"Generating BOM: {bom_path}")
    bom_path.parent.mkdir(parents=True, exist_ok=True)
    bom = pcbnew.BOM()
    bom.SetOutputFile(bom_path.as_posix())
    bom.Generate()
    logging.info(f"BOM written to {bom_path}")
