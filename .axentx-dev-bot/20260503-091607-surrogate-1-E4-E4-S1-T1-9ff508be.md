# SLA thresholds (configurable)

    return datetime.now(timezone.utc)


def _ensure_utc(dt: Optional[datetime]) -> Optional[datetime]:
    if dt is None:
        return None
    if dt.tzinfo is None:
        return dt.replace(tzinfo=timezone.utc)
    return dt.astimezone(timezone.utc)


def _resolution_timestamp(request: Dict[str, Any]) -> Optional[datetime]:
    """Return the first closed/resolved timestamp from status_history, or resolved_at/current status."""
    history: List[Dict[str, Any]] = request.get("status_history") or []
    for entry in history:
        status = (entry.get("status") or "").lower()
        if status in {"closed", "resolved", "done"}:
            ts = _ensure_utc(entry.get("timestamp"))
            if ts:
                return ts

    # Fallbacks
    resolved_at = _ensure_utc(request.get("resolved_at"))
    if resolved_at:
        return resolved_at

    status = (request.get("status") or "").lower()
