from . import db
from . import models

def get_sla_report(team_id):
    # Get team data
    team = db.get_team(team_id)
    if team is None:
        return None

    # Get request data
    requests = db.get_requests(team_id)
    sla_report = {}
    sla_report['open'] = len([request for request in requests if request.status == 'open'])
    sla_report['breached'] = len([request for request in requests if request.status == 'breached'])
    sla_report['at_risk'] = len([request for request in requests if request.status == 'at_risk'])
    sla_report['met_sla'] = len([request for request in requests if request.status == 'met_sla'])
    return sla_report

def get_sla_report_csv(team_id):
    # Get team data
    team = db.get_team(team_id)
    if team is None:
        return None

    # Get request data
    requests = db.get_requests(team_id)
    csv_data = []
    for request in requests:
        csv_data.append({
            'id': request.id,
            'status': request.status,
            'created_at': request.created_at,
            'updated_at': request.updated_at,
            'sla': request.sla
        })
    return '\n'.join([','.join([str(value) for value in row.values()]) for row in csv_data])
