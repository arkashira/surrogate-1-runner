  [key: string]: unknown;
};

/**
 * Minimal, deterministic example generator.
 * Replace body with model/ML inference as required.
 */
export const exampleSignalGenerator: SignalGenerator = (inputs) => {
  const { symbol, horizon } = inputs;

  if (!symbol || typeof symbol !== "string") {
    throw new Error("Invalid symbol");
  }
  if (!horizon || typeof horizon !== "string") {
    throw new Error("Invalid horizon");
  }

  const seed = hashString(symbol + horizon);
  const side: SignalOutput["side"] = seed % 2 === 0 ? "long" : "short";
  const price = 100 + (seed % 1000) / 10;
  const confidence = 0.5 + (seed % 50) / 100; // 0.50..0.99

  return {
    signal_id: `sig_${symbol}_${horizon}_${Date.now()}`,
    side,
    price,
    confidence,
  };
};

function hashString(s: string): number {
  let h = 0;
  for (let i = 0; i < s.length; i++) {
    h = Math.imul(h, 31) + s.charCodeAt(i);
  }
  return Math.abs(h);
}
