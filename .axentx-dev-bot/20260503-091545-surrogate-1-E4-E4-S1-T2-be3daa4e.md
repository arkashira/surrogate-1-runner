+ export class NotificationDedup {
+   private cache = new Map<string, number>();
+
+   shouldSend(watcherId: string, requestId: string, status: string, ttlMs: number = 3600000): boolean {
+     const key = `${watcherId}:${requestId}:${status}`;
+     const now = Date.now();
+     const last = this.cache.get(key);
+     if (last !== undefined && now - last < ttlMs) {
+       return false;
+     }
+     this.cache.set(key, now);
+     return true;
+   }
+
+   // For testing / cleanup (optional)

+     const now = Date.now();
+     for (const [key, ts] of this.cache.entries()) {
+       if (now - ts >= ttlMs) this.cache.delete(key);
+     }
+   }
+ }
