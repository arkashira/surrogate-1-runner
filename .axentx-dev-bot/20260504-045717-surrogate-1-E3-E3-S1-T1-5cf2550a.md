#!/usr/bin/env python3
"""
Metrics collector for surrogate-1 ingestion pipeline.
Provides Prometheus-compatible metrics for monitoring worker performance.
"""

import os
import json
import time
import logging
from datetime import datetime
from pathlib import Path
from typing import Dict, Any, Optional

logger = logging.getLogger(__name__)


class MetricsCollector:
    """Collects and exposes Prometheus metrics for ingestion workers."""
    
    def __init__(self, shard_id: int, worker_id: str):
        self.shard_id = shard_id
        self.worker_id = worker_id
        self._start_time: Optional[float] = None
        self._metrics: Dict[str, Any] = {
            'ingestion_duration_seconds': 0.0,
            'records_processed': 0,
            'records_succeeded': 0,
            'records_failed': 0,
            'errors': [],
            'start_time': None,
            'end_time': None,
        }
        # File paths for external collection
        self._metrics_file = Path(f"/tmp/ingest_metrics_{worker_id}.json")
        self._prom_file = Path(f"/tmp/ingest_metrics_{worker_id}.prom")
    
    def start(self):
        """Mark the start of ingestion with high-resolution timing."""
        self._start_time = time.time()
        self._metrics['start_time'] = datetime.utcnow().isoformat()
        logger.info(f"[{self.worker_id}] Ingestion started for shard {self.shard_id}")
    
    def finish(self):
        """Mark the end of ingestion and calculate duration."""
        self._metrics['end_time'] = datetime.utcnow().isoformat()
        if self._start_time:
            self._metrics['ingestion_duration_seconds'] = time.time() - self._start_time
        logger.info(
            f"[{self.worker_id}] Ingestion finished in "
            f"{self._metrics['ingestion_duration_seconds']:.2f}s"
        )
    
    def record_success(self, record_count: int = 1):
        """Record successful record processing."""
        self._metrics['records_processed'] += record_count
        self._metrics['records_succeeded'] += record_count
    
    def record_failure(self, record_count: int = 1, error: str = ""):
