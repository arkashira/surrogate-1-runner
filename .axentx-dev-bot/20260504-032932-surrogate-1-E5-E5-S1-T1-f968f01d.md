"""
Investor vetting service for compliance checking.

This module provides automatic vetting of investor profiles when uploaded,
flags non-compliant investors in the database, and notifies admins of flagged profiles.
"""

from __future__ import annotations

import datetime
import logging
from enum import Enum
from typing import Any, Dict, List, Optional

# Use existing surrogate-1 infrastructure (from Candidate 1)
try:
    from src.db import InvestorProfile, db_session
    from src.notifications import send_email
except Exception as exc:
    raise RuntimeError(
        "Required surrogate-1 modules could not be imported. "
        "Ensure that the project dependencies are installed."
    ) from exc

logger = logging.getLogger(__name__)


class FlagReason(Enum):
    """Reasons for flagging an investor (from Candidate 2's improved enum system)."""
    SANCTIONED_COUNTRY = "sanctioned_country"
    MISSING_REQUIRED_FIELD = "missing_required_field"
    INVALID_INCOME = "invalid_income"
    UNKNOWN = "unknown"


# Configuration - centralized for maintainability
SANCTIONED_COUNTRIES: set = {"North Korea", "Iran", "Syria", "Cuba"}
ADMIN_EMAIL: str = "admin@example.com"  # Could be loaded from config


def _is_compliant(profile_data: Dict[str, Any]) -> tuple[bool, List[FlagReason]]:
    """
    Perform compliance check on investor profile data.

    Args:
        profile_data: Raw profile data from ingestion pipeline.

    Returns:
        Tuple of (is_compliant, list_of_violations)
    """
    violations: List[FlagReason] = []

    # Check country is not sanctioned
    country = profile_data.get("country")
    if not country:
        violations.append(FlagReason.MISSING_REQUIRED_FIELD)
    elif country in SANCTIONED_COUNTRIES:
        violations.append(FlagReason.SANCTIONED_COUNTRY)

    # Check annual income is positive
    income = profile_data.get("annual_income")
    if income is None:
        violations.append(FlagReason.MISSING_REQUIRED_FIELD)
    elif not isinstance(income, (int, float)) or income <= 0:
        violations.append(FlagReason.INVALID_INCOME)

    return (len(violations) == 0, violations)


def _flag_profile(profile: InvestorProfile, reason: FlagReason, details: str) -> None:
    """
    Mark a profile as non-compliant in the database.

    Args:
        profile: The ORM instance representing the investor profile.
        reason: Enum reason for flagging.
        details: Human-readable description of the violation.
    """
    profile.is_compliant = False
    profile.flagged_at = datetime.datetime.utcnow()
    profile.flag_reason = reason.value
    profile.flag_details = details
    db_session.commit()
    logger.info(f"Flagged investor {profile.id}: {reason.value} - {details}")


def _notify_admin(profile: InvestorProfile) -> bool:
    """
    Send an email notification to the system administrator.

    Args:
        profile: The flagged investor profile.

    Returns:
        True if notification sent successfully.
    """
    subject = f"[Surrogate-1] Investor {profile.id} flagged as non-compliant"
    body = (
        f"Investor ID: {profile.id}\n"
        f"Name: {profile.name}\n"
        f"Country: {profile.raw_data.get('country', 'N/A')}\n"
        f"Reason: {profile.flag_reason}\n"
        f"Details: {profile.flag_details}\n"
        f"Timestamp: {profile.flagged_at.isoformat()} UTC\n"
    )
    
    try:
        send_email(to=ADMIN_EMAIL, subject=subject, body=body)
        logger.info(f"Admin notification sent for investor {profile.id}")
        return True
