import os
import docx
from typing import List

class DocxIngestor:
    def __init__(self, output_dir: str):
        self.output_dir = output_dir
        os.makedirs(self.output_dir, exist_ok=True)

    def ingest(self, file_path: str) -> str:
        doc = docx.Document(file_path)
        text = "\n".join([paragraph.text for paragraph in doc.paragraphs])
        output_path = os.path.join(self.output_dir, os.path.basename(file_path) + ".txt")
        with open(output_path, "w") as f:
            f.write(text)
        return output_path

describe('docx_ingest', () => {
  const ingest = require('./docx_ingest');
  const fs = require('fs');
  const path = require('path');

  const testDir = '/tmp/ingest_test';
  const targetDir = '/tmp/ingest_target';

  beforeAll(() => {
    fs.mkdirSync(testDir, { recursive: true });
    fs.mkdirSync(targetDir, { recursive: true });
  });

  afterAll(() => {
    fs.rmdirSync(testDir, { recursive: true });
    fs.rmdirSync(targetDir, { recursive: true });
  });

  test('ingests a valid DOCX file', async () => {
    const src = path.join(testDir, 'sample.docx');
    fs.writeFileSync(src, Buffer.from('dummy content'));
    const result = await ingest(src, targetDir);
    expect(result.format).toBe('docx');
    expect(fs.existsSync(path.join(targetDir, 'sample.docx'))).toBe(true);
  });

  test('returns metadata with size and timestamp', async () => {
    const src = path.join(testDir, 'sample2.docx');
    fs.writeFileSync(src, Buffer.from('more content'));
    const meta = await ingest(src, targetDir);
    expect(meta.size_bytes).toBeGreaterThan(0);
    expect(new Date(meta.ingested_at).toString()).not.toBe('Invalid Date');
  });

  test('throws IngestionError for unsupported extension', async () => {
    const src = path.join(testDir, 'image.png');
    fs.writeFileSync(src, Buffer.from('png data'));
    await expect(ingest(src, targetDir)).rejects.toThrow('IngestionError');
  });

  test('does not overwrite existing file unless overwrite=true', async () => {
    const src = path.join(testDir, 'dup.docx');
    fs.writeFileSync(src, Buffer.from('first'));
    await ingest(src, targetDir);
    fs.writeFileSync(src, Buffer.from('second'));
    await expect(ingest(src, targetDir)).rejects.toThrow('FileAlreadyExists');
    await ingest(src, targetDir, { overwrite: true });
    const content = fs.readFileSync(path.join(targetDir, 'dup.docx'));
    expect(content.toString()).toBe('second');
  });

  test('handles missing file gracefully', async () => {
    const src = path.join(testDir, 'nonexistent.docx');
    await expect(ingest(src, targetDir)).rejects.toThrow('IngestionError');
  });
});
