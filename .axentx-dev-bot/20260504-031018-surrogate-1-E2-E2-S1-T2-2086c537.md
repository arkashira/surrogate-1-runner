```python
import csv
import json
from typing import List, Dict

class ExportService:
    def __init__(self, audit_logs: List[Dict]):
        self.audit_logs = audit_logs

    def export_to_json(self, filename: str):
        """Exports the audit logs to a JSON file."""
        with open(filename, 'w') as f:
            json.dump(self.audit_logs, f, indent=4, default=str)

    def export_to_csv(self, filename: str):
        """Exports the audit logs to a CSV file."""
        if not self.audit_logs:
            return

        keys = self.audit_logs[0].keys()
        with open(filename, 'w', newline='') as f:
            dict_writer = csv.DictWriter(f, keys)
            dict_writer.writeheader()
            dict_writer.writerows(self.audit_logs)

    def export_audit_trail(self, format_type: str, filename: str):
        """
        Routes the export to the correct format handler.
        :param format_type: 'json' or 'csv'
        :param filename: Target file path
        """
        if format_type == 'json':
            self.export_to_json(filename)
        elif format_type == 'csv':
            self.export_to_csv(filename)
        else:
            raise ValueError("Unsupported format type. Use 'json' or 'csv'.")
```
