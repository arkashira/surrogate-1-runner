import { useEffect, useState } from 'react';
import { queueDocumentForOfflineCaching, storeDocumentLocally } from '../offline/registerSW';

const Reader = ({ docId, docContent }) => {
  const [localDocContent, setLocalDocContent] = useState('');

  useEffect(() => {
    // Queue document for offline caching

    // Store a readable copy locally

    // Sync changes when connectivity returns
    const syncChanges = async () => {

      }
    };

    syncChanges();
  }, [docId, docContent]);

  return (
    <div>
      <h1>Document {docId}</h1>
      <p>{localDocContent}</p>

};
