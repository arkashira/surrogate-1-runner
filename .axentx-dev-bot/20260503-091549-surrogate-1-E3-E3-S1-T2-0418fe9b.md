File: `/opt/axentx/surrogate-1/src/routes/requests.ts`

```ts
import { Router, Request, Response, NextFunction } from 'express';
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();
const router = Router();

/**
 * GET /requests/:id/audit
 * Query params:
 *   page     - number (default 1)
 *   perPage  - number (default 50, max 100)
 *
 * Returns audit entries sorted newest-first.
 * Audit entries are immutable (no PUT/PATCH/DELETE endpoints).
 */
router.get('/:id/audit', async (req: Request, res: Response, next: NextFunction) => {
  try {
    const requestId = req.params.id;
    const page = Math.max(1, parseInt(String(req.query.page || '1'), 10));
    const perPage = Math.min(100, Math.max(1, parseInt(String(req.query.perPage || '50'), 10)));
    const skip = (page - 1) * perPage;

    // Validate request exists
    const requestExists = await prisma.request.count({
      where: { id: requestId },
    });

    if (requestExists === 0) {
      return res.status(404).json({ error: 'Request not found' });
    }

    const [entries, total] = await Promise.all([
      prisma.requestAudit.findMany({
        where: { requestId },
        select: {
          id: true,
          timestamp: true,
          actor: true,
          previousStatus: true,
          newStatus: true,
          comment: true,
        },
        orderBy: { timestamp: 'desc' },
        skip,
        take: perPage,
      }),
      prisma.requestAudit.count({ where: { requestId } }),
    ]);

    const totalPages = Math.ceil(total / perPage);

    return res.json({
      data: entries,
      pagination: {
        page,
        perPage,
        total,
        totalPages,
        hasNext: page < totalPages,
        hasPrev: page > 1,
      },
    });
  } catch (err) {
    next(err);
  }
});

export default router;
```

### 3) Record audit on status change (recommended helper)

```ts
async function updateRequestStatus(
  requestId: string,
  newStatus: string,
  actor: string,
  comment?: string
) {
  const request = await prisma.request.findUniqueOrThrow({
    where: { id: requestId },
    select: { status: true },
  });

  const previousStatus = request.status;

  await prisma.$transaction([
    prisma.request.update({
      where: { id: requestId },
      data: { status: newStatus },
    }),
    prisma.requestAudit.create({
      data: {
        requestId,
        actor,
        previousStatus,
        newStatus,
        comment: comment || null,
      },
    }),
  ]);
}
```
