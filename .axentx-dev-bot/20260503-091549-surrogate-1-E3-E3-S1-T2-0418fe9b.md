import { FastifyInstance, FastifyRequest, FastifyReply } from 'fastify';

// Immutable audit model (append-only)
interface AuditEntry {
  id: string;
  requestId: string;
  timestamp: string; // ISO 8601
  actor: string;     // user ID or system actor
  previousStatus?: string;
  newStatus?: string;
  field?: string;
  previousValue?: unknown;
  newValue?: unknown;
  comment?: string;
}

// Ephemeral store (swap with DB in follow-up)
const auditStore: AuditEntry[] = [];

function generateId(): string {
  return Date.now().toString(36) + Math.random().toString(36).slice(2, 7);
}

// Idempotent sample seeder for manual testing
function seedSampleAuditEntries(requestId: string): void {
  if (auditStore.some((e) => e.requestId === requestId)) return;

  const now = new Date().toISOString();
  auditStore.push(
    {
      id: generateId(),
      requestId,
      timestamp: new Date(Date.now() - 400_000).toISOString(),

      previousStatus: undefined,
      newStatus: 'created',
      comment: 'Request created via ingestion pipeline',
    },
    {

    {
      id: generateId(),
      requestId,
      timestamp: now,
      actor: 'system:worker-shard-3',
      previousStatus: 'processing',
      newStatus: 'completed',
      comment: 'Ingestion completed successfully',
    }
  );
}

export default async function requestRoutes(fastify: FastifyInstance): Promise<void> {
  // GET /requests/:id/audit
  fastify.get<{
    Params: { id: string };
    Querystring: { page?: string; perPage?: string };
  }>('/requests/:id/audit', async (request, reply): Promise<void> => {
    const { id } = request.params;
    const page = Math.max(1, parseInt(request.query.page || '1', 10));
    const perPage = Math.min(100, Math.max(1, parseInt(request.query.perPage || '50', 10)));

    // Seed sample data idempotently for manual testing
    seedSampleAuditEntries(id);

    const entries = auditStore
      .filter((e) => e.requestId === id)
      .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());

    const total = entries.length;
    const start = (page - 1) * perPage;
    const end = start + perPage;
    const items = entries.slice(start, end);

    void reply.status(200).send({
      requestId: id,
      page,
      perPage,
      total,
      totalPages: Math.ceil(total / perPage),
      items,
    });
  });

  // Audit entries are immutable.
  // Future internal services should append via domain events or a non-exposed writer.
}
