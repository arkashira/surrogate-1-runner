## 1) Create audit store (append-only, immutable)

File: `/opt/axentx/surrogate-1/src/audit/audit-store.ts`

```ts
import fs from 'fs';
import path from 'path';
import { randomUUID } from 'crypto';

const AUDIT_LOG = path.join(process.cwd(), 'audit', 'audit.log.jsonl');

export interface AuditEntry {
  id: string;
  requestId: string;
  timestamp: string;
  actor: string;
  previousStatus?: string;
  newStatus: string;
  field?: string;
  previousValue?: any;
  newValue?: any;
  comment?: string;
}

export class AuditStore {
  private ensureLogFile() {
    const dir = path.dirname(AUDIT_LOG);
    if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
    if (!fs.existsSync(AUDIT_LOG)) fs.writeFileSync(AUDIT_LOG, '');
  }

  append(entry: Omit<AuditEntry, 'id'>): AuditEntry {
    this.ensureLogFile();
    const id = randomUUID();
    const full: AuditEntry = { id, timestamp: new Date().toISOString(), ...entry };
    fs.appendFileSync(AUDIT_LOG, JSON.stringify(full) + '\n');
    return full;
  }

  getByRequestId(requestId: string, page = 1, perPage = 50) {
    this.ensureLogFile();
    const raw = fs.readFileSync(AUDIT_LOG, 'utf-8').trim();
    const lines = raw ? raw.split('\n').filter(Boolean) : [];
    const entries = lines
      .map((l) => JSON.parse(l) as AuditEntry)
      .filter((e) => e.requestId === requestId)
      .sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());

    const total = entries.length;
    const start = (page - 1) * perPage;
    const end = start + perPage;
    return {
      entries: entries.slice(start, end),
      page,
      perPage,
      total,
    };
  }
}

export const auditStore = new AuditStore();
```
