from __future__ import annotations

import numpy as np
from typing import List
from sentence_transformers import SentenceTransformer

_model: SentenceTransformer | None = None

def get_model() -> SentenceTransformer:
    global _model
    if _model is None:
        _model = SentenceTransformer("all-MiniLM-L6-v2")
    return _model

def encode_chunks(texts: List[str], batch_size: int = 64) -> np.ndarray:
    """
    Encode texts to normalized embeddings (float32).
    Returns array shape (len(texts), 384).
    """
    if not texts:
        return np.empty((0, 384), dtype=np.float32)
    model = get_model()
    embs = model.encode(
        texts,
        batch_size=batch_size,
        show_progress_bar=False,
        convert_to_numpy=True,
        normalize_embeddings=True,
    )
    return embs.astype(np.float32)

from __future__ import annotations

import numpy as np
from dataclasses import dataclass, asdict
from typing import List, Tuple, Optional
from ..embeddings.encoder import encode_chunks

@dataclass
class Chunk:
    id: str
    doc_title: str
    page_number: int
    text: str
    embedding: Optional[np.ndarray] = None  # normalized float32 vector

    def to_dict(self):
        d = asdict(self)
        if self.embedding is not None:
            d["embedding"] = self.embedding.tobytes()
        return d

    @staticmethod
    def from_dict(d):
        if "embedding" in d and isinstance(d["embedding"], bytes):
            d["embedding"] = np.frombuffer(d["embedding"], dtype=np.float32)
        return Chunk(**d)

class VectorStore:
    """
    In-memory vector store for fast search.
    Stores normalized embeddings; search uses dot product (cosine similarity).
    """

    def __init__(self):
        self.chunks: List[Chunk] = []
        self.embeddings: np.ndarray | None = None  # (N, D) float32 normalized

    def add_chunks(self, chunks: List[Chunk]) -> None:
        if not chunks:
            return
        texts = [c.text for c in chunks]
        embs = encode_chunks(texts)
        for c, e in zip(chunks, embs):
            c.embedding = e
        self.chunks.extend(chunks)
        if self.embeddings is None:
            self.embeddings = embs
        else:
            self.embeddings = np.vstack([self.embeddings, embs])

    def search(self, query: str, k: int = 10) -> List[Tuple[Chunk, float]]:
        if self.embeddings is None or len(self.chunks) == 0:
            return []
        q_emb = encode_chunks([query])  # (1, D) normalized
        scores = q_emb @ self.embeddings.T  # (1, N)
        scores = scores[0]
        k = min(k, len(scores))
        # argpartition + sort top-k for speed
        topk_idx = np.argpartition(-scores, k - 1)[:k]
        topk_idx = topk_idx[np.argsort(-scores[topk_idx])]
        results = [(self.chunks[i], float(scores[i])) for i in topk_idx]
        return results
```
