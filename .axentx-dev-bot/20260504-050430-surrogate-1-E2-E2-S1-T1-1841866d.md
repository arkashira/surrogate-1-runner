P&L Engine – One‑Stop Solution

* Reads all CSV files in a directory (non‑recursive).

# --------------------------------------------------------------------------- #
def load_schema(schema_path: Path) -> Dict[str, str]:
    """Load and validate a finance schema JSON file.

    The schema must contain exactly the keys:
        - date_column
        - revenue_column
        - expense_column
    """
    if not schema_path.is_file():
        raise FileNotFoundError(f"Schema file not found: {schema_path}")

    schema = json.loads(schema_path.read_text(encoding="utf-8"))

    required = {"date_column", "revenue_column", "expense_column"}
    missing = required - schema.keys()
    if missing:
        raise ValueError(f"Schema missing required keys: {missing}")

    return schema

# --------------------------------------------------------------------------- #

    return sorted(raw_dir.glob("*.csv"))

    """Read all CSVs as strings and concatenate into a single DataFrame."""
    if not csv_paths:
        return pd.DataFrame()

    dfs = [pd.read_csv(p, dtype=str) for p in csv_paths]
    return pd.concat(dfs, ignore_index=True)

    """Convert a Series to numeric, coercing errors to 0.0."""
    return pd.to_numeric(series, errors="coerce").fillna(0.0)


def process_dataframe(df: pd.DataFrame, schema: Dict[str, str]) -> pd.DataFrame:
    """Validate columns, parse dates, and coerce revenue/expense to numeric."""
    date_col, rev_col, exp_col = (
        schema["date_column"],
        schema["revenue_column"],
        schema["expense_column"],

    # Ensure required columns exist

        if col not in df.columns:
            raise KeyError(f"Column '{col}' not found in input data.")

    # Coerce numeric values

    df["expense"] = _coerce_numeric(df[exp_col])
