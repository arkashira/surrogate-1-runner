def list_files_via_tree() -> list[str]:
    """Single API call: list date folder contents (non-recursive)."""
    api = HfApi(token=HF_TOKEN)
    items = api.list_repo_tree(
        repo_id=REPO,
        path=DATE,
        repo_type="dataset",
        recursive=False,
    )
    files = []
    for it in items:
        if it.type == "file":
            files.append(it.path)
        else:
            sub = api.list_repo_tree(
                repo_id=REPO,
                path=it.path,
                repo_type="dataset",
                recursive=False,
            )
            for s in sub:

    return files
