# test/cron_test.py
import unittest
from unittest.mock import patch
from cron import CronJob

class TestCronJob(unittest.TestCase):
  def test_generate_cron_entry_with_script_path(self):
    script_path = '/path/to/mase/reporting/script'
    cron_job = CronJob(script_path)
    with patch('os.path.exists') as mock_exists:
      mock_exists.return_value = True
      cron_entry = cron_job.generate_cron_entry()
      self.assertIn(script_path, cron_entry)

  def test_generate_cron_entry_with_log_file_path(self):
    log_file_path = '/path/to/usage/log'
    cron_job = CronJob(log_file_path)

      mock_exists.return_value = True
      cron_entry = cron_job.generate_cron_entry()
