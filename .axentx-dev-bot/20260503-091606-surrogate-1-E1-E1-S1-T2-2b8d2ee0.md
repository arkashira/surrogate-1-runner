### 2) lib/sla.py

```python
from datetime import datetime, timedelta
import os
import yaml

DEFAULT_SLA_PATH = os.path.join(os.path.dirname(__file__), "..", "config", "sla_defaults.yaml")


def calculate_sla_target(base_dt: datetime, sla_rule: dict) -> datetime:
    """
    Compute SLA target datetime from a base datetime and an SLA rule.
    Supported rule types:
      - fixed:     add rule['value'] minutes
      - percentage: add (rule['value'] / 100) * 60 minutes
    """
    if not isinstance(sla_rule, dict):
        raise ValueError("SLA rule must be a dict")
    rule_type = sla_rule.get("type")
    value = sla_rule.get("value")

    if rule_type not in ("fixed", "percentage") or not isinstance(value, (int, float)):
        raise ValueError("Invalid SLA rule")

    if rule_type == "fixed":
        return base_dt + timedelta(minutes=float(value))
    else:  # percentage
        # Interpret as percentage of one hour (60 minutes)
        return base_dt + timedelta(minutes=(float(value) / 100.0) * 60.0)


def load_default_sla_rules(path: str = DEFAULT_SLA_PATH):
    """
    Load default SLA rules from YAML.
    Returns list of rule dicts.
    """
    try:
        with open(path, "r") as f:
            cfg = yaml.safe_load(f) or {}
        rules = cfg.get("sla_defaults")
        if not isinstance(rules, list) or not rules:
            raise ValueError("Invalid or missing sla_defaults list")
        return rules
    except Exception as exc:
        # Fail-safe defaults
        return [{"type": "fixed", "value": 30}]
```
