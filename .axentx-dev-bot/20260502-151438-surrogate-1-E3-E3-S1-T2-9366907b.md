export class InterpolationError extends Error {
  constructor(message: string, public path?: string) {
    super(message);
    this.name = 'InterpolationError';
  }
}

/**
 * Resolve dot-path against context.
 * Example: resolvePath({ prev: { price: 100 } }, 'prev.price') => 100
 */
function resolvePath(obj: any, path: string): any {
  if (!path || typeof path !== 'string') return undefined;
  const parts = path.split('.');
  let cur = obj;
  for (const part of parts) {
    if (cur == null || typeof cur !== 'object') return undefined;
    cur = cur[part];
  }
  return cur;
}

/**
 * Recursively interpolate {{path}} tokens in strings using context.
 * Non-string values are returned as-is.
 */
export function interpolate(value: unknown, context: Record<string, unknown>): unknown {
  if (typeof value === 'string') {
    // Match {{...}} tokens, non-greedy
    return value.replace(/\{\{([^{}]+?)\}\}/g, (match, token) => {
      const path = token.trim();
      const resolved = resolvePath(context, path);
      if (resolved === undefined) {
        throw new InterpolationError(`Cannot resolve interpolation token: ${token}`, path);
      }
      // Preserve non-strings as JSON-embedded values (numbers/objects/booleans)
      return typeof resolved === 'string' ? resolved : JSON.stringify(resolved);
    });
  }

  if (Array.isArray(value)) {
    return value.map((v) => interpolate(v, context));
  }

  if (value && typeof value === 'object') {
    const result: Record<string, unknown> = {};
    for (const [k, v] of Object.entries(value)) {
      result[k] = interpolate(v, context);
    }
    return result;
  }

  return value;
}
