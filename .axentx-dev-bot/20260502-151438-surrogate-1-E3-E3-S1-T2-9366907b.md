export class InterpolationError extends Error {
  constructor(
    public readonly token: string,
    public readonly path: string,
    message: string
  ) {
    super(message);
    this.name = 'InterpolationError';
  }
}

function getByPath(obj: any, path: string): any {
  if (!path || typeof path !== 'string') return undefined;
  const parts = path.split('.');
  let cur = obj;
  for (const part of parts) {
    if (cur == null) return undefined;
    cur = cur[part];
  }
  return cur;
}

function replaceTokens(value: string, context: Record<string, any>): string {
  return value.replace(/\{\{([^{}]+)\}\}/g, (_, tokenPath) => {
    const path = tokenPath.trim();
    const resolved = getByPath(context, path);
    if (resolved === undefined) {
      throw new InterpolationError(tokenPath, path, `Missing interpolation value for "{{${path}}}"`);
    }
    return String(resolved);
  });
}

export function interpolateStageInputs(inputs: any, context: Record<string, any>): any {
  if (inputs == null) return inputs;

  // Primitive: string -> interpolate tokens
  if (typeof inputs === 'string') {
    return replaceTokens(inputs, context);
  }

  // Number / boolean / null: pass through
  if (typeof inputs !== 'object') {
    return inputs;
  }

  // Array: recurse
  if (Array.isArray(inputs)) {
    return inputs.map((item) => interpolateStageInputs(item, context));
  }

  // Object: recurse over values
  const result: Record<string, any> = {};
  for (const [key, value] of Object.entries(inputs)) {
    result[key] = interpolateStageInputs(value, context);
  }
  return result;
}
