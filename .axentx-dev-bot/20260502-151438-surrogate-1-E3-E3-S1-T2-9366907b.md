import { invalidRef, missingRef, missingField, nullRef } from "./errors";

export type InterpolationContext = Record<string, Record<string, unknown>>;

const VARIABLE_RE = /\{\{([^{}]+)\}\}/g;

function resolveRef(ref: string, context: InterpolationContext) {
  const parts = ref.trim().split(".");
  if (parts.length !== 2) throw invalidRef(ref);

  const [stageKey, field] = parts;
  const source = context[stageKey];
  if (source == null) throw missingRef(stageKey, field);
  if (!(field in source)) throw missingField(stageKey, field);

  const value = source[field];
  if (value == null) throw nullRef(stageKey, field);
  return value;
}

export function interpolateValue(value: unknown, context: InterpolationContext): unknown {
  if (typeof value !== "string") return value;

  return value.replace(VARIABLE_RE, (_, ref) => {
    const resolved = resolveRef(ref, context);
    // Coerce to string for template replacement; preserve numbers/booleans in structured inputs via JSON parse if needed downstream.
    return String(resolved);
  });
}

export function interpolateStageInputs(
  inputs: Record<string, unknown>,
  context: InterpolationContext
): Record<string, unknown> {
  const result: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(inputs)) {
    result[key] = interpolateValue(value, context);
  }
  return result;
}
