import { InterpolationError } from './errors';

export class Interpolator {
  static interpolate(value: any, context: Record<string, any>): any {
    if (typeof value === 'string') {
      return value.replace(/\{\{([^{}]+)\}\}/g, (_, token) => {
        const [stageName, ...fieldPath] = token.trim().split('.');
        const stageOut = context[stageName];
        if (stageOut === undefined) {
          throw new InterpolationError(
            'UNRESOLVED_STAGE',
            `Referenced stage "${stageName}" not found in context`
          );
        }
        const resolved = fieldPath.reduce((acc, key) => {
          if (acc == null || typeof acc !== 'object' || !(key in acc)) {
            throw new InterpolationError(
              'MISSING_FIELD',
              `Field "${token}" not found in stage "${stageName}" output`
            );
          }
          return acc[key];
        }, stageOut);
        // Preserve original type; only replace the token with the value
        return resolved;
      });
    }

    if (Array.isArray(value)) {
      return value.map((v) => this.interpolate(v, context));
    }

    if (value && typeof value === 'object') {
      const out: Record<string, any> = {};
      for (const [k, v] of Object.entries(value)) {
        out[k] = this.interpolate(v, context);
      }
      return out;
    }

    return value;
  }
}
