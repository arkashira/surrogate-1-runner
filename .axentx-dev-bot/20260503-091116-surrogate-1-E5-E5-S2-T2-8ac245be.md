        setMonthlyActiveUsers(data.monthlyActiveUsers);
      } catch (error) {
        console.error(error);
      }
    };
    fetchAnalyticsData();
  }, []);

    <div>
      <h2>Daily Active Users: {dailyActiveUsers}</h2>

    </div>
  );
};

    # Calculate daily and monthly active users
    today = datetime.today().date()
    daily_active_users = data[data['date'] == today].shape[0]
