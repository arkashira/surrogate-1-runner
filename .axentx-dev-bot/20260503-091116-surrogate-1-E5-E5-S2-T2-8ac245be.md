import pandas as pd
from datetime import datetime, timedelta

def get_analytics_data():
    # Fetch data from database or storage
    data = pd.read_csv('analytics_data.csv')

    # Calculate daily and monthly active users
    today = datetime.today().date()
    daily_active_users = data[data['date'] == today].shape[0]
    monthly_active_users = data[data['date'] >= today - timedelta(days=30)].shape[0]

    return {
        'dailyActiveUsers': daily_active_users,
        'monthlyActiveUsers': monthly_active_users,
    }
