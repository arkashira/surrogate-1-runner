Base = declarative_base()

class Request(Base):
    __tablename__ = "requests"
    id = Column(Integer, primary_key=True, index=True)
    type = Column(String)

    owner = Column(String)
    initial_status = Column(Enum("new", "in_progress", "done"))
    created_at = Column(DateTime)
    updated_at = Column(DateTime)

class RequestCreate(BaseModel):
    type: str
    title: str
    owner: str
    initial_status: str

class RequestUpdate(BaseModel):
    type: Optional[str]
    title: Optional[str]
    owner: Optional[str]
    initial_status: Optional[str]

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

def create_request(db: SessionLocal, request: RequestCreate):
    db_request = Request(
        type=request.type,
        title=request.title,
        owner=request.owner,
        initial_status=request.initial_status,
        created_at=datetime.now(),
        updated_at=datetime.now(),
    )
    db.add(db_request)
    db.commit()
    db.refresh(db_request)
    return db_request

def get_requests(db: SessionLocal, team: Optional[str] = None, status: Optional[str] = None):
    if team:
        return db.query(Request).filter(Request.team == team).all()
    if status:
        return db.query(Request).filter(Request.status == status).all()
    return db.query(Request).all()

def validate_request(request: RequestCreate):
    required_fields = ['type', 'title', 'owner', 'initial_status']
    if not all(field in request for field in required_fields):
