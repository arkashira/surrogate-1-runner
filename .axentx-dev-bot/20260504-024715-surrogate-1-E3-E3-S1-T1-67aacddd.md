export interface Tier {
  name: string;
  price: number;          // monthly price in USD
  features?: string[];
  conversionRate: number; // 0‑1 probability that a user upgrades to this tier
  monthlyChurnRate: number; // 0‑1 probability that a user leaves this tier each month
}

export interface PricingScenarioJSON {
  id: string;
  name: string;
  tiers: Tier[];
  baseUsers: number;
  growthRate: number;      // e.g. 0.05 = 5 % monthly growth
  projectionMonths: number;
  createdAt: string;       // ISO string
}

export class PricingScenario {
  public readonly id: string;
  public readonly name: string;
  public readonly tiers: Tier[];
  public readonly baseUsers: number;
  public readonly growthRate: number;
  public readonly projectionMonths: number;
  public readonly createdAt: Date;

  constructor({
    id,
    name,
    tiers = [],
    baseUsers = 0,
    growthRate = 0,
    projectionMonths = 12,
    createdAt,
  }: Partial<PricingScenarioJSON> & { name: string }) {
    this.id = id ?? PricingScenario.generateId();
    this.name = name;
    this.tiers = tiers.map(t => PricingScenario.validateTier(t));
    this.baseUsers = baseUsers;
    this.growthRate = growthRate;
    this.projectionMonths = projectionMonths;
    this.createdAt = createdAt ? new Date(createdAt) : new Date();
  }

  /* ------------------------------------------------------------------ */

    return `scenario_${Date.now()}_${Math.random().toString(36).slice(2, 11)}`;

    if (!tier.name) throw new Error('Tier must have a name');

    expect(calculateRevenue(scenario)).toBe(0);
  });

  test('calculates correct revenue for normal values', () => {
    const scenario = new PricingCalculator({ name: 'Basic', price: 19.99, projectedUnits: 250 });
    // 19.99 * 250 = 4,997.5
    expect(calculateRevenue(scenario)).toBeCloseTo(4997.5, 2);
  });

  test('handles large numbers without overflow', () => {
    const scenario = new PricingScenario({ name: 'Enterprise', price: 9999.99, projectedUnits: 1_000_000 });
    expect(calculateRevenue(scenario)).toBeCloseTo(9_999_990_000, -2);
  });

  test('throws ValidationError for negative price', () => {
    const scenario = new PricingScenario({ name: 'Bad', price: -10, projectedUnits: 100 });
    expect(() => calculateRevenue(scenario)).toThrow('Price must be non‑negative');
  });

  test('throws ValidationError for negative units', () => {
    const scenario = new PricingScenario({ name: 'Bad', price: 10, projectedUnits: -5 });
    expect(() => calculateRevenue(scenario)).toThrow('Projected units must be non‑negative');
