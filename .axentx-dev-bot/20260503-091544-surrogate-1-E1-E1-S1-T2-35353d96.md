from datetime import datetime
from typing import List, Optional
from uuid import uuid4

# Lightweight in-memory store for demo; replace with DB in production.
_requests_db = []

_VALID_TYPES = {"incident", "task", "bug", "feature"}
_VALID_SLA_TARGETS = {"p0", "p1", "p2", "p3", None}


class Request:
    def __init__(
        self,
        id: str,
        team_id: str,
        type: str,
        title: str,
        owner: str,
        sla_target: Optional[str],
        status: str,
        created_at: datetime,
        public_url: str,
    ):
        self.id = id
        self.team_id = team_id
        self.type = type
        self.title = title
        self.owner = owner
        self.sla_target = sla_target
        self.status = status
        self.created_at = created_at
        self.public_url = public_url

    def to_dict(self):
        return {
            "id": self.id,
            "team_id": self.team_id,
            "type": self.type,
            "title": self.title,
            "owner": self.owner,
            "sla_target": self.sla_target,
            "status": self.status,
            "created_at": self.created_at.isoformat(),
            "public_url": self.public_url,
        }


def _validate_sla_target(sla_target: Optional[str]) -> bool:
    return sla_target in _VALID_SLA_TARGETS


def create_request(
    team_id: str,
    type: str,
    title: str,
    owner: str,
    sla_target: Optional[str],
) -> Request:
    # Validation
    if not team_id or not isinstance(team_id, str):
        raise ValueError("team_id is required")
    if not type or not isinstance(type, str) or type not in _VALID_TYPES:
        raise ValueError(f"type is required and must be one of {_VALID_TYPES}")
    if not title or not isinstance(title, str):
        raise ValueError("title is required")
    if not owner or not isinstance(owner, str):
        raise ValueError("owner is required")
    if not _validate_sla_target(sla_target):
        raise ValueError(f"sla_target must be one of {_VALID_SLA_TARGETS}")

    req_id = str(uuid4())
    created_at = datetime.utcnow()
    public_url = f"https://platform.example.com/requests/{req_id}"
    status = "open"

    req = Request(
        id=req_id,
        team_id=team_id,
        type=type,
        title=title,
        owner=owner,
        sla_target=sla_target,
        status=status,
        created_at=created_at,
        public_url=public_url,
    )
    _requests_db.append(req)
    return req


def list_requests_by_team(team_id: str) -> List[Request]:
    if not team_id or not isinstance(team_id, str):
        raise ValueError("team_id is required")
    team_requests = [r for r in _requests_db if r.team_id == team_id]
    # Default ordering: created_at desc
    team_requests.sort(key=lambda r: r.created_at, reverse=True)
    return team_requests
