# /opt/axentx/surrogate-1/src/report.py
"""
Report generation utilities for surrogate‑1.

The public API consists of three classes:

* :class:`Issue` – a single problem discovered during analysis.
* :class:`ReportSection` – an optional logical grouping of issues.
* :class:`Report` – the top‑level container that can hold a flat list of
  issues *or* a collection of sections (but not both at the same time).

The rendering format is plain‑text with markdown‑style headings, which
works nicely in terminals, CI logs, or simple text files.
"""

from __future__ import annotations

from dataclasses import dataclass, field
from typing import Any, List, Mapping, Dict, Sequence, Iterable, Optional


# --------------------------------------------------------------------------- #
#  Issue – the atomic building block
# --------------------------------------------------------------------------- #
@dataclass
class Issue:
    """A single issue discovered during processing."""

    title: str
    description: str
    location: Optional[str] = None
    metadata: Mapping[str, Any] = field(default_factory=dict)

    # ------------------------------------------------------------------- #
    #  Rendering helpers
    # ------------------------------------------------------------------- #
    def _format_metadata(self) -> str:
        """Return a compact ``key=value`` list or an empty string."""
        if not self.metadata:
            return ""
        return ", ".join(f"{k}={v}" for k, v in self.metadata.items())

    def format(self, idx: Optional[int] = None) -> str:
        """
        Render the issue as a markdown‑style block.

        Parameters
        ----------
        idx: Optional[int]
            If supplied, the issue is prefixed with a numeric heading
            (e.g. ``#### Issue 3``).  When ``None`` a generic heading is used.

        Returns
        -------
        str
            Multi‑line string ready to be concatenated into the final report.
        """
        heading = f"#### Issue {idx}" if idx is not None else "### " + self.title
        lines = [heading, f"{self.description}"]
        if self.location:
            lines.append(f"*Location*: `{self.location}`")
        meta = self._format_metadata()
        if meta:
            lines.append(f"*Metadata*: {meta}")
        return "\n".join(lines)
