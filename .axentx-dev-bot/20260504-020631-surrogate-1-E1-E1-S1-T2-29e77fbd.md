import React, { useEffect, useRef } from 'react';
import {
  MapView,
  MapViewProps,
  Marker,
  MarkerProps,
  PROVIDER_GOOGLE,
} from 'react-native-maps';
import { ViewStyle } from 'react-native';

/**
 * MapViewWrapper
 *
 * A thin wrapper around `react-native-maps`'s `MapView` that safely handles
 * mounting and unmounting lifecycle events to avoid memory-leak crashes on
 * both Android and iOS. All props accepted by `MapView` are re-exposed so
 * the component can be used as a drop-in replacement.
 *
 * This wrapper ensures proper cleanup of native resources during component
 * unmount to prevent memory leaks and crashes.
 *
 * Usage:
 * ```tsx
 * <MapViewWrapper
 *   style={styles.map}
 *   initialRegion={{
 *     latitude: 37.78825,
 *     longitude: -122.4324,
 *     latitudeDelta: 0.0922,
 *     longitudeDelta: 0.0421,
 *   }}
 * >
 *   <Marker coordinate={{ latitude: 37.78825, longitude: -122.4324 }} />
 * </MapViewWrapper>
 * ```
 *
 * @param props All `MapViewProps` supported by `react-native-maps`.
 */
export const MapViewWrapper = React.forwardRef<MapView, MapViewProps>((props, ref) => {
  const internalRef = useRef<MapView>(null);
  const mapRef = (ref as React.MutableRefObject<MapView | null>) ?? internalRef;

  // Lifecycle handling – mount & unmount
  useEffect(() => {
    // No special initialization required for MapView itself,
    // but this hook guarantees a deterministic cleanup point.
    // If any listeners or native modules are attached in the future,
    // they can be safely removed here.

    return () => {
      // Defensive cleanup: ensure any pending animations or
      // location listeners are cancelled to prevent native leaks.
      if (mapRef.current) {
        // `stopAnimation` is a no-op on some platforms but safe to call.
        // The method exists on the native MapView instance.
        // @ts-ignore – method may be undefined on older versions.
        mapRef.current?.stopAnimation?.();
      }
    };
  }, [mapRef]);

  // Forward the ref to the underlying MapView
  return <MapView ref={mapRef} {...props} />;
});

/**
 * Re-export commonly used sub-components so consumers can import them from the
 * wrapper module (e.g., `import { MapViewWrapper, Marker } from ...`).
 */
export { Marker, MarkerProps, PROVIDER_GOOGLE };

export default MapViewWrapper;
