package audit_test

import (
    "context"
    "os"
    "testing"
    "time"

    "github.com/stretchr/testify/assert"
    "github.com/stretchr/testify/mock"

    "opt/axentx/surrogate-1/audit"
    "opt/axentx/surrogate-1/db"
)

// MockDB implements db.Client interface for unit tests
type MockDB struct {
    mock.Mock
}
func (m *MockDB) InsertAudit(ctx context.Context, record map[string]interface{}) error {
    args := m.Called(ctx, record)
    return args.Error(0)
}

func TestLogAudit_Enabled(t *testing.T) {
    // Arrange
    os.Setenv("AUDIT_ENABLED", "true")
    defer os.Unsetenv("AUDIT_ENABLED")

    mockDB := new(MockDB)
    mockDB.On("InsertAudit", mock.Anything, mock.Anything).Return(nil)

    audit.SetDBClient(mockDB) // inject mock

    req := map[string]interface{}{"input": "hello"}
    resp := map[string]interface{}{"output": "world"}
    user := "alice"
    patternVer := "v1.2"

    // Act
    err := audit.LogAudit(context.Background(), req, resp, user, patternVer)

    // Assert
    assert.NoError(t, err)
    mockDB.AssertExpectations(t)
}

func TestLogAudit_Disabled(t *testing.T) {
    // Arrange
    os.Setenv("AUDIT_ENABLED", "false")
    defer os.Unsetenv("AUDIT_ENABLED")

    mockDB := new(MockDB)
    // No expectation set – should not be called

    audit.SetDBClient(mockDB)

    // Act
    err := audit.LogAudit(context.Background(), nil, nil, "", "")

    // Assert
    assert.NoError(t, err)
    mockDB.AssertNotCalled(t, "InsertAudit")
}

func TestLogAudit_MissingDBClient(t *testing.T) {
    // Arrange
    os.Setenv("AUDIT_ENABLED", "true")
    defer os.Unsetenv("AUDIT_ENABLED")

    audit.SetDBClient(nil) // simulate missing client

    // Act
    err := audit.LogAudit(context.Background(), nil, nil, "", "")

    // Assert
    assert.Error(t, err)
    assert.Contains(t, err.Error(), "db client not configured")
}
