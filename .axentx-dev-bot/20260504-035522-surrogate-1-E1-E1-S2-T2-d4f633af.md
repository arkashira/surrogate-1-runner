import React, { useState } from 'react';
import { View, Text, TouchableOpacity, Alert, Modal, TextInput, StyleSheet } from 'react-native';
import Swipeable from 'react-native-gesture-handler/Swipeable';

const NodeItem = ({ node, onDelete, onRename }) => {
  const [isEditing, setIsEditing] = useState(false);
  const [newName, setNewName] = useState(node.name);
  
  const handleDelete = () => {
    Alert.alert(
      'Confirm Delete',
      'Are you sure you want to delete this node?',
      [
        { text: 'Cancel', style: 'cancel' },
        { text: 'OK', onPress: () => onDelete(node.id) },
      ],
      { cancelable: false }
    );
  };

  const handleRename = () => {
    onRename(node.id, newName);
    setIsEditing(false);
  };

  const renderRightActions = () => (
    <TouchableOpacity onPress={handleDelete} style={styles.deleteButton}>
      <Text style={styles.deleteText}>Delete</Text>
    </TouchableOpacity>
  );

  return (
    <Swipeable renderRightActions={renderRightActions}>
      <View style={styles.container}>
        <TouchableOpacity onPress={() => setIsEditing(true)}>
          <Text style={styles.nodeText}>{node.name}</Text>
        </TouchableOpacity>
        <Modal
          animationType="slide"
          transparent={true}
          visible={isEditing}
          onRequestClose={() => setIsEditing(false)}
        >
          <View style={styles.modalView}>
            <TextInput

};

## 1. Acceptance Criteria
- **Inline Rename**: Tapping a node’s name replaces the text with an input field pre‑filled with the current name and a “Save” button.  
- **Delete Swipe**: Swiping left on a node reveals a red “Delete” button.  
- **Delete Confirmation**: Tapping “Delete” opens a modal with “Are you sure?” and “OK”/“Cancel” options.  
- **Delete Callback**: Confirming deletion calls `onDelete(nodeId)` exactly once.  
- **Rename Callback**: Saving a new name calls `onRename(nodeId, newName)` exactly once.  
- **Persistence**: After a rename, the node’s displayed name updates immediately and remains after re‑rendering the component tree.  
- **Edge Cases**:  
  - Swiping right or tapping outside the delete area does not trigger deletion.  
  - Canceling the confirmation modal does not call `onDelete`.

import { render, fireEvent, screen } from '@testing-library/react';
import NodeItem from '../src/components/NodeItem';

const mockOnDelete = jest.fn();
const mockOnRename = jest.fn();

const node = { id: 'node-1', name: 'Initial' };

beforeEach(() => {
  jest.clearAllMocks();
});

test('renders node name and shows input on tap', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  expect(screen.getByText('Initial')).toBeInTheDocument();

  fireEvent.click(screen.getByText('Initial'));
  expect(screen.getByDisplayValue('Initial')).toBeInTheDocument();
  expect(screen.getByText('Save')).toBeInTheDocument();
});

test('calls onRename with new name', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  fireEvent.click(screen.getByText('Initial'));
  fireEvent.change(screen.getByDisplayValue('Initial'), { target: { value: 'Renamed' } });
  fireEvent.click(screen.getByText('Save'));

  expect(mockOnRename).toHaveBeenCalledWith('node-1', 'Renamed');
});

test('swipe left reveals delete button', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  const nodeEl = screen.getByTestId('node-item');

  // Simulate swipe left
  fireEvent.touchStart(nodeEl, { touches: [{ clientX: 200 }] });
  fireEvent.touchMove(nodeEl, { touches: [{ clientX: 50 }] });
  fireEvent.touchEnd(nodeEl);

  expect(screen.getByText('Delete')).toBeInTheDocument();
});

test('delete shows confirmation modal', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  // Reveal delete
  fireEvent.click(screen.getByText('Delete'));
  expect(screen.getByText('Are you sure?')).toBeInTheDocument();
  expect(screen.getByText('OK')).toBeInTheDocument();
  expect(screen.getByText('Cancel')).toBeInTheDocument();
});

test('confirming delete calls onDelete', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  fireEvent.click(screen.getByText('Delete'));
  fireEvent.click(screen.getByText('OK'));
  expect(mockOnDelete).toHaveBeenCalledWith('node-1');
});

test('cancelling delete does not call onDelete', () => {
  render(<NodeItem node={node} onDelete={mockOnDelete} onRename={mockOnRename} />);
  fireEvent.click(screen.getByText('Delete'));
  fireEvent.click(screen.getByText('Cancel'));
  expect(mockOnDelete).not.toHaveBeenCalled();
});
