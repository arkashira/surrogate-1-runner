**Core diagnosis (agreed across candidates)**  
- No persisted, date-scoped manifest → training re-enumerates HF repos via authenticated API every run → 429 risk and non-determinism.  
- No CDN bypass → training uses authenticated calls during data loading → burns quota.  
- No sibling-repo routing → single repo can hit 128 commits/hr cap.  
- Studio lifecycle fragile: recreation instead of reuse; idle-stop kills training without restart logic.

**Chosen approach**  
- Single new module `manifest.py` + minimal `train.py` patch.  
- Deterministic manifest built once per date-folder (top-level only) → stored JSON.  
- Training uses CDN-only downloads (`resolve/main/...`) with zero authenticated calls.  
- Deterministic sibling routing for HF writes; studio reuse with restart-on-idle.  
- Remove unactionable or risky items (Teamspace, `list_repo_tree` with `recursive=False` if unavailable, naive text splitting).

from __future__ import annotations

import json
import hashlib
from pathlib import Path
from typing import Dict, List, Optional, Iterator, Tuple

try:
    from huggingface_hub import HfApi, hf_hub_download
    HF_AVAILABLE = True
except Exception:
    HF_AVAILABLE = False

import requests


HF_API_BASE = "https://huggingface.co"
CDN_BASE = "https://huggingface.co/datasets"


def build_manifest(
    repo: str,
    date_folder: str,
    out_json: str,
    repo_type: str = "dataset",
    recursive: bool = False,
) -> Dict:
    """
    Single authenticated call to list files under repo/date_folder,
    then persist manifest for deterministic, CDN-only training.

    If recursive=False (default), lists only top-level entries.
    Set recursive=True if you need full tree (may be slower/larger).
    """
    if not HF_AVAILABLE:
        raise RuntimeError("huggingface_hub not available")

    api = HfApi()
    tree = api.list_repo_tree(
        repo_id=repo,
        path=date_folder,
        repo_type=repo_type,
        recursive=recursive,
    )

    files = [item.rfilename for item in tree if item.rfilename]
    manifest = {
        "repo": repo,
        "date": date_folder,
        "files": sorted(files),
        "total_files": len(files),
    }

    out_path = Path(out_json)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))
    return manifest


def cdn_download(
    repo: str,
    file_path: str,
    local_path: str,
    chunk_size: int = 8192,
    timeout: int = 60,
) -> str:
    """
    Download via HF CDN (no Authorization header) to bypass API rate limits.
    """
    url = f"{CDN_BASE}/{repo}/resolve/main/{file_path}"
    out = Path(local_path)
    out.parent.mkdir(parents=True, exist_ok=True)

    resp = requests.get(url, stream=True, timeout=timeout)
    resp.raise_for_status()
    with open(out, "wb") as f:
        for chunk in resp.iter_content(chunk_size=chunk_size):
            f.write(chunk)
    return str(out)

    Returns repo_id like: "axentx/surrogate-1-0"
