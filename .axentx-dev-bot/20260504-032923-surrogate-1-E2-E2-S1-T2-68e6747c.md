def send_pitch_submission_email(self, pitch_id: str, deck_url: Optional[str] = None):

from fastapi import FastAPI, File, UploadFile, HTTPException
from sqlalchemy.orm import sessionmaker
from backend.database import engine, Base
from backend.schemas import PitchSubmissionRequest, PitchSubmissionResponse
from backend.models import Provider
from backend.services import s3_service, email_service

app = FastAPI()
Base.metadata.create_all(bind=engine)
SessionLocal = sessionmaker(bind=engine)

@app.post("/pitch")
async def create_pitch(request: PitchSubmissionRequest, file: Optional[UploadFile] = None):
    db = SessionLocal()
    provider = Provider(**request.dict())
    db.add(provider)
    db.commit()

    if file:
        s3_url = s3_service.generate_presigned_upload_url(provider.pitch_id, file.content_type, file.file_size)
        with open(file.filename, "wb") as f:
            f.write(await file.read())
        await file.save("temp_file.pdf")
        email_service.send_pitch_submission_email(provider.pitch_id, s3_url)

    response = PitchSubmissionResponse(**provider.dict())
    return response
