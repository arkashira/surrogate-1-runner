// hf-utils.js
// Lightweight HF helpers: manifest build only (authenticated), data via CDN (unauthed).

export async function listRepoTree({ owner, repo, path, recursive = false }) {
  const token = process.env.HF_TOKEN || '';
  const url = new URL(`https://huggingface.co/api/repos/${owner}/${repo}/tree`);
  url.searchParams.set('path', path || '');
  url.searchParams.set('recursive', String(!!recursive));

  const res = await fetch(url, {
    headers: token ? { Authorization: `Bearer ${token}` } : {}
  });

  if (!res.ok) {
    throw new Error(`HF tree API failed: ${res.status} ${await res.text()}`);
  }
  return res.json();
}

export function pickSiblingRepo({ baseRepo, slug, siblings = 5 }) {
  // Deterministic, stable across runs
  let hash = 0;
  for (let i = 0; i < slug.length; i++) {
    const ch = slug.charCodeAt(i);
    hash = ((hash << 5) - hash) + ch;
    hash |= 0;
  }
  const idx = Math.abs(hash) % siblings;
  return idx === 0 ? baseRepo : `${baseRepo}-sibling-${idx}`;
}
