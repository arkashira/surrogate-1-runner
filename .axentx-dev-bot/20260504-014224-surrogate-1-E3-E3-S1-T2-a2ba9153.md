export interface SuggestionRequest {

}

}

export class ApiClient {
  private readonly API_URL = 'https://api.surrogate-1.com/suggestions';

    try {
      const response: AxiosResponse<SuggestionResponse> = await axios.post(

}

import axios from 'axios';
import { ApiClient } from '../src/services/ApiClient';
import { SuggestionRequest, SuggestionResponse } from '../src/types';

jest.mock('axios');
const mockedAxios = axios as jest.Mocked<typeof axios>;

describe('ApiClient.getSuggestions', () => {
  const client = new ApiClient();
  const request: SuggestionRequest = {
    filePath: '/src/app.ts',
    cursorLine: 42,
    context: 'function foo() { return 1; }',
  };

  afterEach(() => jest.clearAllMocks());

  it('sends POST to correct URL with unchanged payload', async () => {
    const mockResp = { data: { suggestions: ['opt1', 'opt2'] } };
    mockedAxios.post.mockResolvedValueOnce(mockResp);

    await client.getSuggestions(request);

    expect(mockedAxios.post).toHaveBeenCalledWith(
      'https://api.surrogate-1.com/suggestions',
      request
    );
  });

  it('resolves with SuggestionResponse containing suggestions', async () => {
    const mockResp = { data: { suggestions: ['optA', 'optB'] } };
    mockedAxios.post.mockResolvedValueOnce(mockResp);

    const result = await client.getSuggestions(request);

    const expected: SuggestionResponse = { suggestions: ['optA', 'optB'] };
    expect(result).toEqual(expected);
  });

  it('rejects with generic error message on network error', async () => {
    mockedAxios.post.mockRejectedValueOnce(new Error('Network Error'));

    await expect(client.getSuggestions(request)).rejects.toThrow(
      'Failed to fetch suggestions. Please try again later.'

      status: 500,
