### Why these choices?

| Issue | Why it matters | How we solve it |
|-------|----------------|-----------------|
| **Re‑renders** | Every state change forces a new render, which is expensive for a map. | `React.memo` + stable handlers (`useCallback`) + `useRef` for the native map instance. |
| **GC spikes** | Rapid `onRegionChange` events create many temporary objects. | `useRafThrottle` (frame‑based) + a lightweight `throttle` for other events. |
| **Android layout glitches** | Native map can jump during state updates. | Enable `LayoutAnimation` on Android once at module load. |
| **Memory profiling** | Hard to spot leaks in a map‑heavy component. | Optional console markers that can be hooked into Xcode Instruments. |
| **Debounce search/filter** | Not directly used in the wrapper, but useful for consumers. | Exported `debounce` helper for reuse. |

/* 2️⃣  Utility: throttle (ms) – generic, used for region change            */
/* -------------------------------------------------------------------------- */
export const throttle = <T extends (...args: unknown[]) => void>(
  fn: T,
  delay: number,
): T => {
  let last = 0;
  let timeout: NodeJS.Timeout | null = null;

  return ((...args: unknown[]) => {
    const now = Date.now();
    const remaining = delay - (now - last);

    if (remaining <= 0) {
      if (timeout) {
        clearTimeout(timeout);
        timeout = null;
      }
      last = now;
      fn(...args);
    } else if (!timeout) {
      timeout = setTimeout(() => {
        last = Date.now();
        timeout = null;
        fn(...args);
      }, remaining);
    }
  }) as T;
};

/* -------------------------------------------------------------------------- */

/* -------------------------------------------------------------------------- */

  fn: T,
  delay: number,
): T => {
  let timeout: NodeJS.Timeout | null = null;

  return ((...args: unknown[]) => {
    if (timeout) clearTimeout(timeout);
    timeout = setTimeout(() => fn(...args), delay);
  }) as T;
};

/* -------------------------------------------------------------------------- */

      performance.memory.usedJSHeapSize / 1048576,
    );

  }
