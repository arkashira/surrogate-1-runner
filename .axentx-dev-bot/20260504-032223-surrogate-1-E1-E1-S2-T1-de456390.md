# test_engine_resolution_unit.py
import pytest
from rendering.engine import RenderingEngine, ResolutionError

@pytest.fixture
def engine():
    return RenderingEngine()   # default init creates a framebuffer

def test_initial_resolution(engine):
    assert engine.resolution == (1920, 1080)   # assume default

def test_set_valid_4k_resolution(engine):
    engine.set_resolution(3840, 2160)
    assert engine.resolution == (3840, 2160)

def test_set_invalid_width_too_large(engine):
    with pytest.raises(ResolutionError) as exc:
        engine.set_resolution(5000, 1080)
    assert "exceeds maximum width" in str(exc.value)

def test_set_invalid_height_too_large(engine):
    with pytest.raises(ResolutionError):
        engine.set_resolution(1920, 3000)

def test_set_invalid_negative_values(engine):
    with pytest.raises(ResolutionError):
        engine.set_resolution(-1280, 720)

def test_set_invalid_non_integer(engine):
    with pytest.raises(ResolutionError):
        engine.set_resolution(1920.5, 1080)

def test_no_framebuffer_recreation_when_same_resolution(engine, mocker):
    # spy on the private _recreate_framebuffer method
    spy = mocker.spy(engine, "_recreate_framebuffer")
    engine.set_resolution(*engine.resolution)   # same size
    assert spy.call_count == 0

def test_framebuffer_recreation_on_change(engine, mocker):
    spy = mocker.spy(engine, "_recreate_framebuffer")
    engine.set_resolution(2560, 1440)
    assert spy.call_count == 1

def test_logging_on_successful_change(engine, caplog):
    engine.set_resolution(2560, 1440)
    assert any(
        "Resolution changed to 2560x1440" in rec.message and rec.levelname == "INFO"
        for rec in caplog.records
    )

def test_logging_on_failure(engine, caplog):
    with pytest.raises(ResolutionError):

    )
