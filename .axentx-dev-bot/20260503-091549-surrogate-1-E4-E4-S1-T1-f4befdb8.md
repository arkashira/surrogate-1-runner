```typescript
import axios from 'axios';

interface SlackWebhookSender {
  send(message: string): void;
}

class SlackWebhookSenderImpl implements SlackWebhookSender {
  private webhookUrl: string;

  constructor(webhookUrl: string) {
    this.webhookUrl = webhookUrl;
  }

  send(message: string): void {
    const data = {
      text: message,
    };

    axios.post(this.webhookUrl, data)
      .then((response) => {
        console.log(`Notification sent successfully: ${response.status}`);
      })
      .catch((error) => {
        console.error(`Error sending notification: ${error.message}`);
      });
  }
}

export { SlackWebhookSender, SlackWebhookSenderImpl };
```
