  const [errors, setErrors] = useState({});

          return 'Enter a margin between 1-100%';
        }
        return '';
      case 'competitorPrice':
        if (!value || isNaN(parseFloat(value)) || parseFloat(value) <= 0) {
          return 'Enter a valid positive competitor price';
        }
        return '';
      default:
        return '';
    }
  }, []);

  // ─────────────────────────────────────────────────────────────
  // COMPETITIVENESS CALCULATION (Candidate 1 - core insight)

    const cost = parseFloat(formData.costPerUnit);
    const targetMargin = parseFloat(formData.targetMargin);

    if (isNaN(price) || isNaN(competitorPrice) || isNaN(cost) || isNaN(targetMargin)) {
      setCompetitivenessFeedback(null);
      return;
    }

    const priceDiff = ((competitorPrice - price) / competitorPrice) * 100;
