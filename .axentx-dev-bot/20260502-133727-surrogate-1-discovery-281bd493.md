import json, sys
from huggingface_hub import HfApi

def main(repo_id, date_folder, out_path):
    api = HfApi()
    entries = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = []
    for e in entries:
        if e.type == "directory":
            try:
                subentries = api.list_repo_tree(repo_id=repo_id, path=e.path, recursive=False)
                for se in subentries:
                    if se.type == "file":
                        files.append({"path": se.path, "size": se.size})
            except Exception as exc:
                print(f"Skipping {e.path}: {exc}", file=sys.stderr)
        elif e.type == "file":
            files.append({"path": e.path, "size": se.size if hasattr(se, 'size') else None})
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump(files, f, indent=2)
    print(f"Wrote {len(files)} files to {out_path}")

if __name__ == "__main__":
    main(sys.argv[1], sys.argv[2], sys.argv[3])
PY
