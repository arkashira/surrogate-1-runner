import React, { useState, useEffect } from "react";
import PropTypes from "prop-types";

/**
 * BatteryWarningPanel
 *
 * Props
 * -----
 * - warnings: Array<{ level: number, message: string, instruction: string }>
 *   Each warning is shown when its `level` ≤ the current threshold.
 *
 * - data: Array<{ timestamp: number | string, chargeCycles: number, healthPercent: number }>
 *   The most recent element (by timestamp) is used to compute `latestHealth`.
 *
 * - initialThreshold: number (optional) – default threshold (percentage).  Default is 20.
 * - onThresholdChange: (newThreshold: number) => void (optional)
 *
 * The component renders:
 *   * A slider to adjust the threshold.
 *   * A card that appears when `latestHealth < threshold`.
 *   * A list of warnings that match the threshold.
 */
export default function BatteryWarningPanel({
  warnings = [],
  data = [],
  initialThreshold = 20,
  onThresholdChange,
}) {
  /* ---------- Threshold state & persistence ---------- */
  const [threshold, setThreshold] = useState(initialThreshold);

  useEffect(() => {
    const stored = localStorage.getItem("batteryWarningThreshold");
    if (stored !== null) {
      const parsed = parseInt(stored, 10);
      if (!Number.isNaN(parsed)) setThreshold(parsed);
    }
  }, []);

  useEffect(() => {
    localStorage.setItem("batteryWarningThreshold", threshold);
    if (typeof onThresholdChange === "function") onThresholdChange(threshold);
  }, [threshold, onThresholdChange]);

  /* ---------- Latest health calculation ---------- */
  const [latestHealth, setLatestHealth] = useState<number | null>(null);

    if (Array.isArray(data) && data.length > 0) {
