### 1. `bin/build_manifest.py` (run from Mac orchestrator)

```python
#!/usr/bin/env python3
"""
Generate train_manifest.json for today's folder.
Run from Mac (or CI) after rate-limit window clears.
"""
import json
import os
from datetime import datetime, timezone
from huggingface_hub import HfApi

HF_REPO = "datasets/axentx/surrogate-1-training-pairs"
OUT_FILE = "train_manifest.json"

def main():
    api = HfApi(token=os.getenv("HF_TOKEN"))
    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    # Single API call: non-recursive per folder
    entries = api.list_repo_tree(
        repo_id=HF_REPO,
        path=today,
        repo_type="dataset",
        recursive=False,
    )
    files = [e.rfilename for e in entries if e.type == "file"]
    manifest = {
        "date": today,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": [
            {
                "filename": f,
                "cdn_url": f"https://huggingface.co/datasets/{HF_REPO}/resolve/main/{f}",
                "hf_path": f,
            }
            for f in sorted(files)
        ],
    }
    with open(OUT_FILE, "w") as fp:
        json.dump(manifest, fp, indent=2)
    print(f"Wrote {OUT_FILE} with {len(files)} files")

if __name__ == "__main__":
    main()
```
