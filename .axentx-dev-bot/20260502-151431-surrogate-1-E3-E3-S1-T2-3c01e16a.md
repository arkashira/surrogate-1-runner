from __future__ import annotations

import threading
import uuid
from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from enum import Enum
from typing import Callable, Dict, List, Optional, Any


class EventType(str, Enum):
    ORDER_ACK = "order_ack"
    ORDER_FILL = "order_fill"
    ORDER_REJECT = "order_reject"


@dataclass(frozen=True)
class Event:
    event_id: str
    event_type: EventType
    timestamp: datetime
    payload: Dict[str, Any]

    @staticmethod
    def create(event_type: EventType, payload: Dict[str, Any]) -> "Event":
        return Event(
            event_id=str(uuid.uuid4()),
            event_type=event_type,
            timestamp=datetime.now(timezone.utc),
            payload=payload,
        )


class EventStore:
    """
    Simple thread-safe in-memory event store.
    Replace with persistent DB in production without changing adapter interface.
    """

    def __init__(self) -> None:
        self._events: List[Event] = []
        self._lock = threading.Lock()

    def persist(self, event: Event) -> None:
        with self._lock:
            self._events.append(event)

    def list_events(self, event_type: Optional[EventType] = None) -> List[Event]:
        with self._lock:
            if event_type is None:
                return list(self._events)
            return [e for e in self._events if e.event_type == event_type]

    def count(self) -> int:
        with self._lock:
            return len(self._events)


class EventBus:
    """
    Synchronous, ordered event bus.
    Persists before notifying subscribers to guarantee durability-first semantics.
    """

    def __init__(self, store: EventStore) -> None:
        self.store = store
        self._subscribers: Dict[EventType, List[Callable[[Event], None]]] = {}
        self._lock = threading.Lock()

    def subscribe(self, event_type: EventType, handler: Callable[[Event], None]) -> None:
        with self._lock:
            self._subscribers.setdefault(event_type, []).append(handler)

    def publish(self, event: Event) -> None:
        # Persist first (durability), then notify.
        self.store.persist(event)
        with self._lock:
            handlers = list(self._subscribers.get(event.event_type, []))
        for handler in handlers:
            try:
                handler(event)
            except Exception:
                # Production: log and continue.
                pass


# Default global instances for convenience (override via composition if needed).
default_store = EventStore()
default_bus = EventBus(default_store)
