"""
Slack notifier for request status transitions.
- Non-blocking enqueue (async or sync-safe).
- Background worker posts to webhook with retries/backoff.
- Failures are logged and never block status transitions.
"""
import asyncio
import logging
from typing import Optional

try:
    import aiohttp
except ImportError:
    aiohttp = None  # handled at runtime with clear error

logger = logging.getLogger(__name__)

# In-memory queue; replace with Redis/RQ in production if multi-worker.
_SLACK_QUEUE: asyncio.Queue = asyncio.Queue(maxsize=2000)
_WORKER_TASK: Optional[asyncio.Task] = None


def _build_payload(
    request_title: str,
    new_status: str,
    public_url: str,
    sla_state: str,
    assignee_email: Optional[str] = None,
    creator_email: Optional[str] = None,
) -> dict:
    mentions = []
    if assignee_email:
        mentions.append(f"<{assignee_email.strip()}>")
    if creator_email:
        mentions.append(f"<{creator_email.strip()}>")
    mention_text = " ".join(mentions) if mentions else "_No assignee/creator email available_"

    text_lines = [
        "*Request Update*",
        f"> *Title:* {request_title}",
        f"> *Status:* {new_status}",
        f"> *SLA:* {sla_state}",
        f"> *URL:* <{public_url}|Open request>",
        f"> *Participants:* {mention_text}",
    ]
    text = "\n".join(text_lines)

    return {
        "text": text,
        "blocks": [
            {
                "type": "section",
                "text": {"type": "mrkdwn", "text": text},
            },
        ],
    }
