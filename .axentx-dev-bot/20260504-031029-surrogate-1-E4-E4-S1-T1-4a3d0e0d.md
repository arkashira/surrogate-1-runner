# --------------------------------------------------------------
# Concurrency‑check workflow
# --------------------------------------------------------------
#  • Runs on every push to `main` *and* on every PR that targets `main`
#  • Can also be triggered manually (workflow_dispatch)
#  • Installs the project, runs a static‑analysis suite and a profiling
#    agent, fails the job on any *critical* problem, and publishes
#    analysis artifacts + a summary badge that can be embedded in the#    repo README.
# --------------------------------------------------------------

name: Concurrency Check

# -----------------------------------------------------------------# Triggers
# -----------------------------------------------------------------
on:
  push:
    branches: [main]               # run on direct pushes to main
  pull_request:
    branches: [main]               # run on PRs that target main
  workflow_dispatch:               # allow manual runs from the UI

# -----------------------------------------------------------------
# Jobs
# -----------------------------------------------------------------
jobs:
  concurrency-check:
    runs-on: ubuntu-latest
    permissions:
      # Needed for uploading artifacts & for the summary badge step
      contents: read
      actions: read
      checks: write
      pull-requests: write

    steps:
      # -------------------------------------------------------------
      # 1️⃣ Checkout the repository
      # -------------------------------------------------------------
      - name: Checkout repository
        uses: actions/checkout@v4      # -------------------------------------------------------------
      # 2️⃣ Set up Python (use the same version everywhere)
      # -------------------------------------------------------------
      - name: Set up Python 3.11
        uses: actions/setup-python@v4
        with:
          python-version: "3.11"

      # -------------------------------------------------------------
      # 3️⃣ Install project dependencies
      # -------------------------------------------------------------
      - name: Install dependencies        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      # -------------------------------------------------------------
      # 4️⃣ Run static analysis (mypy + flake8)
      # -------------------------------------------------------------
      - name: Run static analyzer
        id: static_analysis
        continue-on-error: true               # we will decide later if we fail
        run: |
          echo "🔎 Running mypy ..."
          if ! python -m mypy src/; then
            echo "MYPY_FAILED=true" >> $GITHUB_ENV
          fi

          echo "🔎 Running flake8 ..."
          if ! python -m flake8 src/; then
            echo "FLAKE8_FAILED=true" >> $GITHUB_ENV
          fi      # -------------------------------------------------------------
      # 5️⃣ Run the profiling agent (cProfile → .prof file)
      # -------------------------------------------------------------
      - name: Run profiling agent
        id: profiling
        continue-on-error: true
        run: |
          echo "📈 Running cProfile ..."
          if ! python -m cProfile -o profile_output.prof src/main.py; then            echo "PROFILE_FAILED=true" >> $GITHUB_ENV
          fi

      # -------------------------------------------------------------
      # 6️⃣ Aggregate findings & decide whether to fail the job
      # -------------------------------------------------------------
      - name: Fail on critical issues
        if: env.MYPY_FAILED == 'true' || env.FLAKE8_FAILED == 'true' || env.PROFILE_FAILED == 'true'
        run: |
          echo "❌ Critical issue detected – failing the workflow."
          exit 1
