package com.axentx.surrogate1.llm;

import com.axentx.surrogate1.exception.UnsupportedLLMProviderException;
import com.theokanning.openai.service.OpenAiService;
import minimax.MiniMax;
import ai.gemini.Gemini;
import cohere.Cohere;
import anthropic.Anthropic;

import java.util.HashMap;
import java.util.Map;

public class LLMProviderFactory {
    private static final Map<String, Class<? extends LLMProvider>> PROVIDERS = new HashMap<>();

    static {
        PROVIDERS.put("CLAUDE", Claude.class);
        PROVIDERS.put("MINIMAX", MiniMax.class);
        PROVIDERS.put("OPENAI", OpenAiService.class);
        PROVIDERS.put("ANTHROPIC", Anthropic.class);
        PROVIDERS.put("COHERE", Cohere.class);
        PROVIDERS.put("GEMINI", Gemini.class);
    }

    public static LLMProvider getProvider(String providerName, String apiKey) {
        Class<? extends LLMProvider> providerClass = PROVIDERS.get(providerName.toUpperCase());
        if (providerClass == null) {
            throw new UnsupportedLLMProviderException("Unsupported LLM provider: " + providerName);
        }
        try {
            return providerClass.getConstructor(String.class).newInstance(apiKey);
        } catch (Exception e) {
            throw new RuntimeException("Error creating LLM provider instance", e);
        }
    }
}
