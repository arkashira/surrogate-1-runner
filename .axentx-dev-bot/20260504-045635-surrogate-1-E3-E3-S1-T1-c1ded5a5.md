// /opt/axentx/surrogate-1/ci-cd-integration.go
package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"os"
	"os/exec"
	"path/filepath"
	"strings"
)

// ---------------------------------------------------------------------
// 1. Data structures
// ---------------------------------------------------------------------

// PolicyViolation is the canonical representation of a rule breach.
type PolicyViolation struct {
	RuleID      string `json:"rule_id"`      // unique identifier, e.g. "SEC001"
	Description string `json:"description"` // human‑readable message
	Severity    string `json:"severity"`    // "low", "medium", "high", "critical"
}

// Policy is an *in‑process* check that can be run without an external binary.
type Policy struct {
	Name      string                                 // friendly name
	Check     func() (passed bool, violation PolicyViolation) // returns false + violation when it fails
	Remediate func() error                           // optional auto‑fix
}

// ---------------------------------------------------------------------
// 2. Configurable hooks (can be swapped in tests or CI environments)
// ---------------------------------------------------------------------

// Notifier is called for every violation.  The default implementation prints
// to stdout, but callers can replace it with a webhook, Slack, GitHub Checks, etc.
var Notifier = func(v PolicyViolation) error {
	fmt.Printf("⚠️ Policy violation: [%s] %s (severity: %s)\n",
		v.RuleID, v.Description, v.Severity)
	return nil
}

// RemediationRunner is invoked for each violation that has a remediation
// script or Go function.  It returns nil on success, non‑nil on failure.
var RemediationRunner = func(v PolicyViolation) error {
	// 1️⃣ Try a Go‑function remediation (if one exists in the policy map)
	if fn, ok := remediationFuncs[v.RuleID]; ok {
		return fn()
	}
	// 2️⃣ Fallback to a shell script under ./remediation/<RuleID>.sh
	script := filepath.Join("remediation", v.RuleID+".sh")
	if _, err := os.Stat(script); os.IsNotExist(err) {
		// No remediation – not an error, just a no‑op.
		fmt.Printf("No remediation script for rule %s; skipping.\n", v.RuleID)
		return nil
	}
	cmd := exec.Command("bash", script)
	cmd.Stdout = os.Stdout
	cmd.Stderr = os.Stderr
	return cmd.Run()
}
