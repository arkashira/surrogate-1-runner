# .github/workflows/test-filter-coverage.yml
name: Test Filter & Coverage

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  workflow_dispatch:
    inputs:
      filter:
        description: 'Optional test filter string (pytest -k syntax)'
        required: false
        default: ''

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt
          pip install pytest pytest-cov

      - name: Run tests with coverage
        env:
          TEST_FILTER: ${{ github.event.inputs.filter }}
        run: |
          if [ -n "$TEST_FILTER" ]; then
            echo "Running tests with filter: $TEST_FILTER"
            pytest --cov=src --cov-report=xml --cov-report=term-missing -k "$TEST_FILTER"
          else
            echo "Running all tests"
            pytest --cov=src --cov-report=xml --cov-report=term-missing
          fi

      - name: Upload coverage report
        uses: actions/upload-artifact@v4
        with:
          name: coverage-report
          path: coverage.xml
```
