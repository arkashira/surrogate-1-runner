"""
Core embedding access and manipulation functions.

Provides interfaces for accessing internal embeddings
from the surrogate-1 dataset pipeline.
"""

from typing import List, Optional, Dict, Any
from dataclasses import dataclass, field
import numpy as np


@dataclass
class EmbeddingVector:
    """Represents an embedding vector with metadata.
    
    Attributes:
        vector: Numpy array of embedding values
        model: Name of the model that generated the embedding
        dimension: Dimensionality of the embedding vector
        metadata: Additional metadata about the embedding
    """
    vector: np.ndarray
    model: str
    dimension: int
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    def __post_init__(self):
        if self.vector.shape[0] != self.dimension:
            raise ValueError(f"Vector dimension {self.vector.shape[0]} does not match declared dimension {self.dimension}")
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert to dictionary representation."""
        return {
            "vector": self.vector.tolist(),
            "model": self.model,
            "dimension": self.dimension,
            "metadata": self.metadata,
        }
    
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "EmbeddingVector":
        """Create EmbeddingVector from dictionary."""
        return cls(
            vector=np.array(data["vector"], dtype=np.float32),
            model=data["model"],
            dimension=data["dimension"],
            metadata=data.get("metadata", {}),
        )
    
    def normalize(self) -> "EmbeddingVector":
        """Return L2-normalized copy of this embedding."""
        norm = np.linalg.norm(self.vector)
        if norm == 0:
            return self
        normalized = EmbeddingVector(
            vector=self.vector / norm,
            model=self.model,
            dimension=self.dimension,
            metadata=self.metadata.copy()
        )
        return normalized

        """Compute cosine similarity with another embedding."""

            raise ValueError("Cannot compute similarity between embeddings of different dimensions")
