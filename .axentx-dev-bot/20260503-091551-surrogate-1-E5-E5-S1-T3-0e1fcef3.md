from datetime import datetime, timedelta
from typing import List, Dict
from ..database import db

def get_weekly_active_tracked_requests(team_id: int) -> int:
    # Retrieve the count of distinct requests with activity in the last 7 days
    query = """
        SELECT COUNT(DISTINCT request_id) 
        FROM request_activity 
        WHERE team_id = :team_id AND timestamp > :last_week
    """
    last_week = datetime.now() - timedelta(days=7)
    result = db.execute(query, {"team_id": team_id, "last_week": last_week})
    return result.fetchone()[0]

def get_open_at_risk_breached_counts(team_id: int) -> Dict[str, int]:
    # Retrieve the counts of open, at risk, and breached requests
    query = """
        SELECT status, COUNT(*) 
        FROM requests 
        WHERE team_id = :team_id 
        GROUP BY status
    """
    result = db.execute(query, {"team_id": team_id})
    counts = {}
    for row in result.fetchall():
        counts[row[0]] = row[1]
    return counts

def get_recent_activity_timeline(team_id: int) -> List[Dict[str, str]]:
    # Retrieve the recent activity timeline
    query = """
        SELECT request_id, timestamp, status 
        FROM request_activity 
        WHERE team_id = :team_id 
        ORDER BY timestamp DESC 
        LIMIT 10
    """
    result = db.execute(query, {"team_id": team_id})
    timeline = []
    for row in result.fetchall():
        timeline.append({
            "request_id": row[0],
            "timestamp": row[1].strftime("%Y-%m-%d %H:%M:%S"),
            "status": row[2]
        })
    return timeline
