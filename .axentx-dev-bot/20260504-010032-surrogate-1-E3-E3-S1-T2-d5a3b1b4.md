<script>
import { ref } from 'vue';
import { deployService, streamLogs, getDeploymentStatus } from '@/api/deploy';

export default {
  name: 'DeployButton',
  setup() {
    const deploying = ref(false);
    const success = ref(false);
    const logs = ref([]);
    const deploymentStatus = ref('');

    const startDeployment = async () => {
      deploying.value = true;
      logs.value = [];
      success.value = false;
      deploymentStatus.value = '';

      try {
        await deployService();
        // Start streaming logs after deployment request
        const logStream = streamLogs((logLine) => {
          logs.value.push(logLine);
        });
        // Cleanup log stream when deployment is complete
        await new Promise((resolve) => {
          setTimeout(resolve, 5000); // Wait 5 seconds for deployment to complete
          logStream();
        });
        // Fetch deployment status after deployment is complete
        const status = await getDeploymentStatus();
        deploymentStatus.value = status;
        success.value = true;
      } catch (err) {
        logs.value.push(`Error: ${err.message}`);
      } finally {
        deploying.value = false;
      }
    };

    return {
      deploying,
      success,
      logs,
      deploymentStatus,
      startDeployment,
    };
  },
};
</script>

/**
 * Calls the backend to trigger deployment.
 * Expects a POST to /api/deploy.
 */
export async function deployService() {
  const resp = await fetch('/api/deploy', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
  });
  if (!resp.ok) {
    const errText = await resp.text();
    throw new Error(`Deploy failed: ${errText}`);
  }
  return resp.json();
}

/**
 * Streams deployment logs via Server-Sent Events.
 * The backend should expose /api/deploy/logs as an SSE endpoint.
 * @param {(line: string) => void} onLine Callback for each log line.
 */
export function streamLogs(onLine) {
  const evtSource = new EventSource('/api/deploy/logs');
  evtSource.onmessage = (e) => {
    onLine(e.data);
  };
  evtSource.onerror = (e) => {
    onLine('Log stream error');
    evtSource.close();
  };
  // Return a function to allow manual cleanup if needed
  return () => evtSource.close();
}

/**
 * Fetches the deployment status from the backend.
 * Expects a GET to /api/status.
 */
export async function getDeploymentStatus() {
  const resp = await fetch('/api/status');
  if (!resp.ok) {
    const errText = await resp.text();
    throw new Error(`Failed to get deployment status: ${errText}`);
  }
  return resp.json();
}
