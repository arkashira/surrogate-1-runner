import os
import json
import smtplib
from email.mime.text import MIMEText
from typing import Optional, List, Dict, Any
import httpx
from datetime import datetime, timedelta

SLACK_WEBHOOK_URL = os.getenv("SLACK_WEBHOOK_URL")
SMTP_HOST = os.getenv("SMTP_HOST", "localhost")
SMTP_PORT = int(os.getenv("SMTP_PORT", "25"))
SMTP_FROM = os.getenv("SMTP_FROM", "notifications@axentx.local")


def get_sla_state(created_at: datetime, sla_hours: Optional[int], current_status: str) -> str:
    """Return SLA state: OK, WARNING, BREACH, or N/A."""
    if not sla_hours or current_status in ("closed", "resolved", "cancelled"):
        return "N/A"
    due = created_at + timedelta(hours=sla_hours)
    now = datetime.utcnow()
    if now > due:
        return "BREACH"
    if now > due - timedelta(hours=1):
        return "WARNING"
    return "OK"

    """Fetch last timeline entry for a request."""
    try:

    except Exception:
        return None


def build_message(
