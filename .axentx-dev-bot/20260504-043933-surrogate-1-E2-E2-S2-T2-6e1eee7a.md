package relay

import (
	"context"
	"math/big"

	"github.com/ethereum/go-ethereum/ethclient"
)

// GasStrategy holds chain‑specific multipliers that are applied to the
// base gas price returned by the RPC node.  The multiplier is a
// floating‑point value that is multiplied with the base gas price
// (in wei) to produce the final gas price that the relay should
// use when signing a transaction.
//
// Example: a multiplier of 1.2 means the relay will pay 20 % more
// gas than the node suggests.  A multiplier of 0.9 will pay 10 % less.
type GasStrategy struct {
	// multipliers maps chain ID to a multiplier.  If a chain ID is
	// missing from the map the default multiplier of 1.0 is used.
	multipliers map[uint64]float64
}

// NewGasStrategy creates a new GasStrategy with the supplied
// multipliers.  The map is copied so that callers can safely reuse
// the original map after passing it in.
func NewGasStrategy(multipliers map[uint64]float64) *GasStrategy {
	// copy the map to avoid accidental mutation
	cp := make(map[uint64]float64, len(multipliers))
	for k, v := range multipliers {
		cp[k] = v
	}
	return &GasStrategy{multipliers: cp}
}

// DefaultGasStrategy returns a strategy that applies a multiplier of
// 1.0 to every chain.  This can be used when no chain‑specific
// adjustments are required.
func DefaultGasStrategy() *GasStrategy {
	return &GasStrategy{multipliers: map[uint64]float64{}}
}
