from __future__ import annotations

from dataclasses import dataclass, asdict
from typing import Any, Dict, Optional
import time


@dataclass
class ConnectorError(Exception):
    """Structured error raised by connectors to avoid crashing orchestrator."""
    code: str
    message: str
    connector: str
    details: Optional[Dict[str, Any]] = None
    timestamp: float = 0.0

    def __post_init__(self) -> None:
        if self.timestamp == 0.0:
            self.timestamp = time.time()

    def to_dict(self) -> Dict[str, Any]:
        return {
            "code": self.code,
            "message": self.message,
            "connector": self.connector,
            "details": self.details or {},
            "timestamp": self.timestamp,
        }


class BaseConnector:
    """
    Standard connector interface.

    Config schema (common fields):
      - name: str
      - enabled: bool

    Subclasses must validate config in init() and raise ConnectorError on invalid config.
    """

    def __init__(self, config: Dict[str, Any]) -> None:
        self.config = config
        self.name = str(config.get("name", self.__class__.__name__))
        self.enabled = bool(config.get("enabled", True))
        self._initialized = False

    def init(self) -> None:
        """
        Initialize connector (validate config, establish connections, warm-up).
        Must raise ConnectorError on failure.
        """
        if not self.enabled:
            raise ConnectorError(
                code="CONNECTOR_DISABLED",
                message=f"Connector '{self.name}' is disabled",
                connector=self.name,
                details={"config": self.config},
            )
        self._initialized = True

    def call(self, payload: Dict[str, Any]) -> Dict[str, Any]:
        """
        Execute connector logic. Must be implemented by subclasses.
        Must raise ConnectorError on failure.
        """
        raise NotImplementedError("call() must be implemented by subclass")

    def health(self) -> Dict[str, Any]:
        """
        Return health status dict. Should not raise.
        """
        return {
            "name": self.name,
            "class": self.__class__.__name__,
            "enabled": self.enabled,
            "initialized": self._initialized,
            "timestamp": time.time(),
            "status": "healthy" if self._initialized and self.enabled else "degraded",
        }
