// Lightweight SLA timer — no external deps
(function () {
  function $(sel) { return document.querySelector(sel); }

  function createTimer(options) {
    var el = options.element;
    if (!el) return;

    var slaSeconds = Number(options.slaTargetSeconds || 0);
    var lastTransitionMs = Number(options.lastTransitionEpoch || 0);
    if (!isFinite(slaSeconds) || !isFinite(lastTransitionMs) || slaSeconds <= 0) {
      el.textContent = '—';
      return;
    }

    // SLA target absolute time (ms)
    var targetMs = lastTransitionMs + slaSeconds * 1000;

    function pad(n) { return String(n).padStart(2, '0'); }

    function formatDuration(ms) {
      var neg = ms < 0;
      var abs = Math.abs(ms);
      var s = Math.floor(abs / 1000);
      var m = Math.floor(s / 60);
      var h = Math.floor(m / 60);
      var d = Math.floor(h / 24);
      s %= 60;
      m %= 60;
      h %= 24;
      if (d > 0) return (neg ? '-' : '') + d + 'd ' + pad(h) + ':' + pad(m) + ':' + pad(s);
      return (neg ? '-' : '') + pad(h) + ':' + pad(m) + ':' + pad(s);
    }

    function tick() {
      var now = Date.now();
      var remainingMs = targetMs - now;

      if (remainingMs <= 0) {
        el.textContent = 'overdue';
        el.classList.add('breach');
        el.classList.remove('ok');
      } else {
        el.textContent = formatDuration(remainingMs);
        el.classList.remove('breach');
        el.classList.add('ok');
      }
    }

    tick();
    var iv = setInterval(tick, 1000);

    // Expose cleanup
    el._slaTimer = { stop: function () { clearInterval(iv); } };
  }

  // Auto-initialize on elements with data-sla-target-seconds and data-last-transition-epoch
  function init() {
    var els = document.querySelectorAll('[data-sla-target-seconds][data-last-transition-epoch]');
    for (var i = 0; i < els.length; i++) {
      var el = els[i];
      createTimer({
        element: el,
        slaTargetSeconds: el.getAttribute('data-sla-target-seconds'),
        lastTransitionEpoch: el.getAttribute('data-last-transition-epoch')
      });
    }
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }

  // Expose for manual init if needed
  window.SlaTimer = { create: createTimer };
})();
```
