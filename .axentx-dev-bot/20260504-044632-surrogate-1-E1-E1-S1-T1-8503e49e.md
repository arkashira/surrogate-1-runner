# healthcheck.test.py

import unittest
from unittest.mock import patch, MagicMock
from healthcheck import HealthCheck, Container

class TestHealthCheck(unittest.TestCase):

    def setUp(self):
        self.health_check = HealthCheck(interval=30, timeout=30)
        self.container = Container(id='abc123', service_name='web')

    @patch('healthcheck.docker')
    def test_detect_unhealthy_container(self, mock_docker):
        mock_container = MagicMock()
        mock_container.attrs = {'State': {'Health': {'Status': 'unhealthy'}}}
        mock_docker.from_env().containers.get.return_value = mock_container

        result = self.health_check.detect_unhealthy_container(self.container)
        self.assertTrue(result)

    @patch('healthcheck.docker')
    def test_report_unhealthy_container(self, mock_docker):
        mock_container = MagicMock()
        mock_container.attrs = {'State': {'Health': {'Status': 'unhealthy', 'Log': [{'Output': 'Error: Connection refused'}]}}}
        mock_docker.from_env().containers.get.return_value = mock_container

        result = self.health_check.report_unhealthy_container(self.container)
        self.assertIn('abc123', result)
        self.assertIn('web', result)
        self.assertIn('Error: Connection refused', result)

    def test_configure_health_check(self):
        self.health_check.configure_health_check(interval=60, timeout=60)
        self.assertEqual(self.health_check.interval, 60)
        self.assertEqual(self.health_check.timeout, 60)

    @patch('healthcheck.logging')
    def test_log_health_check_results(self, mock_logging):
        self.health_check.log_health_check_results('Health check passed')
        mock_logging.info.assert_called_with('Health check passed')

    @patch('healthcheck.requests')
    def test_send_alert(self, mock_requests):
        self.health_check.send_alert('Unhealthy container detected')
        mock_requests.post.assert_called_with('http://alert-endpoint', json={'message': 'Unhealthy container detected'})

if __name__ == '__main__':
    unittest.main()
