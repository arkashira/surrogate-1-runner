MAPE_THRESHOLD: float = 0.10       # 10 %

# ----------------------------------------------------------------------

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(name)s: %(message)s")

    * MAPE back‑testing
    * S3 persistence
    """

        self.mape: Optional[float] = None

    # ------------------------------------------------------------------
    # Training
    # ------------------------------------------------------------------
    def train(self, cost_data: pd.DataFrame) -> "CostForecastModel":
        """
        Train Prophet on the last `DATA_QUERY_DAYS` rows of `cost_data`.

        `cost_data` must contain a datetime column named `date`

            daily_seasonality=True,

        self.model.fit(df)
        logger.info(f"Prophet model trained on {len(df)} records")
        return self

    # ------------------------------------------------------------------
    # Prediction
    # ------------------------------------------------------------------
    def predict(self, periods: int = FORECAST_DAYS) -> pd.DataFrame:
        """
        Return a DataFrame with columns:

        """

            raise RuntimeError("Model must be trained before prediction")

        future = self.model.make_future_dataframe(periods=periods)
        forecast = self.model.predict(future)
