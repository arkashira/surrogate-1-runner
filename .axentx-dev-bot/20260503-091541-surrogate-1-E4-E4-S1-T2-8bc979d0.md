Add a robust, typed endpoint that returns all audit entries for a request, sorted by `created_at` ascending.

```ts
import { Router } from "express";
import { db } from "../db";

const router = Router();

/**
 * GET /requests/:id/audit
 * Returns all audit entries for a request, sorted by time ascending.
 */
router.get("/requests/:id/audit", async (req, res) => {
  const { id } = req.params;

  if (!id) {
    return res.status(400).json({ error: "missing_request_id" });
  }

  try {
    const rows = await db.query(
      `SELECT id, request_id, action, actor, metadata, created_at
       FROM audit_log
       WHERE request_id = $1
       ORDER BY created_at ASC`,
      [id]
    );

    return res.json({
      request_id: id,
      count: rows.length,
      entries: rows.map((r) => ({
        id: r.id,
        action: r.action,
        actor: r.actor,
        metadata: r.metadata,
        created_at: r.created_at,
      })),
    });
  } catch (err) {
    console.error("Failed to fetch audit entries", err);
    return res.status(500).json({ error: "internal_server_error" });
  }
});

export default router;
```

### 2) `/opt/axentx/surrogate-1/src/routes/public.ts`

Fetch request, last-3 audits (ascending for display), and SLA summary in parallel. Keep behavior deterministic and safe for missing rows.

```ts
import { Router } from "express";
import { db } from "../db";

const router = Router();

/**
 * GET /requests/:id
 * Public status page data for a request.
 */
router.get("/requests/:id", async (req, res) => {
  const { id } = req.params;

  if (!id) {
    return res.status(400).json({ error: "missing_request_id" });
  }

  const [request, auditLast3Rows, slaRow] = await Promise.all([
    db.oneOrNone(
      "SELECT id, status, title, created_at, resolved_at FROM requests WHERE id = $1",
      [id]
    ),
    db.query(
      `SELECT id, action, actor, metadata, created_at
       FROM audit_log
       WHERE request_id = $1
       ORDER BY created_at DESC
       LIMIT 3`,
      [id]
    ),
    db.oneOrNone(
      `SELECT
         created_at,
         resolved_at,
         72 AS sla_target_hours,
         CASE WHEN COALESCE(resolved_at, NOW()) > created_at + (72 * INTERVAL '1 hour')
           THEN TRUE ELSE FALSE END AS breached
       FROM requests
       WHERE id = $1`,
      [id]
    ),
  ]);

  if (!request) {
    return res.status(404).json({ error: "not_found" });
  }

  // audit.last3 in ascending order for display
  const auditLast3 = auditLast3Rows.rows
    .map((r) => ({
      id: r.id,
      action: r.action,
      actor: r.actor,
      metadata: r.metadata,
      created_at: r.created_at,
    }))
    .reverse();

  return res.json({
    request,
    audit: {
      last3: auditLast3,
    },
    sla: {
      created_at: slaRow?.created_at ?? request.created_at,
      resolved_at: slaRow?.resolved_at ?? request.resolved_at ?? null,
      sla_target_hours: slaRow?.sla_target_hours ?? 72,
      breached: slaRow?.breached ?? false,
    },
  });
});

export default router;
```
