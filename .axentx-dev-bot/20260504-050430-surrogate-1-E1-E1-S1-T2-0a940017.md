import datetime
import math
from dataclasses import dataclass
from typing import List, Optional
import numpy as np
import yaml
import sqlite3

class CostRecord:
    """Represents a single cost measurement."""
    def __init__(self, resource_id, cost, timestamp):
        self.resource_id = resource_id
        self.cost = cost
        self.timestamp = timestamp

class AnomalyAlert:
    """Represents an anomaly alert to be stored and published."""
    def __init__(self, resource_id, cost_delta, severity, timestamp, message):
        self.resource_id = resource_id
        self.cost_delta = cost_delta
        self.severity = severity
        self.timestamp = timestamp
        self.message = message

def compute_rolling_stats(records: List[CostRecord]) -> Optional[tuple[float, float]]:
    """
    Compute the mean and standard deviation of the last 30 days of cost
    records for a single resource.

    Returns None if there are fewer than 2 records (cannot compute stddev).
    """
    if len(records) < 2:
        return None

    costs = [r.cost for r in records]
    mean = sum(costs) / len(costs)
    variance = sum((c - mean) ** 2 for c in costs) / (len(costs) - 1)
    stddev = math.sqrt(variance)
    return mean, stddev

def detect_anomalies(
    new_record: CostRecord,
    historical_records: List[CostRecord],
    sigma_threshold: float = 3.0,
) -> Optional[AnomalyAlert]:
    """
    Detect if the new_record represents a cost anomaly.

    An anomaly is flagged when the cost deviates > sigma_threshold * stddev
    from the rolling mean of the historical_records (last 30 days).
    """
    stats = compute_rolling_stats(historical_records)
    if stats is None:
        return None

    mean, stddev = stats
    deviation = new_record.cost - mean
    if abs(deviation) > sigma_threshold * stddev:
        severity = "high" if deviation > 0 else "medium"
        alert = AnomalyAlert(
            resource_id=new_record.resource_id,
            cost_delta=deviation,
            severity=severity,
            timestamp=new_record.timestamp,
            message=f"Cost anomaly detected for {new_record.resource_id}",
        )
        return alert
    return None

def store_alert(alert: AnomalyAlert) -> None:
    """
    Persist the alert to the database.

    The alerts table is assumed to have columns:
        id (auto), resource_id, cost_delta, severity, timestamp, message
    """
    conn = sqlite3.connect("alerts.db")
    cur = conn.cursor()
    cur.execute(
        """
        INSERT INTO alerts
        (resource_id, cost_delta, severity, timestamp, message)
        VALUES (?, ?, ?, ?, ?)
        """,
        (
            alert.resource_id,
            alert.cost_delta,
            alert.severity,
            alert.timestamp,
            alert.message,
        ),
    )
    conn.commit()

def publish_to_channel(alert: AnomalyAlert, channel: str = "Surrogate-1") -> None:
    """
    Publish the alert to the specified alerting channel.
    """
