package com.axentx.surrogate1.hardware;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class HardwareDetector {

    public static String detectHardwareConfiguration() {
        StringBuilder hardwareInfo = new StringBuilder();

        try {
            // Detect CPU information
            hardwareInfo.append("CPU Information:\n");
            hardwareInfo.append(executeCommand("cat /proc/cpuinfo"));

            // Detect memory information
            hardwareInfo.append("\nMemory Information:\n");
            hardwareInfo.append(executeCommand("cat /proc/meminfo"));

            // Detect GPU information if available
            hardwareInfo.append("\nGPU Information:\n");
            hardwareInfo.append(executeCommand("lspci | grep -i vga"));

            // Detect disk information
            hardwareInfo.append("\nDisk Information:\n");
            hardwareInfo.append(executeCommand("lsblk"));

        } catch (IOException | InterruptedException e) {
            e.printStackTrace();
        }

        return hardwareInfo.toString();
    }

    private static String executeCommand(String command) throws IOException, InterruptedException {
        Process process = Runtime.getRuntime().exec(command);
        BufferedReader reader = new BufferedReader(new InputStreamReader(process.getInputStream()));
        StringBuilder output = new StringBuilder();
        String line;

        while ((line = reader.readLine()) != null) {
            output.append(line).append("\n");
        }

        process.waitFor();
        return output.toString();
    }
}

// HardwareDetectorTest.java

@ExtendWith(MockitoExtension.class)
class HardwareDetectorTest {

    @Mock ProcessExecutor executor;

    @Test
    void detectCPU_returnsValidCpuInfo() throws Exception {
        // Arrange
        when(executor.execute("lscpu")).thenReturn("Model name: Intel Xeon\nCPU(s): 8\nThread(s) per core: 2\nCPU MHz: 2400");
        HardwareDetector detector = new HardwareDetector(executor);

        // Act
        CpuInfo result = detector.detectCPU();

        // Assert
        assertEquals("Intel Xeon", result.getModelName());
        assertEquals(8, result.getCoreCount());
        assertEquals(16, result.getThreadCount()); // 8 cores * 2 threads
        assertEquals(2400, result.getFrequencyMHz(), 50);
    }

    @Test
    void detectMemory_returnsValidMemoryInfo() throws Exception {
        // Arrange
        when(executor.execute("free", "-b")).thenReturn("Mem: 17179869184 8589934592 8579934592");
        HardwareDetector detector = new HardwareDetector(executor);

        // Act
        MemoryInfo result = detector.detectMemory();

        // Assert
        assertEquals(17179869184L, result.getTotalBytes());
        assertEquals(8589934592L, result.getAvailableBytes(), 100000000);
    }

    @Test
    void detectGPU_returnsEmpty_whenNvidiaSmiNotAvailable() {
        // Arrange
        when(executor.execute("nvidia-smi")).thenThrow(new IOException("Command not found"));
        HardwareDetector detector = new HardwareDetector(executor);

        // Act
        GpuInfo result = detector.detectGPU();

        // Assert
        assertNull(result.getName());
        assertTrue(result.getVramBytes() == null || result.getVramBytes() == 0);
    }

    @Test
    void detectDisk_parsesMultipleMountPoints() throws Exception {
        // Arrange
        String dfOutput = "/dev/sda1 ext4 500000000000 200000000000\n/dev/sdb1 xfs 1000000000000 800000000000";
        when(executor.execute("df", "-B1")).thenReturn(dfOutput);
        HardwareDetector detector = new HardwareDetector(executor);

        // Act
        List<DiskInfo> result = detector.detectDisk();

        // Assert
        assertEquals(2, result.size());
        assertEquals("ext4", result.get(0).getFilesystem());
        assertEquals(500000000000L, result.get(0).getTotalBytes());
    }

    @Test
    void detectCPU_handlesCommandFailure_gracefully() {
        // Arrange
        when(executor.execute("lscpu")).thenThrow(new RuntimeException("Command failed"));
        HardwareDetector detector = new HardwareDetector(executor);

        // Act & Assert
        assertThrows(HardwareDetectionException.class, () -> detector.detectCPU());
    }
}
