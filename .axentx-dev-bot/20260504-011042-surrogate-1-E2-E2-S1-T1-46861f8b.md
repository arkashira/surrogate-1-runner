"""
A minimal real‑time progress dashboard for the surrogate‑1 project.

Features
--------
* FastAPI based HTTP + WebSocket server.
* In‑memory task store with status, deadline, milestone.
* Background task that simulates progress updates.
* Simple HTML page that connects to the WebSocket and renders
  the dashboard in the browser.

Usage
-----
Run the server with:

    uvicorn progress_dashboard:app --host 0.0.0.0 --port 8000

The dashboard is available at http://localhost:8000/dashboard.
"""

import asyncio
import datetime
import json
import uuid
from typing import Dict, List

from fastapi import FastAPI, WebSocket, WebSocketDisconnect, Request
from fastapi.responses import HTMLResponse
from fastapi.staticfiles import StaticFiles
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI(title="Surrogate‑1 Progress Dashboard")

# Allow CORS for local dev
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# Mount static files for the simple dashboard page
app.mount("/static", StaticFiles(directory="static"), name="static")

# --------------------------------------------------------------------------- #
# In‑memory task store
# --------------------------------------------------------------------------- #
class Task:
    def __init__(self, name: str, deadline: datetime.datetime, milestone: str):
        self.id = str(uuid.uuid4())
        self.name = name
        self.status = "Pending"
        self.deadline = deadline
        self.milestone = milestone

    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "name": self.name,
            "status": self.status,
            "deadline": self.deadline.isoformat(),
            "milestone": self.milestone,
        }


# Seed with a few example tasks
TASKS: Dict[str, Task] = {
    t.id: t
    for t in [
        Task(
            "Data ingestion",
            datetime.datetime.utcnow() + datetime.timedelta(hours=4),
            "Phase 1",
        ),
        Task(
            "Model training",
            datetime.datetime.utcnow() + datetime.timedelta(days=1),
            "Phase 2",
        ),
        Task(
            "Evaluation & QA",
            datetime.datetime.utcnow() + datetime.timedelta(days=2),
            "Phase 3",
        ),
    ]
}


# --------------------------------------------------------------------------- #
# WebSocket manager
# --------------------------------------------------------------------------- #
class ConnectionManager:
    def __init__(self):
        self.active_connections: List[WebSocket] = []

    async def connect(self, websocket: WebSocket):
        await websocket.accept()
        self.active_connections.append(websocket)

    def disconnect(self, websocket: WebSocket):
        self.active_connections.remove(websocket)

    async def broadcast(self, message: str):
        for connection in list(self.active_connections):
            try:
                await connection.send_text(message)
            except Exception:
                # Remove broken connections
                self.disconnect(connection)


manager = ConnectionManager()


# --------------------------------------------------------------------------- #
# Background task to simulate progress updates
# --------------------------------------------------------------------------- #

    """Periodically change task status and broadcast updates."""
    while True:
        await asyncio.sleep(5)  # every 5 seconds
        for task in TASKS.values():
            if task.status == "Pending":
                task.status = "In Progress"
                break
            elif task.status == "In Progress":
                task.status = "Completed"
                break
        await manager.broadcast(json.dumps([t.to_dict() for t in TASKS.values()]))
