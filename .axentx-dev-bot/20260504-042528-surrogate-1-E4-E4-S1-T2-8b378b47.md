import io.prometheus.client.Counter;
import io.prometheus.client.Histogram;

public class Metrics {
    private static final Counter apiRequestsTotal = Counter.build()
            .namespace("rat")
            .name("api_requests_total")
            .labelNames("endpoint")
            .help("Total number of API requests")
            .register();

    private static final Histogram apiLatencySeconds = Histogram.build()
            .namespace("rat")
            .name("api_latency_seconds")
            .labelNames("endpoint")
            .help("API latency in seconds")
            .register();

    private static final Counter workerSuccessTotal = Counter.build()
            .namespace("rat")
            .name("worker_success_total")
            .labelNames("worker")
            .help("Total number of successful worker runs")
            .register();

    private static final Counter workerFailureTotal = Counter.build()
            .namespace("rat")
            .name("worker_failure_total")
            .labelNames("worker")
            .help("Total number of failed worker runs")
            .register();

    // Add methods to increment/decrement and observe these metrics
}
