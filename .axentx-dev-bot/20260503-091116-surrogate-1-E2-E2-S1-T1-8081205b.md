from __future__ import annotations

from typing import List
import numpy as np
from sentence_transformers import SentenceTransformer

_MODEL_NAME = "sentence-transformers/all-MiniLM-L6-v2"
_model: SentenceTransformer | None = None


def _get_model() -> SentenceTransformer:
    global _model
    if _model is None:
        _model = SentenceTransformer(_MODEL_NAME)
    return _model


def encode(texts: List[str], normalize: bool = True) -> np.ndarray:
    """
    Encode texts into embeddings.

    Args:
        texts: List of strings.
        normalize: If True, return L2-normalized embeddings (cosine-ready).

    Returns:
        np.ndarray of shape (len(texts), dim)
    """
    if not texts:
        return np.empty((0, 384), dtype=np.float32)

    model = _get_model()
    embeddings = model.encode(texts, convert_to_numpy=True, show_progress_bar=False)
    embeddings = embeddings.astype(np.float32, copy=False)

    if normalize:
        norms = np.linalg.norm(embeddings, axis=1, keepdims=True)
        norms = np.where(norms == 0, 1.0, norms)
        embeddings = embeddings / norms
    return embeddings


def encode_query(query: str) -> np.ndarray:
    """Encode a single query string (normalized)."""
    return encode([query], normalize=True)[0]
```
