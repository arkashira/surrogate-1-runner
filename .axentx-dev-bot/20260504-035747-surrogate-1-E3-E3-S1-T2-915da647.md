import com.axentx.surrogate.model.AuditTrailEntry;
import com.axentx.surrogate.repository.AuditTrailRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class AuditTrailService {

    @Autowired
    private AuditTrailRepository auditTrailRepository;

    public void logAction(String user, String action) {
        AuditTrailEntry entry = new AuditTrailEntry();
        entry.setUser(user);
        entry.setAction(action);
        entry.setTimestamp(new java.util.Date());
        auditTrailRepository.save(entry);
    }

    public List<AuditTrailEntry> getAuditTrail() {
        return auditTrailRepository.findAll();
    }

    public List<AuditTrailEntry> searchAuditTrail(String keyword) {
        return auditTrailRepository.findByUserContainingIgnoreCaseOrActionContainingIgnoreCase(keyword, keyword);
    }
}
