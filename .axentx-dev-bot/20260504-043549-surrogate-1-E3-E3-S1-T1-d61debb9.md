import os
import json
from datetime import datetime

class CommandLogger:
    def __init__(self, log_dir='/audit_logs'):
        self.log_dir = log_dir
        os.makedirs(self.log_dir, exist_ok=True)

    def log_command(self, command, user_identity, session_metadata):
        timestamp = datetime.now().isoformat()
        log_entry = {
            'timestamp': timestamp,
            'command': command,
            'user_identity': user_identity,
            'session_metadata': session_metadata
        }
        log_file = os.path.join(self.log_dir, f"audit_log_{timestamp.replace(':', '-')}.json")
        with open(log_file, 'w') as f:
            json.dump(log_entry, f)

def test_logger_initialization(tmp_path):

    assert logger.log_file == tmp_path / "audit_logs/terminal_commands.log"
    assert logger.log_file.parent.exists()

def test_log_entry_structure(tmp_path):
    logger = CommandLogger(log_dir=tmp_path / "audit_logs")
    logger.log_command(user="alice", session_id="sess123", command="ls", args=["-l"], exit_code=0)
    with open(logger.log_file) as f:
        line = f.readline()
    record = json.loads(line)
    assert set(record.keys()) == {"timestamp","user","session_id","command","args","exit_code"}
    assert record["user"] == "alice"
    assert record["session_id"] == "sess123"
    assert record["command"] == "ls"
    assert record["args"] == ["-l"]
    assert record["exit_code"] == 0
    # timestamp format check
    ts = record["timestamp"]
    # simple ISO8601 check
    assert ts.endswith("Z")
    assert "." in ts

def test_concurrent_logging(tmp_path):
    logger = CommandLogger(log_dir=tmp_path / "audit_logs")
    # spawn 10 threads writing logs
    def worker(i):
        logger.log_command(user=f"user{i}", session_id=f"s{i}", command="echo", args=[str(i)], exit_code=0)
    threads = [threading.Thread(target=worker, args=(i,)) for i in range(10)]
    for t in threads: t.start()
    for t in threads: t.join()
    # verify 10 lines
    with open(logger.log_file) as f:
        lines = f.readlines()
    assert len(lines) == 10
