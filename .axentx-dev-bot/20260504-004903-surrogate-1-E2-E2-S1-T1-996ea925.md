Utility to perform a canary traffic shift on an Istio VirtualService.

The module exposes a single public function :func:`shift_traffic` that
updates the VirtualService to route a percentage of traffic to a new
subset while keeping the rest on the old subset.  The implementation

``kubectl`` is available and the user has the appropriate permissions.

The code is intentionally lightweight – it does **not** depend on the

dependency footprint small and makes the module easy to ship in
container images.
"""

import subprocess
import sys

from typing import Dict, List, Tuple

    """
    Execute a ``kubectl`` command and return ``(returncode, stdout, stderr)``.
    """
    try:
        result = subprocess.run(
            command,
            capture_output=True,
            text=True,
            timeout=timeout,
            check=False,
        )
        return result.returncode, result.stdout.strip(), result.stderr.strip()
    except subprocess.TimeoutExpired as exc:  # pragma: no cover
        return -1, "", f"kubectl command timed out after {timeout}s: {exc}"


def _load_virtual_service(name: str, namespace: str = "default") -> Dict:
    """
    Retrieve the current VirtualService as a Python dict.
    """
    rc, out, err = _run_kubectl(
        ["kubectl", "get", "virtualservice", name, "-n", namespace, "-o", "json"]

    if rc != 0:
