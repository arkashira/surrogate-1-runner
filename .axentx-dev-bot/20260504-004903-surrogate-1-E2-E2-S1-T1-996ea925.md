        result = subprocess.run(
            command,
            capture_output=True,
            text=True,
            timeout=timeout,
            check=False,
        )

    Retrieve the current VirtualService as a Python dict.

import unittest
from unittest.mock import patch, MagicMock
from shift_traffic import shift_traffic, _update_virtual_service, _get_current_weights

class TestShiftTraffic(unittest.TestCase):

    @patch('shift_traffic.subprocess.run')
    def test_shift_traffic_success(self, mock_run):
        mock_run.return_value = MagicMock(returncode=0)
        result = shift_traffic('old', 'new', 10)
        self.assertEqual(result, "Traffic shift successful")

    @patch('shift_traffic.subprocess.run')
    def test_shift_traffic_failure(self, mock_run):
        mock_run.return_value = MagicMock(returncode=1)
        result = shift_traffic('old', 'new', 10)
        self.assertEqual(result, "Error: Traffic shift failed")

    def test_update_virtual_service(self):
        mock_vs = MagicMock()
        _update_virtual_service(mock_vs, 'old', 'new', 10)
        self.assertEqual(mock_vs.spec.http[0].route[0].weight, 90)
        self.assertEqual(mock_vs.spec.http[0].route[1].weight, 10)

    def test_get_current_weights(self):
        mock_vs = MagicMock()
        mock_vs.spec.http[0].route = [
            MagicMock(subset='old', weight=80),
            MagicMock(subset='new', weight=20)
        ]
        old_weight, new_weight = _get_current_weights(mock_vs, 'old', 'new')
        self.assertEqual(old_weight, 80)
        self.assertEqual(new_weight, 20)

if __name__ == '__main__':
    unittest.main()
