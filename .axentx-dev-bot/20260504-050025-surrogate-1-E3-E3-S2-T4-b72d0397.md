"""
Canonical LLM response object and provider contract.
"""

from __future__ import annotations

import json
from dataclasses import dataclass, field
from typing import Dict, Iterable, List, Protocol


# --------------------------------------------------------------------------- #
# 1.  LLMResponse – the single source of truth for all provider responses
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class LLMResponse:
    """
    Immutable, type‑checked representation of an LLM response.

    Attributes
    ----------
    text : str
        The generated text.
    usage : Dict[str, int]
        Token‑usage statistics (e.g. prompt_tokens, completion_tokens, total_tokens).
    finish_reason : str
        Reason the model stopped generating (e.g. 'stop', 'length', 'content_filter').
    """

    text: str
    usage: Dict[str, int] = field(default_factory=dict)
    finish_reason: str = ""

    # ----------------------------------------------------------------------- #
    # Validation helpers – run once at construction time
    # ----------------------------------------------------------------------- #
    def __post_init__(self):
        if not isinstance(self.text, str):
            raise TypeError(f"text must be str, got {type(self.text).__name__}")
        if not isinstance(self.usage, dict):
            raise TypeError(f"usage must be dict, got {type(self.usage).__name__}")
        if not isinstance(self.finish_reason, str):
            raise TypeError(f"finish_reason must be str, got {type(self.finish_reason).__name__}")

    # ----------------------------------------------------------------------- #
    # Convenience methods
    # ----------------------------------------------------------------------- #
    def to_dict(self) -> Dict[str, object]:
        """Return a plain‑dict representation suitable for JSON serialisation."""
        return {"text": self.text, "usage": self.usage, "finish_reason": self.finish_reason}

    def to_json(self) -> str:
        """Return a JSON string representation."""
        return json.dumps(self.to_dict())

    # ----------------------------------------------------------------------- #
    # Equality & representation – useful for tests and debugging
    # ----------------------------------------------------------------------- #
    def __eq__(self, other: object) -> bool:
        if not isinstance(other, LLMResponse):
            return NotImplemented
        return (
            self.text == other.text
            and self.usage == other.usage
            and self.finish_reason == other.finish_reason
        )

    def __repr__(self) -> str:
        return (
            f"LLMResponse(text={self.text!r}, usage={self.usage!r}, "
            f"finish_reason={self.finish_reason!r})"
        )
