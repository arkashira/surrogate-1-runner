const fs = require('fs').promises;
const path = require('path');
const { spawn } = require('child_process');

class AlertEngine {
    constructor(config = {}) {
        this.threshold = config.threshold || 0.05; // Default 5% drift threshold
        this.logDir = config.logDir || '/opt/axentx/surrogate-1/drift-logs';
        this.notifierPath = config.notifierPath || './slack-notifier.py';
        this.timeoutMs = config.timeoutMs || 100; // Max time to generate alert
    }

    /**
     * Generates an alert payload with upstream correlation.
     * @param {Object} driftData - { upstreamService, endpoint, driftPercentage, currentValue, expectedValue }
     * @returns {Object}
     */
    generateAlert(driftData) {
        return {
            timestamp: new Date().toISOString(),
            upstreamService: driftData.upstreamService,
            endpoint: driftData.endpoint,
            driftPercentage: driftData.driftPercentage,
            currentValue: driftData.currentValue,
            expectedValue: driftData.expectedValue
        };
    }

    /**
     * Logs drift data to disk asynchronously.
     * @param {Object} alertPayload
     * @returns {Promise<void>}
     */
    async logToFile(alertPayload) {
        const logFile = path.join(this.logDir, 'drift.log');
        try {
            await fs.appendFile(logFile, JSON.stringify(alertPayload) + '\n');
        } catch (err) {
            console.error('Failed to write to log file:', err);
        }
    }

    /**
     * Triggers external Slack notification using Python script.
     * @param {Object} alertPayload
     * @returns {Promise<void>}
     */
    async notifySlack(alertPayload) {
        return new Promise((resolve, reject) => {
            const child = spawn('python3', [this.notifierPath], {
                stdio: ['pipe', 'inherit', 'inherit']
            });

            child.stdin.write(JSON.stringify(alertPayload));
            child.stdin.end();

            child.on('close', (code) => {
                if (code === 0) {
                    resolve();
                } else {
                    reject(new Error(`Slack notifier exited with code ${code}`));
                }
            });
        });
    }

    /**
     * Main entry point to detect drift and trigger alerts.
     * @param {Object} driftData - { upstreamService, endpoint, driftPercentage, currentValue, expectedValue }
     * @returns {Promise<boolean>} - True if alert was triggered.
     */
    async detectAndAlert(driftData) {
        // Threshold check
        if (driftData.driftPercentage <= this.threshold) {
            return false;
        }

        const alertPayload = this.generateAlert(driftData);

        // Asynchronous tasks to avoid blocking
        const logTask = this.logToFile(alertPayload);
        const notifyTask = this.notifySlack(alertPayload);

        // Ensure we don't exceed 100ms total delay
        const timeoutPromise = new Promise(resolve => setTimeout(resolve, this.timeoutMs));

        try {
            await Promise.race([logTask, notifyTask, timeoutPromise]);
            return true;
        } catch (error) {
            console.error("Error during alert processing:", error.message);
            return false;
        }
    }
}

module.exports = AlertEngine;
```
