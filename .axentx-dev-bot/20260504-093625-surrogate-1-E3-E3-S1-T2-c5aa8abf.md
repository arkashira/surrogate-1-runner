"""
Error handling utilities for the surrogate-1 parser.

This module defines a lightweight exception hierarchy, a decorator for
wrapping parsing functions, and a set of common recovery strategies.
The goal is to provide clear, actionable error messages while keeping
runtime overhead minimal.

Key concepts
-------------
* :class:`ParserError` – Base exception for all parser errors.
* :class:`RecoverableError` – Subclass that carries a suggested recovery
  action.
* :func:`handle_error` – Decorator that catches parser errors, logs a
  concise message, and returns a recovery action.
* Recovery actions – Enumerated via :class:`RecoveryAction` and
  implemented as simple callables.

The module is intentionally self‑contained so it can be imported by any
parser implementation without pulling in heavy dependencies.
"""

import logging
from enum import Enum, auto
from functools import wraps
from typing import Callable, Any, Tuple, Optional

# Configure a module‑level logger.  The caller may override the level.
logger = logging.getLogger(__name__)
if not logger.handlers:
    handler = logging.StreamHandler()
    formatter = logging.Formatter(
        "[%(levelname)s] %(name)s:%(lineno)d - %(message)s"
    )
    handler.setFormatter(formatter)
    logger.addHandler(handler)
    logger.setLevel(logging.INFO)


class RecoveryAction(Enum):
    """
    Enumerated recovery actions that a parser can suggest when a recoverable
    error occurs.

    * ``SKIP`` – Skip the offending record and continue.
    * ``RETRY`` – Retry the operation (typically used for transient I/O).
    * ``USE_DEFAULT`` – Substitute a default value and continue.
    * ``ABORT`` – Stop parsing entirely.
    """

    SKIP = auto()
    RETRY = auto()
    USE_DEFAULT = auto()
    ABORT = auto()


class ParserError(Exception):
    """
    Base exception for all parser errors.

    Parameters
    ----------
    message : str
        Human‑readable description of the error.
    """

    def __init__(self, message: str) -> None:
        super().__init__(message)
        self.message = message

    def __str__(self) -> str:
        return self.message


class RecoverableError(ParserError):
    """
    Exception that carries a recovery action.

    Parameters
    ----------
    message : str
        Human‑readable description of the error.
    action : :class:`RecoveryAction`
        Suggested recovery action.
    """

    def __init__(self, message: str, action: RecoveryAction) -> None:
        super().__init__(message)
        self.action = action

    def __str__(self) -> str:
        return f"{self.message} (Recovery: {self.action.name})"


def handle_error(
    *,
    recoverable_actions: Tuple[RecoveryAction, ...] = (
        RecoveryAction.SKIP,
        RecoveryAction.RETRY,
        RecoveryAction.USE_DEFAULT,
    ),
) -> Callable[[Callable[..., Any]], Callable[..., Any]]:
    """
    Decorator that wraps a parsing function to provide consistent error
    handling.

    The wrapped function should raise :class:`ParserError` or
    :class:`RecoverableError`.  The decorator logs a concise message and
    returns a tuple ``(success: bool, recovery: Optional[RecoveryAction])``.
    ``success`` is ``False`` when an error was caught.  ``recovery`` is
    ``None`` for unrecoverable errors.

    Parameters
    ----------
    recoverable_actions : tuple
        Actions that the decorator will log for recoverable errors.
    """

    def decorator(func: Callable[..., Any]) -> Callable[..., Any]:
        @wraps(func)
        def wrapper(*args, **kwargs) -> Tuple[bool, Optional[RecoveryAction]]:
            try:
                result = func(*args, **kwargs)
                return True, None
            except RecoverableError as e:
                if e.action in recoverable_actions:
                    logger.warning(f"{e}")
                    return False, e.action
                else:
                    logger.error
