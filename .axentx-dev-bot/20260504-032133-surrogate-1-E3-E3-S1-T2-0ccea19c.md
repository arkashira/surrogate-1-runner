def test_timeout_raises_error():
    with Sandbox() as sb:
        with pytest.raises(CommandExecutionError) as exc:
            sb.run(["sleep", "10"], timeout=1)
    assert exc.value.error_type == "timeout"

def test_env_isolation():
    os.environ["LEAK"] = "parent"
    with Sandbox() as sb:

    # parent env unchanged
    assert os.getenv("FOO") is None
    assert os.getenv("LEAK") == "parent"

    with Sandbox() as sb:
