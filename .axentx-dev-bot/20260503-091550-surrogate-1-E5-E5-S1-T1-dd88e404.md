// src/models/request.ts
import { DateTime } from 'luxon';

const TERMINAL_STATUSES = new Set(['closed', 'resolved', 'cancelled', 'rejected']);

export interface RequestAttributes {
  id: string;
  teamId: string;
  currentStatus: string;
  dueAt: Date | null;
  createdAt: Date;
  updatedAt: Date;
  isInternal: boolean;
}

export class Request {
  id: string;
  teamId: string;
  currentStatus: string;
  dueAt: Date | null;
  createdAt: Date;
  updatedAt: Date;
  isInternal: boolean;

  constructor(data: RequestAttributes) {
    this.id = data.id;
    this.teamId = data.teamId;
    this.currentStatus = data.currentStatus;
    this.dueAt = data.dueAt;
    this.createdAt = data.createdAt;
    this.updatedAt = data.updatedAt;
    this.isInternal = data.isInternal;
  }

  /**
   * SLA breach: request is past due and not in a terminal state.
   * - No due date => no SLA target => no breach.
   * - Terminal status => already closed/resolved => no breach.
   * - dueAt in the past (strictly before today) => breach.
   */
  static isSlaBreach(req: Pick<Request, 'currentStatus' | 'dueAt'>): boolean {
    if (!req.dueAt) return false;
    const status = String(req.currentStatus || '').toLowerCase();
    if (TERMINAL_STATUSES.has(status)) return false;
    return DateTime.fromJSDate(req.dueAt).endOf('day') < DateTime.local().startOf('day');
  }

  static isTerminal(req: Pick<Request, 'currentStatus'>): boolean {
    return TERMINAL_STATUSES.has(String(req.currentStatus || '').toLowerCase());
  }

  static isTracked(req: Pick<Request, 'isInternal'>): boolean {
    return !req.isInternal;
  }

  static isActiveLastNDays(req: Pick<Request, 'createdAt' | 'updatedAt'>, nDays: number): boolean {
    const cutoff = DateTime.local().minus({ days: nDays }).toJSDate();
    return req.createdAt >= cutoff || req.updatedAt >= cutoff;
  }
}
