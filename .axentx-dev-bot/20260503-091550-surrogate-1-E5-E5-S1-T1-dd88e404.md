export interface RequestAttributes {
  id: string;
  team_id: string;
  current_status: string;
  sla_target_at: Date | null;
  created_at: Date;
  updated_at: Date;
  // internal-only fields (must not be exposed)
  internal_notes?: string;
  raw_payload?: unknown;
}

export class Request {
  static tableName = 'requests';

  static terminalStatuses = ['closed', 'resolved', 'cancelled', 'done'] as const;

  static isTerminalStatus(status: string) {
    return this.terminalStatuses.includes(status.toLowerCase() as any);
  }

  // Public-safe projection
  static toPublicJSON(record: RequestAttributes) {
    const { internal_notes, raw_payload, ...publicFields } = record;
    return publicFields;
  }

  // Core dashboard query: WAU, open count, SLA breach counts (7d/30d)
  static async getTeamDashboard(knex: Knex, teamId: string) {
    const now = new Date();
    const sevenDaysAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
    const thirtyDaysAgo = new Date(now.getTime() - 30 * 24 * 60 * 60 * 1000);

    // WAU: distinct requests created in last 7 days for this team
    const wauResult = await knex<{ count: string | number }>(this.tableName)
      .where('team_id', teamId)
      .whereBetween('created_at', [sevenDaysAgo, now])
      .countDistinct('id as count')
      .first();
    const wauRequests = Number(wauResult?.count || 0);

    // Open requests (team-scoped, non-terminal)
    const openResult = await knex<{ count: string | number }>(this.tableName)
      .where('team_id', teamId)
      .whereNotIn('current_status', this.terminalStatuses)
      .count('id as count')
      .first();
    const openRequests = Number(openResult?.count || 0);

    // SLA breach: non-terminal AND past SLA target (sla_target_at < now)
    // Count only requests created within the window (windowed breach counts)
    const breachBase = (from: Date, to: Date) =>
      knex<{ count: string | number }>(this.tableName)
        .where('team_id', teamId)
        .whereBetween('created_at', [from, to])
        .whereNotNull('sla_target_at')
        .where('sla_target_at', '<', now)
        .whereNotIn('current_status', this.terminalStatuses);

    const breach7dResult = await breachBase(sevenDaysAgo, now).count('id as count').first();
    const breach30dResult = await breachBase(thirtyDaysAgo, now).count('id as count').first();

    const slaBreachCount = {
      last_7_days: Number(breach7dResult?.count || 0),
      last_30_days: Number(breach30dResult?.count || 0),
    };

    return {
      team_id: teamId,
      wau_requests: wauRequests,
      open_requests: openRequests,
      sla_breach_count: slaBreachCount,
      computed_at: now.toISOString(),
    };
  }
}
