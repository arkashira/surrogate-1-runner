#### ValidationResult Interface

The `ValidationResult` interface will be used to represent a single validation check result. This interface will include the following properties:

* `id`: a unique identifier for the validation result
* `status`: the status of the validation check, which can be either 'pass' or 'fail'
* `reason`: an optional property that explains why a check failed
* `fix`: an optional property that provides a suggested fix for the failure
* `ruleLink`: an optional property that points to the full validation rule documentation
* `guidelineRefs`: an optional property that lists Apple guideline references relevant to the rule

```typescript
interface ValidationResult {
  id: string;
  status: 'pass' | 'fail';
  reason?: string;
  fix?: string;
  ruleLink?: string;
  guidelineRefs?: string[];
}
```

#### ValidationDetails Component

The `ValidationDetails` component will render a detailed view of validation results. This component will accept an array of `ValidationResult` objects as a prop.

```tsx
import React, { useState } from 'react';

interface ValidationDetailsProps {
  results: ValidationResult[];
}

const ValidationDetails: React.FC<ValidationDetailsProps> = ({ results }) => {
  const [showRules, setShowRules] = useState<boolean>(false);

  const toggleRules = () => setShowRules((prev) => !prev);

  return (
    <div style={{ fontFamily: 'Arial, sans-serif', lineHeight: 1.5 }}>
      <h2>Validation Details</h2>

      {results.length === 0 && <p>No validation results to display.</p>}

      {results.map((result) => (
        <div
          key={result.id}
          style={{
            border: '1px solid #ddd',
            borderRadius: '4px',
            padding: '12px',
            marginBottom: '12px',
            backgroundColor: result.status === 'pass' ? '#e6ffed' : '#ffe6e6',
          }}
        >
          <div style={{ display: 'flex', justifyContent: 'space-between' }}>
            <strong>{result.status === 'pass' ? '✅ Passed' : '❌ Failed'}</strong>
            <button
              onClick={toggleRules}
              style={{
                background: 'none',
                border: 'none',
                color: '#0066cc',
                cursor: 'pointer',
                textDecoration: 'underline',
                padding: 0,
              }}
            >
              {showRules ? 'Hide Rules' : 'Show Rules'}
            </button>
          </div>

          {result.status === 'fail' && (
            <>
              {result.reason && (
                <p style={{ margin: '8px 0' }}>
                  <strong>Reason:</strong> {result.reason}
                </p>
              )}
              {result.fix && (
                <p style={{ margin: '8px 0' }}>
                  <strong>Suggested Fix:</strong> {result.fix}
                </p>
              )}
              {result.guidelineRefs && result.guidelineRefs.length > 0 && (
                <p style={{ margin: '8px 0' }}>
                  <strong>Apple Guidelines:</strong>{' '}
                  {result.guidelineRefs.map((ref, idx) => (
                    <span key={idx}>
                      {ref}
                      {idx < result.guidelineRefs!.length - 1 ? ', ' : ''}
                    </span>
                  ))}
                </p>
              )}
            </>
          )}

          {showRules && result.ruleLink && (
            <p style={{ marginTop: '8px' }}>
              <a href={result.ruleLink} target="_blank" rel="noopener noreferrer">
                View Full Validation Rule
              </a>
            </p>
          )}
        </div>
      ))}
    </div>
  );
};

export default ValidationDetails;
```
