import nodemailer from 'nodemailer';
import { format } from 'date-fns';

export interface DigestItem {
  title: string;
  status: string;
  timeInStatus: string;
  publicUrl: string;
}

export interface DigestPayload {
  ownerEmail: string;
  teamName?: string;
  runId: string;
  generatedAt: Date;
  items: DigestItem[];
}

const makeTransporter = () =>
  nodemailer.createTransport({
    host: process.env.SMTP_HOST || 'localhost',
    port: Number(process.env.SMTP_PORT || 587),
    secure: false,
    auth: process.env.SMTP_USER
      ? { user: process.env.SMTP_USER, pass: process.env.SMTP_PASSWORD }
      : undefined,
  });

export async function sendDigestEmail(payload: DigestPayload) {
  const { ownerEmail, teamName, runId, generatedAt, items } = payload;
  const transporter = makeTransporter();

  const subject = `Daily digest: ${items.length} at-risk request(s) for ${teamName || 'your team'}`;
  const dateStr = format(generatedAt, 'yyyy-MM-dd HH:mm:ss');

  const rows = items
    .map(
      (it) => `
      <tr>
        <td style="padding:8px;border-bottom:1px solid #eee">${it.title}</td>
        <td style="padding:8px;border-bottom:1px solid #eee">${it.status}</td>
        <td style="padding:8px;border-bottom:1px solid #eee">${it.timeInStatus}</td>
        <td style="padding:8px;border-bottom:1px solid #eee"><a href="${it.publicUrl}">View</a></td>
      </tr>`
    )
    .join('');

  const html = `
    <p>Daily digest run <code>${runId}</code> generated at ${dateStr}.</p>
    <table style="border-collapse:collapse;width:100%;max-width:600px">
      <thead>
        <tr>
          <th style="text-align:left;padding:8px;border-bottom:2px solid #333">Title</th>
          <th style="text-align:left;padding:8px;border-bottom:2px solid #333">Status</th>
          <th style="text-align:left;padding:8px;border-bottom:2px solid #333">Time in status</th>
          <th style="text-align:left;padding:8px;border-bottom:2px solid #333">URL</th>
        </tr>
      </thead>
      <tbody>
        ${rows || '<tr><td colspan="4" style="padding:16px;text-align:center">No at-risk requests</td></tr>'}
      </tbody>
    </table>
    <p style="color:#666;font-size:12px">Run ID: ${runId}</p>
  `;

  const info = await transporter.sendMail({
    from: process.env.EMAIL_FROM || 'no-reply@axentx.io',
    to: ownerEmail,
    subject,
    html,
    headers: { 'X-Run-Id': runId },
  });

  return { messageId: info.messageId };
}
