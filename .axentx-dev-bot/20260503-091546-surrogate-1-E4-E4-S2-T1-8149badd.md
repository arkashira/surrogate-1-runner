import { CronJob } from 'cron';
import { RequestService } from '../services/requestService';
import { EmailService } from '../services/emailService';
import { Logger } from '../utils/logger';

const requestService = new RequestService();
const emailService = new EmailService();
const logger = new Logger();

const dailyDigestJob = new CronJob('0 0 * * *', async () => {
  try {
    const atRiskRequests = await requestService.getAtRiskRequests();
    const teamOwners = await requestService.getTeamOwners();

    const digests = atRiskRequests.map((request) => {
      const teamOwner = teamOwners.find((owner) => owner.teamId === request.teamId);
      return {
        title: request.title,
        status: request.status,
        timeInStatus: request.timeInStatus,
        publicUrl: request.publicUrl,
        teamOwner: teamOwner.email,
      };
    });

    await emailService.sendDailyDigest(digests);
    logger.info('Daily digest sent successfully');
  } catch (error) {
    logger.error('Error sending daily digest:', error);
  }
});

dailyDigestJob.start();
```
