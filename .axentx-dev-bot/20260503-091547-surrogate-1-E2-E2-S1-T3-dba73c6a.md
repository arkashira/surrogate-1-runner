import { Request } from '../models/request';

// ...

export async function patchRequestStatus(req: Request, res: Response) {
  const requestId = req.params.id;
  const request = await Request.findById(requestId);
  if (!request) {
    return res.status(404).send({ error: 'Request not found' });
  }

  const updatedRequest = await Request.findByIdAndUpdate(
    requestId,
    {
      $set: {
        status: req.body.status,
        notes: req.body.notes,
        updated_at: new Date(),
      },
    },
    { new: true }
  );

  // If the updated_at field is newer than the existing one, return the updated request
  if (updatedRequest.updated_at > request.updated_at) {
    return res.send(updatedRequest);
  } else {
    return res.status(409).send({ error: 'Conflict: concurrent update detected' });
  }
}
```

**`/opt/axentx/surrogate-1/src/routes/public/requestPage.ts`** (update the route to fetch audit trail data along with request status)
```diff
import { Request, Response } from 'express';
import { RequestModel } from '../../models/RequestModel';

export const getRequestStatusPage = async (req: Request, res: Response) => {
  try {
    const { id } = req.params;
    const request = await RequestModel.findById(id);
    
    if (!request) {
      return res.status(404).render('error', { message: 'Request not found' });
    }

    // Ensure audit trail is available in the response
    const statusData = {
      id: request.id,
      status: request.status,
      updatedAt: request.updatedAt,
      auditTrail: (request.auditTrail || []).map(entry => ({
        status: entry.status,
        timestamp: entry.timestamp,
        actor: entry.actor,
        notes: entry.notes || null
      })).reverse() // Most recent first
    };

    return res.render('requestStatus', statusData);
  } catch (error) {
    console.error('Error fetching request status:', error);
    return res.status(500).render('error', { message: 'Internal server error' });
  }
};
```
