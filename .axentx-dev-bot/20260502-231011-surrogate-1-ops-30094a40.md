python3 - "$DATASET_REPO" "$DATE_FOLDER" "$OUT_JSON" <<'PY'
import json, os, sys, time
from huggingface_hub import HfApi

repo, date_folder = sys.argv[1], sys.argv[2]
out_path = sys.argv[3]

api = HfApi()
# One snapshot call; handle 429 with simple retry
for attempt in range(3):
    try:
        tree = api.list_repo_tree(repo_id=repo, path=date_folder, recursive=False)
        break
    except Exception as e:
        if attempt == 2:
            raise
        time.sleep(60 * (attempt + 1))

files = [
    {
        "path": f.path,
        "size": f.size or 0,
        "sha": (f.lfs or {}).get("oid", "") if f.lfs else ""
    }
    for f in tree
]

snapshot = {"date": date_folder, "files": files, "ts": time.time()}
with open(out_path, "w") as f:
    json.dump(snapshot, f)
print(f"Snapshot saved: {out_path} ({len(files)} files)")
PY
