def lazy_read_lines(
    path: str,
    *,
    encoding: str = "utf-8",
    errors: str = "replace",
    chunk_size: int = 1024 * 1024,
) -> Generator[str, None, None]:
    """
    Yield lines from ``path`` lazily.

    Parameters
    ----------
    path: str
        Path to the file to read.
    encoding: str, default ``'utf-8'``
        Text encoding used to decode bytes.
    errors: str, default ``'replace'``
        Error handling scheme for decoding.
    chunk_size: int, default 1 MiB
        Number of bytes to read per iteration.

    Yields
    ------
    str
        The next line from the file, without the trailing newline.
    """
    if not os.path.isfile(path):
        raise FileNotFoundError(path)

    with open(path, "rb") as f:
        leftover: bytes = b""
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                # End of file – flush any remaining data.
                if leftover:
                    yield leftover.decode(encoding, errors)
                break

            buffer = leftover + chunk
            lines = buffer.split(b"\n")
            for line in lines[:-1]:
                yield line.decode(encoding, errors)
            leftover = lines[-1]
```
