import React, { useEffect, useState } from 'react';
import { Form, Input, Switch, Button, message } from 'antd';
import axios from 'axios';

const IntegrationConfig = () => {
  const [form] = Form.useForm();
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [success, setSuccess] = useState(false);

  useEffect(() => {
    fetchConfig();
  }, []);

  const fetchConfig = async () => {
    try {
      const response = await axios.get('/api/integration-config');
      form.setFieldsValue(response.data);
      setLoading(false);
    } catch (err) {
      setError('Failed to load config');
      setLoading(false);
    }
  };

  const onFinish = async (values) => {
    try {
      await axios.post('/api/integration-config', values);
      setSuccess(true);
      message.success('Configuration saved successfully.');
      fetchConfig();
    } catch (err) {
      setError('Failed to save config');
    }
  };

  if (loading) return <p>Loading integration configuration...</p>;

  return (
    <div>
      <h1>AI Compliance Manager Integration</h1>
      {error && <p style={{ color: 'red' }}>{error}</p>}
      {success && <p style={{ color: 'green' }}>Configuration saved successfully.</p>}
      <Form form={form} onFinish={onFinish} layout="vertical">
        <Form.Item name="enabled" label="Enable integration" valuePropName="checked">
          <Switch />
        </Form.Item>
        <Form.Item name="endpoint" label="API Endpoint" rules={[{ required: true, message: 'Please input API Endpoint!' }]}>
          <Input />
        </Form.Item>
        <Form.Item name="apiKey" label="API Key" rules={[{ required: true, message: 'Please input API Key!' }]}>
          <Input />
        </Form.Item>
        <Form.Item>
          <Button type="primary" htmlType="submit">
            Save Configuration
          </Button>
        </Form.Item>
      </Form>
    </div>
  );
};

export default IntegrationConfig;
```

**2. API Endpoint (integration-api.js):**

```javascript
import express from 'express';
import fs from 'fs';
import path from 'path';

const router = express.Router();

const CONFIG_PATH = path.resolve(__dirname, '../../config/integration-config.json');

const readConfig = () => {
  try {
    const data = fs.readFileSync(CONFIG_PATH, 'utf8');
    return JSON.parse(data);
  } catch (err) {
    return { enabled: false, endpoint: '', apiKey: '' };
  }
};

const writeConfig = (config) => {
  fs.writeFileSync(CONFIG_PATH, JSON.stringify(config, null, 2), 'utf8');
};

router.get('/integration-config', (req, res) => {
  const config = readConfig();
  res.json(config);
});

router.post('/integration-config', (req, res) => {
  const { enabled, endpoint, apiKey } = req.body;

  if (typeof enabled !== 'boolean' || typeof endpoint !== 'string' || typeof apiKey !== 'string') {
    return res.status(400).json({ error: 'Invalid payload' });
  }

  const newConfig = { enabled, endpoint, apiKey };
  try {
    writeConfig(newConfig);
    res.json(newConfig);
  } catch (err) {
    res.status(500).json({ error: 'Failed to write config' });
  }
});

export default router;
```
