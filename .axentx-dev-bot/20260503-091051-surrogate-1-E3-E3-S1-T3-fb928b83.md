import fs from 'fs';
import path from 'path';

const STORAGE_ROOT = process.env.DOCUMENT_STORAGE_ROOT || path.resolve(process.cwd(), 'documents');

export interface DocumentFile {
  id: string;
  path: string;
  mime: string;
  size: number;
}

function inferMime(filePath: string): string {
  const ext = path.extname(filePath).toLowerCase();
  switch (ext) {
    case '.pdf': return 'application/pdf';
    case '.epub': return 'application/epub+zip';
    case '.jpg':
    case '.jpeg': return 'image/jpeg';
    case '.png': return 'image/png';
    case '.webp': return 'image/webp';
    default: return 'application/octet-stream';
  }
}

export function resolveDocumentPath(id: string): string {
  const safeId = path.basename(id);
  const candidate = path.join(STORAGE_ROOT, safeId);

  if (fs.existsSync(candidate) && fs.statSync(candidate).isFile()) {
    return candidate;
  }

  const folder = candidate;
  if (fs.existsSync(folder)) {
    const files = fs.readdirSync(folder);
    const preferred = files.find((f) => /\.(pdf|epub|jpg|jpeg|png|webp)$/i.test(f));
    if (preferred) return path.join(folder, preferred);
  }

  return candidate;
}

export function getDocumentFile(id: string): DocumentFile | null {
  const filePath = resolveDocumentPath(id);
  if (!fs.existsSync(filePath) || !fs.statSync(filePath).isFile()) {
    return null;
  }
  return {
    id,
    path: filePath,
    mime: inferMime(filePath),
    size: fs.statSync(filePath).size,
  };
}
```
