    extraction_status TEXT NOT NULL DEFAULT 'success',
    extraction_error TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    
    -- Index for efficient lookups by model_id

);
