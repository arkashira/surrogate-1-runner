```tsx
import React, { useMemo } from 'react';
import './TagCloud.css';

export interface Tag {
  name: string;
  count: number;
}

interface TagCloudProps {
  tags: Tag[];
  selectedTags: string[];
  onTagClick: (tag: string) => void;
  onTagRemove?: (tag: string) => void;
  onTagRename?: (oldName: string, newName: string) => void;
  editable?: boolean;
}

const TagCloud: React.FC<TagCloudProps> = ({
  tags,
  selectedTags,
  onTagClick,
  onTagRemove,
  onTagRename,
  editable = false,
}) => {
  const maxCount = useMemo(() => Math.max(...tags.map((t) => t.count), 1), [tags]);

  const handleKeyDown = (e: React.KeyboardEvent, tag: string) => {
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault();
      onTagClick(tag);
    }
  };

  const handleRename = (oldName: string) => {
    const newName = prompt('Edit tag:', oldName)?.trim();
    if (newName && newName !== oldName && onTagRename) {
      onTagRename(oldName, newName);
    }
  };

  return (
    <div className="tag-cloud" role="list" aria-label="Tag cloud">
      {tags.map((tag) => {
        const selected = selectedTags.includes(tag.name);
        const sizeRatio = tag.count / maxCount;
        const fontSize = 0.85 + sizeRatio * 1.15; // rem range ~0.85–2.0
        return (
          <span
            key={tag.name}
            role="listitem"
            className={`tag-cloud-tag ${selected ? 'selected' : ''}`}
            style={{ fontSize: `${fontSize}rem` }}
            onClick={() => onTagClick(tag.name)}
            onKeyDown={(e) => handleKeyDown(e, tag.name)}
            tabIndex={0}
            aria-pressed={selected}
          >
            {tag.name}
            <span className="tag-cloud-count" aria-hidden="true">
              {tag.count}
            </span>
            {editable && (
              <>
                <button
                  className="tag-cloud-edit"
                  onClick={(e) => {
                    e.stopPropagation();
                    handleRename(tag.name);
                  }}
                  aria-label={`Edit tag ${tag.name}`}
                >
                  ✏
                </button>
                <button
                  className="tag-cloud-remove"
                  onClick={(e) => {
                    e.stopPropagation();
                    if (onTagRemove) onTagRemove(tag.name);
                  }}
                  aria-label={`Remove tag ${tag.name}`}
                >
                  ×
                </button>
              </>
            )}
          </span>
        );
      })}
    </div>
  );
};

export default TagCloud;
```
