**src/services/requestTimelineService.ts**
```ts
import { db } from '../db';
import type { RequestTimelineEntry } from '../types';

export class RequestTimelineService {
  /**
   * List timeline entries for a request scoped to team.
   * Returns entries ordered by occurredAt asc.
   */
  static async listByRequestId(
    requestId: string,
    teamId: string
  ): Promise<RequestTimelineEntry[]> {
    const rows = await db<RequestTimelineEntry>('request_timeline')
      .select(
        'actor_id as actorId',
        'occurred_at as occurredAt',
        'from_status as fromStatus',
        'to_status as toStatus',
        'comment'
      )
      .where({ request_id: requestId, team_id: teamId })
      .orderBy('occurred_at', 'asc');

    return rows;
  }
}
```

import { Router } from 'express';
import { RequestTimelineService } from '../services/requestTimelineService';
import { RequestService } from '../services/requestService';

const router = Router();

/**
 * GET /requests/{id}/timeline
 * Returns timeline entries for a request scoped to the authenticated team.
 * Entries include: actorId, occurredAt, fromStatus, toStatus, comment
 * Ordered by occurredAt asc.
 */
router.get('/:id/timeline', async (req, res, next) => {
  try {
    const requestId = req.params.id;
    const teamId = req.user?.teamId;

    if (!teamId) {
      return res.status(403).json({ error: 'Team context required' });
    }

    // Enforce team scoping: ensure request belongs to team
    const request = await RequestService.getByIdAndTeam(requestId, teamId);
    if (!request) {
      return res.status(404).json({ error: 'Request not found or access denied' });
    }

    const entries = await RequestTimelineService.listByRequestId(requestId, teamId);
    return res.json(entries);
  } catch (err) {
    next(err);
  }
});

export default router;
