// SignatureDriftDetector.test.js
import { SignatureDriftDetector } from '../SignatureDriftDetector';
import { Signature } from '../Signature';
import { DriftDetectorConfig } from '../DriftDetectorConfig';
import { NotificationService } from '../NotificationService';
import { BaselineStore } from '../BaselineStore';

describe('SignatureDriftDetector', () => {
  let detector, mockStore, mockNotifier, config;

  beforeEach(() => {
    mockStore = { getBaseline: jest.fn(), updateBaseline: jest.fn() };
    mockNotifier = { send: jest.fn() };
    config = new DriftDetectorConfig({ alertChannel: 'sre-oncall', retryLimit: 3 });
    detector = new SignatureDriftDetector(mockStore, mockNotifier, config);
  });

  test('detects drift and sends notification within 60s', async () => {
    const baseline = new Signature('GET', '/api/v1/resource', { 'x-auth': 'abc' }, 'hash1');
    mockStore.getBaseline.mockResolvedValue(baseline);

    const current = new Signature('GET', '/api/v1/resource', { 'x-auth': 'xyz' }, 'hash2');

    const start = Date.now();
    await detector.checkForDrift('job-123', current);
    const elapsed = Date.now() - start;

    expect(mockNotifier.send).toHaveBeenCalledWith({
      channel: 'sre-oncall',
      message: expect.stringContaining('drift detected'),
      signature: current,
      jobId: 'job-123'
    });
    expect(elapsed).toBeLessThanOrEqual(60000);
  });

  test('updates baseline when no baseline exists', async () => {
    mockStore.getBaseline.mockResolvedValue(null);
    const current = new Signature('POST', '/api/v1/create', {}, 'hash3');

    await detector.checkForDrift('job-456', current);

    expect(mockStore.updateBaseline).toHaveBeenCalledWith('job-456', current);
    expect(mockNotifier.send).not.toHaveBeenCalled();
  });

  test('retries notification on transient error', async () => {
    mockStore.getBaseline.mockResolvedValue(null);
    const current = new Signature('GET', '/api', {}, 'hash4');

    mockNotifier.send.mockRejectedValueOnce(new Error('network'));
    mockNotifier.send.mockResolvedValueOnce(); // success on retry

    await detector.checkForDrift('job-789', current);

    expect(mockNotifier.send).toHaveBeenCalledTimes(2);
  });

  test('does not alert if signature unchanged', async () => {
    const baseline = new Signature('GET', '/api', {}, 'hash5');
    mockStore.getBaseline.mockResolvedValue(baseline);

    await detector.checkForDrift('job-000', baseline);

    expect(mockNotifier.send).not.toHaveBeenCalled();
  });
});
