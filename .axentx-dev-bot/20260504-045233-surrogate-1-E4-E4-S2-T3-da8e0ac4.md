// frontend/tests/BuildWizard.unit.spec.js
import { render, fireEvent, waitFor, screen } from '@testing-library/react';
import BuildWizard from '@/components/BuildWizard';
import * as compatibilityApi from '@/api/compatibility';

// Mock compatibility API to resolve instantly
jest.mock('@/api/compatibility', () => ({
  checkCompatibility: jest.fn(),
}));

describe('BuildWizard – unit behavior', () => {
  beforeEach(() => {
    compatibilityApi.checkCompatibility.mockResolvedValue({ compatible: true });
  });

  test('renders all eight steps in order', () => {
    render(<BuildWizard />);
    const stepTitles = [
      'Select CPU',
      'Select GPU',
      'Select Motherboard',
      'Select RAM',
      'Select Storage',
      'Select PSU',
      'Select Case',
      'Select Accessories',
    ];
    stepTitles.forEach(title => expect(screen.getByText(title)).toBeInTheDocument());
  });

  test('CPU selection triggers compatibility check', async () => {
    render(<BuildWizard />);
    fireEvent.change(screen.getByLabelText(/CPU/i), {
      target: { value: 'Intel i9-13900K' },
    });
    await waitFor(() => expect(compatibilityApi.checkCompatibility).toHaveBeenCalledWith({
      cpu: 'Intel i9-13900K',
    }));
  });

  test('incompatible selection disables Next button and shows warning', async () => {
    compatibilityApi.checkCompatibility.mockResolvedValue({
      compatible: false,
      reason: 'Socket mismatch',
    });
    render(<BuildWizard />);
    fireEvent.change(screen.getByLabelText(/CPU/i), {
      target: { value: 'AMD Ryzen 9 7950X' },
    });
    fireEvent.change(screen.getByLabelText(/Motherboard/i), {
      target: { value: 'Intel Z690' },
    });
    await waitFor(() => {
      expect(screen.getByText(/Incompatible: Socket mismatch/i)).toBeInTheDocument();
      expect(screen.getByRole('button', { name: /Next/i })).toBeDisabled();
    });
  });
});
