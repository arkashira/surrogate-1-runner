import (

// getSafeVersion returns the suggested safe alternative version for the given vulnerability.
func getSafeVersion(vuln Vulnerability) string {
	if vuln.SafeAlternatives == nil {
		return ""
	}

	// Find the highest version in the safe alternatives map
	highestVersion := ""
	for version := range vuln.SafeAlternatives {
		if version > highestVersion {
			highestVersion = version
		}
	}

	// Return the suggested safe alternative version
	return fmt.Sprintf("Safe alternative: %s", vuln.SafeAlternatives[highestVersion])
}
