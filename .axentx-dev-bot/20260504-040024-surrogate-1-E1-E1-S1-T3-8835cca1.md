// /opt/axentx/surrogate-1/web/src/components/ModelList.test.jsx
import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import ModelList from './ModelList';

const mockModels = [
  {
    id: '1',
    name: 'Alpha',
    version: 'v1.0',
    compliance: 'Pass',
    lastAudit: '2026-05-01T12:00:00Z',
  },
  {
    id: '2',
    name: 'Beta',
    version: 'v2.1',
    compliance: 'Fail',
    lastAudit: '2026-04-20T08:30:00Z',
  },
  {
    id: '3',
    name: 'Gamma',
    version: 'v3.3',
    compliance: 'Pass',
    lastAudit: '2026-05-02T09:15:00Z',
  },
];

describe('ModelList component', () => {
  test('renders all model rows with correct data', () => {
    render(<ModelList models={mockModels} />);
    mockModels.forEach((m) => {
      expect(screen.getByText(m.name)).toBeInTheDocument();
      expect(screen.getByText(m.version)).toBeInTheDocument();
      expect(screen.getByText(m.compliance)).toBeInTheDocument();
      expect(screen.getByText(m.lastAudit)).toBeInTheDocument();
    });
  });

  test('filters by compliance status', () => {
    render(<ModelList models={mockModels} />);
    fireEvent.change(screen.getByLabelText(/status/i), { target: { value: 'Fail' } });
    expect(screen.getByText('Beta')).toBeInTheDocument();
    expect(screen.queryByText('Alpha')).not.toBeInTheDocument();
    expect(screen.queryByText('Gamma')).not.toBeInTheDocument();
  });

  test('filters by model name (case‑insensitive)', () => {
    render(<ModelList models={mockModels} />);
    fireEvent.change(screen.getByLabelText(/search/i), { target: { value: 'ga' } });
    expect(screen.getByText('Gamma')).toBeInTheDocument();
    expect(screen.queryByText('Alpha')).not.toBeInTheDocument();
    expect(screen.queryByText('Beta')).not.toBeInTheDocument();
  });

  test('combined status and name filters', () => {
    render(<ModelList models={mockModels} />);
    fireEvent.change(screen.getByLabelText(/status/i), { target: { value: 'Pass' } });
    fireEvent.change(screen.getByLabelText(/search/i), { target: { value: 'a' } });
    expect(screen.getByText('Alpha')).toBeInTheDocument();
    expect(screen.getByText('Gamma')).toBeInTheDocument();
    expect(screen.queryByText('Beta')).not.toBeInTheDocument();
  });

  test('shows “No models found” when filters match nothing', () => {
    render(<ModelList models={mockModels} />);
    fireEvent.change(screen.getByLabelText(/status/i), { target: { value: 'Fail' } });
    fireEvent.change(screen.getByLabelText(/search/i), { target: { value: 'Omega' } });
    expect(screen.getByText(/no models found/i)).toBeInTheDocument();
  });

  test('does not make network requests on render', () => {
    const fetchSpy = jest.spyOn(global, 'fetch');
    render(<ModelList models={mockModels} />);
    expect(fetchSpy).not.toHaveBeenCalled();
    fetchSpy.mockRestore();
  });
});
