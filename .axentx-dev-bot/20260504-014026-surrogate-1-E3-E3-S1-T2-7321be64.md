"""
Utility to convert OpenAI structured output responses into a LangChain-compatible
tool_calls format.

The function `map_response_to_tool_calls` inspects an OpenAI response dict,
extracts the JSON payload from the `content` field when the response format
is declared as a JSON object, and transforms it into the minimal
LangChain tool call structure:

```json
{
  "tool_calls": [
    {
      "function": {
        "name": "<function_name>",
        "arguments": "<arguments_json_string>"
      }
    }
  ]
}
```

No additional fields are added to the output. If the response does not
contain a JSON-structured format or the content cannot be parsed,
`None` is returned.
"""

import json
import re
from typing import Any, Dict, List, Optional


def _extract_json_from_content(content: str) -> Dict[str, Any]:
    """
    Extract a JSON object from a string. The content is expected to be a JSON
    string, but may contain surrounding whitespace or newlines. If parsing
    fails, a regex fallback is used to locate the first JSON object.
    """
    try:
        return json.loads(content)
    except json.JSONDecodeError:
        # Fallback: find the first '{' and the matching '}'
        match = re.search(r"\{.*\}", content, re.DOTALL)
        if not match:
            raise ValueError("Content does not contain a JSON object")
        try:
            return json.loads(match.group(0))
        except json.JSONDecodeError as exc:
            raise ValueError("Failed to parse JSON from content") from exc


def map_response_to_tool_calls(response: Dict[str, Any]) -> Optional[Dict[str, List[Dict[str, Dict[str, str]]]]]:
    """
    Convert an OpenAI response with a declared JSON output format into a
    LangChain-compatible tool_calls structure.

    Parameters
    ----------
    response : dict
        The raw response from the OpenAI API. Expected keys:
        - `response_format`: dict with a `type` key equal to `"json_object"`.
        - `content`: str containing a JSON object with `name` and `arguments`.

    Returns
    -------
    dict or None
        A dict containing a single key `tool_calls` with a list of tool call
        dictionaries, or `None` if the response does not contain a JSON
        structured format.

    Raises
    ------
    ValueError
        If the content cannot be parsed as JSON or required fields are missing.
    """
    # Verify that the response declares a JSON object format
    resp_fmt = response.get("response_format")
    if not isinstance(resp_fmt, dict) or resp_fmt.get("type") != "json_object":
        return None

    content = response.get("content")
    if content is None:
        raise ValueError("Missing 'content' field in response")

    # Extract the JSON payload from the content
    payload = _extract_json_from_content(content)

    # Ensure required keys are present
    if "name" not in payload or "arguments" not in payload:
        raise ValueError("JSON payload must contain 'name' and 'arguments' keys")

    # Build the minimal LangChain tool call structure
    tool_call = {
        "function": {
            "name": str(payload["name"]),
            "arguments": str(payload["arguments"]),
        }
    }

    return {"tool_calls": [tool_call]}


# Expose the function for external use
__all__ = ["map_response_to_tool_calls"]
