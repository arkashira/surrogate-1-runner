export interface PublicRequestTimelineEntry {
  timestamp: string; // ISO 8601
  status: string;
  message?: string;
}

export interface PublicRequest {
  id: string;
  title: string;
  status: string;
  isDeleted?: boolean;
  sla: {
    targetHours: number;
    dueAt: string | null;
    status: 'on_track' | 'at_risk' | 'breached' | 'fulfilled';
  };
  updatedAt: string; // ISO 8601
  timeline: PublicRequestTimelineEntry[];
}

export interface PublicRequestResponse {
  title: string;
  status: string;
  sla: PublicRequest['sla'];
  updatedAt: string;
  timeline: PublicRequestTimelineEntry[];
}

/**
 * Serialize a domain request into the public-safe shape.
 * Excludes internal/auth fields and sorts timeline chronologically.
 */
export function serializePublicRequest(request: PublicRequest): PublicRequestResponse {
  return {
    title: request.title,
    status: request.status,
    sla: request.sla,
    updatedAt: request.updatedAt,
    timeline: request.timeline
      .slice()
      .sort((a, b) => new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime()),
  };
}
