from cryptography.hazmat.backends import default_backend
import base64

router = APIRouter(prefix="/evidence", tags=["evidence"])

# Configuration
EVIDENCE_DIR = Path("/var/lib/axentx/winguard/evidence")
PUBLIC_KEY_PATH = Path("/etc/axentx/winguard/public_key.pem")
RETENTION_DAYS = 90

def load_public_key():
    """Load and return the public key for signature verification."""
    with PUBLIC_KEY_PATH.open("rb") as key_file:
        public_key = serialization.load_pem_public_key(
            key_file.read(),
            backend=default_backend()
        )
    return public_key

def verify_signature(data: dict, signature: str, public_key):
    """Verify the signature of the evidence data."""
    try:
        # Prepare data for verification (same fields as signing)
        verification_data = {
            "vm_id": data["vm_id"],
            "timestamp": data["timestamp"],
            "policy_version": data["policy_version"],
            "settings_hash": data["settings_hash"]
        }

        # Serialize and encode
        data_bytes = json.dumps(verification_data, sort_keys=True).encode('utf-8')
        signature_bytes = base64.b64decode(signature.encode('utf-8'))

        # Verify
        public_key.verify(
            signature_bytes,
            data_bytes,
            padding.PKCS1v15(),
            hashes.SHA256()
        )
        return True
    except Exception:
        return False

        HTTPException: 404 if evidence not found, 400 if invalid, 500 for other errors
    """
    evidence_file = EVIDENCE_DIR / f"{vm_id}.json"

    # Check file existence
    if not evidence_file.is_file():
        raise HTTPException(status_code=404, detail="Evidence not found")

    try:
        # Read and parse file
        with evidence_file.open("r", encoding="utf-8") as fh:
            evidence_data = json.load(fh)

        # Check retention policy
        file_age = datetime.utcnow() - datetime.fromtimestamp(evidence_file.stat().st_mtime)
        if file_age > timedelta(days=RETENTION_DAYS):
