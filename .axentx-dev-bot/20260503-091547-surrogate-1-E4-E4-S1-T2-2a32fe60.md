// src/lib/sla.ts
// SLA primitives, in-memory stores, and helpers.
// Replace stores with DB adapters in production.

export type SLAState = "ok" | "warning" | "breach";
export type RequestStatus = "open" | "in_progress" | "resolved";

const EPSILON = 1e-9; // avoid floating-point false positives

export interface Request {
  id: string;
  teamId: string;
  title: string;
  createdAt: number; // epoch ms
  slaHours: number;
  status: RequestStatus;
  updatedAt?: number;
}

export interface BreachLog {
  requestId: string;
  teamId: string;
  breachedAt: number; // epoch ms
  elapsedHours: number;
  slaHours: number;
}

// In-memory stores (swap for DB in prod)
export const requests: Request[] = [];
export const breachLogs: BreachLog[] = [];

export function elapsedHours(req: Request): number {
  return (Date.now() - req.createdAt) / (1000 * 60 * 60);
}

export function computeSLAState(req: Request): SLAState {
  const elapsed = elapsedHours(req);
  const ratio = elapsed / req.slaHours;
  if (ratio > 1 + EPSILON) return "breach";
  if (ratio >= 0.8 - EPSILON) return "warning";
  return "ok";
}

export function logBreachIfNew(req: Request): BreachLog | null {
  if (computeSLAState(req) !== "breach") return null;
  const exists = breachLogs.find((b) => b.requestId === req.id);
  if (exists) return exists;
  const log: BreachLog = {
    requestId: req.id,
    teamId: req.teamId,
    breachedAt: Date.now(),
    elapsedHours: elapsedHours(req),
    slaHours: req.slaHours,
  };
  breachLogs.push(log);
  console.warn(
    `[SLA BREACH] request=${req.id} team=${req.teamId} elapsed=${log.elapsedHours.toFixed(2)}h sla=${req.slaHours}h`
  );
  return log;
}

export function activeRequestsForTeam(teamId: string) {
  return requests
    .filter((r) => r.teamId === teamId && r.status !== "resolved")
    .map((r) => ({
      ...r,
      elapsedHours: elapsedHours(r),
      slaState: computeSLAState(r),
    }));
}

export function weeklyCounts(teamId: string) {
  const now = Date.now();
  const oneWeekMs = 7 * 24 * 60 * 60 * 1000;
  const weekStart = now - oneWeekMs;
  const created = requests.filter(
    (r) => r.teamId === teamId && r.createdAt >= weekStart
  ).length;
  const tracked = requests.filter(
    (r) =>
      r.teamId === teamId &&
      (r.createdAt >= weekStart || (r.status !== "resolved" && r.createdAt < weekStart))
  ).length;
  return { created, tracked };
}

// Minimal digest stub: breaches per team (optionally within time window)
export function buildBreachesDigest(teamId?: string, since?: number) {
  let items = breachLogs;
  if (teamId) items = items.filter((b) => b.teamId === teamId);
  if (since) items = items.filter((b) => b.breachedAt >= since);
  return {
    generatedAt: Date.now(),
    teamId: teamId ?? "all",
    since: since ?? 0,
    breaches: items.map((b) => ({
      requestId: b.requestId,
      teamId: b.teamId,
      breachedAt: b.breachedAt,
      elapsedHours: b.elapsedHours,
      slaHours: b.slaHours,
    })),
    count: items.length,
  };
}
