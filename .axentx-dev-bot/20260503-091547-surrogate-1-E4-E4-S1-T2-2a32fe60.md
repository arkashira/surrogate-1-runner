import { Request, RequestWithSLA, WeeklyCounts, BreachLog } from "../types/request";

const requests = new Map<string, Request>();
const breaches: BreachLog[] = [];
const breachedRequestIds = new Set<string>();

function computeSLA(req: Request): RequestWithSLA {
  const created = new Date(req.sla.createdAt || req.createdAt);
  const now = new Date();
  const elapsedMs = now.getTime() - created.getTime();
  const elapsedHours = elapsedMs / (1000 * 60 * 60);
  const targetHours = req.sla.targetHours;
  const timeRemainingHours = Math.max(0, targetHours - elapsedHours);

  let state: SLAState = "ok";
  if (elapsedHours >= targetHours) state = "breach";
  else if (elapsedHours >= targetHours * 0.8) state = "warning";

  return {
    ...req,
    elapsedHours: Number(elapsedHours.toFixed(2)),
    state,
    timeRemainingHours: Number(timeRemainingHours.toFixed(2)),
  };
}

function logBreach(req: RequestWithSLA) {
  if (req.state === "breach" && !breachedRequestIds.has(req.id)) {
    breachedRequestIds.add(req.id);
    breaches.push({
      requestId: req.id,
      teamId: req.teamId,
      breachedAt: new Date().toISOString(),
      elapsedHours: req.elapsedHours,
      targetHours: req.sla.targetHours,
    });
    console.warn(`[SLA_BREACH] request=${req.id} team=${req.teamId} elapsed=${req.elapsedHours}h target=${req.sla.targetHours}h`);
  }
}
