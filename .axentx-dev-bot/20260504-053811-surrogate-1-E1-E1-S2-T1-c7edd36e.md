```python
# /opt/axentx/surrogate-1/src/gpt4_api.py
import re
import os

def validate_gpt4_api_key(api_key):
    pattern = r"^[a-zA-Z0-9_-]{32}$"
    if not re.match(pattern, api_key):
        raise ValueError("Invalid GPT-4 API key")
    return api_key

def get_gpt4_api_key():
    api_key = os.environ.get("GPT4_API_KEY")
    if not api_key:
        raise ValueError("GPT-4 API key is not set")
    return validate_gpt4_api_key(api_key)
```

```python
# tests/test_config_gpt4_key.py
import pytest
from surrogate_1.config import ConfigLoader, ConfigurationError

VALID_KEY = "sk-ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890123456"
INVALID_PREFIX = "skx-ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890123456"
INVALID_LENGTH = "sk-ABC123"

def test_valid_key_parsed_successfully():
    cfg = ConfigLoader({"gpt4_api_key": VALID_KEY})
    assert cfg.get_gpt4_api_key() == VALID_KEY

def test_missing_key_raises_error():
    with pytest.raises(ConfigurationError) as exc:
        ConfigLoader({}).get_gpt4_api_key()
    assert "gpt4_api_key is required" in str(exc.value)

def test_invalid_prefix_raises_error():
    with pytest.raises(ConfigurationError) as exc:
        ConfigLoader({"gpt4_api_key": INVALID_PREFIX}).get_gpt4_api_key()
    assert "does not match required pattern" in str(exc.value)

def test_invalid_length_raises_error():
    with pytest.raises(ConfigurationError) as exc:
        ConfigLoader({"gpt4_api_key": INVALID_LENGTH}).get_gpt4_api_key()
    assert "does not match required pattern" in str(exc.value)

def test_key_not_written_back_to_file(tmp_path):
    cfg_path = tmp_path / "config.yaml"
    cfg_path.write_text(f"gpt4_api_key: {VALID_KEY}\n")
    loader = ConfigLoader.from_file(str(cfg_path))
    assert loader.get_gpt4_api_key() == VALID_KEY
    # ensure file content unchanged
    assert cfg_path.read_text() == f"gpt4_api_key: {VALID_KEY}\n"
```
