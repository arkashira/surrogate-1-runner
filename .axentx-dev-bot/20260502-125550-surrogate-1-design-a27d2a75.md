The system lacks a **persisted, date-scoped manifest**, causing non-deterministic training runs, repeated authenticated API enumeration (429s), mixed-schema ingestion errors, and failure to bypass HF API for CDN-only downloads.

**Core Prescription (Consensus + Actionable):**  
Create a **deterministic manifest → CDN-only training pipeline** with strict schema projection, Lightning Studio reuse, and idempotent artifact outputs.

### `scripts/build_manifest.py`
```python
#!/usr/bin/env python3
"""
Build a date-scoped manifest for a dataset repo.
- Single non-recursive tree call per date folder.
- Validates file extensions and basic schema hints.
- Idempotent: safe to rerun; overwrites same date folder.
"""
import argparse
import json
import os
import sys
from datetime import datetime, timezone

from huggingface_hub import HfApi

ALLOWED_EXTS = {".jsonl", ".json", ".parquet", ".csv"}

def list_date_folder(repo_id: str, date_folder: str):
    api = HfApi()
    try:
        tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
        out = []
        for item in tree:
            if item.path.endswith("/"):
                continue
            ext = os.path.splitext(item.path)[1].lower()
            if ext not in ALLOWED_EXTS:
                continue
            out.append({"path": item.path, "size": item.size})
        return out
    except Exception as e:
        print(f"Error listing {repo_id}/{date_folder}: {e}", file=sys.stderr)
        raise

def build_manifest(repo_id: str, date_str: str):
    # Validate date format early
    try:
        datetime.strptime(date_str, "%Y%m%d")
    except ValueError:
        raise ValueError("date must be YYYYMMDD")

    entries = list_date_folder(repo_id, date_str)
    manifest = {
        "repo_id": repo_id,
        "date": date_str,
        "created_at": datetime.now(timezone.utc).isoformat(),
        "files": entries,
    }
    return manifest

def main():
    parser = argparse.ArgumentParser(description="Build HF CDN manifest for a date folder.")
    parser.add_argument("--repo", required=True, help="HF dataset repo id, e.g. org/dataset")
    parser.add_argument("--date", required=True, help="Date folder name, e.g. 20260501")
    parser.add_argument("--out", required=True, help="Output manifest JSON path")
    args = parser.parse_args()

    os.makedirs(os.path.dirname(os.path.abspath(args.out)), exist_ok=True)
    manifest = build_manifest(args.repo, args.date)

    with open(args.out, "w", encoding="utf-8") as f:
        json.dump(manifest, f, indent=2)
    print(f"Manifest written to {args.out} ({len(manifest['files'])} files)")

if __name__ == "__main__":
    main()
```
