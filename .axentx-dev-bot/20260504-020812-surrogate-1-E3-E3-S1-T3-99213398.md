* **JavaScript/TypeScript tooling** – `eslint` and `prettier` (again, the latest stable releases).

```yaml
# .pre-commit-config.yaml
# --------------------------------------------------------------
# 1️⃣  Global exclude – keeps the repo tidy and the config DRY
# --------------------------------------------------------------
exclude: |
  (?x)(
    __pycache__/
    | \.pyc$
    | vendor/
    | \.git/
    | \.venv/
    | venv/
    | build/
    | dist/
    | \.egg-info/
  )

# --------------------------------------------------------------
# 2️⃣  Repositories & hooks
# --------------------------------------------------------------
repos:
  # ─────────────────────────────────────────────────────────────
  # Python: import sorting
  # ─────────────────────────────────────────────────────────────
  - repo: https://github.com/pycqa/isort
    rev: v5.13.2
    hooks:
      - id: isort
        name: isort (Python)
        files: \.py$
        exclude: __pycache__|vendor

  # Python: formatting
  # ─────────────────────────────────────────────────────────────
  - repo: https://github.com/psf/black

        exclude: __pycache__|vendor

  # ─────────────────────────────────────────────────────────────

  # ─────────────────────────────────────────────────────────────

  # ─────────────────────────────────────────────────────────────

  # ─────────────────────────────────────────────────────────────

    rev: v8.56.0

      - id: eslint
        name: eslint (JavaScript/TypeScript)

        exclude: __pycache__|vendor

  # ─────────────────────────────────────────────────────────────

    rev: v3.3.3

        name: prettier (Code formatter)
        files: \.(js|jsx|ts|tsx|json|md|yaml|yml)$
        exclude: __pycache__|vendor
```
