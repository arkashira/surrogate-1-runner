# -*- coding: utf-8 -*-
"""
Slack integration endpoint for surrogate‑1.

Provides:
* GET  /api/slack/channels – list accessible Slack channels (for UI dropdowns)
* POST /api/slack/post    – post a PDF‑report link to a chosen channel

All token handling is encrypted and abstracted behind the configured
`SECRETS_VAULT` object.
"""

from __future__ import annotations

import logging
from dataclasses import dataclass
from enum import Enum
from typing import Any, Dict, List, Optional

import requests
from flask import Blueprint, current_app, jsonify, request

# --------------------------------------------------------------------------- #
# Blueprint & logger
# --------------------------------------------------------------------------- #
slack_bp = Blueprint("slack", __name__, url_prefix="/api/slack")
log = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Domain models
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class SlackChannel:
    """Minimal representation of a Slack channel."""
    id: str
    name: str
    is_private: bool = False


@dataclass(frozen=True)
class SlackMessagePayload:
    """Payload accepted by chat.postMessage."""
    channel: str
    text: str
    blocks: Optional[List[Dict[str, Any]]] = None


# --------------------------------------------------------------------------- #
# Custom exception – unified error shape for Flask errorhandler
# --------------------------------------------------------------------------- #
class SlackError(RuntimeError):
    """Raised when any Slack‑API call fails."""

    def __init__(self, *, message: str, code: str, status_code: int = 400):
        super().__init__(message)
        self.message = message
        self.code = code
        self.status_code = status_code


# --------------------------------------------------------------------------- #
# Helper: Secrets‑Vault abstraction
# --------------------------------------------------------------------------- #
def _vault() -> Any:
    """Retrieve the configured vault object or raise a clear error."""
    vault = current_app.config.get("SECRETS_VAULT")
    if not vault:
        raise RuntimeError("SECRETS_VAULT not configured in Flask app")
