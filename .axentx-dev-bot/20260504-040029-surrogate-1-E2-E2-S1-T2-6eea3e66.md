// result_display.test.js
import { displayResults } from '../src/result_display';
import { verifyImports } from '../src/verify_imports';
import { run } from '../src/cli';

jest.mock('../src/verify_imports');

describe('displayResults', () => {
  const consoleSpy = jest.spyOn(console, 'log').mockImplementation(() => {});

  afterEach(() => consoleSpy.mockClear());

  test('formats a single successful result', () => {
    const results = [{ importPath: 'foo/bar', status: 'PASS', errorMessage: '' }];
    displayResults(results);
    expect(consoleSpy).toHaveBeenCalledWith(expect.stringContaining('foo/bar'));
    expect(consoleSpy).toHaveBeenCalledWith(expect.stringContaining('PASS'));
  });

  test('formats multiple mixed results', () => {
    const results = [
      { importPath: 'a/b', status: 'PASS', errorMessage: '' },
      { importPath: 'x/y', status: 'FAIL', errorMessage: 'Missing module' },
    ];
    displayResults(results);
    expect(consoleSpy).toHaveBeenCalledTimes(4); // header + 2 rows + separator
    expect(consoleSpy).toHaveBeenCalledWith(expect.stringContaining('FAIL'));
    expect(consoleSpy).toHaveBeenCalledWith(expect.stringContaining('Missing module'));
  });

  test('handles empty result array', () => {
    displayResults([]);
    expect(consoleSpy).toHaveBeenCalledWith('No import statements found.');
  });
});

describe('run', () => {
  const consoleSpy = jest.spyOn(console, 'log').mockImplementation(() => {});
  const exitSpy = jest.spyOn(process, 'exit').mockImplementation(() => {});

  afterEach(() => {
    consoleSpy.mockClear();
    exitSpy.mockClear();
  });

  test('calls verifyImports and displayResults', async () => {
    verifyImports.mockResolvedValue([{ importPath: 'foo', status: 'PASS', errorMessage: '' }]);
    await run(['--path', 'src']);
    expect(verifyImports).toHaveBeenCalledWith('src');
    expect(consoleSpy).toHaveBeenCalled(); // table printed
    expect(exitSpy).toHaveBeenCalledWith(0);
  });

  test('exits with code 1 if any result FAIL', async () => {
    verifyImports.mockResolvedValue([{ importPath: 'bar', status: 'FAIL', errorMessage: 'err' }]);
    await run(['--path', 'src']);
    expect(exitSpy).toHaveBeenCalledWith(1);
  });
});
