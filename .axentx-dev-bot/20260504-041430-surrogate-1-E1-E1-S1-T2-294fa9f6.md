/**
 * Workflow validation utilities.
 *
 * This module defines the shape of a workflow definition and provides a
 * validator that returns a ValidationResult object if the definition does not meet the
 * required schema.
 */

import { z } from 'zod';

// Workflow input schema
export const WorkflowInputSchema = z.object({
  name: z.string().min(1, 'Input name is required'),
  type: z.enum(['string', 'number', 'boolean', 'array', 'object']),
  required: z.boolean().optional().default(false),
  default: z.unknown().optional(),
  description: z.string().optional(),
  schema: z.object().optional(), // Added schema property
});

// Workflow output schema
export const WorkflowOutputSchema = z.object({
  name: z.string().min(1, 'Output name is required'),
  type: z.enum(['string', 'number', 'boolean', 'array', 'object']),
  description: z.string().optional(),
});

// Workflow step schema
export const WorkflowStepSchema = z.object({
  id: z.string().min(1, 'Step ID is required'),
  name: z.string().min(1, 'Step name is required'),
  action: z.string().min(1, 'Action is required'),
  inputs: z.record(z.unknown()).optional(),
  conditions: z
    .array(
      z.object({
        field: z.string(),
        operator: z.enum(['eq', 'neq', 'gt', 'gte', 'lt', 'lte', 'contains', 'exists']),
        value: z.unknown(),
      })
    )
    .optional(),
  onSuccess: z.string().optional(),
  onFailure: z.string().optional(),
});

// Main workflow definition schema
export const WorkflowDefinitionSchema = z.object({
  name: z.string().min(1, 'Workflow name is required').max(100, 'Workflow name must be at most 100 characters'),
  description: z.string().max(500, 'Description must be at most 500 characters').optional(),
  version: z.string().regex(/^\d+\.\d+\.\d+$/, 'Version must be in semver format (e.g., 1.0.0)').optional(),
  inputs: z.array(WorkflowInputSchema).optional().default([]),
  steps: z.array(WorkflowStepSchema).min(1, 'At least one step is required'),
  outputs: z.array(WorkflowOutputSchema).optional().default([]),
  metadata: z
    .object({
      author: z.string().optional(),
      tags: z.array(z.string()).optional(),
      createdAt: z.string().datetime().optional(),
      updatedAt: z.string().datetime().optional(),
    })
    .optional(),
});

// Type exports
export type WorkflowInput = z.infer<typeof WorkflowInputSchema>;
export type WorkflowOutput = z.infer<typeof WorkflowOutputSchema>;
export type WorkflowStep = z.infer<typeof WorkflowStepSchema>;
export type WorkflowDefinition = z.infer<typeof WorkflowDefinitionSchema>;

// Validation result types
export interface ValidationResult {
  valid: boolean;
  errors: ValidationError[];
  warnings: ValidationWarning[];
}

export interface ValidationError {
  path: string;
  message: string;
  code: string;
}

export interface ValidationWarning {
  path: string;
  message: string;
}

// Validator class
export class WorkflowValidator {
  private schema = WorkflowDefinitionSchema;

  /**
   * Validate a workflow definition
   */
  validate(workflow: unknown): ValidationResult {
    const result: ValidationResult = {
      valid: false,
      errors: [],
      warnings: [],
    };

    // Schema validation
    const parseResult = this.schema.safeParse(workflow);

    if (!parseResult.success) {
      result.errors = parseResult.error.errors.map((err) => ({
        path: err.path.join('.'),
        message: err.message,
        code: err.code,
      }));
      return result;
    }

    // Additional business logic validations
    const validated = parseResult.data;
    this.validateStepReferences(validated, result);
    this.validateCircularDependencies(validated, result);
    this.validateInputOutputConsistency(validated, result);

    result.valid = result.errors.length === 0;
    return result;
  }
