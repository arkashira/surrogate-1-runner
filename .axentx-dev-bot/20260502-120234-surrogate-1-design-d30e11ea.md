### 1) One-time manifest build (avoids HF API hot-path)

File: `/opt/axentx/surrogate-1/scripts/build_manifest.py`

```bash
#!/usr/bin/env bash
# Build manifest for one date folder to avoid HF API during training.
# Usage: bash build_manifest.sh <date_folder> <dataset_repo> <out_json>
set -euo pipefail

DATE_FOLDER="${1:-mirror-merged/2026-05-02}"
REPO="${2:-datasets/axentx/surrogate-mirror}"
OUT="${3:-/opt/axentx/surrogate-1/manifest.json}"

python3 - "$DATE_FOLDER" "$REPO" "$OUT" <<'PY'
import json, os, sys
from huggingface_hub import list_repo_tree, hf_hub_download

date_folder = sys.argv[1]
repo_id = sys.argv[2]
out_path = sys.argv[3]

# One API call: non-recursive list for the chosen date folder
tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
files = sorted(f.rfilename for f in tree if f.rfilename.endswith(".parquet"))

# Prefer CDN; include repo fallback for files that require auth
cdn_prefix = f"https://huggingface.co/datasets/{repo_id}/resolve/main/{date_folder}"

manifest = {
    "repo_id": repo_id,
    "date_folder": date_folder,
    "files": files,
    "cdn_prefix": cdn_prefix,
}

os.makedirs(os.path.dirname(out_path) or ".", exist_ok=True)
with open(out_path, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Wrote {len(files)} files to {out_path}")
PY
```

Run once (or in CI before training):

