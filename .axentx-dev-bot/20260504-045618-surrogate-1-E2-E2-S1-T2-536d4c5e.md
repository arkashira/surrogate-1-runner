    // ignore (e.g., private browsing)
  }

import { WalkthroughEngine } from '../src/services/WalkthroughEngine';
import { walkthroughSlice } from '../src/stores/walkthroughStore';
import { configureStore } from '@reduxjs/toolkit';

describe('WalkthroughEngine', () => {
  let store: ReturnType<typeof configureStore>;
  let engine: WalkthroughEngine;

  beforeEach(() => {
    store = configureStore({ reducer: { walkthrough: walkthroughSlice.reducer } });
    engine = WalkthroughEngine.getInstance(store);
  });

  test('starts a walkthrough and sets currentStep to 0', () => {
    engine.start('demoWalkthrough');
    const state = store.getState().walkthrough;
    expect(state.currentWalkthrough).toBe('demoWalkthrough');
    expect(state.currentStep).toBe(0);
  });

  test('advances to next step', () => {
    engine.start('demoWalkthrough');
    engine.next();
    const state = store.getState().walkthrough;
    expect(state.currentStep).toBe(1);
  });

  test('dismisses walkthrough and clears state', () => {
    engine.start('demoWalkthrough');
    engine.dismiss();
    const state = store.getState().walkthrough;
    expect(state.currentWalkthrough).toBeNull();
    expect(state.currentStep).toBeNull();
  });

  test('persist state to localStorage on step change', () => {
    jest.spyOn(Storage.prototype, 'setItem');
    engine.start('demoWalkthrough');
    engine.next();
    expect(localStorage.setItem).toHaveBeenCalledWith(
      'walkthroughState',
      JSON.stringify({ walkthrough: 'demoWalkthrough', step: 1 })
    );
  });

  test('handles missing UI element gracefully', () => {
    const consoleWarn = jest.spyOn(console, 'warn').mockImplementation();
    engine.start('walkthroughWithMissingElement');
    // Assume step 0 references a missing element
    expect(consoleWarn).toHaveBeenCalledWith(
      expect.stringContaining('Element not found for step 0')
    );
    // Engine should skip to next valid step
    const state = store.getState().walkthrough;
    expect(state.currentStep).toBe(1);
  });
});
