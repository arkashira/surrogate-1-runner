# .github/workflows/ci-sandbox.yml
name: CI Sandbox Integration Tests

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  test-sandbox:
    runs-on: ubuntu-latest
    services:
      sandbox:
        image: ghcr.io/your-org/surrogate-sandbox:latest
        ports:
          - 8080:8080
        options: >-
          --health-cmd="curl -f http://localhost:8080/health || exit 1"
          --health-interval=10s
          --health-timeout=5s
          --health-retries=5
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Docker
        uses: docker/setup-buildx-action@v2

      - name: Build sandbox image (if necessary)
        if: github.event_name != 'pull_request_target'
        run: |
          docker build -t surrogate-sandbox .

      - name: Start sandbox container
        run: |
          docker run -d --name sandbox-container -p 8080:8080 surrogate-sandbox
          sleep 10

      - name: Wait for sandbox to be healthy
        run: |
          echo "Waiting for sandbox service to be healthy..."
          for i in {1..30}; do
            if curl -s http://localhost:8080/health | grep -q "ok"; then
              echo "Sandbox is healthy."
              break
            fi
            echo "Still waiting... ($i/30)"
            sleep 5
          done

      - name: Run Datadog query test
        env:
          DD_API_KEY: ${{ secrets.DD_API_KEY }}
          DD_APP_KEY: ${{ secrets.DD_APP_KEY }}
        run: |
          echo "Running sample Datadog query..."
          curl -s -H "DD-API-KEY:${DD_API_KEY}" \
               -H "DD-APPLICATION-KEY:${DD_APP_KEY}" \
               -H "Content-Type: application/json" \
               -X POST "http://localhost:8080/datadog/query" \
               -d '{"query":"avg:system.cpu.user{*} by {host}"}' | jq .

      - name: Stop sandbox container (if necessary)
        if: github.event_name != 'pull_request_target'
        run: |
          docker stop sandbox-container
          docker rm sandbox-container
```
