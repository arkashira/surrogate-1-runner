### 1. **Acceptance Criteria**

- [ ] Workflow data is encrypted at rest using AES-256 with a key derived from a user-provided passcode via PBKDF2 (with at least 100,000 iterations, SHA-256).
- [ ] Encryption occurs automatically before saving; decryption occurs automatically after loading — without requiring changes to calling code.
- [ ] A wrong passcode attempt increments a counter; after 5 failed attempts, the system prompts for user confirmation before wiping the stored workflow data.
- [ ] After successful decryption, workflows are restored to their original state (round-trip integrity).
- [ ] Empty or invalid ciphertext results in a clear decryption failure without data corruption.
- [ ] The system does not store or log the passcode or derived key in any form.

// WorkflowStore.unit.test.ts

import { WorkflowStore } from './WorkflowStore';
import { CryptoService } from '../crypto/CryptoService';

jest.mock('../crypto/CryptoService');

describe('WorkflowStore (unit)', () => {
  let store: WorkflowStore;
  let mockCrypto: jest.Mocked<CryptoService>;

  beforeEach(() => {
    mockCrypto = new CryptoService() as jest.Mocked<CryptoService>;
    store = new WorkflowStore(mockCrypto);
  });

  test('encrypts workflow before saving', async () => {
    const workflow = { id: 'w1', steps: ['A', 'B'] };
    mockCrypto.encrypt.mockResolvedValue('encrypted-data');

    await store.saveWorkflow('user1', workflow, 'pass123');

    expect(mockCrypto.encrypt).toHaveBeenCalledWith(workflow, 'pass123');
    expect(mockCrypto.decrypt).not.toHaveBeenCalled();
  });

  test('decrypts workflow upon loading', async () => {
    const decryptedWorkflow = { id: 'w1', steps: ['A', 'B'] };
    mockCrypto.decrypt.mockResolvedValue(decryptedWorkflow);

    const result = await store.loadWorkflow('user1', 'pass123');

    expect(mockCrypto.decrypt).toHaveBeenCalledWith('stored-ciphertext', 'pass123');
    expect(result).toEqual(decryptedWorkflow);
  });

  test('does not call decrypt if no data exists', async () => {
    mockCrypto.decrypt.mockResolvedValue({ id: 'w1' });
    store['readFromStorage'] = jest.fn().mockResolvedValue(null);

    const result = await store.loadWorkflow('user1', 'pass123');

    expect(mockCrypto.decrypt).not.toHaveBeenCalled();
    expect(result).toBeNull();
  });

  test('increments failed attempt counter on decryption failure', async () => {
    mockCrypto.decrypt.mockRejectedValue(new Error('Invalid passcode'));
    await expect(store.loadWorkflow('user1', 'wrong')).rejects.toThrow();
    expect(store['getFailCount']('user1')).toBe(1);
  });

  test('wipes data after 5 failed attempts, with confirmation prompt', async () => {
    const wipeSpy = jest.spyOn(store, 'wipeAllWorkflowData').mockResolvedValue(undefined);
    mockCrypto.decrypt.mockRejectedValue(new Error('Invalid passcode'));
    const confirmSpy = jest.fn().mockResolvedValue(true); // user confirms wipe
    store['showWipeConfirmation'] = confirmSpy;

    for (let i = 0; i < 5; i++) {
      try {
        await store.loadWorkflow('user1', 'wrong');
      } catch {}
    }

    expect(confirmSpy).toHaveBeenCalled();
    expect(wipeSpy).toHaveBeenCalledWith('user1');
  });

  test('does not wipe if user denies confirmation', async () => {
    const wipeSpy = jest.spyOn(store, 'wipeAllWorkflowData');
    mockCrypto.decrypt.mockRejectedValue(new Error('Invalid passcode'));
    const confirmSpy = jest.fn().mockResolvedValue(false); // deny
    store['showWipeConfirmation'] = confirmSpy;

    for (let i = 0; i < 5; i++) {
      try {
        await store.loadWorkflow('user1', 'wrong');
      } catch {}
    }

    expect(wipeSpy).not.toHaveBeenCalled();
  });
});
