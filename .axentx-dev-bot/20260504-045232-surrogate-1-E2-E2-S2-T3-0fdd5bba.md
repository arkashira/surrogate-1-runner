## 1. Acceptance Criteria
- **AC1** – At least two distinct vendor price records exist for a component, each with a non‑null `price` and a `timestamp` within the last 24 h.  
- **AC2** – `GET /api/components/{id}` returns a field `lowest_price` that equals the minimum of all *current* (timestamp ≤ now) vendor prices for that component.  
- **AC3** – The `lowest_price` value is accompanied by a `price_timestamp` indicating the exact time of the selected price record.  
- **AC4** – `GET /api/components/{id}/prices` returns a JSON array sorted descending by `timestamp`, containing every stored price record for the component.  
- **AC5** – Adding a new price via `POST /api/components/{id}/prices` with a future timestamp does **not** affect the `lowest_price` until the timestamp becomes ≤ now.  
- **AC6** – When a component has only one vendor price, that price is returned as the `lowest_price`.  
- **AC7** – Deleting a price record updates `lowest_price` accordingly without requiring a server restart.

import pytest
from datetime import datetime, timedelta
from surrogate.models import Component, VendorPrice
from surrogate.schemas import PriceResponse

def test_vendor_price_requires_timestamp_and_price():
    with pytest.raises(ValueError):
        VendorPrice(component_id=1, vendor_id=1, price=None, timestamp=datetime.utcnow())

def test_component_lowest_price_calculation():
    now = datetime.utcnow()
    comp = Component(id=1, name="Resistor")
    prices = [
        VendorPrice(component=comp, vendor_id=1, price=0.10, timestamp=now - timedelta(hours=2)),
        VendorPrice(component=comp, vendor_id=2, price=0.08, timestamp=now - timedelta(hours=1)),
        VendorPrice(component=comp, vendor_id=3, price=0.12, timestamp=now - timedelta(days=2)),  # stale
    ]
    # Assume Component.lowest_current_price() filters by timestamp <= now
    lowest = comp.lowest_current_price(reference=now)
    assert lowest.price == 0.08
    assert lowest.vendor_id == 2

def test_lowest_price_ignores_future_timestamp():
    now = datetime.utcnow()
    comp = Component(id=2, name="Capacitor")
    future_price = VendorPrice(
        component=comp, vendor_id=1, price=0.05, timestamp=now + timedelta(hours=5)
    )
    past_price = VendorPrice(
        component=comp, vendor_id=2, price=0.07, timestamp=now - timedelta(hours=1)
    )
    assert comp.lowest_current_price(reference=now).price == 0.07

def test_price_history_sorted_desc():
    now = datetime.utcnow()
    comp = Component(id=3, name="Inductor")
    timestamps = [now - timedelta(days=i) for i in range(5)]
    for i, ts in enumerate(timestamps):
        VendorPrice(component=comp, vendor_id=i, price=0.1 + i, timestamp=ts)
    history = comp.price_history()
    sorted_ts = [p.timestamp for p in history]
    assert sorted_ts == sorted(sorted_ts, reverse=True)

def test_deleting_price_updates_lowest():
    now = datetime.utcnow()
    comp = Component(id=4, name="Diode")
    p1 = VendorPrice(component=comp, vendor_id=1, price=0.20, timestamp=now - timedelta(hours=1))
    p2 = VendorPrice(component=comp, vendor_id=2, price=0.15, timestamp=now - timedelta(hours=2))
    # initial lowest
    assert comp.lowest_current_price(reference=now).price == 0.15
    # delete the cheaper price
    comp.prices.remove(p2)
    assert comp.lowest_current_price(reference=now).price == 0.20
