import time
import pytest
from surrogate_1.services.pattern_matcher import (
    ErrorPattern,
    match_error,
    ERROR_PATTERNS,
)

# ------------------------------------------------------------------
# Helper: create a minimal error string that should match a pattern
def _sample_error(pattern: ErrorPattern) -> str:
    return f"{pattern.error_category}: {pattern.likely_root_cause} - {pattern.resolution_steps}"

# ------------------------------------------------------------------
def test_match_error_returns_correct_pattern():
    pattern = ERROR_PATTERNS[0]  # pick first known pattern
    err_str = _sample_error(pattern)
    result = match_error(err_str)
    assert isinstance(result, ErrorPattern)
    assert result.error_category == pattern.error_category
    assert result.likely_root_cause == pattern.likely_root_cause
    assert result.resolution_steps == pattern.resolution_steps
    assert result.confidence_score == pattern.confidence_score

def test_confidence_based_conflict_resolution():
    # Two patterns that both match the same error string
    pattern_high = ErrorPattern(
        error_category="Network",
        likely_root_cause="Timeout",
        resolution_steps="Restart the router.",
        confidence_score=0.9,
        regex=r"timeout",
    )
    pattern_low = ErrorPattern(
        error_category="Network",
        likely_root_cause="Timeout",
        resolution_steps="Check cables.",
        confidence_score=0.5,
        regex=r"timeout",
    )
    # Insert high first, low second
    patterns = [pattern_low, pattern_high]
    err_str = "Connection timeout occurred."
    result = match_error(err_str, patterns=patterns)
    assert result.confidence_score == 0.9
    assert result.resolution_steps == "Restart the router."

def test_tie_breaker_by_order():
    pattern_a = ErrorPattern(
        error_category="Auth",
        likely_root_cause="Invalid token",
        resolution_steps="Regenerate token.",
        confidence_score=0.7,
        regex=r"invalid token",
    )
    pattern_b = ErrorPattern(
        error_category="Auth",
        likely_root_cause="Invalid token",
        resolution_steps="Check token expiry.",
        confidence_score=0.7,
        regex=r"invalid token",
    )
    patterns = [pattern_a, pattern_b]  # a appears first
    err_str = "User provided an invalid token."
    result = match_error(err_str, patterns=patterns)
    assert result.resolution_steps == "Regenerate token."

def test_no_match_returns_none():
    err_str = "Some unrelated error message."
    result = match_error(err_str)
    assert result is None

def test_markdown_is_valid():
    pattern = ERROR_PATTERNS[0]
    # Simple sanity check: resolution contains at least one Markdown list item
    assert "- " in pattern.resolution_steps
