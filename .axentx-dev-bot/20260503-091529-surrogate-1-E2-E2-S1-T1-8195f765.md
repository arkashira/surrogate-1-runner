// Tag model with CRUD operations (in-memory store for MVP)
export interface Tag {
  id: string;
  name: string;
  slug: string;
  createdAt: Date;
  updatedAt: Date;
  deletedAt: Date | null;
}

class TagModel {
  private store = new Map<string, Tag>();

  private makeSlug(name: string): string {
    return name.toLowerCase().replace(/\s+/g, '-').replace(/[^\w-]+/g, '');
  }

  create(name: string): Tag {
    const id = crypto.randomUUID();
    const slug = this.makeSlug(name);
    const now = new Date();
    const tag: Tag = { id, name, slug, createdAt: now, updatedAt: now, deletedAt: null };
    this.store.set(id, tag);
    return tag;
  }

  rename(id: string, newName: string): Tag | null {
    const tag = this.store.get(id);
    if (!tag || tag.deletedAt) return null;
    tag.name = newName;
    tag.slug = this.makeSlug(newName);
    tag.updatedAt = new Date();
    this.store.set(id, tag);
    return tag;
  }

  delete(id: string): boolean {
    const tag = this.store.get(id);
    if (!tag || tag.deletedAt) return false;
    tag.deletedAt = new Date();
    this.store.set(id, tag);
    return true;
  }

  findById(id: string): Tag | null {
    const tag = this.store.get(id);
    return tag && !tag.deletedAt ? tag : null;
  }

  list(): Tag[] {
    return Array.from(this.store.values()).filter((t) => !t.deletedAt);
  }
}

export const tagModel = new TagModel();
