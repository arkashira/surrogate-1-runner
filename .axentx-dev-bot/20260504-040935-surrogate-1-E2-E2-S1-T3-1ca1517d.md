##### Resource Manager
The resource manager is implemented as an in-memory store that maintains the current allocations for all agents. It uses a `sync.RWMutex` to ensure thread safety during read and write operations.

```go
type ResourceSpec struct {
	CPU    float64 `json:"cpu"`    // number of CPU cores
	Memory int64   `json:"memory"` // memory in MB
}

type allocationStore struct {
	mu          sync.RWMutex
	allocations map[string]ResourceSpec // key = agent ID
}

var store = &allocationStore{
	allocations: make(map[string]ResourceSpec),
}

func GetAllocations() map[string]ResourceSpec {
	store.mu.RLock()
	defer store.mu.RUnlock()
	clone := make(map[string]ResourceSpec, len(store.allocations))
	for k, v := range store.allocations {
		clone[k] = v
	}
	return clone
}

func SetAllocation(agentID string, spec ResourceSpec) {
	store.mu.Lock()
	defer store.mu.Unlock()
	store.allocations[agentID] = spec
}
```

##### HTTP Handlers
HTTP handlers are provided for viewing and adjusting the resource allocations. These handlers support `GET` and `POST` methods for retrieving current allocations and updating them, respectively.

```go
func dashboardHandler(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodGet {
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}
	allocs := GetAllocations()
	w.Header().Set("Content-Type", "application/json")
	if err := json.NewEncoder(w).Encode(allocs); err != nil {
		http.Error(w, "failed to encode response", http.StatusInternalServerError)
	}
}

type adjustRequest struct {
	AgentID string       `json:"agent_id"`
	Spec    ResourceSpec `json:"spec"`
}

func adjustHandler(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "method not allowed", http.StatusMethodNotAllowed)
		return
	}
	var req adjustRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "invalid JSON payload", http.StatusBadRequest)
		return
	}
	if req.AgentID == "" {
		http.Error(w, "agent_id required", http.StatusBadRequest)
		return
	}
	SetAllocation(req.AgentID, req.Spec)
	w.WriteHeader(http.StatusNoContent)
}
```
