"""
Performance monitoring utilities for the multi‑GPU bridge.

This module provides a lightweight, cross‑platform interface for querying
GPU performance metrics.  It prefers the NVIDIA Management Library (NVML)
via the `pynvml` package, but falls back to `nvidia-smi` when the
library is unavailable.  The metrics include:

* GPU name
* GPU utilization (%)
* Memory utilization (%)
* Memory used (MiB)
* Memory total (MiB)
* Temperature (°C)
* Power draw (W)

The :class:`PerformanceMonitor` class exposes a simple API that can be
used by the system‑tray UI or any other component that needs real‑time
GPU statistics.
"""

import json
import subprocess
import sys
import threading
import time
from dataclasses import dataclass
from typing import Callable, Dict, Iterable, List, Optional

# --------------------------------------------------------------------------- #
# Helper utilities
# --------------------------------------------------------------------------- #

def _run_nvidia_smi() -> List[Dict[str, str]]:
    """
    Execute `nvidia-smi` and parse its CSV output.

    Returns a list of dictionaries, one per GPU.
    """
    cmd = [
        "nvidia-smi",
        "--query-gpu=name,utilization.gpu,utilization.memory,memory.used,memory.total,temperature.gpu,power.draw",
        "--format=csv,noheader,nounits",
    ]
    try:
        output = subprocess.check_output(cmd, text=True, stderr=subprocess.DEVNULL)
    except (subprocess.CalledProcessError, FileNotFoundError):
        return []

    metrics = []
    for line in output.strip().splitlines():
        parts = [p.strip() for p in line.split(",")]
        if len(parts) != 7:
            continue
        (
            name,
            gpu_util,
            mem_util,
            mem_used,
            mem_total,
            temp,
            power,
        ) = parts
        metrics.append(
            {
                "name": name,
                "gpu_util": int(gpu_util),
                "mem_util": int(mem_util),
                "mem_used": int(mem_used),
                "mem_total": int(mem_total),
                "temp": int(temp),
                "power": float(power),
            }
        )
    return metrics
