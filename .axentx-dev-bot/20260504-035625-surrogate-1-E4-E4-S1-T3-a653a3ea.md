- **AC1** – The UI must display a table of jobs where each row shows *Job ID*, *Status*, and *Creation Time*.  
- **AC2** – When a new job is created (via the API or another UI action), the job list must refresh within **2 seconds** and include the new entry.  
- **AC3** – If the job API returns an error (HTTP 4xx/5xx or network failure), the UI must show a non‑blocking error banner with the message **“Unable to load jobs. Please try again.”** and must not crash.

// src/components/JobList.test.tsx
import { render, screen, waitFor } from '@testing-library/react';
import JobList from './JobList';
import * as api from '../api/jobs';
import userEvent from '@testing-library/user-event';

jest.mock('../api/jobs');

describe('JobList', () => {
  const mockJobs = [
    { id: '1', status: 'running', createdAt: '2026-05-18T12:00:00Z' },
    { id: '2', status: 'completed', createdAt: '2026-05-17T09:30:00Z' },
  ];

  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('renders jobs table with correct columns', async () => {
    (api.fetchJobs as jest.Mock).mockResolvedValueOnce(mockJobs);
    render(<JobList />);
    await waitFor(() => expect(screen.getByRole('table')).toBeInTheDocument());

    expect(screen.getByText('Job ID')).toBeInTheDocument();
    expect(screen.getByText('Status')).toBeInTheDocument();
    expect(screen.getByText('Creation Time')).toBeInTheDocument();

    mockJobs.forEach(job => {
      expect(screen.getByText(job.id)).toBeInTheDocument();
      expect(screen.getByText(job.status)).toBeInTheDocument();
      expect(screen.getByText(/2026-05/)).toBeInTheDocument();
    });
  });

  test('shows error banner on API failure', async () => {
    (api.fetchJobs as jest.Mock).mockRejectedValueOnce(new Error('Network error'));
    render(<JobList />);
    await waitFor(() => expect(screen.getByText(/Unable to load jobs/i)).toBeInTheDocument());
  });

  test('updates list when new job is added', async () => {
    (api.fetchJobs as jest.Mock)
      .mockResolvedValueOnce(mockJobs)          // initial load
      .mockResolvedValueOnce([...mockJobs, { id: '3', status: 'queued', createdAt: '2026-05-18T12:05:00Z' }]); // after refresh

    render(<JobList />);
    await waitFor(() => expect(screen.getByText('1')).toBeInTheDocument());

    // Simulate external job creation
    userEvent.click(screen.getByRole('button', { name: /Refresh/i }));
    await waitFor(() => expect(screen.getByText('3')).toBeInTheDocument());
  });

  test('collapses Job ID column on small screens', async () => {
    (api.fetchJobs as jest.Mock).mockResolvedValueOnce(mockJobs);
    window.innerWidth = 500;
    window.dispatchEvent(new Event('resize'));
    render(<JobList />);
    await waitFor(() => expect(screen.queryByText('Job ID')).not.toBeInTheDocument());
    expect(screen.getByText('1')).toBeInTheDocument(); // ID still visible via tooltip
  });
});
