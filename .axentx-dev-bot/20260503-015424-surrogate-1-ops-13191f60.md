def shard_assign(slug: str, shard_id: int, shard_total: int) -> bool:

def test_parse_parquet_to_pair(tmp_path):
    tbl = pa.table({"prompt": ["hi"], "response": ["ok"], "extra": [1]})
    p = tmp_path / "x.parquet"
    pq.write_table(tbl, p)
    rows = list(parse_to_pair(p))

def test_parse_jsonl_to_pair(tmp_path):
    p = tmp_path / "x.jsonl"
