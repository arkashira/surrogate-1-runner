```python
import os
from typing import List


def get_registered_accounts() -> List[str]:
    """
    Return a list of account identifiers that should be scanned.

    The identifiers are supplied via the ``AXENTX_ACCOUNTS`` environment
    variable as a comma‑separated list.  Leading/trailing whitespace is
    stripped and empty entries are ignored.

    Returns
    -------
    List[str]
        A list of account IDs.  If the env var is missing or empty an
        empty list is returned.
    """
    raw = os.getenv("AXENTX_ACCOUNTS", "")
    return [acct.strip() for acct in raw.split(",") if acct.strip()]
```

## 1. Acceptance Criteria (measurable)

- **AC1 – Scheduler starts**: On application start, a `BackgroundScheduler` is instantiated, configured with a `CronTrigger` that fires every 6 hours, and the scheduler is started.
- **AC2 – Job registration**: For every account returned by `get_registered_accounts()`, a job is added to the scheduler that invokes `run_scan(account_id)` at the correct interval.
- **AC3 – Log file creation**: Each scan execution writes a log file to `/opt/axentx/surrogate-1/logs/scanner/` with a filename pattern `scan_<account_id>_<YYYYMMDD_HHMMSS>.log`.
- **AC4 – Failure handling**: If `run_scan()` raises an exception, the exception is caught, an error entry is written to the same log file, and the scheduler continues to schedule subsequent runs.
- **AC5 – Health endpoint**: A FastAPI endpoint `/api/v1/scanner/status` returns JSON mapping each `account_id` to the ISO‑8601 UTC timestamp of its next scheduled run.
- **AC6 – Graceful shutdown**: On receiving `SIGTERM` or `SIGINT`, the application shuts down the scheduler cleanly without leaving orphaned threads.
- **AC7 – Time zone correctness**: All timestamps used for scheduling and logging are in UTC (`datetime.datetime.now(datetime.timezone.utc)`).

```python
def test_scheduler_adds_jobs_for_each_account(scheduler):
    scheduler.start()
    assert len(scheduler.scheduler.get_jobs()) == len(scheduler.accounts)
    for job in scheduler.scheduler.get_jobs():
        assert job.trigger.__class__.__name__ == 'CronTrigger'

    scheduler.scheduler._run_job(scheduler.scheduler.get_jobs()[0])  # simulate run
    scheduler.run_scan.assert_called_once_with('acct1')

def test_log_file_created_on_success(tmp_path, scheduler, monkeypatch):
    log_dir = tmp_path / "logs"
    monkeypatch.setenv('LOG_DIR', str(log_dir))
    scheduler.start()
    scheduler.scheduler._run_job(scheduler.scheduler.get_jobs()[0])
    logs = list(log_dir.glob("scan_acct1_*.log"))
    assert len(logs) == 1
    with open(logs[0]) as f:
        content = f.read()
    assert "Scan started" in content
    assert "Scan completed" in content
```
