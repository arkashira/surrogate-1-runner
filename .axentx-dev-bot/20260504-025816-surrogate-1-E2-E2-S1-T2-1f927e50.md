"""
Coverage visualizer module.

This module provides a simple API to generate an HTML coverage report
that highlights uncovered lines in source files.  It is intentionally
light‑weight to keep report generation under the 5 s threshold for
medium‑sized projects.

The module expects coverage data in the following format:

```python
{
    "path/to/file.py": {
        "lines": [1, 2, 3, 4, 5, ...],          # all lines in the file
        "executed": {1, 3, 5, ...}              # set of executed line numbers
    },
    ...
}
```

The `generate_report` function writes an HTML file to `output_path`
and returns the path for convenience.
"""

import os
from pathlib import Path
from typing import Dict, Iterable, Set

from ..utils.html_formatter import format_coverage_html


def _normalize_path(file_path: str) -> str:
    """Return a relative path suitable for display."""
    return os.path.relpath(file_path, start=os.getcwd())


def generate_report(
    coverage_data: Dict[str, Dict[str, Iterable[int] | Set[int]]],
    output_path: str | Path,
    title: str = "Coverage Report",
) -> Path:
    """
    Generate an HTML coverage report.

    Parameters
    ----------
    coverage_data:
        Mapping from file paths to coverage information.
    output_path:
        Destination path for the generated HTML file.
    title:
        Title displayed in the report header.

    Returns
    -------
    Path
        The absolute path to the generated report.
    """
    # Ensure output directory exists
    output_path = Path(output_path).expanduser().resolve()
    output_path.parent.mkdir(parents=True, exist_ok=True)

    # Build the HTML content
    html_body = format_coverage_html(coverage_data)

    # Wrap in a minimal HTML skeleton
    html = f"""<!DOCTYPE html>
<html lang="en">
<head>
<meta charset="utf-8">
<title>{title}</title>
<style>
body {{ font-family: monospace; background: #f5f5f5; }}
table {{ border-collapse: collapse; width: 100%; }}
th, td {{ border: 1px solid #ddd; padding: 4px; }}
th {{ background: #333; color: #fff; }}
.uncovered {{ background: #ffdddd; }}
</style>
</head>
<body>
<h1>{title}</h1>
{html_body}
</body>
</html>"""

    # Write to file
    output_path.write_text(html, encoding="utf-8")
    return output_path
