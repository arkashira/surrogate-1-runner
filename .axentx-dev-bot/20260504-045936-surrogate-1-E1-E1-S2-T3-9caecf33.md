"""
Signature Drift Dashboard – FastAPI implementation
"""

import os
import json
import sqlite3
import threading
import time
from datetime import datetime, timedelta
from typing import List, Optional

from fastapi import FastAPI, HTTPException, Query, Request
from fastapi.responses import JSONResponse, FileResponse
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel, Field

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #

DATABASE_PATH = os.getenv("DATABASE_PATH", "./drift_detections.db")
STATIC_DIR = os.getenv("STATIC_DIR", "./static")
PORT = int(os.getenv("PORT", 8000))
# How often the background worker should poll for new detections (seconds)
POLL_INTERVAL = int(os.getenv("POLL_INTERVAL", 10))

# --------------------------------------------------------------------------- #
# Database helpers
# --------------------------------------------------------------------------- #

def get_db_connection() -> sqlite3.Connection:
    conn = sqlite3.connect(DATABASE_PATH, check_same_thread=False)
    conn.row_factory = sqlite3.Row
    return conn

def init_db() -> None:
    """Create tables if they don't exist."""
    conn = get_db_connection()
    conn.execute(
        """
        CREATE TABLE IF NOT EXISTS detections (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp TEXT NOT NULL,
            pipeline TEXT NOT NULL,
            api_request TEXT NOT NULL,
            severity TEXT NOT NULL CHECK(severity IN ('high','medium','low')),
            details TEXT
        )
        """
    )
    conn.commit()
    conn.close()

# --------------------------------------------------------------------------- #
# Pydantic models

class Detection(BaseModel):

    details: Optional[str] = None
