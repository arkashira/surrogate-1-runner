import numpy as np
from sklearn.linear_model import LinearRegression
import matplotlib.pyplot as plt

class ForecastModel:
    def __init__(self, historical_data):
        self.historical_data = historical_data
        self.model = LinearRegression()

    def train(self):
        X = np.array(range(len(self.historical_data))).reshape(-1, 1)
        y = np.array(self.historical_data)
        self.model.fit(X, y)

    def predict(self, days):
        X_pred = np.array(range(len(self.historical_data), len(self.historical_data) + days)).reshape(-1, 1)
        return self.model.predict(X_pred)

    def plot_forecast(self, forecast, confidence_intervals=None):
        plt.figure(figsize=(10, 6))
        plt.plot(self.historical_data, label='Historical Data')
        plt.plot(range(len(self.historical_data), len(self.historical_data) + len(forecast)), forecast, label='Forecast')
        if confidence_intervals is not None:
            plt.fill_between(range(len(self.historical_data), len(self.historical_data) + len(forecast)),
                            forecast - confidence_intervals,
                            forecast + confidence_intervals,
                            alpha=0.2, label='Confidence Interval')
        plt.xlabel('Days')
        plt.ylabel('Cost')
        plt.title('Cost Forecast')
        plt.legend()
        plt.show()

import pytest
import numpy as np
import pandas as pd
from forecast_engine import ForecastModel

@pytest.fixture
def sample_series():
    # 30 days of monotonically increasing cost data
    dates = pd.date_range(end=pd.Timestamp('2024-04-30'), periods=30)
    values = np.linspace(100, 200, 30)  # linear trend
    return pd.Series(data=values, index=dates)

def test_fit_computes_coefficients(sample_series):
    model = ForecastModel()
    model.fit(sample_series)
    # slope should be approx (200-100)/29
    assert pytest.approx(model.slope, rel=0.01) == (200-100)/29
    assert pytest.approx(model.intercept, rel=0.01) == 100 - model.slope * 0

def test_fit_returns_high_r2(sample_series):
    model = ForecastModel()
    model.fit(sample_series)
    assert model.r2_score >= 0.6

def test_predict_returns_30_values(sample_series):
    model = ForecastModel()
    model.fit(sample_series)
    forecast, lower, upper = model.predict_next_month()
    assert len(forecast) == 30
    assert len(lower) == 30 and len(upper) == 30
    # forecast values must be numeric
    assert all(isinstance(v, float) for v in forecast)

def test_adjust_parameters_overrides_training(sample_series):
    model = ForecastModel()
    model.fit(sample_series)
    model.adjust_parameters(slope=2.0, intercept=50.0)
    forecast, _, _ = model.predict_next_month()
    # first forecast day should equal 2*day_index + 50
    expected_first = 2.0 * 0 + 50.0
    assert pytest.approx(forecast[0]) == expected_first

def test_plot_forecast_creates_figure(sample_series, monkeypatch):
    import matplotlib.pyplot as plt

    assert saved.get('path') == "/tmp/forecast.png"

    model = ForecastModel()
    empty = pd.Series([], dtype=float)
    with pytest.raises(ValueError, match="input series is empty"):
