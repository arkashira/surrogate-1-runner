        type: String,
        required: [true, 'Metric unit is required'],
        trim: true,

        min: [0, 'Target must be a non‑negative number'],

        },
      },
      timeframe: {
        type: String,
        default: null,
        trim: true,
      },
    },
  ],
});

module.exports = mongoose.model('ValidationProject', ValidationProjectSchema);
