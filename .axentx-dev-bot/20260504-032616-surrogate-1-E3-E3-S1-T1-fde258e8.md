name: Test Coverage

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  workflow_dispatch:

jobs:
  test-coverage:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'
          cache: 'pip'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -e ".[test]" pytest pytest-cov coverage

      - name: Run tests with coverage
        run: |
          pytest --cov=src --cov-report=xml --cov-report=term --cov-report=term-missing \
            --junitxml=test-results.xml -v

      - name: Upload coverage to Codecov
        if: github.event_name == 'pull_request' || github.event_name == 'push'
        uses: codecov/codecov-action@v4
        with:
          file: ./coverage.xml
          flags: unittests
          name: surrogate-1
          fail_ci_if_error: false

      - name: Publish Test Results
        if: always()
        uses: dorny/test-reporter@v1
        with:
          name: Test Results
          path: test-results.xml
          reporter: java-junit

      - name: Upload coverage artifact
        uses: actions/upload-artifact@v4
        with:
          name: coverage-report
          path: |
            coverage.xml
            coverage/
          retention-days: 30

      - name: Generate coverage summary
        run: |
          echo "## Coverage Summary" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          coverage report --format=markdown >> $GITHUB_STEP_SUMMARY

      - name: Check minimum coverage threshold
        run: |
          COVERAGE=$(python -c "import xml.etree.ElementTree as ET; tree = ET.parse('coverage.xml'); root = tree.getroot(); print(float(root.get('line-rate', 0)))")
          echo "Coverage: $COVERAGE"
          python -c "import sys; sys.exit(0 if float('$COVERAGE') >= 0.70 else 1)" || {
            echo "❌ Coverage below 70% threshold"
            exit 1
          }
          echo "✅ Coverage meets 70% threshold"

      - name: Generate filtered results
        if: always()
        run: |
          python -c "
          from src.ci_integration import CIIntegration
          ci = CIIntegration()
          ci.load_test_results('test-results.xml')
          ci.load_coverage('coverage.xml')
          filtered = ci.filter_results(status='failed')
          ci.export_results(filtered, 'json', 'filtered-results.json')
          ci.export_results(filtered, 'junit', 'filtered-junit.xml')
          "
        continue-on-error: true

      - name: Upload filtered results artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: filtered-test-results
          path: |
            filtered-results.json
            filtered-junit.xml
          retention-days: 90
