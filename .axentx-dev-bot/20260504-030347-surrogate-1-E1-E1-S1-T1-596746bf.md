/**
 * Docker Compose → Kubernetes manifest converter
 *
 * Usage (CLI):
 *   node tools/compose-parser.js <path-to-docker-compose.yml> [output-dir]
 *
 * The script parses a docker‑compose.yml file and emits a set of Kubernetes
 * YAML manifests (Deployments, Services, PersistentVolumeClaims) that can be
 * applied with `kubectl apply -f`.
 *
 * Exported API:
 *   parseCompose(yamlString) → object
 *   generateK8sManifests(composeObj) → Array<object>
 *
 * Limitations:
 *   - Only a subset of docker‑compose fields are supported (image, ports,
 *     environment, volumes). Unsupported fields generate warnings.
 *   - Persistent storage is mapped to a 1Gi PVC per named volume.
 *   - Service discovery is provided via a ClusterIP Service exposing the
 *     container ports.
 */

const fs = require('fs');
const path = require('path');
const yaml = require('js-yaml');

/**
 * Parse a docker‑compose YAML string into a JavaScript object.
 * Throws on invalid YAML.
 */
function parseCompose(yamlString) {
  const doc = yaml.load(yamlString);
  if (!doc || typeof doc !== 'object' || !doc.services) {
    throw new Error('Invalid docker‑compose file: missing top‑level "services" key.');
  }
  return doc;
}

/**
 * Convert a parsed compose object into an array of Kubernetes manifest objects.
 *
 * @param {object} compose - Result of parseCompose()
 * @returns {Array<object>} - List of K8s resources (Deployments, Services, PVCs)
 */
function generateK8sManifests(compose) {
  const manifests = [];

  const services = compose.services;
  for (const [svcName, svcDef] of Object.entries(services)) {
    // ---- Validation ---------------------------------------------------------
    if (!svcDef.image) {
      throw new Error(`Service "${svcName}" is missing required "image" field.`);
    }

    // ---- Deployment ---------------------------------------------------------
    const deployment = {
      apiVersion: 'apps/v1',
      kind: 'Deployment',
      metadata: {
        name: svcName,
        labels: { app: svcName },
      },
      spec: {
        replicas: svcDef.deploy && svcDef.deploy.replicas ? svcDef.deploy.replicas : 1,
        selector: {
          matchLabels: { app: svcName },
        },
        template: {
          metadata: {
            labels: { app: svcName },
          },
          spec: {
            containers: [
              {
                name: svcName,
                image: svcDef.image,
                env: Object.entries(svcDef.environment || {}).map(([k, v]) => ({
                  name: k,
                  value: String(v),
                })),
                ports: [],
                volumeMounts: [],
              },
            ],
            volumes: [],
          },
        },
      },
    };

    // ---- Ports & Service ----------------------------------------------------
    const containerPorts = [];
    const servicePorts = [];

    if (svcDef.ports) {
      svcDef.ports.forEach((portSpec) => {
        // Port spec can be "HOST:CONTAINER", "CONTAINER", or "HOST:CONTAINER/PROTOCOL"
        const [hostPart, containerPart] = String(portSpec).split(':');
        let containerPortStr, hostPort;
        if (containerPart === undefined) {
          // Only container port provided
          containerPortStr = hostPart;
        } else {
          hostPort = parseInt(hostPart, 10);
          containerPortStr = containerPart.split('/')[0];
