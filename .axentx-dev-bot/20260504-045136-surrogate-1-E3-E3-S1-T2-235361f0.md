from tkinter import ttk, messagebox
from typing import Dict, List, Optional, Tuple
import json

    def __init__(self, root, anomaly_data: Dict):
        self.root = root
        self.anomaly_data = anomaly_data
        self.modal = None
        self.create_modal()
        
    def create_modal(self):
        """Create the remediation modal window"""
        self.modal = tk.Toplevel(self.root)
        self.modal.title("Remediation Steps")
        self.modal.geometry("800x600")
        self.modal.resizable(False, False)
        
        # Modal frame
        frame = ttk.Frame(self.modal, padding="20")
        frame.grid(row=0, column=0, sticky=(tk.W, tk.E, tk.N, tk.S))
        
        # Header
        ttk.Label(frame, text="Remediation Steps", font=("Arial", 16, "bold")).grid(
            row=0, column=0, columnspan=2, pady=(0, 10)
        )
        
        # Anomaly info
