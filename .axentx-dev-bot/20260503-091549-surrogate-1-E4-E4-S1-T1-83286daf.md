# /opt/axentx/surrogate-1/src/queues/notifications.py
from __future__ import annotations

import json
import os
from dataclasses import dataclass, asdict
from typing import Literal, Optional

import redis
from rq import Queue

REDIS_URL = os.getenv("REDIS_URL", "redis://localhost:6379/0")

_queue = None

def get_queue() -> Queue:
    global _queue
    if _queue is None:
        conn = redis.from_url(REDIS_URL)
        _queue = Queue("notifications", connection=conn, default_timeout=3600)
    return _queue

class SLAStatus(str):
    OK = "ok"
    DEGRADED = "degraded"
    OUTAGE = "outage"

@dataclass
class NotificationJob:
    requestId: str
    channel: str  # Slack channel or email list identifier
    template: str  # template key or raw template name
    publicUrl: str
    slaStatus: SLAStatus
    # Optional overrides
    slackWebhookUrl: Optional[str] = None
    emailRecipients: Optional[list[str]] = None
    emailSmtpHost: Optional[str] = None
    emailSmtpPort: Optional[int] = None
    emailFrom: Optional[str] = None

    def to_dict(self):
        return asdict(self)

def enqueue_notification(job: NotificationJob) -> str:
    """
    Enqueue a notification job. Returns job id.
    Retries are handled by RQ (configured in worker).
    """
    q = get_queue()
    job_id = q.enqueue(
        "src.workers.notify.process_notification",
        job.to_dict(),
        retry=None,  # rely on worker-level retry policy or RQ max_retries
        result_ttl=3600,
        failure_ttl=86400,
    )
    return str(job_id)

def enqueue_notification_raw(
    requestId: str,
    channel: str,
    template: str,
    publicUrl: str,
    slaStatus: SLAStatus,
    **kwargs,
) -> str:
    job = NotificationJob(
        requestId=requestId,
        channel=channel,
        template=template,
        publicUrl=publicUrl,
        slaStatus=slaStatus,
        **kwargs,
    )
    return enqueue_notification(job)
```
