CDN_ROOT = "https://huggingface.co/datasets"

def list_date_files(repo: str, date: str) -> list[str]:
    """
    List files under repo/{date}/ (non-recursive + one-level subfolders)
    using minimal tree calls. Returns paths relative to repo root.
    """
    headers = {"Authorization": f"Bearer {os.getenv('HF_TOKEN', '')}"} if os.getenv("HF_TOKEN") else {}

    def _tree(path: str) -> list[dict]:
        url = f"{HF_API}/repos/{repo}/tree/{path}"
        r = requests.get(url, headers=headers, timeout=30)
        if r.status_code == 404:
            return []
        r.raise_for_status()
        return r.json()

    top = _tree(date)
    out = []
    for entry in top:
        if entry.get("type") == "file":
            out.append(f"{date}/{entry['name']}")
        elif entry.get("type") == "tree":
            for se in _tree(entry["path"]):
                if se.get("type") == "file":
                    out.append(f"{entry['path']}/{se['name']}")
    return out

def build_manifest(repo: str, date: str, out_path: Path) -> dict:
    files = list_date_files(repo, date)
    manifest = {
        "repo": repo,
        "date": date,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": files,
        "cdn_root": CDN_ROOT,
    }
    out_path.parent.mkdir(parents=True, exist_ok=True)
    out_path.write_text(json.dumps(manifest, indent=2))

    return manifest
