## 1. Acceptance Criteria  

- **Add/Remove Subagents** – The orchestration API must expose endpoints to register a new subagent and deregister an existing one, returning a 201/204 status respectively.  
- **Task Routing** – When a task is submitted to the API, the system must route it to the correct subagent based on a `subagentId` field in the payload, and return the subagent’s response with a 200 status.  
- **Configuration** – Subagents can be configured with a set of allowed task types via a `config` endpoint; the API must validate that a task’s `type` is permitted before routing.  
- **Standardized Communication** – All subagent interactions must use a JSON payload with the schema `{ taskId, type, payload }` and return `{ taskId, status, result }`.  
- **Error Handling** – If a subagent is unavailable or returns an error, the API must respond with a 502 status and include the original error message.  
- **Security** – All endpoints must require an `Authorization` header containing a valid bearer token; requests without it receive a 401.  
- **Logging** – Each request and response must be logged with a unique correlation ID for traceability.

// api.test.js
const { registerSubagent, deregisterSubagent, routeTask, configureSubagent } = require('./api');
const httpMocks = require('node-mocks-http');

describe('Subagent API', () => {
  test('registerSubagent returns 201 and stores agent', async () => {
    const req = httpMocks.createRequest({
      method: 'POST',
      url: '/subagents',
      body: { id: 'sa-1', endpoint: 'http://localhost:4000' },
      headers: { Authorization: 'Bearer valid' }
    });
    const res = httpMocks.createResponse();
    await registerSubagent(req, res);
    expect(res.statusCode).toBe(201);
    expect(await res._getJSONData()).toEqual({ message: 'registered' });
  });

  test('deregisterSubagent returns 204', async () => {
    // assume sa-1 already registered
    const req = httpMocks.createRequest({
      method: 'DELETE',
      url: '/subagents/sa-1',
      headers: { Authorization: 'Bearer valid' }
    });
    const res = httpMocks.createResponse();
    await deregisterSubagent(req, res);
    expect(res.statusCode).toBe(204);
  });

  test('routeTask forwards to correct subagent', async () => {
    // mock subagent response
    jest.spyOn(global, 'fetch').mockResolvedValue({
      json: () => Promise.resolve({ taskId: 't1', status: 'ok', result: 42 })
    });

    const req = httpMocks.createRequest({
      method: 'POST',
      url: '/tasks',
      body: { taskId: 't1', type: 'calc', payload: { a: 2, b: 3 }, subagentId: 'sa-1' },
      headers: { Authorization: 'Bearer valid' }
    });
    const res = httpMocks.createResponse();
    await routeTask(req, res);
    expect(res.statusCode).toBe(200);
    expect(await res._getJSONData()).toEqual({ taskId: 't1', status: 'ok', result: 42 });
  });

  test('configureSubagent validates task type', async () => {
    const req = httpMocks.createRequest({
      method: 'PUT',
      url: '/subagents/sa-1/config',
      body: { allowedTypes: ['calc'] },
      headers: { Authorization: 'Bearer valid' }
    });
    const res = httpMocks.createResponse();
    await configureSubagent(req, res);
    expect(res.statusCode).toBe(200);
  });

  test('unauthorized request returns 401', async () => {
    const req = httpMocks.createRequest({
      method: 'GET',
      url: '/subagents',
      headers: {}
    });
    const res = httpMocks.createResponse();
    await registerSubagent(req, res);
    expect(res.statusCode).toBe(401);
  });
});
