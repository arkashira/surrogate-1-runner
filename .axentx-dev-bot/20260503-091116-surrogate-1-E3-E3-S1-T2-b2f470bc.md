// TagManager.unit.test.tsx
import { render, screen, fireEvent } from '@testing-library/react';
import TagManager from './TagManager';

describe('TagManager', () => {
  test('renders deterministic suggestions (3–5)', () => {
    render(<TagManager documentId="d1" initialTags={[]} suggestions={['AI','ML','NLP']} />);
    const chips = screen.getAllByRole('button', { name: /add tag/i });
    expect(chips.length).toBeGreaterThanOrEqual(3);
    expect(chips.length).toBeLessThanOrEqual(5);
  });

  test('clicking suggestion adds tag', () => {
    render(<TagManager documentId="d1" initialTags={[]} suggestions={['AI']} />);
    fireEvent.click(screen.getByRole('button', { name: /add tag.*AI/i }));
    expect(screen.getByText('AI')).toBeInTheDocument();
  });

  test('inline rename: Enter commits, Escape cancels', () => {
    render(<TagManager documentId="d1" initialTags={['AI']} suggestions={[]} />);
    fireEvent.click(screen.getByRole('button', { name: /edit tag/i }));
    const input = screen.getByRole('textbox');
    fireEvent.change(input, { target: { value: 'ML' } });
    fireEvent.keyDown(input, { key: 'Enter', code: 'Enter' });
    expect(screen.getByText('ML')).toBeInTheDocument();

    fireEvent.click(screen.getByRole('button', { name: /edit tag/i }));
    const input2 = screen.getByRole('textbox');
    fireEvent.change(input2, { target: { value: 'CV' } });
    fireEvent.keyDown(input2, { key: 'Escape', code: 'Escape' });
    expect(screen.queryByText('CV')).not.toBeInTheDocument();
    expect(screen.getByText('ML')).toBeInTheDocument();
  });

  test('remove tag decreases count', () => {
    render(<TagManager documentId="d1" initialTags={['AI','ML']} suggestions={[]} />);
    fireEvent.click(screen.getByRole('button', { name: /remove tag.*AI/i }));
    expect(screen.queryByText('AI')).not.toBeInTheDocument();
    expect(screen.getByText('ML')).toBeInTheDocument();
  });
});
