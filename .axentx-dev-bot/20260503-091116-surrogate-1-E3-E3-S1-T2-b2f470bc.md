import { DocumentTag } from '../types';

const API_BASE = '/api';

export async function fetchTagsForDocument(documentId: string): Promise<DocumentTag[]> {
  const res = await fetch(`${API_BASE}/documents/${documentId}/tags`);
  if (!res.ok) throw new Error('Failed to fetch tags');
  return res.json();
}

export async function updateTagsForDocument(documentId: string, tags: DocumentTag[]): Promise<DocumentTag[]> {
  const res = await fetch(`${API_BASE}/documents/${documentId}/tags`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(tags),
  });
  if (!res.ok) throw new Error('Failed to update tags');
  return res.json();
}

export async function fetchDocumentsByTag(tag: string): Promise<Array<{ id: string; title: string; tags: DocumentTag[] }>> {
  const res = await fetch(`${API_BASE}/documents?tag=${encodeURIComponent(tag)}`);
  if (!res.ok) throw new Error('Failed to fetch documents by tag');
  return res.json();
}

export async function fetchAllTags(): Promise<Array<{ name: string; count: number }>> {
  const res = await fetch(`${API_BASE}/tags`);
  if (!res.ok) return [];
  return res.json();
}
```
