// TagManager.tsx tests
describe('TagManager', () => {
  it('renders tag suggestions', () => {
    const mockDocument = { id: 1, text: 'Example document' };
    const tagManager = render(<TagManager document={mockDocument} />);
    expect(tagManager.getByTestId('tag-suggestions')).toBeInTheDocument();
  });

  it('adds tag on click', () => {
    const mockDocument = { id: 1, text: 'Example document' };
    const tagManager = render(<TagManager document={mockDocument} />);
    const tagSuggestion = tagManager.getByTestId('tag-suggestion-1');
    fireEvent.click(tagSuggestion);
    expect(tagManager.getByTestId('document-tags')).toContainHTML('Tag 1');
  });

  it('removes tag on delete', () => {
    const mockDocument = { id: 1, text: 'Example document', tags: ['Tag 1'] };
    const tagManager = render(<TagManager document={mockDocument} />);
    const tag = tagManager.getByTestId('document-tag-1');
    fireEvent.click(tag);
    expect(tagManager.queryByTestId('document-tag-1')).not.toBeInTheDocument();
  });
});

// TagBrowse.tsx tests
describe('TagBrowse', () => {
  it('renders document list', () => {
    const mockDocuments = [{ id: 1, text: 'Example document 1' }, { id: 2, text: 'Example document 2' }];
    const tagBrowse = render(<TagBrowse documents={mockDocuments} />);
    expect(tagBrowse.getByTestId('document-list')).toBeInTheDocument();
  });

  it('filters document list by tag', () => {
    const mockDocuments = [{ id: 1, text: 'Example document 1', tags: ['Tag 1'] }, { id: 2, text: 'Example document 2' }];
    const tagBrowse = render(<TagBrowse documents={mockDocuments} />);
    const tagFilter = tagBrowse.getByTestId('tag-filter');
    fireEvent.change(tagFilter, { target: { value: 'Tag 1' } });
    expect(tagBrowse.getByTestId('document-list')).toContainHTML('Example document 1');
    expect(tagBrowse.queryByTestId('document-2')).not.toBeInTheDocument();
  });
});
