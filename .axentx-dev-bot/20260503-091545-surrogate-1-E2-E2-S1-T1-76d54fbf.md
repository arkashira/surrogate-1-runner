import { Router } from 'express';
import { Request } from 'express';
import { Response } from 'express';
import { getPublicRequest } from '../services/publicRequestService';
import { v4 as uuidv4 } from 'uuid';

const publicRequestsRouter = Router();

publicRequestsRouter.get('/requests/:id', async (req: Request, res: Response) => {
  const requestId = req.params.id;
  const request = await getPublicRequest(requestId);
  const publicUrl = uuidv4();
  request.publicUrl = publicUrl;
  // Omit internal-only notes unless explicitly marked visible
  const visibleNotes = request.notes.filter((note) => note.isVisible);
  request.notes = visibleNotes;
  res.json({
    title: request.title,
    currentStatus: request.status,
    sla: request.sla,
    timelineEntries: request.timelineEntries,
    publicUrl: publicUrl,
    notes: visibleNotes,
  });
});

export default publicRequestsRouter;
```
