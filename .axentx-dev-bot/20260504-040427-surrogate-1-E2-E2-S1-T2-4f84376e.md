# This is a basic example of how to run the surrogate-1 workflow in Jenkins

pipeline {
    agent any

    environment {
        SURROGATE_1_VERSION = 'latest'
        DATASET_SLUG = 'axentx/surrogate-1-training-pairs'
        SHARD_ID = '0'  # Change to your desired shard ID
    }

    stages {
        stage('Prepare Environment') {
            steps {
                script {
                    echo "Setting up environment for Surrogate-1 ${env.SURROGATE_1_VERSION}"
                }
            }
        }

        stage('Run Surrogate-1') {
            steps {
                script {
                    echo "Running Surrogate-1 with dataset: ${env.DATASET_SLUG}"
                    echo "Using shard ID: ${env.SHARD_ID}"
                    
                    // Execute the surrogate-1 runner with the specified parameters
                    sh """
                        docker run \
                            -v $(pwd):/workdir \
                            -w /workdir \
                            --env SURROGATE_1_VERSION=${env.SURROGATE_1_VERSION} \
                            --env DATASET_SLUG=${env.DATASET_SLUG} \
                            --env SHARD_ID=${env.SHARD_ID} \
                            axentx/surrogate-1-runner:latest \
                            bin/run.sh
                    """
                }
            }
        }

        stage('Check Results') {
            steps {
                script {
                    echo "Checking results from surrogate-1 run"
                    // Add your custom checks here
                }
            }
        }
    }

    post {
        always {
            echo "Workflow completed for Surrogate-1"
        }
    }
}
