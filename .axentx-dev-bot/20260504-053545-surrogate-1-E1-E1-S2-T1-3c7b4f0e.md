import { GPT4Gateway } from '@/ai_tools/gpt4';
import axios from 'axios';
import { mock } from 'jest-mock-extended';

jest.mock('axios');

describe('GPT4Gateway Unit Tests', () => {
  const validConfig = {
    apiKey: 'sk-test-valid',
    timeoutMs: 2000,
    endpoint: 'https://api.openai.com/v1/chat/completions',
  };

  test('connect() returns session object on success', async () => {
    axios.post.mockResolvedValue({ data: { model: 'gpt-4' } });
    const gateway = new GPT4Gateway(validConfig);
    const session = await gateway.connect();
    expect(session).toHaveProperty('id');
    expect(session).toHaveProperty('createdAt');
  });

  test('connect() throws AuthenticationError on 401', async () => {
    axios.post.mockRejectedValue({ response: { status: 401 } });
    const gateway = new GPT4Gateway(validConfig);
    await expect(gateway.connect()).rejects.toThrow('AuthenticationError');
  });

  test('payload complies with corporate JSON schema', () => {
    const gateway = new GPT4Gateway(validConfig);
    const payload = gateway.buildPayload('Hello');
    const schema = require('@/schemas/gpt4_payload.json');
    const Ajv = require('ajv');
    const validate = new Ajv().compile(schema);
    expect(validate(payload)).toBe(true);
  });

  test('TLS version enforced via axios httpsAgent', () => {
    const gateway = new GPT4Gateway(validConfig);
    const agent = gateway.httpsAgent;
    expect(agent.options.minVersion).toBe('TLSv1.2');
  });

  test('rate‑limit header respected – throttles when remaining = 0', async () => {
    axios.post
      .mockResolvedValueOnce({ headers: { 'x-ratelimit-remaining': '0' } })
      .mockResolvedValue({ data: { model: 'gpt-4' } });

    const gateway = new GPT4Gateway(validConfig);
    const start = Date.now();
    await gateway.sendMessage('test'); // first call consumes limit
    await gateway.sendMessage('test2'); // should wait
    const elapsed = Date.now() - start;
    expect(elapsed).toBeGreaterThanOrEqual(gateway.throttleMs);
  });

  test('no prohibited fields are added to payload', () => {
    const gateway = new GPT4Gateway(validConfig);
    const payload = gateway.buildPayload('sample');
    expect(payload).not.toHaveProperty('user_id');
    expect(JSON.stringify(payload).length).toBeLessThanOrEqual(2048);
  });
