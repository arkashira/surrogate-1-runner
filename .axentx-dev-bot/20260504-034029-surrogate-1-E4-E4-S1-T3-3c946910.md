# test_metrics_exposure.py
import pytest
import requests
from prometheus_client.parser import text_string_to_metric_families

BASE_URL = "http://localhost:8000"

def test_metrics_endpoint_status():
    r = requests.get(f"{BASE_URL}/metrics")
    assert r.status_code == 200

def test_metrics_include_required_labels():
    r = requests.get(f"{BASE_URL}/metrics")
    families = list(text_string_to_metric_families(r.text))
    metric_names = {f.name for f in families}
    required = {
        "task_queue_length",
        "task_latency_seconds",
        "task_success_total",
        "task_failure_total",
        "clarification_loop_count",
    }
    assert required.issubset(metric_names)

    for family in families:
        if family.name in required:
            for sample in family.samples:
                assert "agent_id" in sample.labels
                assert "task_type" in sample.labels

def test_metrics_are_numeric():
    r = requests.get(f"{BASE_URL}/metrics")
    families = list(text_string_to_metric_families(r.text))
    for family in families:
        for sample in family.samples:
            assert isinstance(sample.value, (int, float))

# test_dashboard_json_schema.py
import json
import jsonschema
from jsonschema import validate

SCHEMA_PATH = "grafana_schema.json"  # pre‑downloaded Grafana 10.x schema
DASHBOARD_PATH = "/opt/axentx/surrogate-1/docs/grafana_dashboard.json"

def load_schema():
    with open(SCHEMA_PATH) as f:
        return json.load(f)

def load_dashboard():
    with open(DASHBOARD_PATH) as f:
        return json.load(f)

def test_dashboard_schema_valid():
    schema = load_schema()
    dashboard = load_dashboard()
    validate(instance=dashboard, schema=schema)

def test_dashboard_has_four_panels():
    dashboard = load_dashboard()
    panels = dashboard.get("dashboard", {}).get("panels", [])
    assert len(panels) == 4

def test_panels_use_prometheus_expr_and_datasource():
    dashboard = load_dashboard()
    for panel in dashboard["dashboard"]["panels"]:
        assert "targets" in panel
        for target in panel["targets"]:
            assert "expr" in target
            assert target.get("datasource", {}).get("uid") == "PBFA97B1B506B2093"
