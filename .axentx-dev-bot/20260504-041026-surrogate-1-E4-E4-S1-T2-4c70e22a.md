import os
from typing import Callable, Optional

import jwt
from fastapi import APIRouter, Depends, HTTPException, Request, status
from fastapi.security import HTTPAuthorizationCredentials, HTTPBearer

router = APIRouter()
security = HTTPBearer(auto_error=False)

# Configuration – in production these would be managed securely
JWT_SECRET = os.getenv("JWT_SECRET", "dev-secret")
JWT_ALGORITHM = os.getenv("JWT_ALGORITHM", "HS256")
EXPECTED_AUDIENCE = "surrogate-1"
EXPECTED_SCOPE = "orchestrate"


def verify_jwt(
    credentials: HTTPAuthorizationCredentials = Depends(security),
) -> dict:
    """
    Validate a JWT from the `Authorization: Bearer <token>` header.

    Returns the decoded payload if valid; otherwise raises HTTPException(401).
    """
    if credentials is None or credentials.scheme.lower() != "bearer":
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Missing or malformed Authorization header",
        )

    token = credentials.credentials
    try:
        payload = jwt.decode(
            token,
            JWT_SECRET,
            algorithms=[JWT_ALGORITHM],
            audience=EXPECTED_AUDIENCE,
        )
    except jwt.ExpiredSignatureError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED, detail="Token has expired"
        )
    except jwt.InvalidAudienceError:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail=f"Invalid audience, expected '{EXPECTED_AUDIENCE}'",
        )
    except jwt.InvalidTokenError as exc:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED, detail=str(exc)
        )

    # Verify required scope
    token_scopes = payload.get("scope", "")
    if EXPECTED_SCOPE not in token_scopes.split():
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail=f"Required scope '{EXPECTED_SCOPE}' not present",
        )

    return payload


def auth_dependency(request: Request, payload: dict = Depends(verify_jwt)) -> None:
    """
    FastAPI dependency that injects authentication info into the request.
    Sets `request.state.user_id` for downstream handlers.
    """
    request.state.user_id = payload.get("sub")
    # No return needed; the presence of the dependency enforces auth.
