  try {

      slug: request.slug,
      url: `/r/${request.slug}`, // public URL

  } catch (err) {

**2. Update the `requestService.ts` file to include the newly created request in the request list and sort it by `created_at` in descending order.**

```diff
// /opt/axentx/surrogate-1/src/services/requestService.ts

import { prisma } from '../db';
import { v4 as uuidv4 } from 'uuid';
import slugify from 'slugify';

interface CreateRequestData {
  title: string;
  type: string;
  owner: string;
}

export class RequestService {
  static async createRequest(data: CreateRequestData) {
    const slug = slugify(`${data.title}-${uuidv4().slice(0, 8)}`, { lower: true });

    return await prisma.request.create({
      data: {
        ...data,
        slug,
        status: 'pending', // default initial status
        createdAt: new Date(),
      },
    });
  }

  static async listRequests() {
    return await prisma.request.findMany({
      orderBy: {
        createdAt: 'desc',
      },
    });
  }
}
```
