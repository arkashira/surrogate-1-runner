import { RequestItem, CreateRequestPayload, ValidationError } from '../types/request';

const store: RequestItem[] = [];

function slugify(title: string): string {
  return title
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '')
    .substring(0, 60);
}

function validate(payload: CreateRequestPayload): ValidationError[] {
  const errors: ValidationError[] = [];
  if (!payload.title || !payload.title.trim()) {
    errors.push({ field: 'title', message: 'Title is required' });
  }
  if (!payload.type || !payload.type.trim()) {
    errors.push({ field: 'type', message: 'Type is required' });
  }
  if (!payload.owner || !payload.owner.trim()) {
    errors.push({ field: 'owner', message: 'Owner is required' });
  }
  return errors;
}

export function createRequest(payload: CreateRequestPayload): { request: RequestItem } | { errors: ValidationError[] } {
  const errors = validate(payload);
  if (errors.length > 0) {
    return { errors };
  }

  const id = `req_${Date.now().toString(36)}${Math.random().toString(36).slice(2, 7)}`;
  const titleSlug = slugify(payload.title);
  const slug = `${titleSlug}-${id.slice(0, 8)}`;
  const now = new Date().toISOString();

  const request: RequestItem = {
    id,
    title: payload.title.trim(),
    type: payload.type.trim(),
    owner: payload.owner.trim(),
    status: payload.status || 'open',
    createdAt: now,
    slug,
  };

  // Insert at beginning and keep sorted descending by createdAt
  store.unshift(request);
  store.sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());

  return { request };
}

export function listRequests(): RequestItem[] {
  // Return a stable, descending copy
  return store.slice().sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
}
