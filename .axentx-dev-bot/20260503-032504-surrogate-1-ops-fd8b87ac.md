- Accepts `SHARD_ID`, `SHARD_TOTAL=16`, `DATE`, `HF_TOKEN` via env
- Single `list_repo_tree(path, recursive=False)` per date folder → deterministic shard slice
- Saves file manifest to `manifest/{DATE}/shard{SHARD_ID}.json` (for audit/repro)

- Exits non-zero on unrecoverable errors; logs summary for GitHub Actions

- Aligns with Lightning Studio quota strategy: reuse running Studio for training iterations; zero API calls during data load via pre-saved manifest.

SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))

# ---------- dedup ----------
DEDUP_DB = Path("lib/dedup.py")
if DEDUP_DB.exists():
    # lightweight import of dedup helper
    import importlib.util
    spec = importlib.util.spec_from_file_location("dedup", str(DEDUP_DB))
    dedup_mod = importlib.util.module_from_spec(spec)
    spec.loader.exec_module(dedup_mod)
    seen = dedup_mod.DedupStore()
else:
    class DummyDedup:

def shard_for(slug: str) -> int:
    return int(hashlib.md5(slug.encode()).hexdigest(), 16) % SHARD_TOTAL
