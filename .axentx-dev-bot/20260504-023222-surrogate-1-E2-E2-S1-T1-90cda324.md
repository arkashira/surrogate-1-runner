        results = {}
        for check in checks:

    def _perform_check(self, check: ReferentialIntegrityCheck) -> bool:
        source_table = Table(check.source_table, self.metadata, autoload_with=self.engine)
        target_table = Table(check.target_table, self.metadata, autoload_with=self.engine)

        # Get all source keys
        stmt = select(source_table.c[check.source_key]).distinct()
        source_keys = [row[0] for row in self.engine.execute(stmt)]

        # Check if all source keys exist in target table
        if not source_keys:
            return True

        stmt = select(target_table.c[check.target_key]).where(
            target_table.c[check.target_key].in_(source_keys)
        )
        target_keys = [row[0] for row in self.engine.execute(stmt)]

        missing_keys = set(source_keys) - set(target_keys)
        if missing_keys:
            logger.warning(f"Missing keys in {check.target_table}: {missing_keys}")
        return len(missing_keys) == 0
