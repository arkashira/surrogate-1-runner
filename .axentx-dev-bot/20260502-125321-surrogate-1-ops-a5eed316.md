```python
#!/usr/bin/env python3
"""
Create a deterministic, date-scoped file manifest for a Hugging Face dataset repo.
Usage:
  python3 training_manifest.py <repo_id> <date_folder> [out_json]
Example:
  python3 training_manifest.py datasets/axentx/surrogate-1-data 2026-05-02
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import HfApi

def create_manifest(repo_id: str, date_folder: str, out_path: str) -> dict:
    api = HfApi()
    # Non-recursive listing for the date folder to minimize API usage
    tree = api.list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = sorted(item.rfilename for item in tree if item.type == "file")

    manifest = {"repo": repo_id, "date": date_folder, "files": files}
    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    with open(out_path, "w") as f:
        json.dump(manifest, f, indent=2)
    return manifest

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Usage: training_manifest.py <repo_id> <date_folder> [out_json]")
        sys.exit(1)
    repo_id = sys.argv[1]
    date_folder = sys.argv[2]
    out_path = sys.argv[3] if len(sys.argv) > 3 else f"manifests/{date_folder}/filelist.json"
    m = create_manifest(repo_id, date_folder, out_path)
    print(f"Wrote {len(m['files'])} files to {out_path}")
```
