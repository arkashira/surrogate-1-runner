"""
Encryption utilities for surrogate-1.

This module provides transparent AES‑256 encryption (via Fernet) for data
at rest and ensures that any network communication performed by the
pipeline uses HTTPS (enforced by the `secure_request` wrapper).

The implementation is deliberately simple to aid auditability for GDPR
and HIPAA compliance:

* **Key management** – The symmetric key is supplied via the
  `ENCRYPTION_KEY` environment variable.  The key must be a URL‑safe
  base64‑encoded 32‑byte value, matching Fernet's expectations.
* **Encryption at rest** – `encrypt_file` and `decrypt_file` operate on
  arbitrary files, storing the ciphertext in the same format produced
  by Fernet (nonce + tag + ciphertext, all base64‑encoded).
* **Encryption in transit** – `secure_request` forces HTTPS and verifies
  TLS certificates for all outgoing HTTP calls.

Both functions raise clear exceptions on mis‑configuration, making it
easy for security auditors to verify correct usage.
"""

import os
import base64
import json
from typing import Any, Dict, Optional

import requests
from cryptography.fernet import Fernet, InvalidToken

# --------------------------------------------------------------------------- #
# Key handling
# --------------------------------------------------------------------------- #

def _load_key() -> bytes:
    """
    Load the symmetric encryption key from the ENCRYPTION_KEY environment
    variable.  The key must be a URL‑safe base64‑encoded 32‑byte value.
    """
    key_b64 = os.getenv("ENCRYPTION_KEY")
    if not key_b64:
        raise EnvironmentError(
            "ENCRYPTION_KEY environment variable is not set. "
            "Generate a key with `python -c \"from cryptography.fernet import Fernet; "
            "print(Fernet.generate_key().decode())\"` and export it."
        )
    try:
        # Validate key length by attempting to create a Fernet instance.
        return Fernet(key_b64.encode())._signing_key + Fernet(key_b64.encode())._encryption_key
    except Exception as exc:
        raise ValueError("Invalid ENCRYPTION_KEY format.") from exc


def generate_key() -> str:
    """
    Generate a new Fernet key (URL‑safe base64‑encoded 32‑byte) and return it
    as a string.  Useful for provisioning new environments.
    """
    return Fernet.generate_key().decode()


# --------------------------------------------------------------------------- #
# Core encryptor
# --------------------------------------------------------------------------- #

class DataEncryptor:
    """
    Wrapper around Fernet providing high‑level encrypt/decrypt helpers.
    """

    def __init__(self, key: Optional[bytes] = None):
        if key is None:
            key = _load_key()
        # Fernet expects the full 44‑byte base64 string, not the raw 64‑byte key.
        # Re‑encode the raw bytes back to base64 for Fernet consumption.
        self._fernet = Fernet(base64.urlsafe_b64encode(key))

    def encrypt(self, data: bytes) -> bytes:
        """
        Encrypt arbitrary bytes and return the ciphertext token.
        """
        return self._fernet.encrypt(data)

    def decrypt(self, token: bytes) -> bytes:
        """
        Decrypt a token produced by :meth:`encrypt`.  Raises ``InvalidToken`` on failure.
        """
        return self._fernet.decrypt(token)


# --------------------------------------------------------------------------- #
# File helpers (encryption at rest)
# --------------------------------------------------------------------------- #

def encrypt_file(src_path: str, dst_path: str) -> None:
    """
    Encrypt the file at ``src_path`` and write the ciphertext to ``dst_path``.
    The original file is left untouched.
    """
    encryptor = DataEncryptor()
    with open(src_path, "rb") as src_f:
        plaintext = src_f.read()
    ciphertext = encryptor.encrypt(plaintext)
    with open(dst_path, "wb") as dst_f:
        dst_f.write(ciphertext)
