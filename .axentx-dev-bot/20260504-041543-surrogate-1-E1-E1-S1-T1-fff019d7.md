### 1️⃣ `CreditDashboard.jsx`

```tsx
// src/components/CreditDashboard.jsx
import React, { useEffect, useState } from "react";

/**
 * Credit dashboard component.
 *
 * Expected API shape (GET /api/credits):
 * {
 *   bulkBalance: number,
 *   monthlyBalance: number,
 *   monthlyLimit: number,          // optional
 *   usage: {
 *     "30": [{ date: string, bulkUsed: number, monthlyUsed: number }, ...],
 *     "90": [...],
 *     "365": [...]
 *   }
 * }
 */
export default function CreditDashboard() {
  /* ---------- State ---------- */
  const [bulkBalance, setBulkBalance] = useState<number | null>(null);
  const [monthlyBalance, setMonthlyBalance] = useState<number | null>(null);
  const [monthlyLimit, setMonthlyLimit] = useState<number | null>(null);
  const [usageData, setUsageData] = useState<Record<string, Array<{ date: string; bulkUsed: number; monthlyUsed: number }>>>({});
  const [period, setPeriod] = useState<string>("30");
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  /* ---------- Fetch ---------- */
  useEffect(() => {
    async function fetchCredits() {
      try {
        const resp = await fetch("/api/credits");
        if (!resp.ok) {
          throw new Error(`HTTP ${resp.status}`);
        }
        const data = await resp.json();

        setBulkBalance(data.bulkBalance ?? null);
        setMonthlyBalance(data.monthlyBalance ?? null);
        setMonthlyLimit(data.monthlyLimit ?? null);
        setUsageData(data.usage ?? {});
        setError(null);
      } catch (e) {
        console.error("Failed to load credit data:", e);
        setError("Unable to load credit information.");
      } finally {
        setLoading(false);
      }
    }

    fetchCredits();
  }, []);

  /* ---------- Handlers ---------- */
  const handlePeriodChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    setPeriod(e.target.value);
  };

  /* ---------- Render helpers ---------- */
  const renderUsage = () => {
    const entries = usageData[period] ?? [];
    if (entries.length === 0) {
      return <p>No usage data for the selected period.</p>;
    }

    const bulkTotal = entries.reduce((sum, r) => sum + r.bulkUsed, 0);
    const monthlyTotal = entries.reduce((sum, r) => sum + r.monthlyUsed, 0);

    return (
      <table style={{ width: "100%", borderCollapse: "collapse" }}>
        <thead>
          <tr>
            <th style={tableHeaderStyle}>Date</th>
            <th style={tableHeaderStyle}>Bulk Used</th>
            <th style={tableHeaderStyle}>Monthly Used</th>
          </tr>
        </thead>
        <tbody>
          {entries.map((row, idx) => (
            <tr key={idx}>
              <td style={tableCellStyle}>{row.date}</td>
              <td style={tableCellStyleRight}>{row.bulkUsed}</td>
              <td style={tableCellStyleRight}>{row.monthlyUsed}</td>
            </tr>
          ))}
          <tr>
            <td style={tableCellStyle}><strong>Totals</strong></td>
