// src/lib/queue.ts
import { db } from '../db';
import { jobs as jobsTable } from '../db/schema';
import type { Job } from '../db/schema';

export type JobType = 'notify_status';
export type JobPayload = Record<string, unknown>;

export interface EnqueueOpts {
  priority?: number;
  runAt?: Date;
}

class Queue {
  private pending: Array<{ job: Job; attempts: number }> = [];
  private workers: Array<Promise<void>> = [];
  private running = false;

  constructor(private concurrency = 4) {}

  async enqueue(type: JobType, payload: JobPayload, opts: EnqueueOpts = {}) {
    const runAtMs = opts.runAt ? opts.runAt.getTime() : null;
    const [row] = await db
      .insert(jobsTable)
      .values({
        type,
        payload: JSON.stringify(payload),
        priority: opts.priority ?? 0,
        runAt: runAtMs,
        status: 'pending',
        attempts: 0,
        lastError: null,
      })
      .returning();

    this.pending.push({ job: row, attempts: 0 });
    this.pending.sort((a, b) => (b.job.priority ?? 0) - (a.job.priority ?? 0));
    this.start();
    return row;
  }

  private start() {
    if (this.running) return;
    this.running = true;
    for (let i = 0; i < this.concurrency; i++) {
      this.workers.push(this.worker());
    }
  }

  private async worker() {
    while (this.running) {
      const now = Date.now();
      const item = this.pending.find((p) => !p.job.runAt || p.job.runAt <= now);

      if (!item) {
        await new Promise((r) => setTimeout(r, 200));
        continue;
      }

      // remove from pending while processing
      this.pending = this.pending.filter((p) => p !== item);

      try {
        await this.process(item.job);
        await db.update(jobsTable).set({ status: 'completed' }).where({ id: item.job.id });
      } catch (err: any) {
        const attempts = item.attempts + 1;
        if (attempts >= 3) {
          await db
            .update(jobsTable)
            .set({ status: 'failed', attempts, lastError: err.message })
            .where({ id: item.job.id });
        } else {
          const backoff = Math.pow(2, attempts) * 1000;
          const nextRunAt = Date.now() + backoff;
          await db
            .update(jobsTable)
            .set({
              status: 'pending',
              attempts,
              runAt: nextRunAt,
              lastError: err.message,
            })
            .where({ id: item.job.id });
