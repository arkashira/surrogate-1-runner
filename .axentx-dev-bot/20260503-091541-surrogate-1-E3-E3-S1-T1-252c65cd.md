### 3. Queue (`src/lib/queue.ts`)
```ts
import { db } from '../db';
import { notificationJobs } from '../db/schema';
import type { NotificationJob } from '../types';

const inMemoryQueue: Array<{ job: NotificationJob; id: number }> = [];
let processing = false;

export async function enqueue(job: NotificationJob) {
  const [row] = await db
    .insert(notificationJobs)
    .values({
      requestId: job.requestId,
      oldStatus: job.oldStatus,
      newStatus: job.newStatus,
      payload: job.payload,
      attempts: 0,
    })
    .returning({ id: notificationJobs.id });

  inMemoryQueue.push({ job, id: row.id });
  return row.id;
}

export async function dequeue(): Promise<{ job: NotificationJob; id: number } | null> {
  if (inMemoryQueue.length > 0) return inMemoryQueue.shift()!;

  const rows = await db
    .select()
    .from(notificationJobs)
    .where(notificationJobs.processedAt.isNull())
    .orderBy(notificationJobs.createdAt)
    .limit(1);

  if (rows.length === 0) return null;
  const row = rows[0];
  const job: NotificationJob = {
    requestId: row.requestId,
    oldStatus: row.oldStatus ?? undefined,
    newStatus: row.newStatus,
    payload: row.payload as any,
  };
  return { job, id: row.id };
}

export async function ack(id: number) {
  await db
    .update(notificationJobs)
    .set({ processedAt: new Date() })
    .where(notificationJobs.id.eq(id));
}

export async function fail(id: number, error: string) {
  await db
    .update(notificationJobs)
    .set({ attempts: notificationJobs.attempts.add(1), error })
    .where(notificationJobs.id.eq(id));
}

export async function startWorker(handler: (job: NotificationJob) => Promise<void>, intervalMs = 5000) {
  if (processing) return;
  processing = true;

  async function tick() {
    try {
      const item = await dequeue();
      if (!item) {
        setTimeout(tick, intervalMs);
        return;
      }
      await handler(item.job);
      await ack(item.id);
    } catch (err) {
      // handler should throw to trigger retry/fail; we log and continue polling
      console.error('Worker tick error', err);
    } finally {
      setTimeout(tick, intervalMs);
    }
  }

  tick();
}
