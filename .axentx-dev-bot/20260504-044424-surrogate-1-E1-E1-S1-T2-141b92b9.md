#!/bin/bash
# /opt/axentx/surrogate-1/daemon/health_monitor.sh

# Health Monitor Daemon for Docker Compose
# Polls container health status every 10 seconds and emits JSON logs for unhealthy containers

# Function to log messages in JSON format
log_message() {
    local container_name=$1
    local exit_code=$2
    local timestamp=$(date -u +"%Y-%m-%dT%H:%M:%SZ")

    echo "{\"timestamp\": \"$timestamp\", \"container\": \"$container_name\", \"status\": \"unhealthy\", \"exit_code\": $exit_code}"
}

# Main monitoring loop
while true; do
    # Get container statuses in JSON format
    statuses=$(docker compose ps --format json)

    # Parse JSON and check each container's status
    echo "$statuses" | jq -c '.[] | select(.State == "unhealthy")' | while read -r container; do
        container_name=$(echo "$container" | jq -r '.Name')
        exit_code=$(echo "$container" | jq -r '.ExitCode')

        # Log unhealthy container
        log_message "$container_name" "$exit_code"
    done

    # Sleep for 10 seconds before next poll
    sleep 10
done
