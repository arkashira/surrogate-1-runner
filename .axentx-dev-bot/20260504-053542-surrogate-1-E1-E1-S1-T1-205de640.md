# tests/unit/test_claude_client.py
import pytest
from unittest.mock import patch, MagicMock
from ai_tools.claude import ClaudeClient, ClaudeError

# Helper fixture for a valid config
@pytest.fixture
def valid_config():
    return {
        "api_key": "test-key-123",
        "endpoint": "https://api.claude.ai/v1",
        "proxy": "http://proxy.corp:8080",
        "timeout": 2,
    }

def test_client_initializes_with_config(valid_config):
    client = ClaudeClient(**valid_config)
    assert client.api_key == "test-key-123"
    assert client.endpoint == "https://api.claude.ai/v1"
    assert client.proxy == "http://proxy.corp:8080"

@patch("ai_tools.claude.requests.Session")
def test_successful_health_check(mock_session, valid_config):
    mock_resp = MagicMock(status_code=200, json=lambda: {"status": "ok"})
    mock_session.return_value.get.return_value = mock_resp

    client = ClaudeClient(**valid_config)
    assert client.health_check() is True
    mock_session.return_value.get.assert_called_once_with(
        f"{valid_config['endpoint']}/health",
        headers={"Authorization": f"Bearer {valid_config['api_key']}"},
        proxies={"http": valid_config["proxy"], "https": valid_config["proxy"]},
        timeout=valid_config["timeout"],
    )

@patch("ai_tools.claude.requests.Session")
def test_prompt_returns_formatted_response(mock_session, valid_config):
    mock_resp = MagicMock(
        status_code=200,
        json=lambda: {"completion": "Sure, here's the answer..."}
    )
    mock_session.return_value.post.return_value = mock_resp

    client = ClaudeClient(**valid_config)
    resp = client.send_prompt("What is 2+2?")
    assert isinstance(resp, dict)
    assert "completion" in resp
    assert resp["completion"].startswith("Sure")

@patch("ai_tools.claude.requests.Session")
def test_invalid_api_key_raises_claude_error(mock_session, valid_config):
    mock_resp = MagicMock(status_code=401, text="Unauthorized")
    mock_session.return_value.get.return_value = mock_resp

    client = ClaudeClient(**valid_config)
    with pytest.raises(ClaudeError) as exc:
        client.health_check()
    assert "authentication" in str(exc.value).lower()

@patch("ai_tools.claude.requests.Session")
def test_network_timeout_propagates_as_claude_error(mock_session, valid_config):
    mock_session.return_value.get.side_effect = TimeoutError("timed out")

    client = ClaudeClient(**valid_config)
    with pytest.raises(ClaudeError) as exc:
        client.health_check()
    assert "timeout" in str(exc.value).lower()

def test_stealth_flag_excludes_logging(caplog, valid_config):
    client = ClaudeClient(**valid_config, stealth=True)
    with caplog.at_level("INFO"):
        client._log_request("GET", "/health")
    # No log entry should contain the endpoint URL
    assert not any(valid_config["endpoint"] in record.message for record in caplog.records)
