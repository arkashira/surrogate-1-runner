// Before
import { Request, Response } from 'express';
import { RequestList } from '../models/Request';

export const dashboard = async (req: Request, res: Response) => {
  const requests = await RequestList.find()
    .sort({ updatedAt: -1 })
    .limit(10)
    .select('id title status slaTargetAt slaStatus updatedAt');

  const counts = await getSimpleCounts();

  return res.render('dashboard', { requests, counts });
};

const getSimpleCounts = async () => {
  const requests = await RequestList.find({ updatedAt: { $gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000) } });
  const onTrack = requests.filter(request => request.slaStatus === 'on-track').length;
  const atRisk = requests.filter(request => request.slaStatus === 'at-risk').length;
  const breached = requests.filter(request => request.slaStatus === 'breached').length;

  return { total: requests.length, onTrack, atRisk, breached };
};

// After
import { Request, Response } from 'express';
import { RequestList } from '../models/Request';
import { paginate } from '../utils/pagination';

export const getDashboard = async (req: Request, res: Response) => {
  const requests = await RequestList.find()
    .sort({ updatedAt: -1 })
    .select('id title status slaTargetAt slaStatus updatedAt');

  const paginatedRequests = paginate(requests, req.query.page, 10);
  const counts = await getSimpleCounts();

  return res.render('dashboard', { requests: paginatedRequests, counts });
};
