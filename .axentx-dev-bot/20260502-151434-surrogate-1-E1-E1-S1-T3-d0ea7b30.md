import time
import hashlib
import json
from collections import deque
from typing import Any, Dict, Callable, List, Optional
from audit.store import write_run_record

def deterministic_hash(obj: Any) -> int:
    return int(hashlib.sha256(json.dumps(obj, sort_keys=True).encode()).hexdigest()[:16], 16)

def topological_order(adj: Dict[str, List[str]]) -> List[str]:
    """
    adj: {node: [dependencies]}  (node depends on these)
    Returns execution order: dependencies before dependents.
    Deterministic via sorted queues.
    """
    nodes = set(adj)
    for deps in adj.values():
        nodes.update(deps)
    nodes = sorted(nodes)

    indeg = {n: 0 for n in nodes}
    forward: Dict[str, List[str]] = {n: [] for n in nodes}
    for n, deps in adj.items():
        for d in deps:
            forward[d].append(n)
            indeg[n] = indeg.get(n, 0)
    for n, deps in adj.items():
        indeg[n] = len(deps)

    q = deque([n for n in nodes if indeg[n] == 0])
    order = []
    while q:
        q = deque(sorted(q))
        u = q.popleft()
        order.append(u)
        for v in sorted(forward[u]):
            indeg[v] -= 1
            if indeg[v] == 0:
                q.append(v)
    if len(order) != len(nodes):
        raise ValueError("Cycle detected in DAG")
    return order

def run_dag_deterministic(
    dag_spec: Dict[str, Any],
    inputs: Dict[str, Any],
    node_executors: Dict[str, Callable[[Dict[str, Any], Dict[str, Any]], Dict[str, Any]]],
    run_id: Optional[str] = None,
    deterministic_seed: Optional[int] = None,
) -> Dict[str, Any]:
    """
    dag_spec format:
    {
      "nodes": {
        "research": {"type": "research", "edges": []},
        "signal": {"type": "signal", "edges": ["research"]},
        "risk": {"type": "risk", "edges": ["signal"]},
        "execution": {"type": "execution", "edges": ["risk"]}
      }
    }
    """
    adj = {n: v.get("edges", []) for n, v in dag_spec.get("nodes", {}).items()}
    order = topological_order(adj)

    timestamps: Dict[str, float] = {}
    start = time.perf_counter()

    for node_id in order:
        t0 = time.perf_counter()
        deps = adj.get(node_id, [])
