// /opt/axentx/surrogate-1/src/dependency-validator/parser.js
const fs = require('fs');
const path = require('path');
const { parse } = require('@babel/parser');
const traverse = require('@babel/traverse').default;
const { getImportedModules } = require('./ast-utils');

class DependencyParser {
  constructor() {
    this.dependencies = new Set();
  }

  parseFile(filePath) {
    const content = fs.readFileSync(filePath, 'utf-8');
    const ast = parse(content, {
      sourceType: 'module',
      plugins: ['jsx', 'typescript']
    });

    traverse(ast, {
      ImportDeclaration: (nodePath) => {
        const moduleName = nodePath.node.source.value;
        this.dependencies.add(moduleName);
      },
      CallExpression: (nodePath) => {
        if (nodePath.node.callee.type === 'Import') {
          const moduleName = nodePath.node.arguments[0].value;
          this.dependencies.add(moduleName);
        }
      }
    });
  }

  getDependencies() {
    return Array.from(this.dependencies);
  }
}

module.exports = DependencyParser;

## 1. Acceptance Criteria
- **AC1** – The parser scans every `.js`, `.jsx`, `.ts`, and `.tsx` file under a given root directory and returns a list of *all* import statements found.  
- **AC2** – For each import statement, the parser extracts the module specifier string (e.g., `"react"` or `"./utils"`).  
- **AC3** – The parser cross‑references each extracted module against the `dependencies` and `devDependencies` sections of the nearest `package.json`. Any specifier **not** present in those sections is flagged as a mismatch.  
- **AC4** – The generated report is a JSON file containing:  
  ```json
  {
    "file": "<relative path>",
    "imports": ["moduleA", "moduleB"],
    "missing": ["moduleB"]
  }
  ```  
  and the overall summary includes total files scanned, total imports, and total mismatches.  
- **AC5** – The parser correctly handles the following import patterns:  
  1. ES6 static imports (`import X from "mod"`; `import {a,b} from "mod"`; `import * as X from "mod"`).  
  2. CommonJS `require("mod")`.  
  3. Dynamic `import("mod")`.  
  4. Re‑export syntax (`export * from "mod"`; `export {a} from "mod"`).  
- **AC6** – The tool runs without runtime errors on a codebase containing a mix of JavaScript and TypeScript files, using the project's Babel parser configuration.  
- **AC7** – Execution time scales linearly with the number of source files (no exponential blow‑up) and memory usage stays below 200 MB for a 10 k file repository.
