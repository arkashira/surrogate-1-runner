    class Config:
        env_file = ".env"
        env_file_encoding = "utf-8"

settings = Settings()

## 1. Acceptance Criteria (measurable)

- **API Success**: A GET request to Costinel’s `/v1/tenants/{tenant_id}/monthly` returns a 200 response containing JSON with keys `total_spend`, `breakdown`, and `savings_recommendations`.
- **Data Persistence**: The returned JSON is inserted into `costinel_monthly_snapshot` with columns `tenant_id`, `snapshot_date`, `total_spend`, `breakdown`, `savings_recommendations`, and `fetched_at`. The insert completes within 5 seconds of the API response.
- **Retry Logic**: On network or 5xx/401/403 errors, the client retries up to 3 times with exponential back‑off (1 s, 2 s, 4 s). After the third failure, a structured log entry is created with fields `tenant_id`, `attempts`, `error_type`, `error_message`.
- **Health‑Check**: The `/health/costinel` endpoint returns JSON `{ "status": "OK" }` if the last successful fetch for the tenant is less than 24 h old; otherwise returns `{ "status": "FAIL", "reason": "stale data" }`.

// costinel_client.test.js
const { CostinelClient } = require('../src/api/costinel_client');
const axios = require('axios');
jest.mock('axios');

describe('CostinelClient', () => {
  const tenantId = 'tenant-123';
  const client = new CostinelClient({ baseUrl: 'https://api.costinel.io', apiKey: 'key' });

  afterEach(() => jest.clearAllMocks());

  test('fetches data successfully', async () => {
    const mockResponse = {
      data: {
        total_spend: 12345,
        breakdown: { compute: 5000, storage: 7345 },
        savings_recommendations: ['Reserve Instances']
      }
    };
    axios.get.mockResolvedValue(mockResponse);

    const result = await client.fetchMonthlySnapshot(tenantId);
    expect(result).toEqual(mockResponse.data);
    expect(axios.get).toHaveBeenCalledWith(
      `https://api.costinel.io/v1/tenants/${tenantId}/monthly`,
      { headers: { Authorization: 'Bearer key' } }
    );
  });

  test('retries on network error and logs after 3 attempts', async () => {
    axios.get
      .mockRejectedValueOnce(new Error('ECONNRESET'))
      .mockRejectedValueOnce(new Error('ECONNRESET'))
      .mockRejectedValueOnce(new Error('ECONNRESET'));

    const logSpy = jest.spyOn(console, 'error').mockImplementation(() => {});
    await expect(client.fetchMonthlySnapshot(tenantId)).rejects.toThrow('ECONNRESET');
    expect(axios.get).toHaveBeenCalledTimes(3);
    expect(logSpy).toHaveBeenCalledWith(expect.objectContaining({
      tenant_id: tenantId,
      attempts: 3,
      error_type: 'ECONNRESET',
      error_message: expect.any(String)
    }));
  });

  test('does not retry on 4xx error', async () => {
    axios.get.mockRejectedValueOnce({ response: { status: 401, data: 'Unauthorized' } });

    await expect(client.fetchMonthlySnapshot(tenantId)).rejects.toMatchObject({
      response: { status: 401 }
    });
    expect(axios.get).toHaveBeenCalledTimes(1);
  });
});
