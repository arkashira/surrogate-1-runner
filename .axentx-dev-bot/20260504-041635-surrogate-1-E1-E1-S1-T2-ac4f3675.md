```bash
#!/bin/bash

set -e

# AWS credentials and Cost Explorer settings (use environment variables for security)
AWS_ACCESS_KEY_ID="${AWS_ACCESS_KEY_ID}"
AWS_SECRET_ACCESS_KEY="${AWS_SECRET_ACCESS_KEY}"
COST_EXPLORER_REGION="${COST_EXPLORER_REGION}"
COST_EXPLORER_ACCOUNT_ID="${COST_EXPLORER_ACCOUNT_ID}"

# PostgreSQL database settings
DB_HOST="${DB_HOST}"
DB_PORT="${DB_PORT}"
DB_NAME="${DB_NAME}"
DB_USER="${DB_USER}"
DB_PASSWORD="${DB_PASSWORD}"

# Ingest data from AWS Cost Explorer
aws cost-explorer get-cost-and-usage \
  --access-key-id "$AWS_ACCESS_KEY_ID" \
  --secret-access-key "$AWS_SECRET_ACCESS_KEY" \
  --region "$COST_EXPLORER_REGION" \
  --account-id "$COST_EXPLORER_ACCOUNT_ID" \
  --output json > cost_data.json

# Check AWS command success
if [ $? -ne 0 ]; then
    echo "ERROR: AWS Cost Explorer command failed" >> /var/log/surrogate-1/ingest_cost.log
    exit 1
fi

# Load data into PostgreSQL database
psql -h "$DB_HOST" -p "$DB_PORT" -U "$DB_USER" -d "$DB_NAME" \
  -c "COPY aws_cost FROM '/tmp/cost_data.json' DELIMITER ',' CSV HEADER" >> /var/log/surrogate-1/ingest_cost.log 2>&1

# Check PostgreSQL load success
if [ $? -ne 0 ]; then
    echo "ERROR: Failed to load data into PostgreSQL" >> /var/log/surrogate-1/ingest_cost.log
    exit 1
fi

echo "Ingestion completed successfully" >> /var/log/surrogate-1/ingest_cost.log
```
