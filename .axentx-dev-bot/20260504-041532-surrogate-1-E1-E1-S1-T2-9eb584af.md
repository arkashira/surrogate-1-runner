### 2. `Template` Mongoose Model (`src/models/Template.ts`)

```ts
import mongoose, { Schema, Document } from 'mongoose';

export interface IStep {
  type: string;          // e.g. "action", "approval"
  name: string;
  params?: Record<string, any>;
}

export interface ITemplate extends Document {
  name: string;
  description?: string;
  steps: IStep[];
  parameters?: Record<string, any>;
  createdAt: Date;
  updatedAt: Date;
}

const StepSchema = new Schema<IStep>(
  {
    type: { type: String, required: true },
    name: { type: String, required: true },
    params: { type: Schema.Types.Mixed, default: {} },
  },
  { _id: false }
);

const TemplateSchema = new Schema<ITemplate>(
  {
    name: { type: String, required: true, unique: true, trim: true },
    description: { type: String, default: '' },
    steps: { type: [StepSchema], default: [] },
    parameters: { type: Schema.Types.Mixed, default: {} },
  },
  { timestamps: true }
);

export default mongoose.model<ITemplate>('Template', TemplateSchema);
```

* **`unique: true`** guarantees no duplicate names.
* **`trim: true`** removes accidental whitespace.
* **`default: {}`** for `params` and `parameters` keeps the JSON shape consistent.
