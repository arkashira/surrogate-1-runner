import (

// Signature represents the captured request signature with optimized structure
type Signature struct {
	Method  string            `json:"method"`
	Path    string            `json:"path"`
	Headers map[string]string `json:"headers"`
	Body    interface{}       `json:"body,omitempty"` // Only structure, not values
}

// NewSignatureMiddleware creates an optimized HTTP middleware for request signature capture
func NewSignatureMiddleware(serviceName, deploymentHash string, rdb *redis.Client) func(http.Handler) http.Handler {
	redisKey := serviceName + ":" + deploymentHash

	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, req *http.Request) {
			start := time.Now()

			// Initialize signature with basic request info
			sig := Signature{
				Method:  req.Method,
				Path:    req.URL.Path,
				Headers: make(map[string]string),
			}

			// Capture non-sensitive headers
			for name, values := range req.Header {
				if strings.EqualFold(name, "Authorization") || strings.EqualFold(name, "Cookie") {
					continue
				}
				sig.Headers[name] = strings.Join(values, ",")
			}

			// Capture body structure for JSON requests
			if req.Body != nil && req.ContentLength != 0 && strings.Contains(req.Header.Get("Content-Type"), "application/json") {
				bodyBytes, err := io.ReadAll(req.Body)
				if err == nil && len(bodyBytes) > 0 {
					// Restore body for downstream handlers
					req.Body = io.NopCloser(bytes.NewBuffer(bodyBytes))

					var generic interface{}
					if json.Unmarshal(bodyBytes, &generic) == nil {
						sig.Body = extractShape(generic)
					}
				} else {
					req.Body = io.NopCloser(bytes.NewReader(bodyBytes))
				}
			}

			// Store in Redis with timeout context
			ctx, cancel := context.WithTimeout(context.Background(), 100*time.Millisecond)
			defer cancel()

			if err := rdb.LPush(ctx, redisKey, toJSON(sig)).Err(); err != nil {
				log.Printf("[ERROR] failed to store signature: %v", err)
			}

			// Call next handler
			next.ServeHTTP(w, req)

			// Measure and log latency
			elapsed := time.Since(start)
			if elapsed > 2*time.Millisecond {
				log.Printf("[WARN] signature middleware latency %s exceeds 2ms", elapsed)
			}
		})
	}
}

// extractShape returns the structure of JSON data without values
func extractShape(v interface{}) interface{} {
	switch val := v.(type) {
	case map[string]interface{}:
		shape := make(map[string]interface{}, len(val))
		keys := make([]string, 0, len(val))
		for k := range val {
			keys = append(keys, k)
		}
		sort.Strings(keys) // Ensure deterministic ordering
		for _, k := range keys {
			shape[k] = extractShape(val[k])
		}
		return shape
	case []interface{}:
		if len(val) == 0 {
			return []interface{}{}
		}
		return []interface{}{extractShape(val[0])}
	default:
		return strings.TrimPrefix(strings.TrimPrefix(strings.TrimPrefix(fmt.Sprintf("%T", val), "*"), "main."), "json.")
	}
}

// toJSON converts the signature to JSON string with error handling
func toJSON(sig Signature) string {
	data, err := json.Marshal(sig)
	if err != nil {
		log.Printf("[ERROR] failed to marshal signature: %v", err)
		return "{}"
	}
	return string(data)
}
