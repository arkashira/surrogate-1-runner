// /opt/axentx/surrogate-1/src/storage/requests.ts
import { writeFileSync, readFileSync, existsSync, mkdirSync } from 'fs';
import { join } from 'path';
import { v4 as uuidv4 } from 'uuid';

const DATA_DIR = join(process.cwd(), 'data');
const DB_PATH = join(DATA_DIR, 'requests.json');

export interface TrackedRequest {
  id: string; // UUIDv4
  title: string;
  type: string;
  owner: string;
  slaTarget: string; // ISO 8601 timestamp
  state: 'open' | 'in_progress' | 'resolved' | 'closed';
  createdAt: string;
  updatedAt: string;
}

function ensureDataDir() {
  if (!existsSync(DATA_DIR)) mkdirSync(DATA_DIR, { recursive: true });
}

function loadRequests(): TrackedRequest[] {
  ensureDataDir();
  if (!existsSync(DB_PATH)) return [];
  try {
    const raw = readFileSync(DB_PATH, 'utf8');
    return JSON.parse(raw);
  } catch {
    return [];
  }
}

function saveRequests(requests: TrackedRequest[]) {
  ensureDataDir();
  writeFileSync(DB_PATH, JSON.stringify(requests, null, 2), 'utf8');
}

export function createRequest(input: {
  title: string;
  type: string;
  owner: string;
  slaTarget: string;
}): TrackedRequest {
  const now = new Date().toISOString();
  const req: TrackedRequest = {
    id: uuidv4(),
    title: input.title,
    type: input.type,
    owner: input.owner,
    slaTarget: input.slaTarget,
    state: 'open',
    createdAt: now,
    updatedAt: now,
  };
  const all = loadRequests();
  all.push(req);
  saveRequests(all);
  return req;
}

export function getRequest(id: string): TrackedRequest | undefined {
  const all = loadRequests();
  return all.find((r) => r.id === id);
}
