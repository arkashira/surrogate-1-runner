import Cocoa          // pulls in CoreGraphics, AppKit, etc.
import Carbon         // for virtual‑key constants (optional but clearer)

/// Errors that can be thrown while synthesising a Cmd + V event.
public enum CGEventCmdVError: LocalizedError {
    /// The CGEvent could not be allocated.
    case eventCreationFailed
    /// The event could not be posted (unlikely – CGEvent.post has no return value,
    /// but we keep the case for future extensions, e.g. using Accessibility APIs).
    case postFailed

    public var errorDescription: String? {
        switch self {
        case .eventCreationFailed: return "Failed to create a CGEvent."
        case .postFailed:          return "Failed to post the CGEvent to the system."
        }
    }
}

/// Core helper that builds and posts the two required CGEvents
/// (key‑down and key‑up) for the **⌘ V** shortcut.
public struct CGEventCmdV {

    /// Virtual‑key code for the “V” key (ANSI layout).  
    /// Using the constant from `<Carbon/Events.h>` makes the intent obvious.
    private static let vKeyCode: CGKeyCode = CGKeyCode(kVK_ANSI_V)

    /// Sends a synthetic **Cmd + V** to the system.
    ///
    /// The function:
    /// 1. Creates a *key‑down* event for “V” with the command flag set.  
    /// 2. Creates a *key‑up*   event for “V” with the command flag set.  
    /// 3. Posts both events to the **HID** event tap (`.cghidEventTap`), which is the
    ///    same path the real keyboard uses.
    ///
    /// - Throws: ``CGEventCmdVError`` if any step fails.
    public static func sendPaste() throws {
        // ---------- key‑down ----------
        guard let down = CGEvent(keyboardEventSource: nil,
                                 virtualKey: vKeyCode,
                                 keyDown: true) else {
            throw CGEventCmdVError.eventCreationFailed
        }
        down.flags = .maskCommand

        // ---------- key‑up ----------
        guard let up = CGEvent(keyboardEventSource: nil,
                               virtualKey: vKeyCode,
                               keyDown: false) else {
            throw CGEventCmdVError.eventCreationFailed
        }
        up.flags = .maskCommand

        // ---------- post ----------
        let tap = CGEventTapLocation.cghidEventTap
        down.post(tap: tap)
        up.post(tap: tap)

        // CGEvent.post does not return a status. If posting ever throws,
        // it would surface as a runtime exception, which we let propagate.
        // For extra safety you could query the Accessibility API afterwards.
    }
}
