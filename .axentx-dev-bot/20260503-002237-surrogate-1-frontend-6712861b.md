### 1) Add snapshot script

`bin/snapshot.sh`

```bash
#!/usr/bin/env bash
# bin/snapshot.sh
# Usage: HF_TOKEN=... ./bin/snapshot.sh <repo> <date> [out.jsonl]
#
# Produces a deterministic manifest for one date folder:
# [
#   {"path":"YYYY-MM-DD/file1.parquet","cdn_url":"https://huggingface.co/datasets/.../resolve/main/YYYY-MM-DD/file1.parquet","size":12345},
#   ...
# ]
#
# Notes:
# - Uses HF API (non-recursive) once per date folder.
# - CDN URLs bypass auth and API rate limits during training.
# - Deterministic sort by path for reproducible sharding.

set -euo pipefail

REPO="${1:-axentx/surrogate-1-training-pairs}"
DATE="${2:-$(date +%Y-%m-%d)}"
OUT="${3:-snapshot-${DATE}.jsonl}"

echo "snapshot: repo=${REPO} date=${DATE} out=${OUT}" >&2

# Requires: pip install huggingface_hub jq
python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import os, json, sys
from huggingface_hub import HfApi

repo = sys.argv[1]
date = sys.argv[2]
out_path = sys.argv[3]

api = HfApi(token=os.environ.get("HF_TOKEN"))
# Non-recursive: list top-level objects inside the date folder
entries = api.list_repo_tree(repo=repo, path=date, recursive=False)

rows = []
for e in entries:
    if getattr(e, "type", None) != "file":
        continue
    path = f"{date}/{e.path}"
    rows.append({
        "path": path,
        "cdn_url": f"https://huggingface.co/datasets/{repo}/resolve/main/{path}",
        "size": getattr(e, "size", 0)
    })

# Deterministic ordering
rows.sort(key=lambda x: x["path"])

with open(out_path, "w", encoding="utf-8") as f:
    for r in rows:
        f.write(json.dumps(r, ensure_ascii=False) + "\n")

print(f"snapshot: wrote {len(rows)} entries to {out_path}")
PY

echo "snapshot: done -> ${OUT}" >&2
```

Make executable:

```bash
chmod +x bin/snapshot.sh
```
