	"bufio"

	"io"
	"sort"
	"strconv"
	"strings"

// Public data structures

// Record is the canonical representation of a row that can be processed
// by the meshing utilities.  It is deliberately minimal – only the fields
// required by the two public functions are present.  If a downstream
// package needs more information it can embed Record or keep a parallel
// structure.
//
//   * When reading CSV we fill Category and Value.
//   * When grouping arbitrary slices we fill ID and Value.
//   * The fields are independent – the functions use only the ones they
//     need, so a Record can be partially populated.
type Record struct {
	// Category is the grouping key for CSV aggregation.
	Category string

	// ID is the deterministic ordering key for slice partitioning.
	// It is ignored by GroupByCategory.
	ID string

	// Value is the numeric payload used by both algorithms.
	Value float64
}

type GroupResult struct {
	Category string
	Count    int
	Sum      float64
	Avg      float64
}

// 1. CSV aggregation – deterministic, streaming, O(N) memory
