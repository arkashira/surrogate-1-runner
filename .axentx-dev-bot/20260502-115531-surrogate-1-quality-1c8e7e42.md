"""
Surrogate-1 training: CDN-only, manifest-driven, schema-safe.
- Uses pre-computed file_list.json (one per date folder)

- Reuses Lightning Studio; idempotent start + checkpointing

import json
import os
import io
import hashlib
import pyarrow as pa
import pyarrow.parquet as pq
import requests
from pathlib import Path
from typing import Iterator, Dict, Any, List

# Optional Lightning integration (do not fail if absent)

    from lightning_sdk import Lightning, Teamspace, Studio, Machine

# ── Configuration ──────────────────────────────────────────────────────
HF_DATASET_REPO = os.getenv("HF_DATASET_REPO", "datasets/your-org/surrogate-mirror")
MANIFEST_PATH = Path(os.getenv("MANIFEST_PATH", "file_list.json"))
CACHE_DIR = Path(os.getenv("CACHE_DIR", ".cache/cdn"))
CACHE_DIR.mkdir(parents=True, exist_ok=True)

# Date-scoped manifest recommended: file_list-2026-04-29.json
CDN_ROOT = f"https://huggingface.co/datasets/{HF_DATASET_REPO}/resolve/main"

# Training checkpoint (lightweight)
CHECKPOINT_PATH = Path(os.getenv("CHECKPOINT_PATH", ".cache/checkpoint.json"))


# ── Manifest ───────────────────────────────────────────────────────────
def load_manifest() -> List[str]:
    if not MANIFEST_PATH.exists():
        raise FileNotFoundError(
            f"Manifest not found: {MANIFEST_PATH}. Generate once (e.g., on Mac) with:\n"
            "  from huggingface_hub import list_repo_tree\n"
            "  import json\n"
            "  repo='datasets/your-org/surrogate-mirror'\n"
            "  folder='batches/mirror-merged/2026-04-29'\n"
            "  files=[f.r['path'] for f in list_repo_tree(repo, path=folder, recursive=False)]\n"
            "  json.dump(files, open('file_list.json','w'), indent=2)"
        )
    with open(MANIFEST_PATH) as f:
        data = json.load(f)
    if not isinstance(data, list) or not all(isinstance(x, str) for x in data):
        raise ValueError("Manifest must be a list of strings (relative paths to parquet files).")
    return data


# ── CDN fetch + cache ─────────────────────────────────────────────────
def _cache_key(path: str) -> str:
    return hashlib.sha256(path.encode()).hexdigest() + ".parquet"


def cdn_fetch(path: str) -> bytes:
    cache_file = CACHE_DIR / _cache_key(path)
    if cache_file.exists():
        return cache_file.read_bytes()

    url = f"{CDN_ROOT}/{path}"
    r = requests.get(url, timeout=30)
    r.raise_for_status()
    cache_file.write_bytes(r.content)
    return r.content


# ── Schema projection ─────────────────────────────────────────────────
def project_to_pair(buf: bytes) -> Iterator[Dict[str, str]]:

    cols = {c.lower(): c for c in table.column_names}

    response_col = cols.get("response")
