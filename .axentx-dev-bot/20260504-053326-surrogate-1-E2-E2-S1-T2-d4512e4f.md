import * as fs from 'fs';

describe('SettingsManager', () => {

    threshold: 0.8,
    windowSize: 7,
    sensitivity: 'medium',
  };

  const customSettings = {
    threshold: 0.9,
    windowSize: 14,
    sensitivity: 'high',
  };

  const malformedJson = '{ "threshold": 0.9, "windowSize": 14, "sensitivity": }';

  beforeEach(() => {
    jest.clearAllMocks();
  });

  it('should use default settings when config file is absent', async () => {
    (fs.promises.readFile as jest.Mock).mockImplementationOnce(() => {
      throw new Error('File not found');
    });

    const settingsManager = new SettingsManager('/path/to/config.json');
    await settingsManager.load();

    expect(settingsManager.getSettings()).toEqual(defaultSettings);
  });

  it('should load valid custom settings correctly', async () => {
    (fs.promises.readFile as jest.Mock).mockResolvedValueOnce(
      JSON.stringify(customSettings)
    );

    const settingsManager = new SettingsManager('/path/to/config.json');
    await settingsManager.load();

    expect(settingsManager.getSettings()).toEqual(customSettings);
  });

  it('should fall back to defaults when JSON is malformed', async () => {
    (fs.promises.readFile as jest.Mock).mockResolvedValueOnce(malformedJson);

    const settingsManager = new SettingsManager('/path/to/config.json');
    await settingsManager.load();

    expect(settingsManager.getSettings()).toEqual(defaultSettings);
  });

  it('should log errors when file cannot be parsed', async () => {
    const consoleSpy = jest.spyOn(console, 'error').mockImplementation(() => {});

    const settingsManager = new SettingsManager('/path/to/config.json');
