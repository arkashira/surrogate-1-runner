import os
import time
from typing import Optional, Dict, Any

import httpx

logger = logging.getLogger(__name__)

SLACK_WEBHOOK_URL = os.getenv("SLACK_WEBHOOK_URL")
SLACK_TIMEOUT = float(os.getenv("SLACK_TIMEOUT", "5.0"))
SLACK_MAX_RETRIES = int(os.getenv("SLACK_MAX_RETRIES", "2"))


def _mention_texts(emails):
    if not emails:
        return ""
    mentions = []
    for email in emails:
        email = email.strip()
        if email:
            mentions.append(f"<{email}>")
    return " ".join(mentions)


def _sla_badge(sla_state: Optional[str]) -> str:
    if not sla_state:
        return "N/A"
    return sla_state.upper()

    title: str,
    new_status: str,
    public_url: str,
    sla_state: Optional[str],
    assignee_email: Optional[str] = None,
    creator_email: Optional[str] = None,
) -> bool:
    """
    Post a status update to Slack via webhook.
    Returns True if message was accepted (HTTP 2xx), False otherwise.
    Never raises.
    """
    if not SLACK_WEBHOOK_URL:
        logger.warning("SLACK_WEBHOOK_URL not configured; skipping Slack notification")
        return False

    emails = [e for e in [assignee_email, creator_email] if e]
    mentions = _mention_texts(emails)

    text = (
        f"*Request Update*\n"
        f"*{title}*\n"
        f"Status: `{new_status}`\n"
        f"SLA: `{_sla_badge(sla_state)}`\n"
        f"URL: {public_url}"
    )
    if mentions:
        text = f"{mentions}\n{text}"

    payload: Dict[str, Any] = {"text": text}

    for attempt in range(1, SLACK_MAX_RETRIES + 1):
        try:

                timeout=SLACK_TIMEOUT,
                headers={"Content-Type": "application/json"},
            )
            if 200 <= resp.status_code < 300:
                logger.info(
                    "slack_notification_sent",
                    extra={
                        "title": title,
                        "status": new_status,
                        "public_url": public_url,
                        "sla_state": sla_state,
                        "attempt": attempt,
                    },
                )
                return True

                    },
                )
        except httpx.TimeoutException:
            logger.warning(
