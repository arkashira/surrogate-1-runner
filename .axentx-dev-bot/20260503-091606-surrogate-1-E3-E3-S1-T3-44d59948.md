import logging
import os
import json
import time
from typing import Optional, Dict, Any

import httpx

logger = logging.getLogger(__name__)

SLACK_WEBHOOK_URL = os.getenv("SLACK_WEBHOOK_URL")


def _format_mentions(emails):
    if not emails:
        return ""
    return " ".join(f"<{e}>" for e in emails if e)


def post_status_update(
    request_title: str,
    new_status: str,
    public_url: str,
    sla_state: str,
    assignee_email: Optional[str] = None,
    creator_email: Optional[str] = None,
    request_id: Optional[int] = None,
) -> bool:
    """
    Post a status update to Slack via incoming webhook.

    Behavior:
    - Never raises. Failures are logged and return False.
    - Uses short timeout and one retry to avoid blocking transitions.
    - Returns True only when Slack responds 2xx.
    """
    if not SLACK_WEBHOOK_URL:
        logger.warning(
            "slack_webhook_missing",
            extra={
                "request_id": request_id,
                "request_title": request_title,
                "new_status": new_status,
            },
        )
        return False

    mentions = _format_mentions([e for e in [assignee_email, creator_email] if e])
    text = (
        f"*Request:* {request_title}\n"
        f"*Status:* {new_status}\n"
        f"*SLA:* {sla_state}\n"
        f"*URL:* {public_url}"
    )
    if mentions:
        text = f"{mentions} {text}"

    payload = {"text": text}
    max_retries = 1
    timeout = 5.0

    for attempt in range(max_retries + 1):
        try:
            resp = httpx.post(SLACK_WEBHOOK_URL, json=payload, timeout=timeout)
            resp.raise_for_status()
            logger.info(
                "slack_status_update_sent",
                extra={
                    "request_id": request_id,
                    "request_title": request_title,
                    "new_status": new_status,
                    "public_url": public_url,
                    "sla_state": sla_state,
                    "attempt": attempt,
                    "status_code": resp.status_code,
                },
            )
            return True
        except Exception as exc:
            is_final = attempt == max_retries
            logger.warning(
                "slack_status_update_failed",
                extra={
                    "request_id": request_id,
                    "request_title": request_title,
                    "new_status": new_status,
                    "public_url": public_url,
                    "sla_state": sla_state,
                    "attempt": attempt,
                    "max_retries": max_retries,
                    "is_final": is_final,
                    "error": str(exc),
                    "error_type": type(exc).__name__,
                    "status_code": getattr(getattr(exc, "response", None), "status_code", None),
                },
                exc_info=is_final,
            )
            if is_final:
                return False
            time.sleep(0.5)

    return False
```
