KiCAD 9 compatible adapter for Surrogate-1.

Provides a thin compatibility layer over the KiCAD Python API (pcbnew)
so that the core Surrogate-1 routing logic can operate unchanged across
KiCAD 8 and KiCAD 9 releases.

The adapter:
* Detects the running KiCAD version.
* Normalises API differences (e.g., board retrieval, layer handling).
* Exposes a stable interface used by the rest of Surrogate-1.
"""

import sys
import logging
from pathlib import Path

# KiCAD's Python API is provided via the `pcbnew` module.
# Import lazily to avoid import errors when the plugin is inspected
# outside of KiCAD.
try:
    import pcbnew
except Exception as e:  # pragma: no cover
    pcbnew = None
    logging.error("Failed to import pcbnew module: %s", e)

logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)

    """
    Return the KiCAD version string, e.g. "8.0.0" or "9.0.1".
    If the version cannot be detected, return "unknown".
    """
    if pcbnew is None:
        return "unknown"
    try:
        # KiCAD 9 exposes the version via pcbnew.Version()
        return pcbnew.Version()
    except AttributeError:
        # KiCAD 8 used pcbnew.GetVersion()
        try:
            return pcbnew.GetVersion()
        except Exception:
            return "unknown"


KICAD_VERSION = _detect_kicad_version()
logger.info("Surrogate-1 detected KiCAD version: %s", KICAD_VERSION)


class KiCadAdapter:
    """
    High‑level wrapper around KiCAD's pcbnew API.
