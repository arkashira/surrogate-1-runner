## 1️⃣ `guidance.js` – Unified Guidance Engine  

```javascript
// /opt/axentx/surrogate-1/src/guidance.js
/**
 * Guidance Engine
 * ----------------
 * Provides:
 *   • Persistent dismissal tracking (localStorage + in‑memory fallback)
 *   • UI rendering of a tooltip / pop‑over (no external CSS framework required)
 *   • A simple API that can be called directly by the onboarding controller
 *     or by any other part of the app.
 *
 * Public API
 * ----------
 *   showGuidance(id, message, anchor?) → { dismiss() }
 *   hideGuidance(id)
 *   isDismissed(id) → boolean
 *   dismiss(id)          // persists & removes UI
 *   resetDismissals()    // clears storage (dev / reset button)
 */

const STORAGE_KEY = 'surrogate1.guidance.dismissed';

// ---------------------------------------------------------------------------
// 1️⃣ Persistence helpers – work even when localStorage is blocked (e.g. incog.).
// ---------------------------------------------------------------------------
let _memoryStore = new Set(); // fallback when localStorage throws

function _loadDismissed() {
  try {
    const raw = window.localStorage?.getItem(STORAGE_KEY);
    return raw ? new Set(JSON.parse(raw)) : new Set(_memoryStore);
  } catch (_) {
    // localStorage unavailable – use in‑memory cache
    return new Set(_memoryStore);
  }
}

function _saveDismissed(set) {
  try {
    window.localStorage?.setItem(STORAGE_KEY, JSON.stringify([...set]));
  } catch (_) {
    // persist to in‑memory store only
    _memoryStore = new Set(set);
  }
}

// ---------------------------------------------------------------------------
// 2️⃣ Core state helpers
// ---------------------------------------------------------------------------
export function isDismissed(id) {
  return _loadDismissed().has(id);
}

/**
 * Persist dismissal *and* remove any visible UI.
 */
export function dismiss(id) {
  const dismissed = _loadDismissed();
  dismissed.add(id);
  _saveDismissed(dismissed);
  hideGuidance(id);
}

/**
 * Remove the tooltip element from the DOM (no persistence).
 */
export function hideGuidance(id) {
  const el = document.querySelector(`[data-guidance-id="${id}"]`);
  if (el) el.remove();
}

/**
 * Completely wipe all dismissals – handy for testing or a “Reset tour” button.
 */
export function resetDismissals() {
  _saveDismissed(new Set());
}

// guidance.test.js
import { showGuidance, hideGuidance, isDismissed, resetDismissals } from './guidance';

describe('guidance.js', () => {
  beforeEach(() => {
    localStorage.clear();
    jest.restoreAllMocks();
  });

  test('showGuidance renders tooltip and attaches to element', () => {
    const el = document.createElement('div');
    el.id = 'test-btn';
    document.body.appendChild(el);

    showGuidance({ targetId: 'test-btn', message: 'Hello' });

    const tooltip = document.querySelector('.guidance-tooltip');
    expect(tooltip).toBeTruthy();
    expect(tooltip.textContent).toContain('Hello');
    expect(tooltip.parentElement).toBe(el);
  });

  test('dismissal is stored in localStorage', () => {
    showGuidance({ targetId: 'test-btn', message: 'Hello', dismissKey: 'step1' });
    const btn = document.querySelector('.guidance-dismiss');
    btn.click();

    expect(localStorage.getItem('dismissed-step1')).toBe('true');
    expect(isDismissed('step1')).toBe(true);
  });

  test('fallback to in-memory when localStorage unavailable', () => {
    jest.spyOn(Storage.prototype, 'setItem').mockImplementation(() => { throw new Error('no storage'); });

    showGuidance({ targetId: 'test-btn', message: 'Hello', dismissKey: 'step2' });
    const btn = document.querySelector('.guidance-dismiss');
    btn.click();

    expect(isDismissed('step2')).toBe(true);
  });

  test('hideGuidance removes tooltip and updates state', () => {
    showGuidance({ targetId: 'test-btn', message: 'Hello', dismissKey: 'step3' });
    hideGuidance('step3');

    expect(document.querySelector('.guidance-tooltip')).toBeNull();
    expect(isDismissed('step3')).toBe(true);
  });

  test('resetDismissals clears all dismissals', () => {
    localStorage.setItem('dismissed-step1', 'true');
    resetDismissals();
    expect(isDismissed('step1')).toBe(false);
  });
});
