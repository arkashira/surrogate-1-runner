    SSHConnectionConfig config = new SSHConnectionConfig();
    
    for (String cipher : allowedCiphers) {
        assertTrue(config.isCipherAllowed(cipher), "Cipher " + cipher + " should be allowed");
    }
    
    assertFalse(config.isCipherAllowed("des-cbc"), "Weak cipher should be rejected");

}
