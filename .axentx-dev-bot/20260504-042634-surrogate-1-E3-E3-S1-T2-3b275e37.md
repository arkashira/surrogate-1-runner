import json
import requests
from typing import Dict, Optional, Any

class IntegrationAPI:
    """
    A comprehensive API integration class for managing connections with development tools.
    Handles configuration, tool integration, status tracking, and error handling.
    """

    def __init__(self, config_file: str):
        """
        Initialize the IntegrationAPI with a configuration file path.

        Args:
            config_file: Path to JSON configuration file
        """
        self.config_file = config_file
        self.config = self.load_config()

    def load_config(self) -> Dict[str, Any]:
        """
        Load configuration from JSON file.

        Returns:
            Dictionary containing configuration data

        Raises:
            FileNotFoundError: If config file doesn't exist
            json.JSONDecodeError: If config file is malformed
        """
        try:
            with open(self.config_file, 'r') as f:
                return json.load(f)
        except FileNotFoundError:
            raise FileNotFoundError(f"Configuration file not found at {self.config_file}")
        except json.JSONDecodeError:
            raise json.JSONDecodeError(f"Invalid JSON in configuration file {self.config_file}")

    def integrate_with_tool(self, tool_name: str, user_data: Dict[str, Any]) -> Optional[Dict[str, Any]]:
        """
        Integrate with a specific tool using its configuration.

        Args:
            tool_name: Name of the tool to integrate with
            user_data: Data to send to the tool's API

        Returns:
            Response from the tool's API if successful, None otherwise
        """
        tool_config = self.config.get(tool_name)
        if not tool_config:
            return None

        try:
            api_url = tool_config['api_url']
            api_key = tool_config['api_key']
            headers = {'Authorization': f'Bearer {api_key}'}

            response = requests.post(
                api_url,
                headers=headers,
                json=user_data,
                timeout=10
            )

            if response.status_code == 200:
                return response.json()
            return None

        except requests.exceptions.RequestException as e:
            print(f"Integration error with {tool_name}: {str(e)}")
            return None

    def get_tool_integration_status(self, tool_name: str) -> Optional[str]:
        """
        Get the integration status of a specific tool.

        Args:
            tool_name: Name of the tool to check

        Returns:
            Integration status if tool exists, None otherwise
        """
        tool_config = self.config.get(tool_name)
        return tool_config.get('integration_status') if tool_config else None

    def update_tool_integration_status(self, tool_name: str, status: str) -> bool:
        """
        Update the integration status of a specific tool.

        Args:
            tool_name: Name of the tool to update
            status: New status to set

        Returns:
            True if update was successful, False otherwise
        """
        tool_config = self.config.get(tool_name)
        if not tool_config:
            return False

        tool_config['integration_status'] = status

        try:
            with open(self.config_file, 'w') as f:
                json.dump(self.config, f, indent=2)
            return True
        except IOError as e:
            print(f"Error updating configuration: {str(e)}")
            return False

    def get_available_tools(self) -> Dict[str, Dict[str, Any]]:
        """
        Get all available tools from configuration.

        Returns:
            Dictionary of all configured tools
        """
        return {k: v for k, v in self.config.items() if isinstance(v, dict)}
