import os
import json
import csv
import uuid
from datetime import datetime
from flask import Flask, request, jsonify, send_file
import matplotlib.pyplot as plt
import pandas as pd
from io import BytesIO

app = Flask(__name__)

# In-memory storage for benchmark results
benchmark_results = {}

class BenchmarkController:
    @staticmethod
    def run_benchmark(duration, gpu_mode):
        """Simulate benchmark execution"""
        benchmark_id = str(uuid.uuid4())
        results = {
            "id": benchmark_id,
            "timestamp": datetime.now().isoformat(),
            "duration": duration,
            "gpu_mode": gpu_mode,
            "fps_data": []
        }
        
        # Generate simulated FPS data
        for i in range(duration):
            if gpu_mode == "single":
                fps = 30 + (i % 10)  # Simulate 30-40 FPS
            else:  # multi-GPU
                fps = 60 + (i % 15)  # Simulate 60-75 FPS
            
            results["fps_data"].append({
                "timestamp": i,
                "fps": fps
            })
        
        # Store results
        benchmark_results[benchmark_id] = results
        return benchmark_id
    
    @staticmethod
    def get_latest_results():
        """Retrieve latest single and multi-GPU results"""
        single_gpu = None
        multi_gpu = None
        
        for result in benchmark_results.values():
            if result["gpu_mode"] == "single" and (single_gpu is None or 
                datetime.fromisoformat(result["timestamp"]) > datetime.fromisoformat(single_gpu["timestamp"])):
                single_gpu = result
            elif result["gpu_mode"] == "multi" and (multi_gpu is None or 
                datetime.fromisoformat(result["timestamp"]) > datetime.fromisoformat(multi_gpu["timestamp"])):
                multi_gpu = result
        
        return {
            "single_gpu": single_gpu,
            "multi_gpu": multi_gpu
        }
    
    @staticmethod
    def export_results(format_type):
        """Export results in PNG or CSV format"""
        results = BenchmarkController.get_latest_results()
        
        if format_type == "png":
            return BenchmarkController._generate_plot(results)
        elif format_type == "csv":
            return BenchmarkController._generate_csv(results)
        else:
            raise ValueError("Unsupported format")
    
    @staticmethod
    def _generate_plot(results):
        """Generate FPS comparison plot"""
        plt.figure(figsize=(10, 6))
        
        if results["single_gpu"]:
            single_data = results["single_gpu"]["fps_data"]
            plt.plot([d["timestamp"] for d in single_data],

                    'b-', label='Single GPU')
        
        if results["multi_gpu"]:
            multi_data = results["multi_gpu"]["fps_data"]
            plt.plot([d["timestamp"] for d in multi_data], 
                    [d["fps"] for d in multi_data], 
                    'r-', label='Multi GPU')
        
        plt.title('GPU Performance Comparison')
        plt.xlabel('Time (seconds)')

        plt.legend()
        plt.grid(True)
        
        # Save to BytesIO

        plt.close()
        
        return img
