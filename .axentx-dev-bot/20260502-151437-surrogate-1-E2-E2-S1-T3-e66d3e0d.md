export interface ConnectorError extends Error {
  code: string;
  details?: unknown;
}

export function createConnectorError(code: string, message: string, details?: unknown): ConnectorError {
  const err = new Error(message) as ConnectorError;
  err.code = code;
  err.details = details;
  return err;
}

export type HealthStatus = 'healthy' | 'degraded' | 'unhealthy';

export interface HealthResult {
  status: HealthStatus;
  latencyMs?: number;
  details?: Record<string, unknown>;
}

export interface Connector {
  /** Establish/verify connection to the backend. */
  connect(): Promise<void>;

  /** Generic call to invoke backend methods in a uniform way. */
  call<T = unknown>(method: string, params?: Record<string, unknown>): Promise<T>;

  /** Health check for the connector. */
  health(): Promise<HealthResult>;
}

type Order = {
  id?: string;
  symbol: string;
  side: 'buy' | 'sell';
  quantity: number;
  price?: number;
};

export type SimulatedOrderRecord = Order & {
  id: string;
  status: 'open' | 'filled' | 'cancelled';
};

export class SimulatedBroker implements Connector {
  private connected = false;
  private orders = new Map<string, SimulatedOrderRecord>();

  async connect(): Promise<void> {
    await new Promise((r) => setTimeout(r, 5));
    this.connected = true;
  }

  async call<T = unknown>(method: string, params: Record<string, unknown> = {}): Promise<T> {
    this.ensureConnected();

    switch (method) {
      case 'place_order': {
        const order = this.validateOrder(params);
        const id = this.generateId();
        const record: SimulatedOrderRecord = { ...order, id, status: 'open' };
        this.orders.set(id, record);
        return { id, status: 'accepted' } as T;
      }

      case 'cancel': {
        const id = this.requireString(params, 'id');
        const existing = this.orders.get(id);
        if (!existing) {
          throw createConnectorError('order_not_found', `Order ${id} not found`, { id });
        }
        if (existing.status !== 'open') {
          throw createConnectorError('order_not_cancellable', `Order ${id} is ${existing.status}`, { id, status: existing.status });
        }
        existing.status = 'cancelled';
        return { id, status: 'cancelled' } as T;
      }

        return Array.from(this.orders.values()) as T;
