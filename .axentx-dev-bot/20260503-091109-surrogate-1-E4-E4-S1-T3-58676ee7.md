import sqlite3
import os
from datetime import datetime, timezone

DB_PATH = os.environ.get("AXENTX_DB", "data/events.db")

def init_db():
    os.makedirs(os.path.dirname(DB_PATH), exist_ok=True)
    conn = sqlite3.connect(DB_PATH)
    conn.execute("""
        CREATE TABLE IF NOT EXISTS events (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id TEXT,
            anon_id TEXT,
            event_name TEXT NOT NULL,
            document_id TEXT,
            timestamp TEXT NOT NULL
        )
    """)
    conn.execute("CREATE INDEX IF NOT EXISTS idx_event_name ON events(event_name)")
    conn.execute("CREATE INDEX IF NOT EXISTS idx_timestamp ON events(timestamp)")
    conn.commit()
    conn.close()

def insert_event(user_id=None, anon_id=None, event_name, document_id=None, timestamp=None):
    if timestamp is None:
        timestamp = datetime.now(timezone).isoformat()
    conn = sqlite3.connect(DB_PATH)
    conn.execute(
        "INSERT INTO events (user_id, anon_id, event_name, document_id, timestamp) VALUES (?, ?, ?, ?, ?)",
        (user_id, anon_id, event_name, document_id, timestamp)
    )
    conn.commit()
    conn.close()

def query_summary(days=30):
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    cur = conn.cursor()

    # daily mau and opens per user for last N days
    cur.execute("""
        SELECT
            DATE(timestamp) AS day,
            COUNT(DISTINCT COALESCE(user_id, anon_id)) AS mau,
            CAST(COUNT(CASE WHEN event_name = 'document_open' THEN 1 END) AS FLOAT) /
            NULLIF(COUNT(DISTINCT COALESCE(user_id, anon_id)), 0) AS avg_opens_per_user
        FROM events
        WHERE DATE(timestamp) >= DATE('now', ? || ' days')
        GROUP BY DATE(timestamp)
        ORDER BY day
    """, (-days,))
    rows = [dict(r) for r in cur.fetchall()]

    # overall 30d totals
    cur.execute("""
        SELECT
            COUNT(DISTINCT COALESCE(user_id, anon_id)) AS mau_30d,
            CAST(COUNT(CASE WHEN event_name = 'document_open' THEN 1 END) AS FLOAT) /
            NULLIF(COUNT(DISTINCT COALESCE(user_id, anon_id)), 0) AS avg_opens_per_user_30d
        FROM events
        WHERE DATE(timestamp) >= DATE('now', ? || ' days')
    """, (-days,))
    total = dict(cur.fetchone())

    conn.close()
    return {
        "mau_30d": total["mau_30d"] or 0,
        "avg_opens_per_user_30d": round(total["avg_opens_per_user_30d"] or 0, 2),
        "daily_series": rows
    }
```
