class ConfigError(Exception):
    """Raised when configuration validation fails."""

class Config:
    """Handles loading, validating, and saving sandbox configuration."""

    def __init__(self, config_path: Path | str | None = None):
        self.path = Path(config_path) if config_path else Path.home() / ".surrogate1_config.json"
        self._data: Dict[str, Any] = {}
        self.load()

    @property
    def data(self) -> Dict[str, Any]:
        """Current configuration dictionary (read-only)."""
        return self._data.copy()

    def load(self) -> None:
        """Load configuration from file, falling back to defaults."""
        if self.path.is_file():
            try:
                with self.path.open("r", encoding="utf-8") as f:
                    loaded = json.load(f)
                self._data = {**DEFAULT_CONFIG, **loaded}
            except (json.JSONDecodeError, OSError) as exc:
                raise ConfigError(f"Failed to read config file {self.path}: {exc}") from exc
        else:
            self._data = DEFAULT_CONFIG.copy()

        self.validate()

        """Validate current configuration values."""
        cfg = self._data

        if not isinstance(cfg.get("shard_id"), int) or not (0 <= cfg["shard_id"] < cfg["worker_count"]):
            raise ConfigError("shard_id must be an integer between 0 and worker_count-1")

            raise ConfigError("dataset_url must be a valid HTTP/HTTPS URL")

        if not isinstance(cfg.get("output_path"), str) or not cfg["output_path"]:
