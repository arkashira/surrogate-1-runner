# /opt/axentx/surrogate-1/scripts/snapshot_file_list.sh
#!/usr/bin/env bash
set -euo pipefail
REPO="datasets/axentx/surrogate-1-mirror"
FOLDER="batches/mirror-merged/$(date +%Y-%m-%d)"
OUT="/opt/axentx/surrogate-1/file_list.json"

python3 -c "
import json, os, itertools
from huggingface_hub import HfApi
api = HfApi()

def list_all_files(repo, folder):
    # Recursive enumeration to capture nested parquet files
    entries = api.list_repo_tree(repo, path=folder, recursive=True)
    return sorted(e.path for e in entries if e.type == 'file' and e.path.endswith('.parquet'))

files = list_all_files('$REPO', '$FOLDER')
os.makedirs(os.path.dirname('$OUT'), exist_ok=True)
with open('$OUT', 'w') as f:
    json.dump({'repo': '$REPO', 'folder': '$FOLDER', 'files': files}, f, indent=2)
print(f'Snapshot saved: {len(files)} files')
"

### 2. Ingestion: project schema and write canonical parquet

```python
# /opt/axentx/surrogate-1/ingest_project_to_parquet.py
#!/usr/bin/env python3
"""
Project raw mirror files to {prompt, response} only and write canonical parquet.
Usage: python3 ingest_project_to_parquet.py <input.parquet> <output.parquet>
"""
import sys
import pandas as pd
import pyarrow as pa
import pyarrow.parquet as pq
from pathlib import Path

COLUMNS = {"prompt", "response"}

def project(in_path: str, out_path: str) -> None:
    df = pq.read_table(in_path).to_pandas()
    keep = [c for c in df.columns if c in COLUMNS]
    if not keep:
        raise ValueError(f"No canonical columns found in {in_path}. Available: {list(df.columns)}")
    df = df[keep].rename(columns={k: k for k in keep})
    df["prompt"] = df["prompt"].astype(str)
    df["response"] = df["response"].astype(str)
    df = df.dropna(subset=["prompt", "response"]).reset_index(drop=True)
    table = pa.Table.from_pandas(df, preserve_index=False)
    Path(out_path).parent.mkdir(parents=True, exist_ok=True)
    pq.write_table(table, out_path, compression="zstd")
    print(f"Projected {len(df)} rows -> {out_path}")

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: python3 ingest_project_to_parquet.py <input.parquet> <output.parquet>")
        sys.exit(1)
    project(sys.argv[1], sys.argv[2])
```
