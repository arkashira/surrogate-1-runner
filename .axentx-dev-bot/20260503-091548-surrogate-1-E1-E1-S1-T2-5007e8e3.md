// routes/status.test.ts
import { createRequest, getStatusPage } from '../routes/status';
import { RequestRepository } from '../repositories/RequestRepository';

describe('POST /requests', () => {
  test('creates request and returns uuid within 200ms', async () => {
    const payload = { title: 'Incident A', type: 'incident', owner: 'team-x', slaTarget: 3600 };
    const start = Date.now();
    const res = await createRequest(payload);
    const elapsed = Date.now() - start;
    expect(res.status).toBe(201);
    expect(res.body).toHaveProperty('uuid');
    expect(typeof res.body.uuid).toBe('string');
    expect(elapsed).toBeLessThanOrEqual(200);
  });

  test('returns 400 for missing required fields', async () => {
    const res = await createRequest({ title: '' });
    expect(res.status).toBe(400);
    expect(res.body).toHaveProperty('error', 'validation failed');
    expect(Array.isArray(res.body.details)).toBe(true);
  });
});

describe('GET /status/:uuid', () => {
  test('renders unbranded status page with title, state, owner, slaTarget', async () => {
    const req = await RequestRepository.create({ title: 'Incident A', type: 'incident', owner: 'team-x', slaTarget: 3600 });
    const res = await getStatusPage(req.uuid);
    expect(res.status).toBe(200);
    expect(res.headers['content-type']).toContain('text/html');
    expect(res.text).toContain(req.title);
    expect(res.text).toContain(req.state);
    expect(res.text).toContain(req.owner);
    expect(res.text).toContain(String(req.slaTarget));
    expect(res.text).not.toMatch(/secret|internal|debug/i);
  });

  test('returns 404 for unknown uuid', async () => {
    const res = await getStatusPage('00000000-0000-0000-0000-000000000000');
    expect(res.status).toBe(404);
    expect(res.body).toHaveProperty('error', 'not found');
  });
});
