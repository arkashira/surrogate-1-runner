"""
Test Runner Module for surrogate-1 CI/CD integration.

This module provides a command‑line entry point that can be invoked by CI/CD
runners. It discovers and executes unit tests, collects structured results,
and outputs them as JSON. Configuration can be supplied via environment
variables or a YAML/JSON config file.

Supported configuration sources (in order of precedence):

1. Environment variables
   - SURROGATE_TEST_CONFIG   : Path to a YAML or JSON config file.
   - SURROGATE_TEST_MODULES : Comma‑separated list of test module names.
   - SURROGATE_TEST_VERBOSITY: Integer verbosity level for unittest.
   - SURROGATE_TEST_OUTPUT   : Path to write the JSON results file.
   - SURROGATE_TEST_LOGLEVEL : Logging level (e.g., DEBUG, INFO).

2. Config file (YAML or JSON)
   - modules   : List of test module names.
   - verbosity : Integer verbosity level.
   - output    : Path to write the JSON results file.

If no modules are specified, the default unittest discovery
(`unittest.defaultTestLoader.discover`) is used.
"""

import os
import sys
import json
import yaml
import argparse
import logging
import datetime
import unittest
from typing import List, Dict, Any, Optional

# --------------------------------------------------------------------------- #
# Logging configuration
# --------------------------------------------------------------------------- #
LOG_LEVEL = os.getenv("SURROGATE_TEST_LOGLEVEL", "INFO").upper()
logging.basicConfig(
    level=getattr(logging, LOG_LEVEL, logging.INFO),
    format="%(asctime)s [%(levelname)s] %(message)s",
)
logger = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Custom Test Result to capture structured data
# --------------------------------------------------------------------------- #
class StructuredTestResult(unittest.TextTestResult):
    """
    Extends unittest.TextTestResult to expose test outcomes as a dictionary.
    """

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.start_time = datetime.datetime.utcnow()
        self.end_time = None

    def addSuccess(self, test):
        super().addSuccess(test)

    def addFailure(self, test, err):
        super().addFailure(test, err)

    def addError(self, test, err):
        super().addError(test, err)

    def addSkip(self, test, reason):
        super().addSkip(test, reason)

    def addExpectedFailure(self, test, err):
        super().addExpectedFailure(test, err)

    def addUnexpectedSuccess(self, test):
        super().addUnexpectedSuccess(test)

    def stopTestRun(self):
        super().stopTestRun()
        self.end_time = datetime.datetime.utcnow()

    def get_summary(self) -> Dict[str, Any]:
        """
        Return a dictionary summarizing the test run.
        """
        return {
            "testsRun": self.testsRun,
            "failures": len(self.failures),
            "errors": len(self.errors),
            "skipped": len(self.skipped),
            "expectedFailures": len(self.expectedFailures),
            "unexpectedSuccesses": len(self.unexpectedSuccesses),
            "startTime": self.start_time.isoformat() + "Z",
            "endTime": self.end_time.isoformat() + "Z" if self.end_time else None,
            "durationSeconds": (
                (self.end_time - self.start_time).total_seconds()
                if self.end_time
                else None
            ),
        }

# --------------------------------------------------------------------------- #
# Configuration handling
# --------------------------------------------------------------------------- #
def load_config_from_file(path: str) -> Dict[str, Any]:
    """
    Load configuration from a YAML or JSON file.
    """
    logger.debug(f"Loading config file: {path}")
    with open(path, "r", encoding="utf-8") as f:
        if path.lower().endswith((".yaml", ".yml")):
            return yaml.safe_load(f) or {}

from surrogate_1.test_runner import load_config, merge_configs

def test_load_config_yaml(tmp_path):

    log_level: INFO
    output: results.json

    cfg_file.write_text(yaml_content)

    yaml_content = "log_level: INFO"
    cfg_file = tmp_path / "config.yaml"
    cfg_file.write_text(yaml_content)
    os.environ["SURROGATE_CONFIG"] = str(cfg_file)
    os.environ["SURROGATE_LOG_LEVEL"] = "DEBUG"
    cfg = load_config()

def test_merge_configs():
    file_cfg = {"log_level": "INFO", "output": "out.json"}
