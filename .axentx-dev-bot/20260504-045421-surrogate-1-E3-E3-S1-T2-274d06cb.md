def test_engine_initialization(user_profiles):
    engine = RecommendationEngine(user_profiles)
    assert engine.user_profiles == user_profiles
    assert isinstance(engine, RecommendationEngine)

def test_generate_recommendations_sorted(user_profiles):
    engine = RecommendationEngine(user_profiles)
    recs = engine.generate_recommendations()
    assert isinstance(recs, list)
    assert all(isinstance(r, Recommendation) for r in recs)
    savings = [r.savings for r in recs]
    assert savings == sorted(savings, reverse=True)

def test_get_weekly_recommendations_top_three(user_profiles):
    engine = RecommendationEngine(user_profiles)
    weekly = engine.get_weekly_recommendations()
    assert len(weekly) <= 3
    savings = [r.savings for r in weekly]
    assert savings == sorted(savings, reverse=True)

def test_recommendation_fields(user_profiles):
    engine = RecommendationEngine(user_profiles)
    recs = engine.generate_recommendations()
    for r in recs:
        assert r.action
        assert r.description
        assert r.savings > 0
