```javascript
// src/emulator/middleware/rateLimit.js
const rateLimit = require('express-rate-limit');

const createRateLimiter = (maxRequests = 1200, windowMs = 3600000) => {
  return rateLimit({
    windowMs, // Time window in milliseconds
    max: maxRequests, // Max requests per window
    standardHeaders: true, // Return rate limit info in `RateLimit-*` headers
    legacyHeaders: false, // Disable `X-RateLimit-*` headers
    message: {
      errors: [
        {
          message: 'API rate limit exceeded',
        }
      ]
    },
    // Explicitly count all requests (success and failure) against the limit
    skipSuccessfulRequests: false,
    skipFailedRequests: false,
    // Custom handler to ensure consistent JSON error structure
    handler: (req, res, next, options) => {
      res.status(options.statusCode).json({
        errors: [
          {
            message: 'API rate limit exceeded',
          }
        ]
      });
    }
  });
};

module.exports = {
  createRateLimiter
};
```

```js
const rateLimit = require('../../src/emulator/middleware/rateLimit');
const express = require('express');
const request = require('supertest');

describe('rateLimit middleware', () => {
  let app;
  const limit = 5; // 5 requests per minute for test

  beforeEach(() => {
    app = express();
    app.use(express.json());
    app.use(rateLimit({ limit, windowMs: 60 * 1000 }));
    app.post('/api/v1/series', (req, res) => res.status(202).json({ status: 'ok' }));
  });

  test('allows requests below the limit', async () => {
    for (let i = 0; i < limit; i++) {
      const res = await request(app).post('/api/v1/series').send([{ metric: 'test' }]);
      expect(res.status).toBe(202);
      expect(res.headers['x-ratelimit-limit']).toBe(String(limit));
    }
  });

  test('blocks requests exceeding the limit', async () => {
    for (let i = 0; i < limit; i++) {
      await request(app).post('/api/v1/series').send([{ metric: 'test' }]);
    }
    const res = await request(app).post('/api/v1/series').send([{ metric: 'test' }]);
    expect(res.status).toBe(429);
    expect(res.body).toEqual({ error: 'API rate limit exceeded' });
  });

  test('resets after windowMs', async () => {
    jest.useFakeTimers();
    for (let i = 0; i < limit; i++) {
      await request(app).post('/api/v1/series').send([{ metric: 'test' }]);
    }
    jest.advanceTimersByTime(60 * 1000 + 1); // advance past window
    const res = await request(app).post('/api/v1/series').send([{ metric: 'test' }]);
    expect(res.status).toBe(202);
    jest.useRealTimers();
  });
});
```
