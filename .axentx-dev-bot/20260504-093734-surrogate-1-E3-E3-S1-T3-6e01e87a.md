**3. Integration Tests** *(using pytest + requests for HTTP service)*  

```python
# tests/integration/test_large_stream.py
import io
import time
import pytest
import requests

BASE_URL = os.getenv("SURROGATE_API_URL", "http://localhost:8000")
STREAM_ENDPOINT = f"{BASE_URL}/upload"

@pytest.fixture(scope="session")
def one_gb_stream():
    # Generate a 1 GB stream lazily to avoid loading into RAM
    class OneGBGenerator(io.RawIOBase):
        def __init__(self):
            self.remaining = 1 * 1024 * 1024 * 1024  # bytes

        def readinto(self, b):
            if self.remaining <= 0:
                return 0
            chunk = min(len(b), self.remaining)
            b[:chunk] = b'A' * chunk
            self.remaining -= chunk
            return chunk

    return io.BufferedReader(OneGBGenerator())

def test_1gb_stream_completes_within_30min(one_gb_stream):
    """AC‑2 – stream 1 GB and ensure it finishes ≤ 1800 s with 2xx."""
    start = time.time()
    response = requests.post(
        STREAM_ENDPOINT,
        data=one_gb_stream,
        headers={"Content-Type": "application/octet-stream"},
        timeout=1900,               # slightly above limit to catch hangs
    )
    elapsed = time.time() - start

    assert response.status_code // 100 == 2, "Service did not return success"
    assert elapsed <= 1800, f"Stream took {elapsed}s > 30min"
```
