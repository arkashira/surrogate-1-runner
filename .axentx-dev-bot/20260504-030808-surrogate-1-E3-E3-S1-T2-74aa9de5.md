const router = express.Router();
const { Solution, User } = require('../models');

/**
 * Middleware that pretends authentication has populated req.user.

 */
function mockAuth(req, res, next) {
  // For demo/testing purposes we allow a header `x-user-id`.
  const userId = req.header('x-user-id') || 'anonymous';
  req.user = { id: userId };
  next();
}

/**
 * GET /solutions/:id/votes
 * Returns the total vote count for a solution and, if the request is
 * authenticated, the current user's vote.
 */
router.get('/solutions/:id/votes', mockAuth, async (req, res) => {
  const { id } = req.params;
  const solution = await Solution.findByPk(id, {
    include: [{ model: User }],
  });

  if (!solution) {
    return res.status(404).json({ error: 'Solution not found.' });
  }

  const userVote = solution.UserVotes.find(vote => vote.userId === req.user.id)?.vote || 0;
  res.json({
    solutionId: id,
    totalVotes: solution.totalVotes,
    userVote,
  });
});

/**
 * POST /solutions/:id/vote
 * Body: { vote: 1 | -1 }
 * Records an up-vote (1) or down-vote (-1) from the authenticated user.
 * If the user has already voted, the previous vote is replaced.
 */
router.post('/solutions/:id/vote', mockAuth, express.json(), async (req, res) => {
  const { id } = req.params;
  const { vote } = req.body;
  const userId = req.user.id;

  if (vote !== 1 && vote !== -1) {
    return res.status(400).json({ error: 'Invalid vote value; must be 1 or -1.' });
  }

  const solution = await Solution.findByPk(id, {
    include: [{ model: User, as: 'UserVotes' }],
  });

  if (!solution) {
    return res.status(404).json({ error: 'Solution not found.' });
  }

  let userVoteRecord = solution.UserVotes.find(vote => vote.userId === userId);

  if (userVoteRecord) {
    if (userVoteRecord.vote === vote) {
      return res.status(400).json({ message: 'You have already voted on this solution.' });
    }

    userVoteRecord.vote = vote;
    await userVoteRecord.save();
  } else {
    userVoteRecord = await Solution.create({ solutionId: id, userId, vote });
    await userVoteRecord.save();
  }

  solution.totalVotes += vote - (userVoteRecord.vote || 0);
  await solution.save();

  res.json({
    solutionId: id,
    totalVotes: solution.totalVotes,
    userVote: vote,
  });
});

/**
 * DELETE /solutions/:id/vote
 * Removes the authenticated user's vote for the solution.
 */
router.delete('/solutions/:id/vote', mockAuth, async (req, res) => {
  const { id } = req.params;
  const userId = req.user.id;

  const solution = await Solution.findByPk(id, {
    include: [{ model: User, as: 'UserVotes' }],
  });

  if (!solution) {
    return res.status(404).json({ error: 'Solution not found.' });
  }

  const userVoteRecord = solution.UserVotes.find(vote => vote.userId === userId);

  if (!userVoteRecord) {
    return res.status(404).json({ error: 'No existing vote to remove.' });
  }

  solution.totalVotes -= userVoteRecord.vote;
  await userVoteRecord.destroy();
  await solution.save();

  res.json({
    solutionId: id,
    totalVotes: solution.totalVotes,
    userVote: 0,
  });
});

module.exports = router;
