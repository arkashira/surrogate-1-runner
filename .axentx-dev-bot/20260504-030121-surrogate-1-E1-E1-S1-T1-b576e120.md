import threading
import queue
import time
import numpy as np
import sounddevice as sd

# Constants
TARGET_RMS = 0.7071  # Corresponds to -3 dBFS for a full‑scale sine wave; using 0 dBFS (max) would clip.
MAX_GAIN_CHANGE_PER_SEC = 6.0  # dB per second, limits rapid gain swings when AGC changes.
MAX_GAIN = 10.0  # Prevent runaway amplification.
MIN_GAIN = 0.1   # Prevent excessive attenuation.

def _rms(samples: np.ndarray) -> float:
    """Root‑mean‑square of a 1‑D int16 array, returned as a float in the range [0, 1]."""
    if samples.size == 0:
        return 0.0
    # Normalize to [-1, 1] before RMS calculation.
    norm = samples.astype(np.float32) / 32768.0
    return np.sqrt(np.mean(norm ** 2))

def _db_to_linear(db: float) -> float:
    """Convert decibels to a linear amplitude factor."""
    return 10 ** (db / 20.0)

def _linear_to_db(linear: float) -> float:
    """Convert a linear amplitude factor to decibels."""
    # Guard against log of zero.
    if linear <= 0:
        return -float('inf')
    return 20.0 * np.log10(linear)

def _compute_gain(current_rms: float, target_rms: float, prev_gain: float, dt: float) -> float:
    """
    Compute a gain factor that moves the RMS towards the target while limiting
    the rate of change to MAX_GAIN_CHANGE_PER_SEC.
    """
    if current_rms <= 0:
        return prev_gain  # No signal – keep previous gain.

    desired_gain = target_rms / current_rms
    # Clamp desired gain to allowed range.
    desired_gain = np.clip(desired_gain, MIN_GAIN, MAX_GAIN)

    # Limit gain change per second.
    max_change = _db_to_linear(MAX_GAIN_CHANGE_PER_SEC * dt)
    gain_change = np.clip(desired_gain / prev_gain, 1 / max_change, max_change)

    new_gain = prev_gain * gain_change
    return np.clip(new_gain, MIN_GAIN, MAX_GAIN)

def _apply_gain(samples: np.ndarray, gain: float) -> np.ndarray:
    """Apply linear gain to int16 samples and clip to int16 range."""
    # Convert to float, apply gain, then clip and cast back.
    scaled = samples.astype(np.float32) * gain
    clipped = np.clip(scaled, -32768, 32767)
    return clipped.astype(np.int16)

class AudioCapture:
    """
    Cross‑platform microphone capture that yields 16 kHz, 16‑bit PCM
    with gain normalisation (RMS ≈ TARGET_RMS).  The public interface
    is an iterator yielding raw ``bytes`` objects ready for downstream
    transcription services.
    """

    def __init__(
        self,
        samplerate: int = 16000,
        block_duration: float = 0.01,  # 10 ms blocks → 160 samples
        target_rms: float = TARGET_RMS,
    ):
        self.samplerate = samplerate
        self.block_size = int(self.samplerate * block_duration)
        self.target_rms = target_rms

        self._audio_q: queue.Queue[np.ndarray] = queue.Queue(maxsize=20)
        self._output_q: queue.Queue[bytes] = queue.Queue(maxsize=20)

        self._stream = None
        self._running = threading.Event()
        self._worker_thread = None

        # Gain state
        self._prev_gain = 1.0
        self._last_ts = None

    # --------------------------------------------------------------------- #
    # SoundDevice callback
    # --------------------------------------------------------------------- #
    def _callback(self, indata, frames, time_info, status):
        """SoundDevice stream callback – pushes raw int16 blocks into a queue."""
        if status:
            # Non‑fatal warnings are logged; we keep running.
            print(f"[AudioCapture] Stream status: {status}")

        # Ensure correct dtype and shape.
        block = np.copy(indata[:, 0])  # mono channel
        try:
            self._audio_q.put_nowait(block)
        except queue.Full:
            # Drop oldest block to keep latency low.
            try:
                self._audio_q.get_nowait()
                self._audio_q.put_nowait(block)
            except queue.Empty:
                pass  # Should never happen.
