import json
import csv
import os
from typing import Dict, List, Tuple

class ComparisonEngine:
    def __init__(self, test_run1: Dict, test_run2: Dict):
        """
        Initialize the ComparisonEngine with two test runs.

        Args:
        test_run1 (Dict): The first test run.
        test_run2 (Dict): The second test run.
        """
        self.test_run1 = test_run1
        self.test_run2 = test_run2

    def diff_view(self) -> List[Tuple]:
        """
        Show differences between two test runs.

        Returns:
        List[Tuple]: A list of tuples containing the key, value from test_run1, and value from test_run2.
        """
        differences = []
        for key in set(self.test_run1.keys()) | set(self.test_run2.keys()):
            if key not in self.test_run1:
                differences.append((key, None, self.test_run2[key]))
            elif key not in self.test_run2:
                differences.append((key, self.test_run1[key], None))
            elif self.test_run1[key] != self.test_run2[key]:
                differences.append((key, self.test_run1[key], self.test_run2[key]))
        return differences

    def impact_analysis(self) -> Dict:
        """
        Analyze the impact of changes on test outcomes.

        Returns:
        Dict: A dictionary containing the key, and a dictionary with 'before' and 'after' values.
        """
        impact = {}
        for key in set(self.test_run1.keys()) | set(self.test_run2.keys()):
            if key not in self.test_run1:
                impact[key] = {"before": None, "after": self.test_run2[key]}
            elif key not in self.test_run2:
                impact[key] = {"before": self.test_run1[key], "after": None}
            elif self.test_run1[key] != self.test_run2[key]:
                impact[key] = {"before": self.test_run1[key], "after": self.test_run2[key]}
        return impact

    def export_json(self, data: Dict, export_path: str) -> None:
        """
        Export comparison results to a JSON file.

        Args:
        data (Dict): The data to be exported.
        export_path (str): The path to the export file.
        """
        with open(export_path, 'w') as f:
            json.dump(data, f, indent=4)

    def export_csv(self, data: Dict, export_path: str) -> None:
        """
        Export comparison results to a CSV file.

        Args:
        data (Dict): The data to be exported.
        export_path (str): The path to the export file.
        """
        with open(export_path, 'w', newline='') as csvfile:
            writer = csv.writer(csvfile)
            writer.writerow(["Key", "Before", "After"])
            for key, value in data.items():
                writer.writerow([key, str(value["before"]), str(value["after"])])
