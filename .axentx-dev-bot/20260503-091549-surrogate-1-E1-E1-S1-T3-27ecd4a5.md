import { Router } from 'express';
import { createRequest } from '../services/requestService';
import { validateRequest } from '../utils/validation';

const router = Router();

router.post('/', async (req, res) => {
  const { title, type, owner, slaTarget } = req.body;
  const validationErrors = validateRequest({ title, type, owner, slaTarget });
  if (validationErrors.length > 0) {
    return res.status(400).json({ errors: validationErrors });
  }
  const request = await createRequest({ title, type, owner, slaTarget });
  res.json({
    id: request.id,
    publicUrl: `/requests/${request.id}`,
    currentStatus: request.status,
    createdAt: request.createdAt,
  });
});

router.get('/team/:team', async (req, res) => {
  const team = req.params.team;
  const requests = await getTeamRequests(team);
  res.json(requests);
});

export default router;

```typescript
import { Request } from '../models/request';

async function createRequest({ title, type, owner, slaTarget }) {
  const request = new Request({ title, type, owner, slaTarget });
  await request.save();
  return {
    id: request.id,
    publicUrl: `/requests/${request.id}`,
    currentStatus: request.status,
    createdAt: request.createdAt,
  };
}

async function getTeamRequests(team: string) {
  const requests = await Request.find({ team })
    .sort({ createdAt: -1 })
    .skip(0)
    .limit(10);
  return requests.map((request) => ({
    id: request.id,
    publicUrl: `/requests/${request.id}`,
    currentStatus: request.status,
    createdAt: request.createdAt,
  }));
}

export { createRequest, getTeamRequests };
```

function validateRequest({ title, type, owner, slaTarget }) {
  const errors = [];
  if (!title) {
    errors.push('Title is required');
  }
  if (!type) {
    errors.push('Type is required');
  }
  if (!owner) {
    errors.push('Owner is required');
  }
  if (slaTarget && typeof slaTarget !== 'number') {
    errors.push('SLA target must be a number');
  }
  return errors;
}

export { validateRequest };
