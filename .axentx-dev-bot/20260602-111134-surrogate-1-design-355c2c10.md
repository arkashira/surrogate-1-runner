```python
import pyarrow.parquet as pq
import json
from typing import List, Dict
import os
from pathlib import Path

class ParquetDatasetLoader:
    def __init__(self, config_path: str = "config/file_list.json"):
        with open(config_path) as f:
            self.config = json.load(f)
        self.validate_config()

    def validate_config(self):
        """Validate dataset configuration"""
        if not isinstance(self.config.get("sources"), list):
            raise ValueError("Sources must be a list in config")

    def load_dataset(self) -> pq.ParquetDataset:
        """Load dataset from configured sources"""
        dataset = pq.ParquetDataset(
            [Path(src["local_path"]) for src in self.config["sources"]],
            validate_schema=not self.config.get("skip_schema_validation", False)
        )
        return dataset

    def get_schema(self) -> Dict:
        """Get dataset schema for validation"""
        return self.config.get("expected_schema", {})
```

```python
def test_verify_checksum_match(downloader):
    """Test checksum verification passes when correct."""
    data = b"test data"
    file_path = "test.parquet"
    expected_hash = hashlib.sha256(data).hexdigest()
    with open(file_path, "wb") as f:
        f.write(data)
    assert downloader.verify_checksum(file_path, expected_hash) is True
    os.remove(file_path)

def test_verify_checksum_mismatch(downloader):
    """Test checksum verification fails when incorrect."""
    data = b"test data"
    file_path = "test.parquet"
    wrong_hash = "0" * 64  # invalid hash
    with open(file_path, "wb") as f:
        f.write(data)
    assert downloader.verify_checksum(file_path, wrong_hash) is False
    os.remove(file_path)
```
