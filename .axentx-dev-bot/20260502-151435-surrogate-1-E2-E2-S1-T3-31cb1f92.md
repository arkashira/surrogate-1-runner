from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, Optional
import uuid


class OrderSide(str, Enum):
    BUY = "buy"
    SELL = "sell"


class OrderType(str, Enum):
    LIMIT = "limit"
    MARKET = "market"
    STOP = "stop"
    STOP_LIMIT = "stop_limit"


class OrderStatus(str, Enum):
    NEW = "new"
    ACKED = "acked"
    PARTIALLY_FILLED = "partially_filled"
    FILLED = "filled"
    REJECTED = "rejected"
    CANCELLED = "cancelled"


class EventType(str, Enum):
    ORDER_SUBMITTED = "order_submitted"
    ORDER_ACKED = "order_acked"
    ORDER_FILLED = "order_filled"
    ORDER_REJECTED = "order_rejected"
    ORDER_CANCELLED = "order_cancelled"


@dataclass(frozen=True)
class OrderEvent:
    """Normalized order lifecycle event."""
    event_id: str
    event_type: EventType
    order_id: str
    client_order_id: str
    symbol: str
    side: OrderSide
    order_type: OrderType
    quantity: float
    price: Optional[float]
    status: OrderStatus
    timestamp: datetime
    broker: str
    metadata: Dict[str, Any] = field(default_factory=dict)
    filled_quantity: float = 0.0
    avg_fill_price: Optional[float] = None
    rejection_reason: Optional[str] = None
    remaining_quantity: float = field(init=False)

    def __post_init__(self) -> None:
        object.__setattr__(self, "remaining_quantity", max(0.0, self.quantity - self.filled_quantity))
        if self.timestamp.tzinfo is None:
            object.__setattr__(self, "timestamp", self.timestamp.replace(tzinfo=timezone.utc))

    def to_dict(self) -> Dict[str, Any]:
        return {

            "event_type": self.event_type.value,
            "order_id": self.order_id,

            "side": self.side.value,

    sub = OrderEvent.submitted(order_id="ord-1", broker="simulated", symbol="AAPL", quantity=100, side="buy", price=150.0)

    assert ack.updated_at >= ack.created_at

def test_filled_updates_remaining_quantity():
