import { render, screen, fireEvent } from '@testing-library/react';
import BuildGuidance from '../src/components/BuildGuidance';

jest.mock('../src/utils/recommendationEngine');

describe('BuildGuidance', () => {
  beforeEach(() => {
    engine.generateRecommendations.mockClear();
  });

  test('renders form fields and submit button', () => {
    render(<BuildGuidance />);
    expect(screen.getByLabelText(/needs/i)).toBeInTheDocument();
    expect(screen.getByLabelText(/budget/i)).toBeInTheDocument();
    expect(screen.getByRole('button', { name: /get recommendations/i })).toBeInTheDocument();
  });

  test('calls engine with correct args on submit', () => {
    engine.generateRecommendations.mockReturnValue([{ title: 'CPU', price: 200 }]);

    render(<BuildGuidance />);
    fireEvent.change(screen.getByLabelText(/needs/i), { target: { value: 'gaming' } });
    fireEvent.change(screen.getByLabelText(/budget/i), { target: { value: '500' } });
    fireEvent.click(screen.getByRole('button', { name: /get recommendations/i }));

    expect(engine.generateRecommendations).toHaveBeenCalledWith('gaming', 500);
  });

  test('shows error when budget is empty', () => {
    render(<BuildGuidance />);
    fireEvent.change(screen.getByLabelText(/needs/i), { target: { value: 'gaming' } });
    fireEvent.click(screen.getByRole('button', { name: /get recommendations/i }));

    expect(screen.getByText(/please enter a budget/i)).toBeInTheDocument();
    expect(engine.generateRecommendations).not.toHaveBeenCalled();
  });

  test('shows error when needs are empty', () => {
    render(<BuildGuidance />);
    fireEvent.change(screen.getByLabelText(/budget/i), { target: { value: '300' } });
    fireEvent.click(screen.getByRole('button', { name: /get recommendations/i }));

    expect(screen.getByText(/please describe your needs/i)).toBeInTheDocument();
    expect(engine.generateRecommendations).not.toHaveBeenCalled();
  });

  test('updates recommendations on re‑submit', () => {
    engine.generateRecommendations
      .mockReturnValueOnce([{ title: 'GPU', price: 300 }])
      .mockReturnValueOnce([{ title: 'CPU', price: 200 }]);

    render(<BuildGuidance />);
    const needsInput = screen.getByLabelText(/needs/i);
    const budgetInput = screen.getByLabelText(/budget/i);
    const button = screen.getByRole('button', { name: /get recommendations/i });

    fireEvent.change(needsInput, { target: { value: 'gaming' } });
    fireEvent.change(budgetInput, { target: { value: '300' } });
    fireEvent.click(button);

    expect(screen.getByText(/gpu/i)).toBeInTheDocument();

    fireEvent.change(budgetInput, { target: { value: '200' } });
    fireEvent.click(button);

    expect(screen.getByText(/cpu/i)).toBeInTheDocument();
  });
});
