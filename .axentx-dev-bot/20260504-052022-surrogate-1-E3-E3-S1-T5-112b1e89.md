from unittest.mock import patch, mock_open
import os
from surrogate.templates import list_templates, apply_template

    @patch('os.listdir')

    @patch('os.path.join', return_value='templates/library/reddit.yaml')
    def test_apply_template(self, mock_join, mock_file):
        with patch('builtins.input', return_value='y'):
            result = apply_template('reddit.yaml')
            self.assertTrue(result)

    @patch('os.path.join', return_value='templates/library/reddit.yaml')

        with patch('builtins.input', return_value='n'):
            result = apply_template('reddit.yaml')
            self.assertFalse(result)
            mock_file.assert_not_called()

if __name__ == '__main__':
    unittest.main()

import os

def list_templates():
    template_dir = 'templates/library'
    templates = [f for f in os.listdir(template_dir) if f.endswith('.yaml')]
    return templates

def apply_template(template_name):
    template_path = os.path.join('templates/library', template_name)
    with open(template_path, 'r') as file:
        template_content = file.read()

    print(f"Template content:\n{template_content}")
    confirmation = input("Do you want to apply this template to surrogate.yaml? (y/n): ").strip().lower()

    if confirmation == 'y':
        with open('surrogate.yaml', 'w') as file:
            file.write(template_content)
        return True
    return False
