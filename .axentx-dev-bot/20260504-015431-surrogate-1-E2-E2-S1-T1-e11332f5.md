def _run_hook(repo_dir: Path, file_path: str, file_content: str) -> subprocess.CompletedProcess:
    """Helper to stage a file and run the pre-commit hook."""
    target_file = repo_dir / file_path
    target_file.parent.mkdir(parents=True, exist_ok=True)
    target_file.write_text(file_content)
    
    subprocess.run(["git", "add", file_path], cwd=repo_dir, check=True, capture_output=True)
    
    return subprocess.run(
        ["git", "commit", "--no-verify", "-m", "test commit"],
        cwd=repo_dir,
        capture_output=True,
        env={**os.environ, "GIT_DIR": str(repo_dir / ".git")}
    )

    """Test that hook accepts correctly formatted files."""
    clean_content = "def hello():\n    print('Hello, World!')\n"
    
    result = _run_hook(git_repo, "valid.py", clean_content)
    
    assert result.returncode == 0, f"Hook should pass for valid file: {result.stderr.decode()}"


def test_hook_fails_on_trailing_whitespace(git_repo):

    bad_content = "def hello():\n    print('Hello, World!')   \n"
