Prometheus integration module for AlertForge.

This module provides a lightweight client for querying Prometheus
and assembling a unified view of metrics.  It is intentionally
minimal to keep the integration fast and easy to configure.

Usage
-----
>>> from integrations.prometheus import PrometheusClient, PrometheusConfig
>>> cfg = PrometheusConfig(endpoint="http://localhost:9090")
>>> client = PrometheusClient(cfg)
>>> metrics = client.query(["up", "node_cpu_seconds_total"])
>>> print(metrics)
{'up': 1, 'node_cpu_seconds_total': 12345.67}
"""

from __future__ import annotations

import json
import logging
from dataclasses import dataclass, field
from typing import Dict, Iterable, List, Optional

import requests
from requests import Response

log = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #

@dataclass
class PrometheusConfig:
    """
    Configuration for connecting to a Prometheus server.

    Attributes
    ----------
    endpoint : str
        Base URL of the Prometheus HTTP API (e.g. "http://localhost:9090").
    timeout : float
        Request timeout in seconds.
    headers : Dict[str, str]
        Optional HTTP headers to send with each request.
    """
    endpoint: str
    timeout: float = 5.0
    headers: Dict[str, str] = field(default_factory=dict)


# --------------------------------------------------------------------------- #
# Client
# --------------------------------------------------------------------------- #

class PrometheusClient:
    """
    Simple Prometheus HTTP API client.

    The client supports instant queries and range queries.  It
    returns the result in a Python-friendly format.
    """

    def __init__(self, config: PrometheusConfig):
        self.config = config
        self.base_url = config.endpoint.rstrip("/")
        log.debug("PrometheusClient initialized with endpoint %s", self.base_url)

    def _request(self, path: str, params: Dict[str, str]) -> Response:
        url = f"{self.base_url}/api/v1{path}"
        log.debug("Requesting %s with params %s", url, params)
        resp = requests.get(url, params=params, timeout=self.config.timeout, headers=self.config.headers)
        resp.raise_for_status()
        return resp

    def _parse_result(self, resp_json: Dict) -> Dict:
        """
        Convert Prometheus API JSON to a simple dict.

        For instant queries, returns {metric_name: value}.
        For range queries, returns {metric_name: [value1, value2, ...]}.
        """
        result_type = resp_json["data"]["resultType"]
        results = resp_json["data"]["result"]
        parsed: Dict = {}

        if result_type == "vector":
            for item in results:
                metric_name = item["metric"].get("__name__")
                if metric_name:
                    parsed[metric_name] = float(item["value"][1])
        elif result_type == "matrix":

                metric_name = item["metric"].get("__name__")
                if metric_name:
                    parsed[metric_name] = [float(v[1]) for v in item["values"]]
        else:
            log.warning("Unsupported result type: %s", result_type)
