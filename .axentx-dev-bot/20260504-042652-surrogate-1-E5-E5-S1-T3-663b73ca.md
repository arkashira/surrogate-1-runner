// Framework: Spring Boot Test with MockMvc

@WebMvcTest(DashboardController.class)
class DashboardControllerTest {

    @Autowired
    private MockMvc mockMvc;

    @MockBean
    private RecommendationService recommendationService;

    // AC1: Returns HTTP 200
    @Test
    void getRecommendations_returns200() throws Exception {
        when(recommendationService.getRecommendations())
            .thenReturn(List.of());

        mockMvc.perform(get("/api/dashboard/recommendations"))
            .andExpect(status().isOk());
    }

    // AC2: Returns valid JSON array with expected fields
    @Test
    void getRecommendations_returnsValidJsonArray() throws Exception {
        List<Map<String, Object>> mockRecommendations = List.of(
            Map.of(
                "id", "rec-001",
                "title", "Stop idle instances",
                "description", "Terminate unused EC2 instances",
                "potentialSavings", 150.00,
                "priority", "HIGH"
            )
        );
        when(recommendationService.getRecommendations())
            .thenReturn(mockRecommendations);

        mockMvc.perform(get("/api/dashboard/recommendations"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$[0].id").value("rec-001"))
            .andExpect(jsonPath("$[0].title").value("Stop idle instances"))
            .andExpect(jsonPath("$[0].potentialSavings").value(150.00))
            .andExpect(jsonPath("$[0].priority").value("HIGH"));
    }

    // AC4: Handles empty list gracefully
    @Test
    void getRecommendations_emptyList_returnsEmptyArray() throws Exception {
        when(recommendationService.getRecommendations())
            .thenReturn(List.of());

        mockMvc.perform(get("/api/dashboard/recommendations"))
            .andExpect(status().isOk())
            .andExpect(content().json("[]"));
    }

    // AC5: Handles null gracefully
    @Test
    void getRecommendations_nullList_returnsEmptyArray() throws Exception {
        when(recommendationService.getRecommendations())
            .thenReturn(null);

        mockMvc.perform(get("/api/dashboard/recommendations"))
            .andExpect(status().isOk())
            .andExpect(content().json("[]"));
    }

    // AC6: No security-sensitive fields exposed
    @Test
    void getRecommendations_noSensitiveDataInResponse() throws Exception {
        // Assume service returns data with sensitive fields
        when(recommendationService.getRecommendations())
            .thenReturn(List.of(Map.of("internalCost", 500.0, "apiKey", "secret")));

        mockMvc.perform(get("/api/dashboard/recommendations"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$[0]").doesNotHaveJsonPath())
            .andExpect(content().not(containsString("internalCost")))
            .andExpect(content().not(containsString("apiKey")));
    }
}
