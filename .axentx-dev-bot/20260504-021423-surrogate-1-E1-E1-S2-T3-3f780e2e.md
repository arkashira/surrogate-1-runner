// src/validation_framework/filtering.ts
/**
 * Filtering utilities for benchmark data.
 *
 * The benchmark data is an array of objects where each object represents a startup
 * and contains numeric metrics such as marketSize, teamExperience, funding, etc.
 *
 * The `filterBenchmarks` function allows callers to filter this array by providing
 * a set of filter criteria. Each criterion can be optional and has a minimum and maximum value.
 * Additionally, users can provide custom predicates to filter the data.
 *
 * Example usage:
 * ```ts
 * const filtered = filterBenchmarks(benchmarks, {
 *   marketSize: { min: 1_000_000 },
 *   teamExperience: { min: 5 },
 *   funding: { max: 5_000_000 },
 *   customPredicate: (b) => b.someCustomProperty === 'someValue'
 * });
 * ```
 */

export type Benchmark = Record<string, any>;

export interface FilterCriteria {
  [key: string]: {
    min?: number;
    max?: number;
  };
}

export type Predicate<T> = (item: T) => boolean;

/**
 * Filters an array of benchmarks using the provided filter criteria and custom predicates.
 *
 * @param data - Array of benchmark objects to filter.
 * @param criteria - Filter criteria object.
 * @param predicates - Array of custom predicate functions. All predicates must return true for an item to be included.
 * @returns A new array containing only the benchmarks that satisfy all filter criteria and custom predicates.
 */
export function filterBenchmarks<T extends Benchmark>(
  data: T[],
  criteria: FilterCriteria = {},
  predicates: Predicate<T>[] = []
): T[] {
  if (!Array.isArray(data)) {
    throw new TypeError('data must be an array');
  }

  const filteredData = data.filter((item) => {
    for (const [key, value] of Object.entries(criteria)) {
      const metric = item[key];
      if (
        (value.min !== undefined && metric < value.min) ||
        (value.max !== undefined && metric > value.max)
      ) {
        return false;
      }
    }

    for (const pred of predicates) {
      if (typeof pred !== 'function' || !pred(item)) {
        return false;
      }
    }
    return true;
  });

  return filteredData;
}
