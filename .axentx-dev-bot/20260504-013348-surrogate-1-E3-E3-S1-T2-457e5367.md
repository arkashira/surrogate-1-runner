# /opt/axentx/surrogate-1/routes/dashboard.py
from flask import Blueprint, render_template, redirect, url_for
from flask_login import current_user

dashboard_bp = Blueprint('dashboard', __name__)

@dashboard_bp.route('/dashboard/locks')
def locks_dashboard():
    if not current_user.is_authenticated:
        return redirect(url_for('auth.login'))
    return render_template('locks_dashboard.html')
```

        <!-- Locks will be displayed here -->
    </div>
    <script>
        // WebSocket connection will be established here
        const socket = new WebSocket('ws://' + window.location.host + '/ws/locks');

        socket.onmessage = function(event) {
            const data = JSON.parse(event.data);
            // Update the locks container with the received data
            document.getElementById('locks-container').innerHTML = data.map(lock => `
                <div class="lock">
                    <p>Document: ${lock.document}</p>
                    <p>Section: ${lock.section}</p>
                    <p>Owner: ${lock.owner}</p>
                    <p>TTL: ${lock.ttl}</p>
                </div>
            `).join('');
        };
    </script>
</body>
</html>
