from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Dict, List, Optional, Set
from uuid import uuid4


class RuleKind(str, Enum):
    MAX_NOTIONAL = "max_notional"
    MAX_ORDER_SIZE = "max_order_size"
    BLOCK_COUNTRY = "block_country"
    BLOCK_CURRENCY = "block_currency"


@dataclass(frozen=True)
class RiskRule:
    """
    Per-pipeline risk rule.
    """
    pipeline_id: str
    kind: RuleKind
    # For limit-type rules
    limit_value: Optional[float] = None
    # For block-type rules
    blocked_values: Set[str] = field(default_factory=set)
    enabled: bool = True
    metadata: Dict[str, str] = field(default_factory=dict)

    def validate(self) -> None:
        if self.kind in (RuleKind.MAX_NOTIONAL, RuleKind.MAX_ORDER_SIZE):
            if self.limit_value is None or self.limit_value <= 0:
                raise ValueError(f"{self.kind} requires limit_value > 0")
        if self.kind in (RuleKind.BLOCK_COUNTRY, RuleKind.BLOCK_CURRENCY):
            if not self.blocked_values:
                raise ValueError(f"{self.kind} requires blocked_values")


@dataclass(frozen=True)
class RiskViolation:
    pipeline_id: str
    rule_kind: RuleKind
    rule_limit_value: Optional[float]
    actual_value: Optional[float]
    blocked_value: Optional[str]
    message: str
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))


@dataclass(frozen=True)
class Rejection:
    event_id: str
    pipeline_id: str
    rule_violated: RuleKind
    details: RiskViolation
    reason: str
    timestamp: datetime = field(default_factory=lambda: datetime.now(timezone.utc))

    def to_dict(self) -> Dict:
        return {
            "event_id": self.event_id,
            "pipeline_id": self.pipeline_id,
            "rule_violated": self.rule_violated,
            "reason": self.reason,
            "details": {
                "rule_kind": self.details.rule_kind,
                "rule_limit_value": self.details.rule_limit_value,
                "actual_value": self.details.actual_value,
                "blocked_value": self.details.blocked_value,
                "message": self.details.message,
                "timestamp": self.details.timestamp.isoformat(),
            },
            "timestamp": self.timestamp.isoformat(),
        }
```
