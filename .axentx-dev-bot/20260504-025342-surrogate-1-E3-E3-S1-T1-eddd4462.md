  - `userId` (string)  
  - `beforeValue` (string/number)  
  - `afterValue` (string/number)

// src/api/audit.test.js
const request = require('supertest');
const app = require('../app'); // Express instance
const db = require('../db');   // Mockable DB module

describe('GET /api/audit', () => {
  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('returns 401 if no auth token', async () => {
    const res = await request(app).get('/api/audit?cellId=123');
    expect(res.status).toBe(401);
  });

  test('returns 400 if cellId missing', async () => {
    const res = await request(app)
      .get('/api/audit')
      .set('Authorization', 'Bearer valid-token');
    expect(res.status).toBe(400);
    expect(res.body.error).toMatch(/cellId required/);
  });

  test('returns 400 if cellId invalid format', async () => {
    const res = await request(app)
      .get('/api/audit?cellId=invalid')
      .set('Authorization', 'Bearer valid-token');
    expect(res.status).toBe(400);
  });

  test('returns audit records sorted by timestamp', async () => {
    const fakeRecords = [
      { timestamp: '2024-05-01T10:00:00Z', userId: 'u1', beforeValue: 10, afterValue: 12 },
      { timestamp: '2024-05-02T11:00:00Z', userId: 'u2', beforeValue: 12, afterValue: 15 }
    ];
    db.query.mockResolvedValue(fakeRecords);

    const res = await request(app)
      .get('/api/audit?cellId=123')
      .set('Authorization', 'Bearer valid-token');

    expect(res.status).toBe(200);
    expect(res.body).toEqual(fakeRecords);
    expect(db.query).toHaveBeenCalledWith(
      expect.stringContaining('SELECT'),
      ['123']
    );
  });

  test('handles DB errors with 500', async () => {
    db.query.mockRejectedValue(new Error('DB failure'));

    const res = await request(app)
      .get('/api/audit?cellId=123')
      .set('Authorization', 'Bearer valid-token');

    expect(res.status).toBe(500);
    expect(res.body.error).toMatch(/internal server error/);
  });
});
