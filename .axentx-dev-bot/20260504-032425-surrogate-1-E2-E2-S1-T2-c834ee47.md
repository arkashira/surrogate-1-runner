//! Integration tests for the clipboard “AX → CGEvent Cmd+V” fallback.
//!
//! The public function under test is `clipboard::paste_with_fallback()`.
//! It should:
//!   1. Try the native AX paste first.
//!   2. If that fails, synthesize a Cmd+V keystroke via CGEvent.
//!   3. Log a message that contains the substring `"CGEvent Cmd+V"`.
//!   4. Return `Ok(())` when the fallback succeeds.
//!
//! The tests below verify each of those contracts and also measure the
//! reliability of the CGEvent path.

use std::sync::Once;
use log::{Level, Record};
use rand::{thread_rng, Rng};

static INIT_LOGGER: Once = Once::new();

/// Initialise a logger **once** for the whole test suite.  The logger
/// writes to a thread‑local vector so we can later inspect the messages.
fn init_test_logger() {
    INIT_LOGGER.call_once(|| {
        // `is_test(true)` makes env_logger quiet unless a test explicitly
        // enables RUST_LOG.
        let _ = env_logger::builder()
            .is_test(true)
            .filter_level(Level::Info.to_level_filter())
            .try_init();
    });
}

/// Capture all log records emitted while `f` runs and return them as a `Vec<String>`.

    use std::cell::RefCell;

    thread_local! {
        static LOGS: RefCell<Vec<String>> = RefCell::new(Vec::new());
    }

    // A tiny logger that pushes formatted messages into the thread‑local vec.
    struct TestLogger;
    impl log::Log for TestLogger {
        fn enabled(&self, metadata: &log::Metadata) -> bool {
            metadata.level() <= log::Level::Info
        }
        fn log(&self, record: &Record) {
            if self.enabled(record.metadata()) {
                LOGS.with(|c| c.borrow_mut().push(format!("{}", record.args())));
            }
        }
        fn flush(&self) {}
    }

    // Swap the global logger temporarily.
    let previous = log::logger();
    log::set_logger(&TestLogger).unwrap();
    log::set_max_level(log::LevelFilter::Info);

    f();

    // Restore the original logger.
    log::set_logger(Box::leak(Box::new(previous))).unwrap();

    LOGS.with(|c| c.borrow().clone())
}
