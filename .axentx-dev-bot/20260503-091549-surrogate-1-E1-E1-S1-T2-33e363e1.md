**Implementation:**

#### request.ts (lines 12-20)

```diff
// Before
import { Request } from './request.entity';

export const requestValidator = (req: Request) => {
  return true; // No validation
};

// After
import { Request } from './request.entity';

export const requestValidator = (req: Request) => {
  if (!req.title || !req.type || !req.owner) {
    return { error: 'Missing required fields' };
  }
  if (typeof req.slaTarget !== 'number' || req.slaTarget < 0) {
    return { error: 'Invalid SLA target' };
  }
  return true;
};
```

#### requests.ts (lines 34-42)

```diff

export const createRequest = async (req: Request) => {
  const newRequest = await db.createRequest(req);
  return { id: newRequest.id, publicUrl: newRequest.publicUrl };
};

// After
import { Request } from './request.entity';

export const createRequest = async (req: Request) => {
  try {
    const validatedReq = requestValidator(req);
    if (validatedReq.error) {
      throw new Error(validatedReq.error);
    }
    const newRequest = await db.createRequest(req);
    return {
      id: newRequest.id,
      publicUrl: newRequest.publicUrl,
      created_at: newRequest.createdAt,
      current_status: newRequest.status,
    };
  } catch (error) {
    throw error;
  }
};

export const getRequests = async (team: string) => {
  const requests = await db.getRequests(team);
  return requests.map((request) => ({
    id: request.id,
    publicUrl: request.publicUrl,
    created_at: request.createdAt,
    current_status: request.status,
  }));
};
```
