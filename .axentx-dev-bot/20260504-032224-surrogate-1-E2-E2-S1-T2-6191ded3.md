"""
Thread‑safe, file‑backed JSON storage for user feedback.

Features taken from the two drafts and improved:

* **Configurable location** – defaults to a ``data/feedback.json`` sibling of this file,
  but can be overridden with the ``FEEDBACK_PATH`` environment variable.
* **Atomic writes** – a temporary ``*.tmp`` file is written then renamed.
* **Robust loading** – missing or corrupted files are treated as an empty list
  (never raise for the caller).
* **Strict validation** – rating is clamped to the 1‑5 range, ``user_id`` must be a
  non‑empty string, and ``comments`` is optional.
* **ISO‑8601 UTC timestamps** – human‑readable and sortable.
* **Thread‑level lock** – protects concurrent reads/writes inside a single process.
* **Clear public API** – ``save_feedback`` and ``get_all_feedback``.
"""

import json
import os
import threading
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, List, Optional

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #
# Allow the caller (e.g. a Docker entrypoint or test harness) to point the
# store at a different location without touching the source.
_ENV_VAR = "FEEDBACK_PATH"

_default_dir = Path(__file__).resolve().parent / "data"
_FEEDBACK_FILE = Path(os.getenv(_ENV_VAR, _default_dir / "feedback.json"))

# Ensure the directory hierarchy exists up‑front – this is cheap and avoids
# race conditions later when the first write happens.
_FEEDBACK_FILE.parent.mkdir(parents=True, exist_ok=True)

# --------------------------------------------------------------------------- #
# Concurrency primitive
# --------------------------------------------------------------------------- #
_lock = threading.Lock()

# --------------------------------------------------------------------------- #
# Internal helpers
# --------------------------------------------------------------------------- #
def _ensure_file_exists() -> None:
    """Create an empty JSON list file if it does not already exist."""
    if not _FEEDBACK_FILE.exists():
        _FEEDBACK_FILE.write_text("[]", encoding="utf-8")


def _load_all() -> List[Dict[str, Any]]:
    """Read the whole feedback list from disk.

    Returns an empty list if the file is missing, unreadable or not a JSON list.
    """
    if not _FEEDBACK_FILE.exists():
        return []

    try:
        with _FEEDBACK_FILE.open("r", encoding="utf-8") as f:
            data = json.load(f)
            return data if isinstance(data, list) else []
    except (json.JSONDecodeError, OSError):
        # Corrupted or inaccessible file – start fresh rather than crash.
        return []


def _persist_all(entries: List[Dict[str, Any]]) -> None:
    """Write *entries* atomically to disk."""
    tmp_path = _FEEDBACK_FILE.with_suffix(".tmp")
    with tmp_path.open("w", encoding="utf-8") as f:
        json.dump(entries, f, ensure_ascii=False, indent=2)
    tmp_path.replace(_FEEDBACK_FILE)


def _now_iso8601() -> str:
    """Current UTC time in ISO‑8601 format (e.g. ``2026-05-14T12:34:56Z``)."""
    return datetime.now(timezone.utc).replace(microsecond=0).isoformat().replace("+00:00", "Z")


def _validate_rating(rating: int) -> int:
    """Clamp rating to the allowed 1‑5 range."""
    return max(1, min(5, rating))


def _validate_user_id(user_id: str) -> str:
    if not isinstance(user_id, str) or not user_id.strip():
        raise ValueError("user_id must be a non‑empty string")
    return user_id.strip()
