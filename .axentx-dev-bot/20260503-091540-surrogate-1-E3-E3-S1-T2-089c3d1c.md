import { fetch } from 'node-fetch';

class SlackWebhook {
  private url: string;

  constructor(url: string) {
    this.url = url;
  }

  async postMessage(message: string) {
    const options = {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ text: message }),
    };

    try {
      const response = await fetch(this.url, options);
      if (!response.ok) {
        throw new Error(`Error posting message to Slack webhook: ${response.status}`);
      }
    } catch (error) {
      console.error(error);
    }
  }
}
```
