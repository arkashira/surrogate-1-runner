from functools import wraps
from flask import request, g, abort
from typing import Callable

def assert_team_access(team_id: str) -> None:
    """
    Ensure the current user (g.user) is authorized to access the given team_id.
    Raises 403 if not authorized.
    """
    user = getattr(g, "user", None)
    if not user:
        abort(401, "Authentication required")

    user_teams = getattr(user, "team_ids", [])
    # Allow if user is member of the requested team
    if team_id not in user_teams:
        abort(403, "Access to this team is not permitted")

def require_team_scope(f: Callable) -> Callable:
    """
    Decorator for routes with <team_id> parameter.
    Validates team access before proceeding.
    """
    @wraps(f)
    def decorated(*args, **kwargs):
        team_id = kwargs.get("team_id")
        if not team_id:
            abort(400, "team_id is required in route")
        assert_team_access(str(team_id))
        return f(*args, **kwargs)
    return decorated
```
