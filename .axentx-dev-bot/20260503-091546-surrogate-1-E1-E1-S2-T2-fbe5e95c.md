// src/models/requestTimeline.ts

export type TimelineEventType =
  | 'created'
  | 'updated'
  | 'status_changed'
  | 'comment_added'
  | 'sla_breach'
  | 'resolved';

export interface RequestTimelineEntry {
  id: string;          // internal only
  requestId: string;   // internal only
  publicId: string;    // safe lookup key
  eventType: TimelineEventType;
  label: string;       // human-friendly
  timestamp: string;   // ISO 8601
  actor?: { name?: string };
  metadata?: Record<string, unknown>;
}

const timelineStore = new Map<string, RequestTimelineEntry[]>();

export function getTimelineByPublicId(publicId: string): Omit<RequestTimelineEntry, 'id' | 'requestId'>[] {
  return (timelineStore.get(publicId) || []).map(({ id, requestId, ...safe }) => safe);
}

export function addTimelineEntry(entry: RequestTimelineEntry): void {
  const existing = timelineStore.get(entry.publicId) || [];
  existing.push(entry);
  existing.sort((a, b) => new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime());
  timelineStore.set(entry.publicId, existing);
}

// Seed safe dev data
if (timelineStore.size === 0) {
  const now = Date.now();
  addTimelineEntry({
    id: 'int-001',
    requestId: 'req-001',
    publicId: 'pub-abc123',
    eventType: 'created',
    label: 'Request created',
