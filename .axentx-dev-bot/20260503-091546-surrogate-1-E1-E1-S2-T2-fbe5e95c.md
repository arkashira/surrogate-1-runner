// /opt/axentx/surrogate-1/src/models/requestTimeline.ts
import { Entity, PrimaryColumn, Column, Index, CreateDateColumn } from 'typeorm';

export enum TimelineEventType {
  CREATED = 'created',
  STATUS_UPDATED = 'status_updated',
  COMMENT_ADDED = 'comment_added',
  SLA_BREACH_WARNING = 'sla_breach_warning',
  RESOLVED = 'resolved',
}

@Entity('request_timeline')
export class RequestTimeline {
  @PrimaryColumn('uuid')
  id!: string;

  @Column('uuid')
  requestId!: string; // internal request id (not exposed publicly)

  @Column('varchar', { length: 64 })
  @Index('idx_request_timeline_public_id')
  publicId!: string; // public identifier used in URLs

  @Column('enum', { enum: TimelineEventType })
  eventType!: TimelineEventType;

  @Column('varchar', { length: 32 })
  status!: string; // machine status (e.g. 'open', 'in_progress', 'closed')

  @Column('varchar', { length: 64 })
  statusLabel!: string; // human-friendly label (e.g. 'In Progress')

  @Column('jsonb', { nullable: true })
  details?: Record<string, any>;

  @CreateDateColumn()
  createdAt!: Date;
}
```
