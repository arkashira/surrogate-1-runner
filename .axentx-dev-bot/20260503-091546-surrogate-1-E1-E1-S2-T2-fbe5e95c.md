import { z } from 'zod';

export const RequestTimelineEventType = [
  'CREATED',
  'STATUS_UPDATED',
  'COMMENT_ADDED',
  'SLA_BREACH_WARNED',
  'RESOLVED',
] as const;
export type RequestTimelineEventType = (typeof RequestTimelineEventType)[number];

export const RequestTimelineSchema = z.object({
  id: z.string().cuid(),
  requestId: z.string().cuid(),
  publicId: z.string(),
  eventType: z.enum(RequestTimelineEventType),
  status: z.string(),
  label: z.string(),
  metadata: z.record(z.any()).optional(),
  createdAt: z.date(),
});
export type RequestTimeline = z.infer<typeof RequestTimelineSchema>;

// Public-safe projection (exclude internal IDs)
export const PublicRequestTimelineEntrySchema = z.object({
  eventType: z.enum(RequestTimelineEventType),
  status: z.string(),
  label: z.string(),
  metadata: z.record(z.any()).optional(),
  createdAt: z.date(),
});
export type PublicRequestTimelineEntry = z.infer<typeof PublicRequestTimelineEntrySchema>;

## 3) Service: requestTimelineService.ts

```ts
// /opt/axentx/surrogate-1/src/services/requestTimelineService.ts
import { prisma } from '../db';
import { PublicRequestTimelineEntrySchema } from '../models/requestTimeline';
import type { PublicRequestTimelineEntry } from '../models/requestTimeline';

const STATUS_LABEL_MAP: Record<string, string> = {
  OPEN: 'Open',
  IN_PROGRESS: 'In Progress',
  PENDING: 'Pending',
  RESOLVED: 'Resolved',
  CLOSED: 'Closed',
};

export async function getPublicTimelineByPublicId(publicId: string) {
  const request = await prisma.request.findUnique({
    where: { publicId },
    select: {
      title: true,
      status: true,
      slaHours: true,
      updatedAt: true,
    },
  });

  if (!request) {
    return null;
  }

  const events = await prisma.requestTimeline.findMany({
    where: { publicId },
    orderBy: { createdAt: 'asc' },
  });

  const timeline: PublicRequestTimelineEntry[] = events
    .map((e) =>
      PublicRequestTimelineEntrySchema.parse({
        eventType: e.eventType,
        status: e.status,
        label: e.label,
        metadata: e.metadata,
        createdAt: e.createdAt,
      })
    )
    .filter(Boolean);

  return {
    title: request.title,
    status: request.status,
    statusLabel: STATUS_LABEL_MAP[request.status] || request.status,
    sla: request.slaHours ? `${request.slaHours}h` : null,
    timeline,
    lastUpdated: request.updatedAt,
  };
}
```
