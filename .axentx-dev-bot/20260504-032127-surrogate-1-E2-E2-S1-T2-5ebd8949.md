	Provider    string    `json:"provider"`
	Payload     string    `json:"payload"`
}

type TaskStatus struct {
	TaskID     string    `db:"task_id"`
	Status     string    `db:"status"`
	LastUpdate time.Time `db:"last_update"`
}

// ParseCIPayload parses the CI payload and updates task status
func ParseCIPayload(payload string) error {
	var ciStatus CIStatus
	if err := json.Unmarshal([]byte(payload), &ciStatus); err != nil {
		return fmt.Errorf("failed to unmarshal payload: %v", err)
	}

	// Extract status based on provider
	status, err := extractStatus(ciStatus.Provider, ciStatus.Payload)
	if err != nil {
		return fmt.Errorf("failed to extract status: %v", err)
	}

	// Update task status in database
	db, err := connectDB()
	if err != nil {
		return fmt.Errorf("failed to connect to database: %v", err)
	}
	defer db.Close()

	_, err = db.Exec(
		"UPDATE tasks SET status = ?, last_update = ? WHERE task_id = ?",
		status,
		time.Now(),
		ciStatus.JobID,
	)
	if err != nil {
		return fmt.Errorf("failed to update task status: %v", err)
	}

	// Notify assignee if build failed
	if status == "failure" {
		if err := notifyAssignee(ciStatus.JobID); err != nil {

}
