import os
import logging
import time
from typing import Dict, Any, Optional
import requests

logger = logging.getLogger(__name__)

_SLACK_WEBHOOK_URL = os.getenv("SLACK_WEBHOOK_URL")
_SLACK_CHANNEL = os.getenv("SLACK_CHANNEL", "#status-updates")
_EMAIL_TO_SLACK_CACHE: Dict[str, Optional[str]] = {}


def _email_to_slack_mention(email: Optional[str]) -> Optional[str]:
    """Best-effort email -> Slack handle mapping.
    Returns '@username' if plausible, else None.
    """
    if not email:
        return None
    if email in _EMAIL_TO_SLACK_CACHE:
        return _EMAIL_TO_SLACK_CACHE[email]

    try:
        local = email.split("@")[0].lower().strip()
        if local and len(local) <= 32 and local.replace(".", "").replace("_", "").isalnum():
            mention = f"@{local}"
            _EMAIL_TO_SLACK_CACHE[email] = mention
            return mention
    except Exception:
        pass
    _EMAIL_TO_SLACK_CACHE[email] = None
    return None


def _post_with_retry(payload: Dict[str, Any], max_attempts: int = 3) -> bool:
    """POST JSON to Slack webhook with exponential backoff and jitter."""
    if not _SLACK_WEBHOOK_URL:
        logger.warning("SLACK_WEBHOOK_URL not configured; skipping Slack notification.")
        return False

    backoff = 1
    for attempt in range(1, max_attempts + 1):
        try:
            resp = requests.post(_SLACK_WEBHOOK_URL, json=payload, timeout=10)
            if resp.status_code == 200:
                logger.info("Slack notification sent (attempt %s)", attempt)
                return True
            logger.warning(
                "Slack webhook returned %s (attempt %s): %s",
                resp.status_code,
                attempt,
                resp.text[:200],
            )
        except Exception as exc:
            logger.warning("Slack post failed (attempt %s): %s", attempt, exc)

        if attempt < max_attempts:
            jitter = 0.1 * backoff * (2 ** (attempt - 1))
            sleep_time = backoff + jitter
            logger.info("Retrying Slack notification in %.1fs", sleep_time)
            time.sleep(sleep_time)
            backoff *= 2

    logger.error("Slack notification failed after %s attempts", max_attempts)
    return False


class SlackNotifier:
    @staticmethod
    def notify_status_change(
        *,
        request_title: str,
        new_status: str,
        public_url: str,
        requester_email: Optional[str] = None,
        sla_timer: Optional[str] = None,
        request_id: Optional[str] = None,
    ) -> bool:
        """Post status change to Slack (non-blocking best-effort)."""
        mention = _email_to_slack_mention(requester_email)
        header = f"{mention} " if mention else ""
        status_emoji = {
            "open": ":large_blue_circle:",
            "in_progress": ":large_yellow_circle:",
            "resolved": ":large_green_circle:",
            "closed": ":white_check_mark:",
            "cancelled": ":x:",
        }.get(new_status.lower(), ":black_circle:")

        text = f"{header}Request *{request_title}* {status_emoji} status changed to *{new_status}*."
        if sla_timer:
            text += f"\n⏱ SLA: {sla_timer}"
        text += f"\n🔗 <{public_url}|View request>"
        if request_id:
            text += f" (ID: `{request_id}`)"

        payload = {"channel": _SLACK_CHANNEL, "text": text, "mrkdwn": True}
        ok = _post_with_retry(payload, max_attempts=3)
        if not ok:
            logger.error(
                "Failed to notify Slack for request=%s status=%s",
                request_id or request_title,
                new_status,
            )
        return ok
`
