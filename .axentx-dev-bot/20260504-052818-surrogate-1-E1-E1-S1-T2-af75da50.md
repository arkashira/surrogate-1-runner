from prometheus_client import start_http_server, Gauge
import time

class WorkerMetrics:
    def __init__(self, port=8000):
        self.port = port
        self.worker_status = Gauge('worker_status', 'Status of the worker', ['worker_id'])
        self.dataset_size = Gauge('dataset_size', 'Size of the dataset being processed', ['worker_id'])

    def start_metrics_server(self):
        start_http_server(self.port)

    def update_worker_status(self, worker_id, status):
        self.worker_status.labels(worker_id=worker_id).set(status)

    def update_dataset_size(self, worker_id, size):
        self.dataset_size.labels(worker_id=worker_id).set(size)

# Example usage
if __name__ == "__main__":
    metrics = WorkerMetrics()
    metrics.start_metrics_server()
    while True:
        # Update metrics here
        time.sleep(1)

- **Dataset Size Gauge** – The `dataset_size_bytes` gauge matches the worker’s reported dataset size in bytes.

- **Prometheus Scrape** – A Prometheus client can scrape the endpoint without errors and receives all metrics within 5 s.  
- **Error Handling** – Invalid worker IDs (empty strings, `None`) or negative dataset sizes raise a `ValueError` and do not corrupt the metrics state.

# test_metrics.py
import pytest
from src.metrics.prometheus_metrics import WorkerMetrics, Worker
from src.metrics.exceptions import MaxWorkersExceeded, MinWorkersViolation

@pytest.fixture
def metrics():
    return WorkerMetrics(min_workers=1, max_workers=5)

def test_register_worker(metrics):
    w = Worker(id="w1", dataset_size=1024)
    metrics.register_worker(w)
    assert metrics.worker_count == 1
    assert metrics.get_metric("worker_status", {"worker_id": "w1"}) == 1

def test_unregister_worker(metrics):
    w = Worker(id="w2", dataset_size=2048)
    metrics.register_worker(w)
    metrics.unregister_worker("w2")
    assert metrics.worker_count == 0
    with pytest.raises(KeyError):
        metrics.get_metric("worker_status", {"worker_id": "w2"})

def test_max_workers_exceeded(metrics):
    for i in range(metrics.max_workers):
        metrics.register_worker(Worker(id=f"w{i}", dataset_size=100))
    with pytest.raises(MaxWorkersExceeded):
        metrics.register_worker(Worker(id="overflow", dataset_size=50))

def test_min_workers_violation(metrics):
    w = Worker(id="w1", dataset_size=100)
    metrics.register_worker(w)
    metrics.unregister_worker("w1")
    with pytest.raises(MinWorkersViolation):
        metrics.unregister_worker("w1")  # already below min

def test_invalid_worker_id(metrics):
    with pytest.raises(ValueError):
        metrics.register_worker(Worker(id="", dataset_size=100))
    with pytest.raises(ValueError):
        metrics.register_worker(Worker(id=None, dataset_size=100))

def test_negative_dataset_size(metrics):
    with pytest.raises(ValueError):
        metrics.register_worker(Worker(id="w1", dataset_size=-10))

def test_status_update(metrics):
    w = Worker(id="w1", dataset_size=500)
    metrics.register_worker(w)
    w.status = "stopped"
    metrics.update_metrics()
    assert metrics.get_metric("worker_status", {"worker_id": "w1"}) == 0

def test_dataset_size_update(metrics):
    w = Worker(id="w1", dataset_size=500)
    metrics.register_worker(w)
    w.dataset_size = 1500
    metrics.update_metrics()
    assert metrics.get_metric("dataset_size_bytes", {"worker_id": "w1"}) == 1500
