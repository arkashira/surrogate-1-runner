from datetime import datetime, timedelta
from typing import Dict

def get_wau_requests_per_team(requests: Dict) -> Dict:
    """
    Calculate the weekly active users (WAU) requests per team.

    Args:
    requests (Dict): A dictionary of requests with team IDs and timestamps.

    Returns:
    Dict: A dictionary with team IDs as keys and the number of WAU requests as values.
    """
    wau_requests_per_team = {}
    now = datetime.now()
    one_week_ago = now - timedelta(days=7)

    for request in requests:
        team_id = request['team_id']
        timestamp = datetime.strptime(request['timestamp'], '%Y-%m-%d %H:%M:%S')

        if team_id not in wau_requests_per_team:
            wau_requests_per_team[team_id] = 0

        if timestamp >= one_week_ago:
            wau_requests_per_team[team_id] += 1

    return wau_requests_per_team

def get_debug_requests(requests: Dict) -> Dict:
    """
    Get the debug requests.

    Args:
    requests (Dict): A dictionary of requests with team IDs and timestamps.

    Returns:
    Dict: A dictionary with the debug requests.
    """
    debug_requests = {}
    for request in requests:
        team_id = request['team_id']
        timestamp = datetime.strptime(request['timestamp'], '%Y-%m-%d %H:%M:%S')
        if timestamp >= datetime.now() - timedelta(days=1):
            debug_requests[team_id] = request
    return debug_requests
```
