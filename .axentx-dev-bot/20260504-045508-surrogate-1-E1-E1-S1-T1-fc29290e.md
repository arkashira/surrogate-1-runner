import PropTypes from "prop-types";

 * UI for filtering a catalogue of items.
 *
 * Props
 * -----
 * categories:   [{ id: string, name: string }]
 * specs:        [{ id: string, name: string, type: "radio"|"checkbox",
 *                options: [{ id: string, label: string }] }]
 * priceRange:   { min: number, max: number }   // absolute bounds
 * onChange:     (filter: FilterState) => void   // called *debounced*
 *
 * FilterState shape
 * -----------------
 * {
 *   categoryIds: string[],
 *   price: { min: number, max: number },

 * }
 */

  categories,
  specs,
  priceRange,
  onChange,

  // ---------- internal UI state ----------

  // We debounce at 200 ms – fast enough for UI feel, slow enough to avoid
  // a cascade of renders while the user types a number.
  const emitChange = useCallback(
    debounce((state) => onChange(state), 200),
