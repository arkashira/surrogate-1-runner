// src/models/request.ts
import { v4 as uuidv4 } from 'uuid';

export type RequestType = 'incident' | 'task' | 'change' | 'question';
export type RequestStatus = 'open' | 'in_progress' | 'resolved' | 'closed';

export interface SlaTarget {
  hours: number;
}

export interface RequestRecord {
  id: string;
  teamId: string;
  title: string;
  type: RequestType;
  owner: string; // user id or email
  slaTarget: SlaTarget;
  publicUrl: string;
  status: RequestStatus;
  createdAt: Date;
  updatedAt: Date;
}

const store: RequestRecord[] = [];

export function createRequest(input: {
  teamId: string;
  title: string;
  type: RequestType;
  owner: string;
  slaTarget: SlaTarget;
  publicUrl: string;
}): RequestRecord {
  const now = new Date();
  const record: RequestRecord = {
    id: uuidv4(),
    teamId: input.teamId,
    title: input.title,
    type: input.type,
    owner: input.owner,
    slaTarget: input.slaTarget,
    publicUrl: input.publicUrl,
    status: 'open',
    createdAt: now,
    updatedAt: now,
  };
  store.push(record);
  return record;
}

export function listRequestsByTeam(teamId: string): RequestRecord[] {
  return store
    .filter((r) => r.teamId === teamId)
    .sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime());
}

export function findById(id: string): RequestRecord | undefined {
  return store.find((r) => r.id === id);
}
