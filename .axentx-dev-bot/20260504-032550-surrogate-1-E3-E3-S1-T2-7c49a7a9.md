### `sandbox_cli.py`

```python
#!/usr/bin/env python3
"""
sandbox_cli.py – A lightweight sandbox manager.

Features
--------
* Create, list, inspect, delete, and clean up sandbox environments.
* Each sandbox lives in its own UUID directory under a configurable root.
* Metadata (created_at, last_activity, status, api_endpoint) is stored in
  `metadata.json` inside the sandbox.
* A minimal HTTP server exposes sandbox metadata at `/sandbox/<env_id>/api`.
* Automatic cleanup of environments that have been idle for a configurable
  number of hours (default 24h).

Usage
-----
    sandbox_cli.py <subcommand> [options]

Subcommands
-----------
create    – Create a new sandbox.
list      – List all sandboxes.
inspect   – Show details of a sandbox.
delete    – Delete a sandbox (interactive unless --force).
cleanup   – Remove sandboxes idle for > N hours.
api       – Start the HTTP API server.

Environment Variables
---------------------
SANDBOX_ROOT   – Root directory for sandboxes (default: /tmp/sandboxes)
SANDBOX_PORT   – Port for the API server (default: 8080)
"""

import argparse
import json
import os
import shutil
import sys
import uuid
from datetime import datetime, timedelta
from pathlib import Path
from typing import Dict, Optional

# --------------------------------------------------------------------------- #
# Configuration (can be overridden by env vars)
# --------------------------------------------------------------------------- #
SANDBOX_ROOT = Path(os.getenv("SANDBOX_ROOT", "/tmp/sandboxes")).resolve()
API_PORT = int(os.getenv("SANDBOX_PORT", "8080"))
CLEANUP_THRESHOLD_HOURS = 24  # default threshold for stale sandboxes


# --------------------------------------------------------------------------- #
# Utility helpers
# --------------------------------------------------------------------------- #
def ensure_root() -> Path:
    SANDBOX_ROOT.mkdir(parents=True, exist_ok=True)
    return SANDBOX_ROOT


def env_path(env_id: str) -> Path:
    return SANDBOX_ROOT / env_id


def meta_path(env_id: str) -> Path:
    return env_path(env_id) / "metadata.json"


def load_meta(env_id: str) -> Optional[Dict]:
    p = meta_path(env_id)
    if p.exists():
        with p.open() as f:
            return json.load(f)
    return None


def save_meta(env_id: str, data: Dict) -> None:
    p = meta_path(env_id)
    with p.open("w") as f:
        json.dump(data, f, indent=2)


def update_last_activity(env_id: str) -> None:
    meta = load_meta(env_id)
    if meta:
        meta["last_activity"] = datetime.utcnow().isoformat()
        save_meta(env_id, meta)

import unittest
import json
import tempfile

from unittest.mock import patch
from subprocess import run
from datetime import datetime, timedelta

class TestSandboxCLI(unittest.TestCase):

    @patch('os.path.exists')
    def test_create_sandbox(self, mock_exists, mock_mkdir):

            mock_exists.return_value = False

    @patch('os.listdir')
    @patch('json.load')

            mock_load.return_value = {'env_id': 'sandbox1', 'created_at': datetime.now()}
            run(['sandbox_cli.py', 'list', '--root', root])
