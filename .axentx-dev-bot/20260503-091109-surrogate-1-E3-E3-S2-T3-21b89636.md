// utils/isFormElement.js
export function isFormElement(el) {
  if (!el || !(el instanceof Element)) return false;
  if (['INPUT', 'TEXTAREA', 'SELECT'].includes(el.tagName)) return true;
  if (el.isContentEditable) return true;
  return isFormElement(el.parentElement);
}

// handlers/keyboardShortcuts.js
export function handleGlobalKeydown(event, { searchInputId = '#search-input', previousFocusRef }) {
  if (event.key === '/' && !isFormElement(document.activeElement)) {
    event.preventDefault();
    const search = document.querySelector(searchInputId);
    if (search) {
      previousFocusRef.current = document.activeElement;
      search.focus();
    }
    return;
  }

  if (event.key === 'Escape') {
    const search = document.querySelector(searchInputId);
    if (search && document.activeElement === search) {
      event.preventDefault();
      const prev = previousFocusRef.current;
      search.blur();
      if (prev && typeof prev.focus === 'function') prev.focus();
    }
  }
}
