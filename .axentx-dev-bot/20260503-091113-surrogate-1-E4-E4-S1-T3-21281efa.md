    svc.upsert(uri="file:///a.pdf", title="Doc A", preview="new")

    assert second["preview"] == "new"
    assert second["accessed_at"] > first["accessed_at"]

def test_get_all_returns_descending_by_accessed_at(svc):

    items = svc.get_all()
    assert len(items) == 2
    assert items[0]["uri"] == "file:///b.pdf"  # most recent first
    assert items[1]["uri"] == "file:///a.pdf"
