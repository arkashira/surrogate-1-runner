# src/services/recent.py
from __future__ import annotations

import json
import os
import time
from pathlib import Path
from typing import Any, Dict, List, Optional
from datetime import datetime, timezone

try:
    from PyPDF2 import PdfReader
except ImportError:
    PdfReader = None  # optional preview support

RECENT_FILE = Path.home() / ".axentx" / "recent.jsonl"
MAX_ITEMS = 100
TTL_DAYS = 30


class RecentItemsService:
    def __init__(self, path: Path = RECENT_FILE) -> None:
        self.path = path
        self.path.parent.mkdir(parents=True, exist_ok=True)

    def _now_ts(self) -> float:
        return datetime.now(timezone.utc).timestamp()

    def _is_expired(self, item: Dict[str, Any]) -> bool:
        ts = item.get("added_at", 0)
        return (self._now_ts() - ts) > (TTL_DAYS * 86400)

    def add(self, uri: str, title: str, mime: str) -> None:
        # Load existing
        items = self._load_raw()
        # Remove older entry for same uri if present
        items = [it for it in items if it.get("uri") != uri]
        # Prepend new
        new_item = {
            "uri": uri,
            "title": title,
            "mime": mime,
            "added_at": self._now_ts(),
            "preview_available": self._can_preview(mime, uri),
        }
        items.insert(0, new_item)
        # Truncate
        items = items[:MAX_ITEMS]
        self._write_all(items)

    def list(self, limit: int = 20) -> List[Dict[str, Any]]:
        items = self._load_raw()
        out = []
        for it in items:
            if self._is_expired(it):
                continue
            out.append(
                {
                    "uri": it["uri"],
                    "title": it["title"],
                    "mime": it["mime"],
                    "added_at": it["added_at"],
                    "preview_available": it.get("preview_available", False),
                }
            )
            if len(out) >= limit:
                break
        return out

    def clear(self) -> None:
        if self.path.exists():
            self.path.unlink()
