### 1) Deterministic file-list cache (ingest)

- **Single non-recursive tree call per date folder** (no recursive enumeration).  
- **Project to strict schema** (`prompt`, `response`) and store as parquet in `{date}/{slug}.parquet`.

- **No extra metadata columns** in parquet that would force downstream schema reconciliation.

```python
#!/usr/bin/env python3
"""
Ingest HF dataset mirror -> project to {prompt,response}, store as parquet,
and emit file_list.json for CDN-only training.
"""
import json
import os
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import list_repo_tree, hf_hub_download

HF_REPO = os.getenv("HF_DATASET_REPO", "datasets/surrogate-1-mirror")
DATE_FOLDER = os.getenv("INGEST_DATE_FOLDER", datetime.now(timezone.utc).strftime("%Y-%m-%d"))
OUT_DIR = Path(os.getenv("OUT_DIR", "backend/batches/mirror-merged"))
FILE_LIST_PATH = OUT_DIR / DATE_FOLDER / "file_list.json"

def project_to_prompt_response(src_path, dst_path):
    """
    Implement per source format.
    Must produce parquet with at least columns: prompt (str), response (str).
    """
    import pandas as pd
    # Placeholder: adapt to real schema (jsonl, csv, etc.)
    df = pd.read_json(src_path, lines=True) if src_path.suffix == ".jsonl" else pd.read_parquet(src_path)
    # Keep only required fields; drop extras to guarantee stable schema
    df = df[["prompt", "response"]].reset_index(drop=True)
    df.to_parquet(dst_path, index=False)

def run_ingest() -> None:
    OUT_DIR.mkdir(parents=True, exist_ok=True)
    date_dir = OUT_DIR / DATE_FOLDER
    date_dir.mkdir(parents=True, exist_ok=True)

    entries = list_repo_tree(
        repo_id=HF_REPO,
        path=DATE_FOLDER,
        recursive=False,
        repo_type="dataset",
    )

    file_paths = []
    for entry in entries:
        if entry.type != "file":
            continue
        rpath = f"{DATE_FOLDER}/{entry.path}"
        local_src = hf_hub_download(
            repo_id=HF_REPO,
            filename=rpath,
            repo_type="dataset",
            cache_dir=str(date_dir / ".cache_raw"),
        )
        slug = Path(entry.path).stem
        out_parquet = date_dir / f"{slug}.parquet"
        project_to_prompt_response(Path(local_src), out_parquet)
        file_paths.append(str(out_parquet.relative_to(OUT_DIR.parent.parent)))

    FILE_LIST_PATH.parent.mkdir(parents=True, exist_ok=True)
    with open(FILE_LIST_PATH, "w") as f:
        json.dump(sorted(file_paths), f, indent=2)

    print(f"Wrote {len(file_paths)} files -> {FILE_LIST_PATH}")

if __name__ == "__main__":
    run_ingest()
```
