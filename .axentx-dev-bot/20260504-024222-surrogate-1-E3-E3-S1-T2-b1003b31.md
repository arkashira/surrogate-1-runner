"""
Correlation Engine for Surrogate‑1 post‑mortem reports.

The engine can be used in two ways:

1. **High‑level functional API** – call :func:`generate_postmortem_report`
   with a list of decision artifacts and a dict describing the incident.
   This is perfect for scripts, CI jobs or the dashboard.

2. **Object‑oriented, injectable API** – instantiate :class:`CorrelationEngine`
   with concrete ``IncidentRepository`` and ``ArtifactRepository`` objects and
   call :meth:`CorrelationEngine.correlate`.  This is ideal for production
   code where data lives in a database or external service and for unit tests
   where you supply in‑memory fakes.

Both APIs share the same core logic (timestamp normalisation, summary
generation, error handling) so the output format is identical.
"""

from __future__ import annotations

import json
from dataclasses import dataclass
from datetime import datetime
from typing import Any, Dict, List, Mapping, Protocol, Sequence, TypedDict

# --------------------------------------------------------------------------- #
# Helper – timestamp normalisation
# --------------------------------------------------------------------------- #
def _normalize_timestamp(ts: Any) -> str | None:
    """
    Convert *any* timestamp representation to an ISO‑8601 string (UTC).

    Accepted inputs:
    - ``datetime`` (naïve or aware – will be treated as UTC)
    - epoch seconds (int/float)
    - ISO‑8601 string (with or without a trailing ``Z``)
    - ``None`` – returns ``None`` (caller can decide how to render)

    If parsing fails, the original value is stringified so the downstream
    dashboard can still display something useful.
    """
    if ts is None:
        return None
    if isinstance(ts, datetime):
        # Force UTC – if tz‑aware, convert; if naïve, assume UTC
        if ts.tzinfo:
            ts = ts.astimezone(datetime.timezone.utc).replace(tzinfo=None)
        return ts.isoformat() + "Z"
    if isinstance(ts, (int, float)):
        return datetime.utcfromtimestamp(ts).isoformat() + "Z"
    if isinstance(ts, str):
        try:
            # Handles both "2024-05-13T12:00:00" and "2024-05-13T12:00:00Z"
            return datetime.fromisoformat(ts.rstrip("Z")).isoformat() + "Z"
        except ValueError:
            pass
    # Fallback – keep something readable
    return str(ts)
