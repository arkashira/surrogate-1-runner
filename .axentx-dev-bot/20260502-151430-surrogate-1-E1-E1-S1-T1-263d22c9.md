from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, List, Optional

from pydantic import BaseModel, Field, field_validator


class PipelineState(str, Enum):
    RESEARCH = "research"
    SIGNAL = "signal"
    RISK = "risk"
    EXECUTION = "execution"
    FAILED = "failed"
    COMPLETED = "completed"


class InvalidTransitionError(ValueError):
    """Raised when a pipeline state transition is invalid."""


_ALLOWED_TRANSITIONS: Dict[PipelineState, List[PipelineState]] = {
    PipelineState.RESEARCH: [PipelineState.SIGNAL, PipelineState.FAILED],
    PipelineState.SIGNAL: [PipelineState.RISK, PipelineState.FAILED],
    PipelineState.RISK: [PipelineState.EXECUTION, PipelineState.FAILED],
    PipelineState.EXECUTION: [PipelineState.COMPLETED, PipelineState.FAILED],
    PipelineState.FAILED: [],
    PipelineState.COMPLETED: [],
}


def validate_transition(current: PipelineState, next_state: PipelineState) -> None:
    if next_state not in _ALLOWED_TRANSITIONS.get(current, []):
        raise InvalidTransitionError(
            f"Invalid transition {current.value} -> {next_state.value}"
        )


class PipelineEvent(BaseModel):
    pipeline_id: str
    from_state: Optional[PipelineState]
    to_state: PipelineState
    timestamp: str  # ISO-8601 UTC
    metadata: Dict[str, Any] = Field(default_factory=dict)

    @classmethod
    def create(
        cls,
        pipeline_id: str,
        from_state: Optional[PipelineState],
        to_state: PipelineState,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> "PipelineEvent":
        return cls(
            pipeline_id=pipeline_id,
            from_state=from_state,
            to_state=to_state,
            timestamp=datetime.now(timezone.utc).isoformat(),
            metadata=metadata or {},
        )


class PipelineStage(BaseModel):
    name: str
    state: PipelineState
    config: Dict[str, Any] = Field(default_factory=dict)


class PipelineDefinition(BaseModel):
    id: str
    name: str
    stages: List[PipelineStage]
    config: Dict[str, Any] = Field(default_factory=dict)

    @field_validator("stages")
    @classmethod
    def validate_stage_states(cls, v: List[PipelineStage]) -> List[PipelineStage]:
        names = [s.name for s in v]
        if len(set(names)) != len(names):
            raise ValueError("Duplicate stage names in pipeline definition")
        return v
```
