const fs = require('fs');
const path = require('path');

class ImportScanner {
  constructor(projectRoot) {
    this.projectRoot = projectRoot;
    this.packageJsonPath = path.join(projectRoot, 'package.json');
    this.imports = new Map(); // file -> Set of import paths
    this.usedIdentifiers = new Map(); // file -> Set of identifiers
    this.errors = [];
  }

  scanDirectory(dir, baseDir = dir) {
    const entries = fs.readdirSync(dir, { withFileTypes: true });
    
    for (const entry of entries) {
      const fullPath = path.join(dir, entry.name);
      
      if (entry.isDirectory()) {
        this.scanDirectory(fullPath, baseDir);
      } else if (entry.isFile()) {
        const ext = path.extname(entry.name).toLowerCase();
        const isJsFile = ['.js', '.jsx', '.ts', '.tsx', '.mjs', '.cjs'].includes(ext);
        
        if (isJsFile) {
          this.scanFile(fullPath, baseDir);
        }
      }
    }
  }

  scanFile(filePath, baseDir) {
    const relativePath = path.relative(baseDir, filePath);
    const content = fs.readFileSync(filePath, 'utf-8');
    
    // Extract imports
    this.extractImports(content, relativePath);
    
    // Extract used identifiers
    this.extractUsedIdentifiers(content, relativePath);
  }

  extractImports(content, relativePath) {
    const importRegex = /import\s+(?:(?:\w+\s*,\s*)?\w+\s*,\s*)?from\s+['"]([^'"]+)['"]/g;
    const namedImportRegex = /import\s+\{([^}]+)\}\s+from\s+['"]([^'"]+)['"]/g;
    const defaultImportRegex = /import\s+(\w+)\s+from\s+['"]([^'"]+)['"]/g;
    const sideEffectImportRegex = /import\s+['"]([^'"]+)['"]/g;
    
    const imports = new Set();
    
    let match;
    while ((match = importRegex.exec(content)) !== null) {
      imports.add(match[1]);
    }
    
    while ((match = namedImportRegex.exec(content)) !== null) {
      imports.add(match[2]);
    }
    
    while ((match = defaultImportRegex.exec(content)) !== null) {
      imports.add(match[2]);
    }
    
    while ((match = sideEffectImportRegex.exec(content)) !== null) {
      imports.add(match[1]);
    }
    
    this.imports.set(relativePath, imports);
  }

  extractUsedIdentifiers(content, relativePath) {
    const used = new Set();
    
    // Match identifier usage (simplified regex)
    const identifierRegex = /\b([a-zA-Z_$][a-zA-Z0-9_$]*)\b/g;
    
    let match;
    while ((match = identifierRegex.exec(content)) !== null) {
      const identifier = match[1];
      
      // Skip keywords and import-related words
      const keywords = ['import', 'from', 'export', 'default', 'const', 'let', 'var', 
                       'function', 'class', 'if', 'else', 'for', 'while', 'switch', 'case',
                       'break', 'continue', 'return', 'try', 'catch', 'finally', 'throw',
                       'new', 'this', 'true', 'false', 'null', 'undefined', 'typeof',
                       'void', 'delete', 'in', 'of', 'extends', 'super', 'await', 'async'];
      
      if (!keywords.includes(identifier) && identifier.length > 0) {
        used.add(identifier);
      }
    }
    
    this.usedIdentifiers.set(relativePath, used);
  }

  validateImports() {
    const packageJson = JSON.parse(fs.readFileSync(this.packageJsonPath, 'utf-8'));
    const dependencies = new Set([...packageJson.dependencies || {}, ...packageJson.devDependencies || {}]);
    
    for (const [filePath, imports] of this.imports) {
      const usedIdentifiers = this.usedIdentifiers.get(filePath) || new Set();
      
      // Check for missing imports
      for (const importPath of imports) {
        const packageName = this.getPackageNameFromPath(importPath);
        if (packageName && !dependencies.has(packageName)) {
          this.errors.push({
            type: 'missing',
            file: filePath,
            import: importPath,
            message: `Missing import: ${importPath} (not in package.json)`
          });
        }
      }
