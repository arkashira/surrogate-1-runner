"""
Data Sync Job Template Generator - Production-Ready

Generates Kubernetes Job manifests with init containers for migrating
container data (bind mounts and named volumes) to PersistentVolumes.
"""

from typing import Optional
from dataclasses import dataclass, field
import yaml


@dataclass
class VolumeSource:
    """Represents a volume source from docker compose."""
    name: str
    type: str  # "bind" or "named_volume"
    host_path: Optional[str] = None  # For bind mounts
    container_path: str = "/data"


@dataclass
class DataSyncJobSpec:
    """Specification for a data sync job."""
    job_name: str
    namespace: str = "default"
    source_volumes: list[VolumeSource] = field(default_factory=list)
    target_pvc_name: str = ""
    target_container_path: str = "/data"
    rsync_image: str = "docker.io/rsync/rsync:latest"
    init_container_image: str = "docker.io/busybox:latest"
    storage_size: str = "10Gi"
    storage_class: str = "local-path"
    labels: dict = field(default_factory=dict)
    backoff_limit: int = 4
    ttl_seconds_after_finished: int = 3600
    # NEW: For accessing Docker named volumes (requires privileged pod + Docker socket)
    docker_socket_path: str = "/var/run/docker.sock"


# =============================================================================
# FIX #1: Use proper PVC with VolumeName reference instead of kubectl wait
# =============================================================================
def generate_pv_manifest(spec: DataSyncJobSpec) -> dict:
    """Generate PersistentVolume manifest for the target."""
    pv_name = f"{spec.job_name}-pv"
    
    return {
        "apiVersion": "v1",
        "kind": "PersistentVolume",
        "metadata": {
            "name": pv_name,
            "labels": {
                "app": spec.job_name,
                "sync-job": "source",  # Label for PVC selector
                **spec.labels
            }
        },
        "spec": {
            "capacity": {"storage": spec.storage_size},
            "accessModes": ["ReadWriteOnce"],
            "storageClassName": spec.storage_class,
            "persistentVolumeReclaimPolicy": "Retain",
            "hostPath": {"path": f"/mnt/data/{spec.job_name}"}
        }
    }


def generate_pvc_manifest(spec: DataSyncJobSpec) -> dict:
    """Generate PersistentVolumeClaim manifest with proper selector."""
    pvc_name = spec.target_pvc_name or f"{spec.job_name}-pvc"
    
    return {
        "apiVersion": "v1",
        "kind": "PersistentVolumeClaim",
        "metadata": {
            "name": pvc_name,
            "namespace": spec.namespace,
            "labels": {
                "app": spec.job_name,
                "sync-job": "target",  # Different label to avoid self-binding
                **spec.labels
            }
        },
        "spec": {
            "accessModes": ["ReadWriteOnce"],
            "storageClassName": spec.storage_class,
            "resources": {"requests": {"storage": spec.storage_size}},
            # FIX: Use volumeName to explicitly bind to PV
            "volumeName": f"{spec.job_name}-pv"
        }
    }
