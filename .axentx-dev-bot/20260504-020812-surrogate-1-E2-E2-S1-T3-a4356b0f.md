dev-setup:
	@echo "🔧 Installing Python dependencies..."

	@echo "✅ Dependencies installed."

	@echo "⚠️  Skipping pre‑commit hook installation (NO_HOOK=1)."

	chmod +x .git/hooks/pre-commit
	@echo "✅ Pre‑commit hook installed."
