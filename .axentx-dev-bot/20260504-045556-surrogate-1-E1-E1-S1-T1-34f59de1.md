package main

import (
	"encoding/json"
	"fmt"
	"io/ioutil"
	"log"
	"net/http"
	"os"
	"time"

	"github.com/gorilla/mux"
)

// APIInterceptor is responsible for capturing API request/response signatures
type APIInterceptor struct {
	router *mux.Router
	store  *ContractStore
}

// NewAPIInterceptor returns a new instance of APIInterceptor
func NewAPIInterceptor(store *ContractStore) *APIInterceptor {
	return &APIInterceptor{
		router: mux.NewRouter(),
		store:  store,
	}
}

// Start starts the API interceptor
func (a *APIInterceptor) Start() {
	a.router.Use(func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			startTime := time.Now()
			next.ServeHTTP(w, r)
			endTime := time.Now()
			duration := endTime.Sub(startTime)

			// Capture request/response signatures
			signature := &Signature{
				Request:  r,
				Response: w,
				Duration: duration,
			}

			// Store the signature
			a.store.StoreSignature(signature)
		})
	})

	log.Fatal(http.ListenAndServe(":8080", a.router))
}

// Signature represents an API request/response signature
type Signature struct {
	Request  *http.Request
	Response http.ResponseWriter
	Duration time.Duration
}

// ContractStore is responsible for storing API request/response signatures
type ContractStore struct {
	contractsDir string
}

// NewContractStore returns a new instance of ContractStore
func NewContractStore(contractsDir string) *ContractStore {
	return &ContractStore{
		contractsDir: contractsDir,
	}
}

// StoreSignature stores an API request/response signature
func (c *ContractStore) StoreSignature(signature *Signature) {
	// Marshal the signature to JSON
	signatureJSON, err := json.Marshal(signature)
	if err != nil {
		log.Println(err)
		return
	}

	// Store the signature in the contracts directory
	contractFile := fmt.Sprintf("%s/contract-%d.json", c.contractsDir, time.Now().UnixNano())
	err = ioutil.WriteFile(contractFile, signatureJSON, 0644)
	if err != nil {
		log.Println(err)
		return
	}
}

func main() {
	contractsDir := "/opt/axentx/surrogate-1/contracts"
	store := NewContractStore(contractsDir)
	interceptor := NewAPIInterceptor(store)
	interceptor.Start()
}

const fs = require('fs');
const path = require('path');

class ContractStore {
  constructor(contractsDir) {
    this.contractsDir = contractsDir;
  }

  storeSignature(signature) {
    // Marshal the signature to JSON
    const signatureJSON = JSON.stringify(signature);

    // Store the signature in the contracts directory
    const contractFile = path.join(this.contractsDir, `contract-${Date.now()}.json`);
    fs.writeFileSync(contractFile, signatureJSON);
  }

  compareSignatures(baselineSignature, liveSignature) {
    // Compare the baseline signature with the live signature
    // Return true if the signatures match within 50ms, false otherwise
    const durationDiff = Math.abs(baselineSignature.duration - liveSignature.duration);
    return durationDiff <= 50;
  }
}

module.exports = ContractStore;
