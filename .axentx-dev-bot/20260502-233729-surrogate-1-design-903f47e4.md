### Step 1: Pre-flight snapshot generation

Create a new script `bin/snapshot.sh` that generates a pre-flight snapshot of the dataset:

```bash
#!/bin/bash

# Define dataset repo and path
REPO="axentx/surrogate-1-training-pairs"
PATH="public"

# Define snapshot directory
SNAPSHOT_DIR="batches/snapshot"

# Create snapshot directory if it doesn't exist
mkdir -p "$SNAPSHOT_DIR"

# Download and process each file in the dataset
for date in $(curl -s https://huggingface.co/datasets/$REPO/resolve/main/$PATH | jq -r '.[] | @base64'); do
  echo "$date"
  file=$(echo "$date" | base64 --decode)
  slug=$(echo "$file" | jq -r '.slug')
  filename=$(echo "$file" | jq -r '.filename')
  jsonl=$(echo "$file" | jq -r '.jsonl')
  slug_hash=$(echo "$slug" | sha256sum | cut -d' ' -f1)
  snapshot_path="$SNAPSHOT_DIR/$slug_hash/$filename"
  mkdir -p "$snapshot_path"
  curl -s https://huggingface.co/datasets/$REPO/resolve/main/$PATH/$jsonl | jq -r '.[] | @base64' | base64 --decode > "$snapshot_path/$filename.jsonl"
done
```
