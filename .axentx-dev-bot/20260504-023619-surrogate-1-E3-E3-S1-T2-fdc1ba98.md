import csv
import io
import pdfkit
from datetime import datetime

def generate_report(format):
    # Fetch data from database or other sources
    data = [
        {'timestamp': '2022-01-01 12:00:00', 'user_id': '123', 'change_log': 'Remediation applied'},
        {'timestamp': '2022-01-02 13:00:00', 'user_id': '456', 'change_log': 'Compliance check performed'},
    ]

    if format == 'pdf':
        html = '<html><body><h1>Audit Report</h1><table><tr><th>Timestamp</th><th>User ID</th><th>Change Log</th></tr>'
        for row in data:
            html += f"<tr><td>{row['timestamp']}</td><td>{row['user_id']}</td><td>{row['change_log']}</td></tr>"
        html += '</table></body></html>'
        pdf = pdfkit.from_string(html, False)
        return io.BytesIO(pdf)
    elif format == 'csv':
        output = io.StringIO()
        csv_writer = csv.writer(output)
        csv_writer.writerow(['Timestamp', 'User ID', 'Change Log'])
        for row in data:
            csv_writer.writerow([row['timestamp'], row['user_id'], row['change_log']])
        return io.StringIO(output.getvalue())
    else:
        return "Invalid format", 400
```
