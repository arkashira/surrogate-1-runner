        logging.error(f"Error extracting text from PDF: {str(e)}")
        return None

def update_processing_status(status, error_message=None):
    """
    Updates the processing_status in the database.
    
    Args:
        status (str): New status ('ready' or 'failed').
        error_message (str, optional): Error message if status is 'failed'. Defaults to None.
    """
    processing_status = ProcessingStatus.query.get(1)  # Assuming the ID is 1
    processing_status.status = status
    if error_message:
        processing_status.error_message = error_message
    processing_status.save()

def extract_text_using_library(pdf_file):
    # Implementation of text extraction using a library like PyPDF2 or pdfplumber
    # For example, using PyPDF2:
    import PyPDF2
    pdf = PyPDF2.PdfFileReader(pdf_file)
    text = ''
    for page in range(pdf.numPages):
        text += pdf.getPage(page).extractText()
    return text
