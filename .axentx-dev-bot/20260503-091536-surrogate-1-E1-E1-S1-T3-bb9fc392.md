import { Pool, PoolClient, QueryResult } from 'pg';

export interface RequestAttrs {
  id?: number;
  publicId?: string; // UUID string
  title: string;
  status: 'pending' | 'processing' | 'completed' | 'failed';
  createdAt?: Date;
  updatedAt?: Date;
}

export class RequestModel {
  private pool: Pool;

  constructor(poolOrOptions: Pool | { connectionString?: string }) {
    if (poolOrOptions instanceof Pool) {
      this.pool = poolOrOptions;
    } else {
      this.pool = new Pool(poolOrOptions);
    }
  }

  async create(attrs: Omit<RequestAttrs, 'id' | 'publicId' | 'createdAt' | 'updatedAt'>): Promise<RequestAttrs> {
    const res: QueryResult = await this.pool.query(
      `INSERT INTO requests (title, status)
       VALUES ($1, $2)
       RETURNING id, public_id, title, status, created_at, updated_at`,
      [attrs.title, attrs.status]
    );
    const row = res.rows[0];
    return rowToAttrs(row);
  }

  async findByPublicId(publicId: string): Promise<RequestAttrs | null> {
    const res: QueryResult = await this.pool.query(
      `SELECT id, public_id, title, status, created_at, updated_at
       FROM requests
       WHERE public_id = $1`,
      [publicId]
    );
    if (res.rowCount === 0) return null;
    return rowToAttrs(res.rows[0]);
  }
}

function rowToAttrs(row: any): RequestAttrs {
  return {
    id: row.id,
    publicId: row.public_id,
    title: row.title,
    status: row.status,
    createdAt: row.created_at,
    updatedAt: row.updated_at,
  };
}

import { Router } from 'express';
import { Pool } from 'pg';
import { RequestModel } from '../models/request';

const router = Router();
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

// POST /requests
// Body: { title: string, status: string }
// Returns 201 { publicUrl: string }
router.post('/requests', async (req, res) => {
  const { title, status } = req.body;
  if (!title || !status) {
    return res.status(400).json({ error: 'title and status required' });
  }

  const model = new RequestModel(pool);
  try {
    const created = await model.create({ title, status });
    const publicUrl = `${req.protocol}://${req.get('host')}/public/requests/${created.publicId}`;
    return res.status(201).json({ publicUrl });
  } catch (err) {
    console.error('Failed to create request', err);
    return res.status(500).json({ error: 'internal server error' });
  }
});

export default router;
