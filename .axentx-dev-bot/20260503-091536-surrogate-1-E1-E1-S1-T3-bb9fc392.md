import { PoolClient, QueryResult } from 'pg';

export interface Request {
  id: number;
  title: string;
  publicId: string; // UUID string
  status: string;
  createdAt: Date;
  updatedAt: Date;
  lastUpdated?: Date;
  slaDeadline?: Date;
}

export class RequestModel {
  constructor(private pool: PoolClient) {}

  async create(title: string): Promise<Request> {
    const res: QueryResult = await this.pool.query(
      `INSERT INTO requests (title)
       VALUES ($1)
       RETURNING id, title, public_id, status, created_at, updated_at, last_updated, sla_deadline`,
      [title]
    );
    const row = res.rows[0];
    return {
      id: row.id,
      title: row.title,
      publicId: row.public_id,
      status: row.status,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      lastUpdated: row.last_updated,
      slaDeadline: row.sla_deadline,
    };
  }

  async findByPublicId(publicId: string): Promise<Request | null> {
    const res: QueryResult = await this.pool.query(
      `SELECT id, title, public_id, status, created_at, updated_at, last_updated, sla_deadline
       FROM requests
       WHERE public_id = $1`,
      [publicId]
    );
    if (res.rows.length === 0) return null;
    const row = res.rows[0];
    return {
      id: row.id,
      title: row.title,
      publicId: row.public_id,
      status: row.status,
      createdAt: row.created_at,
      updatedAt: row.updated_at,
      lastUpdated: row.last_updated,
      slaDeadline: row.sla_deadline,
    };
  }
}
```

Why this is correct + actionable:
- Uses DB default for UUID generation (no extra dependency, no race risk).
- `publicId` is returned on create and populated from `public_id` column.
- `findByPublicId` uses the unique index and returns `null` for missing IDs (enables 404).
