from datetime import datetime, timedelta
from ..models.request import Request

class RequestRepository:
    def get_last_30_days_requests(self):
        # Calculate the date 30 days ago
        thirty_days_ago = datetime.now() - timedelta(days=30)
        
        # Retrieve requests from the last 30 days
        requests = Request.query.filter(Request.created_at >= thirty_days_ago).all()
        
        # Initialize counters for status and SLA states
        status_counts = {'on-track': 0, 'at-risk': 0, 'breached': 0}
        sla_states = {'on-track': 0, 'at-risk': 0, 'breached': 0}
        
        # Count requests by status and SLA state
        for request in requests:
            if request.status == 'on-track':
                status_counts['on-track'] += 1
            elif request.status == 'at-risk':
                status_counts['at-risk'] += 1
            else:
                status_counts['breached'] += 1
            
            if request.sla_state == 'on-track':
                sla_states['on-track'] += 1
            elif request.sla_state == 'at-risk':
                sla_states['at-risk'] += 1
            else:
                sla_states['breached'] += 1
        
        # Return the request data with status counts and SLA states
        return {
            'requests': requests,
            'status_counts': status_counts,
            'sla_states': sla_states
        }
