```python
import logging
from datetime import datetime
import os

class RetryLogger:
    def __init__(self, task_id, max_retries=3, backoff_base=2, max_backoff=60):
        """
        Initialize the RetryLogger with task_id, max_retries, backoff_base, and max_backoff.

        Args:
            task_id (str): The ID of the task being logged.
            max_retries (int, optional): Maximum number of retry attempts. Defaults to 3.
            backoff_base (int, optional): Base value for exponential backoff. Defaults to 2.
            max_backoff (int, optional): Maximum backoff time in seconds. Defaults to 60.
        """
        self.task_id = task_id
        self.max_retries = max_retries
        self.backoff_base = backoff_base
        self.max_backoff = max_backoff
        self.retry_count = 0
        self.log_file = f"/var/log/axentx/surrogate-1/retry_logs/{task_id}.log"
        os.makedirs(os.path.dirname(self.log_file), exist_ok=True)
        self.logger = logging.getLogger(f"retry_logger_{task_id}")
        self.logger.setLevel(logging.INFO)
        handler = logging.FileHandler(self.log_file)
        formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)

    def log_retry_attempt(self, error_message):
        """
        Log a retry attempt with the given error message.

        Args:
            error_message (str): The error message to log.

        Returns:
            str: The log message.
        """
        self.retry_count += 1
        log_message = f"Retry attempt {self.retry_count} of {self.max_retries} for task {self.task_id}. Error: {error_message}"
        self.logger.info(log_message)
        return log_message

    def should_retry(self):
        """
        Check if the maximum number of retries has been reached.

        Returns:
            bool: True if retries are allowed, False otherwise.
        """
        return self.retry_count < self.max_retries

    def get_backoff_time(self):
        """
        Calculate the backoff time based on the retry count and configured parameters.

        Returns:
            int: The backoff time in seconds.
        """
        return min(self.backoff_base ** self.retry_count, self.max_backoff)
```
