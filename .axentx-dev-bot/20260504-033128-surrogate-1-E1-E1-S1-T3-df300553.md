## EventQueue.kt

package common

import java.util.concurrent.ConcurrentLinkedQueue
import java.util.concurrent.atomic.AtomicLong

data class Event(
    val id: Long,
    val timestamp: Long,
    val type: String
)

class EventQueue {
    private val queue: ConcurrentLinkedQueue<Event> = ConcurrentLinkedQueue()
    private val idCounter: AtomicLong = AtomicLong(0)

    fun addEvent(type: String): Event {
        val event = Event(
            id = idCounter.incrementAndGet(),
            timestamp = System.currentTimeMillis(),
            type = type
        )
        queue.add(event)
        return event
    }

    fun getEvents(): List<Event> = queue.toList()

    fun clearEvents() {
        queue.clear()
    }
    
    fun size(): Int = queue.size
}
```
