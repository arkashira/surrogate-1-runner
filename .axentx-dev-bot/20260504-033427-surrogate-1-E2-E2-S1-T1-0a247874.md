"""
Cost forecasting module.

The public API is the :class:`CostForecaster` class.  It expects a list of
historical records, each a mapping with keys ``date`` (ISO‑8601 string) and
``cost`` (float).  The class exposes:

* ``forecast(days_ahead)`` – returns a list of forecast dictionaries.
* ``update_with_latest(latest_cost)`` – append a new daily cost and
  automatically advance the internal “current date”.
"""

from __future__ import annotations

import datetime
import math
from dataclasses import dataclass
from typing import Dict, Iterable, List, Mapping

# --------------------------------------------------------------------------- #
# Helper utilities
# --------------------------------------------------------------------------- #
def _to_date(d: str | datetime.date) -> datetime.date:
    """Coerce a string or date into a ``datetime.date``."""
    if isinstance(d, datetime.date):
        return d
    return datetime.datetime.strptime(d, "%Y-%m-%d").date()


def _confidence_factor(n: int) -> float:
    """
    Simple confidence metric: 1.0 for 30+ points, 0.5 for 10–29, 0.2 for 7–9,
    0.0 for <7.
    """
    if n >= 30:
        return 1.0
    if n >= 10:
        return 0.5
    if n >= 7:
        return 0.2
    return 0.0


# --------------------------------------------------------------------------- #
# Core class
# --------------------------------------------------------------------------- #
@dataclass
class CostForecaster:
    """
    Forecast future daily costs using a hybrid moving‑average + linear‑trend
    model.

    Parameters
    ----------
    historical_data : Iterable[Mapping[str, float]]
        Each mapping must contain ``date`` (ISO‑8601 string) and ``cost``.
    """
    historical_data: Iterable[Mapping[str, float]]

    def __post_init__(self) -> None:
        # Convert to a sorted list of tuples (date, cost)
        self._data: List[tuple[datetime.date, float]] = sorted(
            (( _to_date(rec["date"]), float(rec["cost"])) for rec in self.historical_data),
            key=lambda x: x[0],
        )
        if len(self._data) < 7:
            raise ValueError("At least 7 days of historical data are required.")
        # Current date is the last date in the data set
        self.current_date: datetime.date = self._data[-1][0]

    # --------------------------------------------------------------------- #
    # Forecasting helpers
    # --------------------------------------------------------------------- #
    def _moving_average(self, window: int = 7) -> float:
        """Return the simple moving average of the last ``window`` days."""
        return sum(cost for _, cost in self._data[-window:]) / window

    def _trend_slope(self) -> float:
        """
        Linear regression slope (cost per day) over the entire data set.
        Uses timestamps as the independent variable.
        """
        n = len(self._data)
        if n < 2:
            return 0.0
        xs = [datetime.datetime.combine(d, datetime.time.min).timestamp() for d, _ in self._data]
        ys = [cost for _, cost in self._data]
        mean_x = sum(xs) / n
        mean_y = sum(ys) / n
        num = sum((x - mean_x) * (y - mean_y) for x, y in zip(xs, ys))
        den = sum((x - mean_x) ** 2 for x in xs)
        return num / den if den else 0.0

    def _predict_for_day(self, target: datetime.date) -> float:
        """Predict cost for a single future date."""
