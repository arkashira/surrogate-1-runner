Designed for 10k+ tasks/sec throughput with <5ms event loop blocking.
"""

from dataclasses import dataclass, field, asdict
from typing import Any, Callable, Optional, Dict, List
from contextlib import asynccontextmanager
import logging

class Task:
    """Represents a task in the queue."""
    id: str
    payload: Dict[str, Any]
    priority: int = 0
    created_at: float = field(default_factory=time.time)
    retries: int = 0
    max_retries: int = 3
    status: str = "pending"
    result: Optional[Any] = None
    error: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "Task":
        return cls(**data)


class TaskQueue:
    """
    Async task queue with Redis backend.
    Implements at-least-once delivery with acknowledgment pattern.
    """

    def __init__(
        self,
        redis_url: str = "redis://localhost:6379",
        queue_name: str = "axentx_tasks",
        max_queue_size: int = 5000,
        retry_delay: float = 1.0,
    ):
        self.redis_url = redis_url
        self.queue_name = queue_name
        self.max_queue_size = max_queue_size
        self.retry_delay = retry_delay
        self._redis: Optional[asyncio.StreamWriter] = None
        self._running = False
        self._workers: List[asyncio.Task] = []
        self._lock = asyncio.Lock()
        self._stats = {
            "enqueued": 0,
            "processed": 0,
            "failed": 0,
            "backpressure_events": 0,
        }

    async def connect(self) -> None:
        """Establish Redis connection."""
        import aioredis

        self._redis = await aioredis.from_url(
            self.redis_url,
            encoding="utf-8",
            decode_responses=True,
        )
        logger.info("Connected to Redis")

    async def disconnect(self) -> None:
        """Close Redis connection."""
        if self._redis:
            await self._redis.close()
            self._redis = None
        logger.info("Disconnected from Redis")

    @asynccontextmanager
    async def connection(self):
        """Context manager for Redis connection."""
        if not self._redis:
            await self.connect()
        try:
            yield self._redis
        finally:
            pass

    async def enqueue(self, payload: Dict[str, Any], priority: int = 0) -> str:
        """
        Add a task to the queue.
        Applies back-pressure if queue exceeds max size.
        Returns task ID.
        """
        task_id = str(uuid.uuid4())
        task = Task(id=task_id, payload=payload, priority=priority)

        async with self._lock:
            # Check queue length for back-pressure
            queue_len = await self._redis.llen(self.queue_name)
            if queue_len >= self.max_queue_size:
                self._stats["backpressure_events"] += 1
                logger.warning(
                    f"Back-pressure: queue length {queue_len} exceeds threshold {self.max_queue_size}"
                )
                # Push to lower priority sub-queue for back-pressure handling
                await self._redis.lpush(
                    f"{self.queue_name}:backpressure",
                    json.dumps(task.to_dict()),
                )
                return task_id

            # Add to main queue with priority (lower number = higher priority)
            await self._redis.zadd(
                self.queue_name,
                {task_id: task.priority * 1000 - task.created_at},
            )
            await self._redis.expire(self.queue_name, 86400)  # 24h TTL
            self._stats["enqueued"] += 1

        logger.debug(f"Enqueued task {task_id}")
