from fastapi import FastAPI, Query
from typing import List, Optional
from app.models import Step
from app.database import get_steps

app = FastAPI()

@app.get("/steps", response_model=List[Step])
async def search_steps(q: Optional[str] = Query(None)):
    steps = get_steps()
    if q:
        steps = [step for step in steps if q.lower() in step.title.lower() or q.lower() in step.description.lower()]
        steps.sort(key=lambda step: (q.lower() not in step.title.lower(), step.title.lower().count(q.lower()) + step.description.lower().count(q.lower())), reverse=True)
    return steps
```
