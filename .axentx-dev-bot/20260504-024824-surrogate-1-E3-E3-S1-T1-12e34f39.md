#!/usr/bin/env python3
"""
Surrogate‑1 request signature diff detector.

The tool compares two JSON files that each contain an array of
signature objects.  A signature is considered a match if the HTTP
method and path are identical **and** the required headers and JSON
schema are exactly the same.  Any deviation results in a FAILED
check.

Usage
-----
    detector.py <baseline.json> <incoming.json>

The script prints either ``PASS`` or ``FAIL`` to stdout and exits
with status 0 for PASS and 1 for FAIL.  All mismatches are printed to
stderr, one per line, prefixed with ``- ``.

The implementation is intentionally lightweight to keep per‑batch
latency well below 200 ms for typical dataset sizes.
"""

from __future__ import annotations

import argparse
import json
import sys
from pathlib import Path
from typing import Any, Dict, List, Tuple

# --------------------------------------------------------------------------- #
# 1.  Data loading & validation
# --------------------------------------------------------------------------- #

def load_signatures(path: Path) -> List[Dict[str, Any]]:
    """
    Load a list of signature objects from a JSON file.

    Raises
    ------
    RuntimeError
        If the file cannot be read or the JSON is not an array.
    """
    try:
        with path.open("r", encoding="utf-8") as f:
            data = json.load(f)
    except Exception as exc:  # pragma: no cover
        raise RuntimeError(f"Failed to read {path}: {exc}") from exc

    if not isinstance(data, list):
        raise RuntimeError(f"Signature file {path} must contain a JSON array")

    # Basic key‑presence validation – helps catch malformed data early.
    required_keys = {"method", "path", "required_headers", "json_schema"}
    for idx, sig in enumerate(data, start=1):
        if not isinstance(sig, dict):
            raise RuntimeError(f"Signature #{idx} in {path} is not an object")
        missing = required_keys - sig.keys()
        if missing:
            raise RuntimeError(
                f"Signature #{idx} in {path} missing keys: {', '.join(sorted(missing))}"
            )
    return data


# --------------------------------------------------------------------------- #
# 2.  Normalisation helpers
# --------------------------------------------------------------------------- #

def signature_key(sig: Dict[str, Any]) -> Tuple[str, str]:
    """
    Return a canonical key for a signature: (METHOD, PATH).

    The method is normalised to upper‑case to avoid case‑sensitivity
    surprises.
    """
    return (sig["method"].upper(), sig["path"])


def signatures_by_key(
    sigs: List[Dict[str, Any]]
) -> Dict[Tuple[str, str], Dict[str, Any]]:
    """
    Build a lookup dictionary keyed by (method, path).

    If duplicate keys exist, the last one wins – this mirrors the
    behaviour of the original implementation and keeps the code
    simple.
    """
    return {signature_key(sig): sig for sig in sigs}
