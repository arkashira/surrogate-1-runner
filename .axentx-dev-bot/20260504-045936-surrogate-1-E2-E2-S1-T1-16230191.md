from surrogate_1.detection import SignatureDetector, RequestMetadata, DriftEvent, DriftType
from hashlib import sha256
import pytest
from datetime import datetime

def test_signature_computation():
    metadata = RequestMetadata(
        pipeline_name="user-auth-pipeline",
        api_endpoint="/login",
        method="POST",
        headers={"Content-Type": "application/json"},
        body='{"username": "alice", "password": "secret"}'
    )
    expected_hash = sha256(metadata.body.encode()).hexdigest()[:16]
    assert len(expected_hash) == 16
    assert expected_hash == metadata.get_signature()

def test_drift_event_to_dict():
    event = DriftEvent(
        drift_type=DriftType.SIGNATURE_DRIFT,
        timestamp=datetime(2026, 5, 3, 10, 0, 0),
        pipeline_name="user-auth-pipeline",
        api_request="/login",
        details={"old_sig": "a1b2c3d4", "new_sig": "e5f6g7h8"}
    )
    output = event.to_dict()
    assert output["drift_type"] == "SIGNATURE_DRIFT"
    assert output["timestamp"] == "2026-05-03T10:00:00"
    assert output["pipeline_name"] == "user-auth-pipeline"
    assert output["api_request"] == "/login"
    assert "old_sig" in output["details"]

def test_signature_drift_detection():
    detector = SignatureDetector()
    old_metadata = RequestMetadata(pipeline_name="test-pipe", api_endpoint="/data", method="GET", headers={}, body='{"v":1}')
    new_metadata = RequestMetadata(pipeline_name="test-pipe", api_endpoint="/data", method="GET", headers={}, body='{"v":2}')

    detector.register_baseline("test-pipe", "/data", old_metadata.get_signature())

    drift_event = detector.detect(new_metadata)
    assert drift_event is not None
    assert drift_event.drift_type == DriftType.SIGNATURE_DRIFT
    assert drift_event.pipeline_name == "test-pipe"
    assert drift_event.api_request == "/data"

def test_no_drift_when_signatures_match():
    detector = SignatureDetector()
    metadata = RequestMetadata(pipeline_name="test-pipe", api_endpoint="/data", method="GET", headers={}, body='{"v":1}')
    detector.register_baseline("test-pipe", "/data", metadata.get_signature())
    assert detector.detect(metadata) is None

def test_detect_handles_missing_baseline():
    detector = SignatureDetector()
    metadata = RequestMetadata(pipeline_name="unknown", api_endpoint="/api", method="POST", headers={}, body="{}")
    with pytest.raises(KeyError):
        detector.detect_strict(metadata)  # strict mode expects baseline
    assert detector.detect(metadata) is None  # non-strict returns None
