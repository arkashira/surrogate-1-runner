"""
Cloud Pricing Integration module for retrieving current instance pricing data.
This module provides a unified interface to fetch on-demand prices for various cloud services (AWS, GCP, etc.) using their respective APIs. It is designed to be lightweight, flexible, and extensible for supporting multiple cloud providers.

Prerequisites:
- boto3 or googleapiclient must be installed and configured with appropriate credentials.
- The cloud region used for pricing queries should be set via the respective environment variables or explicitly passed.

The public API of this module is the `get_instance_pricing` function, which returns a dictionary mapping instance type names to their on-demand price per hour in USD.
"""

import os
from typing import Dict, List

import boto3
from googleapiclient.discovery import build
from googleapiclient.errors import HttpError

# Cache the pricing client per cloud provider and region to avoid repeated initialisation
_PRICING_CLIENTS: Dict[str, Dict[str, boto3.client | build]] = {}

def _get_pricing_client(cloud_provider: str, region: str) -> boto3.client | build:
    """Return a cached cloud pricing client for the given cloud provider and region."""
    if cloud_provider not in _PRICING_CLIENTS:
        _PRICING_CLIENTS[cloud_provider] = {}
    if region not in _PRICING_CLIENTS[cloud_provider]:
        _PRICING_CLIENTS[cloud_provider][region] = {}
    if cloud_provider == "aws":
        _PRICING_CLIENTS[cloud_provider][region] = _get_aws_pricing_client(region)
    elif cloud_provider == "gcp":
        _PRICING_CLIENTS[cloud_provider][region] = _get_gcp_pricing_client(region)
    else:
        raise NotImplementedError(f"Unsupported cloud provider: {cloud_provider}")
    return _PRICING_CLIENTS[cloud_provider][region]

def _get_aws_pricing_client(region: str) -> boto3.client:
    """Return a boto3 AWS pricing client for the given region."""
    return boto3.client("pricing", region_name=region)

def _get_gcp_pricing_client(region: str) -> build:
    """Return a Google Cloud Platform pricing client for the given region."""
    return build("cloudresourcemanager", "v1", cache_discovery=False)

def get_instance_pricing(
    cloud_provider: str,
    instance_types: List[str],
    region: str = None,
    currency: str = "USD",
) -> Dict[str, float]:
    """
    Retrieve on-demand pricing for a list of cloud service instances.

    Parameters
    ----------
    cloud_provider : str
        Cloud provider name (e.g., "aws", "gcp").
    instance_types : List[str]
        List of instance type identifiers (e.g., ["t3.micro", "m5.large"]).
    region : str, optional
        Cloud region to query the pricing API. Defaults to the value of
        the respective environment variable or "us-east-1".
    currency : str, optional
        Currency code for the returned prices. Defaults to "USD".

    Returns
    -------
    Dict[str, float]
        Mapping from instance type to its on-demand price per hour.

    Raises
    ------
    RuntimeError
        If the pricing API call fails or the requested instance type
        is not found in the response.
    """
    if not region:
        if cloud_provider == "aws":
            region = os.getenv("AWS_DEFAULT_REGION", "us-east-1")
        elif cloud_provider == "gcp":
            region = os.getenv("GOOGLE_CLOUD_REGION", "us-central1")
        else:
            raise NotImplementedError(f"Unsupported cloud provider: {cloud_provider}")

    client = _get_pricing_client(cloud_provider, region)

    # Build the filter for the requested instance types
    filters = [
        {
            "Type": "TERM_MATCH",
            "Field": "instanceType",
            "Value": instance_type,
        }
        for instance_type in instance_types
    ]
