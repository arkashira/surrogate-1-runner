## 1. Acceptance Criteria  

- **AC1** – A Slack webhook URL can be entered, saved, and retrieved for a specific workspace.  
- **AC2** – When an issue is created, assigned, or closed, a Slack message is posted to the configured webhook.  
- **AC3** – The Slack message payload contains:  
  - The issue title.  
  - The action type (created / assigned / closed).  
  - A clickable link that opens the Surrogate‑1 rule page for the issue.  
- **AC4** – If no webhook is configured, no Slack request is sent and no error is thrown.  
- **AC5** – The UI displays validation errors for an empty or malformed URL.  
- **AC6** – The UI shows a success toast when the webhook is saved successfully.

// tests/unit/SlackConfig.spec.js
import { shallowMount } from '@vue/test-utils'
import SlackConfig from '@/ui/SlackConfig.vue'
import * as api from '@/api/slackConfig'

jest.mock('@/api/slackConfig')

describe('SlackConfig.vue', () => {
  beforeEach(() => {
    api.getWebhook.mockResolvedValue({ url: '' })
    api.saveWebhook.mockResolvedValue({ success: true })
  })

  it('fetches existing webhook on mount', async () => {
    const wrapper = shallowMount(SlackConfig)
    await wrapper.vm.$nextTick()
    expect(api.getWebhook).toHaveBeenCalled()
    expect(wrapper.vm.webhookUrl).toBe('')
  })

  it('validates non-empty URL', async () => {
    const wrapper = shallowMount(SlackConfig)
    wrapper.setData({ webhookUrl: '' })
    await wrapper.find('button.save').trigger('click')
    expect(wrapper.vm.error).toBe('Webhook URL cannot be empty')
  })

  it('validates URL format', async () => {
    const wrapper = shallowMount(SlackConfig)
    wrapper.setData({ webhookUrl: 'not-a-url' })
    await wrapper.find('button.save').trigger('click')
    expect(wrapper.vm.error).toBe('Invalid URL format')
  })

  it('saves webhook when valid', async () => {
    const wrapper = shallowMount(SlackConfig)
    wrapper.setData({ webhookUrl: 'https://hooks.slack.com/services/ABC/DEF/GHI' })
    await wrapper.find('button.save').trigger('click')
    expect(api.saveWebhook).toHaveBeenCalledWith('https://hooks.slack.com/services/ABC/DEF/GHI')
    expect(wrapper.vm.successToast).toBe(true)
  })

  it('does not call API if URL unchanged', async () => {
    api.getWebhook.mockResolvedValue({ url: 'https://hooks.slack.com/services/ABC/DEF/GHI' })
    const wrapper = shallowMount(SlackConfig)
    await wrapper.vm.$nextTick()
    await wrapper.find('button.save').trigger('click')
    expect(api.saveWebhook).not.toHaveBeenCalled()
  })
})

import { sendSlackMessage } from '@/api/slackService'
import axios from 'axios'

jest.mock('axios')
const mockedAxios = axios as jest.Mocked<typeof axios>

describe('sendSlackMessage', () => {
  const webhook = 'https://hooks.slack.com/services/XYZ'
  const payload = { text: 'test' }

  it('posts to webhook and returns success', async () => {
    mockedAxios.post.mockResolvedValue({ status: 200 })
    const res = await sendSlackMessage(webhook, payload)
    expect(mockedAxios.post).toHaveBeenCalledWith(webhook, payload)
    expect(res).toBe(true)
  })

  it('returns false on non-200 status', async () => {
    mockedAxios.post.mockResolvedValue({ status: 500 })
    const res = await sendSlackMessage(webhook, payload)
    expect(res).toBe(false)
  })

  it('catches network errors and returns false', async () => {
    mockedAxios.post.mockRejectedValue(new Error('network'))
    const res = await sendSlackMessage(webhook, payload)
    expect(res).toBe(false)
  })
})
