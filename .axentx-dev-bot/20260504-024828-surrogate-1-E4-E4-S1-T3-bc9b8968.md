"""
Dashboard Access Control Module

Provides fine‑grained access control for dashboards.
Supports:
* Individual users
* Teams
* Three access levels (viewer, editor, admin)
* Public dashboards
"""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Dict, List, Optional, Set

# --------------------------------------------------------------------------- #
#  Enums
# --------------------------------------------------------------------------- #

class AccessLevel(Enum):
    """Access levels for dashboard sharing."""
    VIEWER = "viewer"
    EDITOR = "editor"
    ADMIN = "admin"

    @staticmethod
    def priority(level: "AccessLevel") -> int:
        """Higher number = higher privilege."""
        return {"viewer": 1, "editor": 2, "admin": 3}[level.value]


class PrincipalType(Enum):
    """Type of principal that can be granted access."""
    USER = "user"
    TEAM = "team"


# --------------------------------------------------------------------------- #
#  Dataclasses
# --------------------------------------------------------------------------- #

@dataclass
class Principal:
    """Represents a user or team that can access dashboards."""
    id: str
    name: str
    type: PrincipalType
    email: Optional[str] = None

    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "name": self.name,
            "type": self.type.value,
            "email": self.email,
        }


@dataclass
class DashboardShare:
    """Represents a shared dashboard with access control."""
    dashboard_id: str
    owner_id: str
    shared_with: Dict[str, AccessLevel] = field(default_factory=dict)  # user_id -> level
    team_access: Dict[str, AccessLevel] = field(default_factory=dict)  # team_id -> level
    created_at: datetime = field(default_factory=datetime.utcnow)
    updated_at: datetime = field(default_factory=datetime.utcnow)
    is_public: bool = False

    # --------------------------------------------------------------------- #
    #  Grant / Revoke
    # --------------------------------------------------------------------- #
    def grant_user_access(self, user_id: str, level: AccessLevel) -> None:
        self.shared_with[user_id] = level
        self.updated_at = datetime.utcnow()

    def revoke_user_access(self, user_id: str) -> bool:
        if user_id in self.shared_with:
            del self.shared_with[user_id]
            self.updated_at = datetime.utcnow()
            return True
        return False

    def grant_team_access(self, team_id: str, level: AccessLevel) -> None:
        self.team_access[team_id] = level
        self.updated_at = datetime.utcnow()

    def revoke_team_access(self, team_id: str) -> bool:
        if team_id in self.team_access:
            del self.team_access[team_id]
            self.updated_at = datetime.utcnow()
            return True
        return False
