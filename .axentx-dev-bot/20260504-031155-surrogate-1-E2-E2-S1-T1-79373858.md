"""
Coverage Report Generator for surrogate-1

Generates HTML and JSON coverage reports with gap analysis.
Supports scheduled CLI execution.
"""

import argparse
import json
import os
import sys
from dataclasses import dataclass, field
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional, Tuple

import coverage
from coverage import Coverage


@dataclass
class ModuleCoverage:
    """Coverage data for a single module."""
    module_path: str
    module_name: str
    total_lines: int
    covered_lines: int
    uncovered_lines: List[int] = field(default_factory=list)
    line_coverage_pct: float = 0.0


@dataclass
class CoverageReport:
    """Complete coverage report with gap analysis."""
    generated_at: str
    total_modules: int
    total_lines: int
    total_covered: int
    overall_coverage_pct: float
    modules: List[ModuleCoverage] = field(default_factory=list)
    gaps: List[Dict] = field(default_factory=list)


class CoverageReporter:
    """Generates coverage reports with gap analysis."""

    def __init__(self, coverage_data: Optional[Dict] = None):
        """
        Initialize reporter.
        
        Args:
            coverage_data: Pre-computed coverage data dict, or None to run coverage
        """
        self.coverage_data = coverage_data or {}
        self.report = CoverageReport(
            generated_at=datetime.now().isoformat(),
            total_modules=0,
            total_lines=0,
            total_covered=0,
            overall_coverage_pct=0.0,
            modules=[],
            gaps=[]
        )

    def load_coverage_data(self, coverage_file: str) -> None:
        """Load coverage data from coverage.py output file."""
        if not os.path.exists(coverage_file):
            raise FileNotFoundError(f"Coverage file not found: {coverage_file}")

        with open(coverage_file, 'r') as f:
            self.coverage_data = json.load(f)

    def run_coverage(self, source: str, omit: Optional[List[str]] = None) -> None:
        """Run coverage on source directory."""
        cov = Coverage()
        cov.source(source)
        if omit:
            cov.omit(omit)
        cov.run()
        self.coverage_data = cov.data
