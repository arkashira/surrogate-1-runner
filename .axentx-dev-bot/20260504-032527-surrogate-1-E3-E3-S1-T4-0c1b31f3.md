### ✅ Final Answer: Optimized CI Workflow for Sandbox Testing

#### 📁 File: `.github/workflows/ci.yml`
```yaml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  workflow_dispatch:

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Cache Docker layers
        uses: actions/cache@v4
        with:
          path: /tmp/.buildx-cache
          key: ${{ runner.os }}-buildx-${{ github.sha }}
          restore-keys: |
            ${{ runner.os }}-buildx-

      - name: Build sandbox image
        run: |
          docker build -t surrogate-1-sandbox:latest -f Dockerfile .

      - name: Start sandbox with docker-compose
        run: |
          docker compose up -d

      - name: Wait for health check
        run: |
          echo "Waiting for sandbox to be ready..."
          for i in {1..30}; do
            if curl -sf http://localhost:8080/health > /dev/null 2>&1; then
              echo "Sandbox is ready"
              exit 0
            fi
            echo "Waiting... ($i/30)"
            sleep 2
          done
          echo "Health check failed after 60 seconds"
          docker compose logs
          exit 1

      - name: Run integration tests
        run: |
          docker compose exec -T app go test -v ./...

      - name: Collect logs on failure
        if: failure()
        run: docker compose logs

      - name: Stop and remove sandbox
        if: always()
        run: docker compose down --volumes --remove-orphans
```
