package com.axentx.surrogate1.model;

import jakarta.persistence.*;
import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.NotNull;
import java.time.Instant;
import java.util.Objects;

/**
 * Records every distinct API request signature that the system has seen.
 * The table is append‑only – rows are never updated or deleted.
 */
@Entity
@Table(name = "api_signature_history",
       indexes = {@Index(name = "idx_signature_recorded", columnList = "signature, recorded_at")})
public final class ApiSignatureHistory {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "signature", nullable = false, length = 512)
    @NotBlank
    private final String signature;

    @Column(name = "recorded_at", nullable = false, updatable = false)
    @NotNull
    private final Instant recordedAt;

    /** JPA only */
    protected ApiSignatureHistory() {
        this.signature = null;
        this.recordedAt = null;
    }

    public ApiSignatureHistory(@NotBlank String signature, @NotNull Instant recordedAt) {
        this.signature = Objects.requireNonNull(signature, "signature");
        this.recordedAt = Objects.requireNonNull(recordedAt, "recordedAt");
    }

    // ----- getters ---------------------------------------------------------
    public Long getId() { return id; }
    public String getSignature() { return signature; }
    public Instant getRecordedAt() { return recordedAt; }
}
