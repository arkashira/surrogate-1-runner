import time
from surrogate_1.connectors.brokers.simulated import SimulatedBrokerAdapter, Order, Fill

def test_market_order_fills_immediately_at_mid():
    broker = SimulatedBrokerAdapter()
    order = Order(symbol="AAPL", side="BUY", qty=100, type="MARKET")
    t0 = time.perf_counter()
    fill = broker.send_order(order)
    elapsed = (time.perf_counter() - t0) * 1000
    assert isinstance(fill, Fill)
    assert fill.symbol == "AAPL"
    assert fill.qty == 100
    assert fill.price > 0
    assert elapsed < 200  # p95 target; local sanity

def test_limit_order_fills_when_price_touches():
    broker = SimulatedBrokerAdapter()
    broker.set_price("AAPL", 150.0)
    order = Order(symbol="AAPL", side="BUY", qty=100, type="LIMIT", limit=150.0)
    fill = broker.send_order(order)
    assert isinstance(fill, Fill)
    assert fill.price == 150.0

def test_limit_order_no_fill_when_price_not_touched():
    broker = SimulatedBrokerAdapter()
    broker.set_price("AAPL", 149.0)
    order = Order(symbol="AAPL", side="BUY", qty=100, type="LIMIT", limit=150.0)
    fill = broker.send_order(order)
    assert fill is None or getattr(fill, "filled", False) is False

def test_cancel_prevents_fill():
    broker = SimulatedBrokerAdapter()
    order = Order(symbol="AAPL", side="BUY", qty=100, type="LIMIT", limit=200.0)
    oid = broker.send_order(order).order_id
    broker.cancel(oid)
    broker.set_price("AAPL", 200.0)
    filled = broker.query(oid)
    assert filled is None or filled.filled is False

def test_query_returns_fill_state():
    broker = SimulatedBrokerAdapter()
    order = Order(symbol="AAPL", side="BUY", qty=50, type="MARKET")
    fill = broker.send_order(order)
    q = broker.query(fill.order_id)
    assert q.order_id == fill.order_id
    assert q.filled is True
