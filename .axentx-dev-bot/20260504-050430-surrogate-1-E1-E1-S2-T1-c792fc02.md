```python
import io
import json
import pytest
from fastapi import UploadFile, HTTPException
from finance_routes import ingest_csv, MAX_FILE_SIZE_BYTES

@pytest.fixture
def mock_db(monkeypatch):
    """Replace the real DB session with an in‑memory mock."""
    class DummySession:
        def __init__(self):
            self.records = []
        def add_all(self, objs):
            self.records.extend(objs)
        def commit(self):
            pass
    dummy = DummySession()
    monkeypatch.setattr('finance_routes.get_db_session', lambda: dummy)
    return dummy

def make_upload_file(content: str, filename: str = "test.csv") -> UploadFile:
    stream = io.BytesIO(content.encode())
    return UploadFile(filename=filename, file=stream, content_type="text/csv")

def test_ingest_happy_path(mock_db):
    csv_content = "date,amount,category\n2023-01-01,100,Revenue\n2023-01-02,50,Expense"
    upload = make_upload_file(csv_content)
    response = ingest_csv(upload)          # sync function for unit test
    assert response.status_code == 200
    payload = json.loads(response.body)
    assert payload["status"] == "success"
    assert payload["records_inserted"] == 2
    # Verify DB interaction
    assert len(mock_db.records) == 2
    assert mock_db.records[0].amount == 100

def test_ingest_oversized_file(mock_db):
    # Generate a payload just over the limit (5 MB + 1 byte)
    oversized_content = "a," * (MAX_FILE_SIZE_BYTES // 2 + 1)  # rough approximation
    upload = make_upload_file(oversized_content, filename="big.csv")
    with pytest.raises(HTTPException) as exc:
        ingest_csv(upload)
    assert exc.value.status_code == 413
    assert "exceeds 5 MB" in exc.value.detail.lower()

def test_ingest_malformed_csv(mock_db):
    malformed = "date,amount,category\n2023-01-01,not_a_number,Revenue"
    upload = make_upload_file(malformed)
    with pytest.raises(HTTPException) as exc:
        ingest_csv(upload)
    assert exc.value.status_code == 400
    assert "malformed csv" in exc.value.detail.lower()
```
