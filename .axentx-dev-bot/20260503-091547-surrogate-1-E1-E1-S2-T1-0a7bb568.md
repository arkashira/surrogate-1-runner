```python
from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime, timezone
from enum import Enum
from typing import Dict, FrozenSet, Optional, Set


class RequestStatus(str, Enum):
    SUBMITTED = "submitted"
    IN_PROGRESS = "in_progress"
    RESOLVED = "resolved"
    CLOSED = "closed"


# Deterministic valid forward transitions
VALID_TRANSITIONS: Dict[RequestStatus, FrozenSet[RequestStatus]] = {
    RequestStatus.SUBMITTED: frozenset({RequestStatus.IN_PROGRESS}),
    RequestStatus.IN_PROGRESS: frozenset({RequestStatus.RESOLVED, RequestStatus.CLOSED}),
    RequestStatus.RESOLVED: frozenset({RequestStatus.CLOSED}),
    RequestStatus.CLOSED: frozenset(),
}


@dataclass(frozen=True)
class StatusHistoryEntry:
    status: RequestStatus
    changed_at: datetime
    changed_by: Optional[str] = None
    reason: Optional[str] = None


class StatusTransitionError(Exception):
    pass


class RequestNotFoundError(StatusTransitionError):
    pass


class InvalidTransitionError(StatusTransitionError):
    pass


def validate_transition(current: RequestStatus, next_status: RequestStatus) -> None:
    """
    Raise InvalidTransitionError if transition is invalid.
    Idempotent: current == next_status is allowed.
    """
    if current == next_status:
        return
    allowed = VALID_TRANSITIONS.get(current, frozenset())
    if next_status not in allowed:
        raise InvalidTransitionError(
            f"Invalid transition {current.value} -> {next_status.value}. "
            f"Allowed: {[s.value for s in allowed]}"
        )


def next_status_sequence() -> Dict[RequestStatus, Set[RequestStatus]]:
    """Return mutable adjacency map for tests/inspection."""
    return {k: set(v) for k, v in VALID_TRANSITIONS.items()}
```

class TestValidateTransition:
    @pytest.mark.parametrize("state", list(RequestStatus))
    def test_idempotent_allowed(self, state):
        assert validate_transition(state, state) is True

    def test_submitted_to_in_progress(self):
        assert validate_transition(RequestStatus.SUBMITTED, RequestStatus.IN_PROGRESS) is True

    def test_in_progress_to_resolved(self):
        assert validate_transition(RequestStatus.IN_PROGRESS, RequestStatus.RESOLVED) is True

    def test_in_progress_to_closed(self):
        assert validate_transition(RequestStatus.IN_PROGRESS, RequestStatus.CLOSED) is True

    def test_submitted_to_resolved_disallowed(self):
        with pytest.raises(InvalidTransitionError) as exc:
            validate_transition(RequestStatus.SUBMITTED, RequestStatus.RESOLVED)
        assert "allowed transitions" in str(exc.value).lower()

    def test_resolved_to_closed_disallowed(self):
        with pytest.raises(InvalidTransitionError):
            validate_transition(RequestStatus.RESOLVED, RequestStatus.CLOSED)

    def test_closed_to_any_disallowed(self):
        with pytest.raises(InvalidTransitionError):

        try:

        except InvalidTransitionError as e:
            msg = str(e)
            assert "SUBMITTED" in msg or "allowed" in msg
