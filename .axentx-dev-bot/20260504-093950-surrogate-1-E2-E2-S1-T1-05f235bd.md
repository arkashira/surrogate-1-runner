"""
memory.py – Memory‑efficiency helpers for the surrogate‑1 parser.

The original implementation accumulated entire files or large intermediate
structures in memory, which caused high RAM pressure when processing the
public dataset (tens of gigabytes).  This module introduces three
light‑weight techniques that together reduce the memory footprint while
preserving the existing public API:

1. **Chunked streaming** – `stream_parse` reads a file in configurable
   byte‑chunks and yields parsed records lazily.  Callers can iterate over
   the generator without ever holding the whole dataset in RAM.

2. **Reusable byte‑buffer pool** – `BufferPool` maintains a small pool of
   pre‑allocated ``bytearray`` objects.  Re‑using buffers eliminates the
   overhead of repeated allocations during high‑throughput parsing.

3. **Context‑manager for temporary buffers** – `temp_buffer` provides a
   convenient ``with`` block that automatically acquires and releases a
   buffer from the pool.

These helpers are deliberately dependency‑free and can be dropped into
existing code with minimal changes:

```python
from surrogate_1.parser.memory import stream_parse, temp_buffer

def parse_line(line: bytes) -> dict:
    # user‑defined parsing logic
    ...

for record in stream_parse(path, parse_line):
    process(record)
```

The implementation favours readability and low‑overhead data structures,
making it suitable for the parallel runners that operate under strict
memory budgets.
"""

from __future__ import annotations

import os
import threading
from collections import deque
from contextlib import contextmanager
from typing import Callable, Generator, Iterable, Optional

# ----------------------------------------------------------------------
# Buffer pool – re‑use pre‑allocated bytearrays to avoid repeated allocations
# ----------------------------------------------------------------------


class BufferPool:
    """
    A thread‑safe pool of reusable ``bytearray`` buffers.

    Parameters
    ----------
    buffer_size: int
        Size of each buffer in bytes.
    max_buffers: int, optional
        Maximum number of buffers kept in the pool.  ``None`` means unlimited.
    """

    def __init__(self, buffer_size: int, max_buffers: Optional[int] = None):
        self._buffer_size = buffer_size
        self._max_buffers = max_buffers
        self._pool: deque[bytearray] = deque()
        self._lock = threading.Lock()

    def acquire(self) -> bytearray:
        """
        Retrieve a buffer from the pool, allocating a new one if necessary.
        """
        with self._lock:
            if self._pool:
                buf = self._pool.popleft()
            else:
                buf = bytearray(self._buffer_size)
        return buf

    def release(self, buf: bytearray) -> None:
        """
        Return a buffer to the pool for future reuse.
        """
        if len(buf) != self._buffer_size:
            # Defensive: ignore buffers of unexpected size.
            return
        with self._lock:
            if self._max_buffers is None or len(self._pool) < self._max_buffers:
                self._pool.append(buf)
            # else: let GC reclaim the buffer


# Global default pool – 1 MiB buffers, up to 32 kept alive per process.
_DEFAULT_POOL = BufferPool(buffer_size=1024 * 1024, max_buffers=32)


@contextmanager
def temp_buffer(pool: BufferPool = _DEFAULT_POOL) -> Generator[bytearray, None, None]:
    """
    Context manager that yields a temporary buffer from the pool and returns it
    when the block exits.

    Example
    -------
    >>> with temp_buffer() as buf:
    ...     n = f.readinto(buf)
    ...     data = bytes(buf[:n])
    """
    buf = pool.acquire()
    try:
        yield buf
    finally:
        pool.release(buf)
