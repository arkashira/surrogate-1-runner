```bash
#!/usr/bin/env bash
# list-date-files.sh
# Usage: ./list-date-files.sh <date> [output.json]
# Example: ./list-date-files.sh 2026-05-02 file-list.json
#
# Uses HF API list_repo_tree (non-recursive) for a single date folder
# to avoid recursive pagination on large repos. Emits array of paths
# relative to the date folder.

set -euo pipefail

DATE="${1:-$(date -u +%Y-%m-%d)}"
OUT="${2:-file-list.json}"
REPO="axentx/surrogate-1-training-pairs"

# If HF_TOKEN is set, use it; if not, tree listing for public repos
# often works without auth, but token avoids stricter unauthed limits.
echo "Listing files for ${REPO}/tree/main/${DATE} ..."

python3 - "$REPO" "$DATE" "$OUT" <<'PY'
import os, json, sys
from huggingface_hub import HfApi

repo_id = sys.argv[1]
date_folder = sys.argv[2]
out_path = sys.argv[3]

api = HfApi(token=os.getenv("HF_TOKEN", None))
# List top-level items in the date folder (non-recursive)
items = api.list_repo_tree(
    repo_id=repo_id,
    path=date_folder,
    repo_type="dataset",
    recursive=False,
)

files = []
for item in items:
    if item.type == "file":
        # store path relative to repo root (training will use resolve/main/...)
        files.append(item.path)

# Sort for deterministic ordering across runs
files.sort()

with open(out_path, "w", encoding="utf-8") as f:
    json.dump({"date": date_folder, "files": files}, f, indent=2)

print(f"Wrote {len(files)} files to {out_path}")
PY

echo "Done: ${OUT}"
```

Make executable:
```bash
chmod +x bin/list-date-files.sh
```

## 2) `lib/cdn_download.py` (new)

```python
# lib/cdn_download.py
import time
import requests
from pathlib import Path
from typing import Optional

CDN_TEMPLATE = "https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def cdn_download(
    repo: str,
    path: str,
    out_path: Path,
    max_retries: int = 5,
    backoff: float = 1.0,
) -> Path:
    """
    Download a dataset file via public CDN (no auth).
    Retries on transient failures.
    """
    url = CDN_TEMPLATE.format(repo=repo, path=path)
    out_path.parent.mkdir(parents=True, exist_ok=True)

    for attempt in range(1, max_retries + 1):
        try:
            resp = requests.get(url, stream=True, timeout=30)
            resp.raise_for_status()
            with open(out_path, "wb") as f:
                for chunk in resp.iter_content(chunk_size=8192):
                    f.write(chunk)
            return out_path
        except Exception as exc:
            if attempt == max_retries:
                raise
            sleep_sec = backoff * (2 ** (attempt - 1))
            print(f"Download failed ({exc}), retry {attempt}/{max_retries} in {sleep_sec}s: {url}")
            time.sleep(sleep_sec)
    raise RuntimeError("Unreachable")
```
