// src/lib/validation.ts
export interface ValidationError {
  field: string;
  message: string;
}

export function validateCreateRequest(body: any): ValidationError[] {
  const errors: ValidationError[] = [];

  if (!body.type || typeof body.type !== 'string' || body.type.trim() === '') {
    errors.push({ field: 'type', message: 'type is required and must be a non-empty string' });
  }

  if (!body.title || typeof body.title !== 'string' || body.title.trim() === '') {
    errors.push({ field: 'title', message: 'title is required and must be a non-empty string' });
  }

  if (!body.owner || typeof body.owner !== 'string' || body.owner.trim() === '') {
    errors.push({ field: 'owner', message: 'owner is required and must be a non-empty string' });
  }

  if (body.slaTargetHours === undefined || body.slaTargetHours === null) {
    errors.push({ field: 'slaTargetHours', message: 'slaTargetHours is required' });
  } else if (typeof body.slaTargetHours !== 'number' || Number.isNaN(body.slaTargetHours)) {
    errors.push({ field: 'slaTargetHours', message: 'slaTargetHours must be a number' });
  } else if (body.slaTargetHours <= 0 || body.slaTargetHours > 8760) {
    // max 1 year in hours as sanity bound
    errors.push({ field: 'slaTargetHours', message: 'slaTargetHours must be > 0 and <= 8760' });
  }

  return errors;
}

// src/models/request.ts
export type RequestStatus = 'open' | 'in_progress' | 'resolved' | 'closed';

export interface Request {
  id: string;
  type: string;
  title: string;
  owner: string;
  slaTargetHours: number;
  status: RequestStatus;
  createdAt: string; // ISO8601
  updatedAt: string; // ISO8601
  team?: string; // optional team grouping for metrics
}

export interface RequestWithSLA extends Request {
  slaDeadline: string; // ISO8601
}

export function computeSLADeadline(createdAt: string, slaTargetHours: number): string {
  const d = new Date(createdAt);
  d.setHours(d.getHours() + slaTargetHours);
  return d.toISOString();
}
