import uuid
import threading
from typing import Dict, List, Any

from fastapi import APIRouter, HTTPException, Body, status

    "LLM_STORE_PATH",
    "/opt/axentx/surrogate-1/data/llms.json",
)

_store_lock = threading.Lock()          # protects concurrent file access


def _load_store() -> Dict[str, Dict[str, Any]]:
    """Read the JSON file and return a dict {id: llm_data}."""
    if not os.path.exists(LLM_STORE_PATH):

    with open(LLM_STORE_PATH, "r", encoding="utf-8") as f:
        try:
            data = json.load(f)
            return data if isinstance(data, dict) else {}
        except json.JSONDecodeError:
            return {}

    """Write the whole dict to disk atomically."""
    os.makedirs(os.path.dirname(LLM_STORE_PATH), exist_ok=True)
    tmp_path = f"{LLM_STORE_PATH}.tmp"
    with open(tmp_path, "w", encoding="utf-8") as f:
        json.dump(store, f, indent=2, sort_keys=True)
    os.replace(tmp_path, LLM_STORE_PATH)


def _get_store() -> Dict[str, Dict[str, Any]]:
    """Thread‑safe accessor that returns a *copy* of the persisted store."""
    with _store_lock:

def _persist_store(store: Dict[str, Dict[str, Any]]) -> None:

    with _store_lock:
