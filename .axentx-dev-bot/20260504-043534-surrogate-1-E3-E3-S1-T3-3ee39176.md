import datetime
from .models import WorkflowHistory

def log_workflow_history(workflow_id, status, details=None):
    """
    Logs the workflow history entry for a given workflow ID with its status and optional details.
    
    :param workflow_id: Unique identifier for the workflow.
    :param status: Current status of the workflow (e.g., 'started', 'completed', 'failed').
    :param details: Optional details about the workflow status.
    """
    history_entry = WorkflowHistory(
        workflow_id=workflow_id,
        status=status,
        details=details,
        timestamp=datetime.datetime.utcnow()
    )
    history_entry.save()

def get_workflow_history(workflow_id):
    """
    Retrieves the history of a specific workflow.
    
    :param workflow_id: Unique identifier for the workflow.
    :return: List of WorkflowHistory objects representing the workflow's history.
    """
    return WorkflowHistory.objects.filter(workflow_id=workflow_id).order_by('timestamp')

def delete_workflow_history(workflow_id):
    """
    Deletes the history of a specific workflow.
    
    :param workflow_id: Unique identifier for the workflow.
    """
    WorkflowHistory.objects.filter(workflow_id=workflow_id).delete()
