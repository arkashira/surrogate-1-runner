- Single API call to list one date folder via `list_repo_tree(recursive=False)` → deterministic shard assignment by hash(slug)
- Downloads only assigned files via **HF CDN bypass** (`https://huggingface.co/datasets/.../resolve/main/...`) with no Authorization header to avoid 429 rate limits
- Streams, normalizes per-schema, dedups via central md5 store (`lib/dedup.py`), projects to `{prompt, response}`
- Writes `batches/public-merged/{DATE}/shard{N}-{HHMMSS}.jsonl` (one file per shard per run)
- Exits 0 on success; non-zero on fatal error (GitHub Actions will retry)

Why this is the highest-value incremental improvement:
- Eliminates `load_dataset(streaming=True)` mixed-schema `pyarrow.CastError` (known failure mode)
- Bypasses HF API rate limits during data loading (CDN tier has much higher limits)
- Keeps parallel 16-shard architecture and RAM isolation (7 GB/runner)
- Deterministic sharding prevents commit collisions and enables idempotent retries
- Fits <2h: small focused Python script + minimal workflow tweak

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1-training-pairs.

Usage (GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 DATE=2026-05-03 \
  HF_TOKEN=hf_xxx REPO_ID=axentx/surrogate-1-training-pairs \
  python bin/dataset-enrich.py
"""

import json
import os
import sys
import hashlib
import time
from datetime import datetime, timezone
from pathlib import Path
from typing import Dict, Any, List, Optional

import httpx

# Local dedup store
sys.path.insert(0, str(Path(__file__).parent))
try:
    from lib.dedup import DedupStore  # type: ignore
except Exception:
    DedupStore = None  # optional; if missing, dedup is skipped

HF_API_BASE = "https://huggingface.co/api"
HF_CDN_BASE = "https://huggingface.co/datasets"

DEFAULT_REPO = "axentx/surrogate-1-training-pairs"
RETRY_WAIT = 360  # seconds after 429
MAX_RETRIES = 3


def _now() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")


def _slug_from_path(path: str) -> str:
    """Stable slug used for sharding and dedup."""
    return path.strip("/").replace("/", "--")


def _hash_shard(slug: str, shard_total: int) -> int:
    """Deterministic shard assignment."""
    digest = int(hashlib.sha256(slug.encode()).hexdigest(), 16)
    return digest % shard_total


def list_date_files(
    repo_id: str,
    date_folder: str,
    token: Optional[str],
) -> List[str]:
    """
    Single API call to list files in a date folder (non-recursive).
    Retries on 429 with exponential-ish backoff.
    """
    headers = {"Authorization": f"Bearer {token}"} if token else {}
    url = f"{HF_API_BASE}/datasets/{repo_id}/tree/{date_folder}"
    params = {"recursive": "false"}

    for attempt in range(MAX_RETRIES):
        try:
            resp = httpx.get(url, headers=headers, params=params, timeout=30.0)
            if resp.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{_now()}] Rate limited 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            resp.raise_for_status()
            items = resp.json()
            # Expect list of {"path": "...", "type": "file"}
            files = [item["path"] for item in items if item.get("type") == "file"]
            print(f"[{_now()}] Listed {len(files)} files in {date_folder}", file=sys.stderr)
            return files
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 429:
                wait = RETRY_WAIT
                print(f"[{_now()}] Rate limited 429, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
