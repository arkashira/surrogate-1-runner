/**
 * Unified test suite that validates synthetic error responses against
 * Datadog’s public error format (https://docs.datadoghq.com/api/latest/errors/).
 *
 * The suite:
 *   • Compiles two compatible schemas (a minimal one and the full “errors” array).
 *   • Generates a synthetic payload for a given HTTP status code.
 *   • Runs AJV validation for each payload.
 *   • Fails with a clear diff if the payload does not conform.
 *
 * The test framework is Jest, but only the `expect(...).toBeTruthy()` line
 * would need to change to use Chai/Mocha.
 */

const Ajv = require('ajv');
const addFormats = require('ajv-formats');

// ---------------------------------------------------------------------------
// 1️⃣  JSON‑Schema definitions
// ---------------------------------------------------------------------------

// Minimal schema – matches the first candidate (error + status)
const minimalSchema = {
  type: 'object',
  required: ['error', 'status'],
  additionalProperties: false,
  properties: {
    error: {
      type: 'object',
      required: ['code', 'message'],
      additionalProperties: false,
      properties: {
        code: { type: 'string' },
        message: { type: 'string' },
        details: { type: 'object' },
      },
    },
    status: { type: 'integer', minimum: 400, maximum: 599 },
  },
};

// Full Datadog “errors” array schema – matches the second candidate
const fullDatadogSchema = {
  type: 'object',
  required: ['errors'],
  additionalProperties: false,
  properties: {
    errors: {
      type: 'array',
      minItems: 1,
      items: {
        type: 'object',
        required: ['title', 'detail'],
        additionalProperties: false,
        properties: {
          title: { type: 'string' },
          detail: { type: 'string' },
          source: {
            type: 'object',
            required: ['pointer'],
            additionalProperties: false,
            properties: {
              pointer: { type: 'string' },
            },
          },
          meta: {
            type: 'object',
            additionalProperties: true,
          },
        },
      },
    },
  },
};

// ---------------------------------------------------------------------------
// 2️⃣  AJV compiler (single instance, two validators)
// ---------------------------------------------------------------------------
const ajv = new Ajv({ allErrors: true, strict: false });
addFormats(ajv);
const validateMinimal = ajv.compile(minimalSchema);
const validateFull = ajv.compile(fullDatadogSchema);
