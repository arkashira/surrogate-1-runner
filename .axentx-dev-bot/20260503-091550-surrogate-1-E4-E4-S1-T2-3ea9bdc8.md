# workers/notify.py
import json
import logging
import smtplib
from email.message import EmailMessage
from typing import Any, Dict, List

import requests

from config import Config
from models.request import Request

logger = logging.getLogger(__name__)


def format_slack_message(request: Request, transition: Dict[str, str]) -> Dict[str, Any]:
    old_status = transition.get("from", "unknown")
    new_status = transition.get("to", "unknown")
    actor = transition.get("by", "system")
    url = request.public_status_url(Config.PUBLIC_STATUS_BASE)

    text = (
        f"*{request.title}* status updated\n"
        f"> {old_status} → *{new_status}* (by {actor})\n"
        f"<{url}|View public status>"
    )

    return {
        "channel": Config.SLACK_CHANNEL,
        "text": text,
        "blocks": [
            {
                "type": "section",
                "text": {"type": "mrkdwn", "text": text},
            },
        ],
    }


def send_slack(payload: Dict[str, Any]) -> None:
    if not Config.SLACK_WEBHOOK_URL:
        logger.warning("SLACK_WEBHOOK_URL not configured — skipping Slack post")
        return

    try:
        resp = requests.post(Config.SLACK_WEBHOOK_URL, json=payload, timeout=10)
        resp.raise_for_status()
