# tasks.py
from celery import shared_task
from .models import Request
from .utils import send_notification

@shared_task
def enqueue_notification(request_id, transition_details, recipients):
    """Enqueue a notification job with request_id, transition details, and recipients"""
    request = Request.objects.get(id=request_id)
    send_notification.delay(request, transition_details, recipients)
```

# notify.py
import os
from celery import shared_task
from django.core.mail import send_mail
from .utils import send_slack_notification, send_email_notification

@shared_task
def send_notifications(request, transition_details, recipients):
    """Background worker that sends notifications to Slack and emails to stakeholders"""
    send_slack_notification.delay(request, transition_details)
    send_email_notification.delay(request, transition_details, recipients)

def send_slack_notification(request, transition_details):
    """Send a formatted message to configured Slack webhook/channel per team"""
    # Assuming we have a Slack webhook URL and channel configured in the settings
    slack_webhook_url = os.environ.get('SLACK_WEBHOOK_URL')
    slack_channel = os.environ.get('SLACK_CHANNEL')
    # Format the message
    message = f"Request {request.id} has transitioned to {transition_details['new_status']}"
    # Send the notification to Slack
    import requests
    requests.post(slack_webhook_url, json={'text': message, 'channel': slack_channel})

def send_email_notification(request, transition_details, recipients):
    """Send email to request owner and subscribers with public status link"""
    # Assuming we have a mail server configured in the settings
    mail_server = os.environ.get('MAIL_SERVER')
    mail_port = os.environ.get('MAIL_PORT')
    mail_username = os.environ.get('MAIL_USERNAME')
    mail_password = os.environ.get('MAIL_PASSWORD')
    # Format the email
    subject = f"Request {request.id} has transitioned to {transition_details['new_status']}"
    body = f"Please visit the public status page for details: {request.get_status_url()}"
    # Send the email
    send_mail(subject, body, mail_username, recipients, mail_server, mail_port, mail_password)
```
