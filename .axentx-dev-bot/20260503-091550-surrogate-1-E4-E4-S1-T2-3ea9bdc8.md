def send_slack_notification(request, transition_details):

    message = f"Request {request.id} has transitioned to {transition_details['new_status']}"
    # Send the notification to Slack
    import requests
    requests.post(slack_webhook_url, json={'text': message, 'channel': slack_channel})

def send_email_notification(request, transition_details, recipients):
    """Send email to request owner and subscribers with public status link"""

    mail_password = os.environ.get('MAIL_PASSWORD')
    # Format the email

    body = f"Please visit the public status page for details: {request.get_status_url()}"
    # Send the email
    send_mail(subject, body, mail_username, recipients, mail_server, mail_port, mail_password)
