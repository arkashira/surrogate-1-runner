package logging

import (
	"crypto/sha256"
	"encoding/csv"
	"encoding/json"
	"fmt"
	"io"
	"io/ioutil"
	"log"
	"os"
	"path/filepath"
	"sync"
	"time"

	"gopkg.in/yaml.v2"
)

// Config holds logging configuration loaded from logging.yaml.
type Config struct {
	LogDir        string   `yaml:"log_dir"`        // Directory where logs are stored
	RetentionDays int      `yaml:"retention_days"` // Number of days to keep logs
	ExportFormats []string `yaml:"export_formats"` // Supported export formats (json, csv)
}

// LogEntry represents a single AI request log entry.
type LogEntry struct {
	Timestamp   time.Time `json:"timestamp"`    // UTC timestamp of the request
	UserID      string    `json:"user_id"`      // Identifier of the user making the request
	Model       string    `json:"model"`        // Model name used for the request
	PayloadHash string    `json:"payload_hash"` // SHA256 hash of the request payload
}

// logger is a singleton that manages writing log entries.
type logger struct {
	cfg   Config
	mu    sync.Mutex
	file  *os.File
	enc   *json.Encoder
	ready bool
}

var (
	globalLogger = &logger{}
	onceLoad     sync.Once
)

// init loads configuration and prepares the logger.
func init() {
	onceLoad.Do(func() {
		if err := loadConfig(); err != nil {
			log.Printf("AI logger config load error: %v", err)
			return
		}
		if err := globalLogger.rotateFile(); err != nil {
			log.Printf("AI logger file init error: %v", err)
			return
		}
		// Perform cleanup of old logs on startup.
		if err := CleanupOldLogs(); err != nil {
			log.Printf("AI logger cleanup error: %v", err)
		}
		globalLogger.ready = true
	})
}

// loadConfig reads /opt/axentx/surrogate-1/config/logging.yaml into globalLogger.cfg.
func loadConfig() error {
	cfgPath := "/opt/axentx/surrogate-1/config/logging.yaml"
	data, err := ioutil.ReadFile(cfgPath)
	if err != nil {
		return fmt.Errorf("read config: %w", err)
	}
	var cfg Config
	if err := yaml.Unmarshal(data, &cfg); err != nil {
		return fmt.Errorf("unmarshal yaml: %w", err)
	}
	// Ensure defaults
	if cfg.LogDir == "" {
		cfg.LogDir = "/var/log/axentx/ai"
	}
	if cfg.RetentionDays <= 0 {
		cfg.RetentionDays = 365 // default 1 year
	}
	if len(cfg.ExportFormats) == 0 {
		cfg.ExportFormats = []string{"json", "csv"}
	}
	globalLogger.cfg = cfg
	return nil
}

// rotateFile ensures the logger is writing to today's log file.
func (l *logger) rotateFile() error {
	l.mu.Lock()
	defer l.mu.Unlock()

	if l.file != nil {
		_ = l.file.Close()
	}
	filename := fmt.Sprintf("ai_%s.log", time.Now().UTC().Format("20060102"))
	fullPath := filepath.Join(l.cfg.LogDir, filename)

	if err := os.MkdirAll(l.cfg.LogDir, 0755); err != nil {
		return fmt.Errorf("mkdir log dir: %w", err)
	}
	f, err := os.OpenFile(fullPath, os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0644)
	if err != nil {
		return fmt.Errorf("open log file: %w", err)
	}
	l.file = f
	l.enc = json.NewEncoder(f)
	return nil
}

// LogAIRequest records an AI request with the required fields.
func LogAIRequest(userID, model string, payload []byte) error {
	if !globalLogger.ready {
		return fmt.Errorf("ai logger not initialized")
	}
	// Rotate if day changed
	if err := globalLogger.rotateIfNeeded(); err != nil {
		return err
	}
	hash := sha256.Sum256(payload)
	entry := LogEntry{
		Timestamp:   time.Now().UTC(),
		UserID:      userID,
		Model:       model,
		PayloadHash: fmt.Sprintf("%x", hash[:]),
	}
	globalLogger.mu.Lock()
	defer globalLogger.mu.Unlock()
	if err := globalLogger.enc.Encode(entry); err != nil {
		return fmt.Errorf("encode log entry: %w", err)
	}
	return nil
}

// rotateIfNeeded checks if the current log file matches today's date.
func (l *logger) rotateIfNeeded() error {
	expected := fmt.Sprintf("ai_%s.log", time.Now().UTC().Format("20060102"))
	current := ""
	if l.file != nil {
		current = filepath.Base(l.file.Name())
	}
	if expected != current {
		return l.rotateFile()
	}
	return nil
}
