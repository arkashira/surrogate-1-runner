    def url_must_be_unique(cls, v, values, **kwargs):

class SiteMap(BaseModel):
    root: SiteMapNode

    @validator('root')
    def check_unique_urls(cls, root, values, **kwargs):
        urls = set()

                raise ValueError(f'Duplicate URL found: {node.url}')
            urls.add(node.url)
            if node.children:
                stack.extend(node.children)
