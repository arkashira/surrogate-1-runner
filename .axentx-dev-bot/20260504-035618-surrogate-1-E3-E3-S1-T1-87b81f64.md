# Build a lightweight Python image for CI/CD synthetic data generation.
# The image installs dependencies, copies the source, and exposes a
# single entrypoint that runs the surrogate module.  It is designed to be
# invoked from CI runners via `docker build` / `docker compose up`.

FROM python:3.11-slim

# Set working directory
WORKDIR /app

# Install build dependencies (only if needed by any wheels)
RUN apt-get update && apt-get install -y --no-install-recommends \
        build-essential \
    && rm -rf /var/lib/apt/lists/*

# Copy and install Python dependencies
COPY requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt

# Copy the entire source tree
COPY . .

# Default command: run the surrogate CLI.  The CLI will read
# configuration from environment variables or a config file.
ENTRYPOINT ["python", "-m", "surrogate"]
CMD ["--help"]
```

```yaml
# /opt/axentx/surrogate-1/docker-compose.yml
# Simple compose file to run the surrogate image in CI/CD environments.
# It builds the image from the Dockerfile in the current directory,
# passes any environment variables defined in the runner, and mounts
# the source directory for debugging or local overrides.

version: "3.8"

services:
  surrogate:
    build:
      context: .
      dockerfile: Dockerfile
    # Environment variables can be set in the CI runner or via an
    # env_file.  The surrogate CLI will read these to configure
    # synthetic data generation.
    environment:
      - SURROGATE_CONFIG=/app/config.yaml
      - SURROGATE_LOG_LEVEL=INFO
    # Mount the source tree for local debugging; remove in production.
    volumes:
      - .:/app
    # Override the default command if needed; otherwise the Dockerfile
    # entrypoint will run the surrogate module.
