"""
Report Export Module for Disk Geometry Data
Supports CSV, JSON, and PDF export formats.
"""

import csv
import json
import os
from datetime import datetime
from typing import Any, Dict, List, Optional
from pathlib import Path

try:
    from reportlab.lib.pagesizes import letter
    from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
    from reportlab.lib.units import inch
    from reportlab.platypus import SimpleDocTemplate, Table, TableStyle, Paragraph, Spacer
    REPORTLAB_AVAILABLE = True
except ImportError:
    REPORTLAB_AVAILABLE = False


class ReportExporter:
    """Handles export of disk geometry reports in multiple formats."""

    def __init__(self, data: List[Dict[str, Any]]):
        """
        Initialize with disk geometry data.
        
        Args:
            data: List of disk geometry dictionaries
        """
        self.data = data
        self.export_timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.output_dir = Path("/opt/axentx/surrogate-1/reports")
        self.output_dir.mkdir(parents=True, exist_ok=True)

    def _get_disk_geometry_columns(self) -> List[str]:
        """Extract column names from disk geometry data."""
        if not self.data:
            return []
        return list(self.data[0].keys())

    def _format_value(self, value: Any) -> str:
        """Format a value for display in reports."""
        if value is None:
            return "N/A"
        if isinstance(value, (int, float)):
            return str(value)
        return str(value)

    def export_csv(self, filename: Optional[str] = None) -> str:
        """
        Export data to CSV format.
        
        Args:
            filename: Optional custom filename (defaults to timestamp-based)
            
        Returns:
            Path to the exported CSV file
        """
        if filename is None:
            filename = f"disk_geometry_{self.export_timestamp}.csv"
        
        filepath = self.output_dir / filename
        with open(filepath, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=self._get_disk_geometry_columns())
            writer.writeheader()
            writer.writerows(self.data)
        
        return str(filepath)

    def export_json(self, filename: Optional[str] = None, pretty: bool = True) -> str:
        """
        Export data to JSON format.
        
        Args:
            filename: Optional custom filename
            pretty: Whether to format with indentation
            
        Returns:
            Path to the exported JSON file
        """
        if filename is None:
            filename = f"disk_geometry_{self.export_timestamp}.json"
        
        filepath = self.output_dir / filename
        with open(filepath, 'w', encoding='utf-8') as f:
            if pretty:
                json.dump(self.data, f, indent=2, ensure_ascii=False)
            else:
                json.dump(self.data, f, ensure_ascii=False)
        
        return str(filepath)

    def export_pdf(self, filename: Optional[str] = None) -> str:
        """
        Export data to PDF format.
        
        Args:
            filename: Optional custom filename
            
        Returns:
            Path to the exported PDF file
            
        Raises:
            ImportError: If reportlab is not installed
        """
        if not REPORTLAB_AVAILABLE:
            raise ImportError("reportlab is required for PDF export. Install with: pip install reportlab")

            filename = f"disk_geometry_{self.export_timestamp}.pdf"
