import { NotificationService, SlackProvider, EmailProvider } from '../src/notification/service';

describe('NotificationService', () => {

  });

  test('registerProvider registers a provider', () => {
    const slack = new SlackProvider({ url: 'https://hooks.slack.com/...'});
    service.registerProvider('slack', slack);
    expect(service.providers.slack).toBe(slack);
  });

  test('sendNotification dispatches to enabled providers', async () => {
    const mockSlack = { send: jest.fn().mockResolvedValue(true) };
    const mockEmail = { send: jest.fn().mockResolvedValue(true) };

    service.registerProvider('slack', mockSlack);
    service.registerProvider('email', mockEmail);

    const payload = { taskId: 42, title: 'Fix bug', status: 'assigned', assignee: 'alice' };
    await service.sendNotification('workflow-1', payload);

    expect(mockSlack.send).toHaveBeenCalledWith(expect.objectContaining(payload));
    expect(mockEmail.send).toHaveBeenCalledWith(expect.objectContaining(payload));
  });

  test('sendNotification logs error but continues', async () => {
    const failing = { send: jest.fn().mockRejectedValue(new Error('boom')) };
    const passing = { send: jest.fn().mockResolvedValue(true) };

    service.registerProvider('fail', failing);
    service.registerProvider('pass', passing);

    const payload = { taskId: 1 };
    await expect(service.sendNotification('wf', payload)).resolves.not.toThrow();

    expect(failing.send).toHaveBeenCalled();
    expect(passing.send).toHaveBeenCalled();
  });
});
