const CDN_ROOT = 'https://huggingface.co/datasets';

// Exponential backoff with jitter; maxBackoff in seconds
async function retryFetch(url: string, opts: RequestInit = {}, maxRetries = 3, maxBackoff = 30) {
  let lastErr: any;
  for (let i = 0; i <= maxRetries; i++) {
    try {
      const res = await fetch(url, { ...opts, credentials: 'omit' });

      // 429: respect Retry-After when present; otherwise backoff
      if (res.status === 429) {
        const retryAfterRaw = res.headers.get('retry-after');
        const retryAfter = Math.min(
          maxBackoff,
          Number.isFinite(Number(retryAfterRaw)) ? Number(retryAfterRaw) : 30 * (1 + Math.random())
        );
        await new Promise((r) => setTimeout(r, retryAfter * 1000));
        continue;
      }

      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return res;
    } catch (err) {
      lastErr = err;
      if (i < maxRetries) {
        const backoff = Math.min(maxBackoff, (2 ** i) * 1000 + Math.random() * 500);
        await new Promise((r) => setTimeout(r, backoff));
      }
    }
  }
  throw lastErr;
}

export function resolveCdnUrl({ repo, revision = 'main', filePath }: { repo: string; revision?: string; filePath: string }) {
  // filePath may contain slashes; encode each segment for safety
  const encodedPath = filePath.split('/').map(encodeURIComponent).join('/');
  return `${CDN_ROOT}/${repo}/${revision}/${encodedPath}`;
}

export async function fetchDatasetFile({ repo, revision = 'main', filePath }: { repo: string; revision?: string; filePath: string }) {
  const url = resolveCdnUrl({ repo, revision, filePath });
  const res = await retryFetch(url, {}, 3, 30);
  return res;
}
