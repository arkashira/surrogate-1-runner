```python
def test_forecast_accuracy(historical_data):
    engine = MovingAverageForecastEngine(historical_data)
    forecasts = engine.generate(days_ahead=7)

    # Calculate MAPE for the forecasts
    mape = calculate_mape(forecasts, historical_data[-7:])
    assert mape < 5.0

def calculate_mape(forecasts, actuals):
    # Calculate MAPE between forecasts and actuals
    errors = [
        abs((forecast["predicted_amount"] - actual.amount) / actual.amount)
        for forecast, actual in zip(forecasts, actuals)
    ]
    mape = sum(errors) / len(errors) * 100
    return mape
```
