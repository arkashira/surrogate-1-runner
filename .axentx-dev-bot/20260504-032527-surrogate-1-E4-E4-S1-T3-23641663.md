package sdk

import (
	"bytes"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"os"
	"time"
)

// ---------------------------------------------------------------------
// Configuration
// ---------------------------------------------------------------------

// Config holds SDK configuration.
type Config struct {
	BaseURL string        // e.g. http://localhost:8080
	APIKey  string        // optional bearer token
	Timeout time.Duration // HTTP client timeout
}

// DefaultConfig reads environment variables and falls back to defaults.
func DefaultConfig() *Config {
	return &Config{
		BaseURL: getEnv("SURROGATE_API_URL", "http://localhost:8080"),
		APIKey:  getEnv("SURROGATE_API_KEY", ""),
		Timeout: parseDuration(getEnv("SURROGATE_TIMEOUT", "30s")),
	}
}

// ---------------------------------------------------------------------
// Client
// ---------------------------------------------------------------------

// Client is a thin wrapper around http.Client that knows the API.
type Client struct {
	baseURL    string
	apiKey     string
	httpClient *http.Client
}

// NewClient creates a new SDK client.
func NewClient(cfg *Config) *Client {
	return &Client{
		baseURL: cfg.BaseURL,
		apiKey:  cfg.APIKey,
		httpClient: &http.Client{
			Timeout: cfg.Timeout,
		},
	}
}

// ---------------------------------------------------------------------
// Data models
// ---------------------------------------------------------------------

// WorkerStatus represents the status of a worker runner.
type WorkerStatus struct {
	WorkerID   string    `json:"worker_id"`
	ShardID    int       `json:"shard_id"`
	Status     string    `json:"status"` // idle, running, completed, failed
	Records    int       `json:"records_processed"`
	StartedAt  time.Time `json:"started_at,omitempty"`
	CompletedAt time.Time `json:"completed_at,omitempty"`
	Error      string    `json:"error,omitempty"`
}

// JobRequest represents a new processing job.
type JobRequest struct {
	ShardID   int               `json:"shard_id"`
	ShardHash string            `json:"shard_hash"`
	Dataset   string            `json:"dataset"`
	Options   map[string]string `json:"options,omitempty"`
}

// JobResponse is the API response for a job.

	JobID   string `json:"job_id"`
	Status  string `json:"status"`
	Message string `json:"message"`
}

// UploadRequest represents a dataset upload request.

	ShardID  int    `json:"shard_id"`
	Content  string `json:"content"`
	Format   string `json:"format"` // json, parquet, csv
	Compress bool   `json:"compress"`
}

// UploadResponse is the API response for an upload.
type UploadResponse struct {
	UploadID string `json:"upload_id"`
	URL      string `json:"url"`
	Size     int64  `json:"size"`
}

func TestDoMockRequest_Success(t *testing.T) {
    // Mock server
    ts := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        if r.URL.Path != "/mock" || r.Method != http.MethodPost {
            t.Fatalf("Unexpected request")
        }
        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(http.StatusOK)
        w.Write([]byte(`{"status":"ok","data":"hello"}`))
    }))
    defer ts.Close()

    client := NewClient(ts.URL, "key")
    resp, err := client.DoMockRequest(context.Background(), map[string]string{"msg":"hi"})
    if err != nil { t.Fatalf("unexpected error: %v", err) }
    if resp.Status != "ok" || resp.Data != "hello" {
        t.Errorf("unexpected response: %+v", resp)
    }
}

func TestDoMockRequest_BadStatus(t *testing.T) {
    ts := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.WriteHeader(http.StatusBadRequest)
        w.Write([]byte(`{"error":"bad"}`))
    }))
    defer ts.Close()

    client := NewClient(ts.URL, "key")
    _, err := client.DoMockRequest(context.Background(), nil)
    if err == nil || !strings.Contains(err.Error(), "400") {
        t.Errorf("expected HTTP 400 error, got %v", err)
    }
}
