 * Error handler class for managing MapView errors

  constructor() {
    this.errorListeners = [];
    this.errorLog = [];
    this.maxLogSize = 100;

      maxRetries: 3,
      backoffMs: 2000,
    });

      backoffMs: 500,

      backoffMs: 0,
