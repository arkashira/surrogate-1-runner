// ...

class RequestService {
  // ...

  async getRequest(teamId: string, requestId: string): Promise<Request> {
    // ...
    const request = await this.getRequestFromDatabase(teamId, requestId);
    // Compute risk label
    const slaTarget = request.slaTarget;
    const currentTime = Date.now() / 1000;
    const timeInCurrentStatusSeconds = currentTime - request.updatedAt;
    const riskLabel = this.computeRiskLabel(slaTarget, timeInCurrentStatusSeconds);
    // Update request record with timeInCurrentStatusSeconds and riskLabel
    request.timeInCurrentStatusSeconds = timeInCurrentStatusSeconds;
    request.riskLabel = riskLabel;
    return request;
  }

  private computeRiskLabel(slaTarget: number, timeInCurrentStatusSeconds: number): string {
    if (!slaTarget) {
      return 'unknown';
    }
    if (timeInCurrentStatusSeconds <= slaTarget) {
      return 'ok';
    } else if (timeInCurrentStatusSeconds <= slaTarget * 2) {
      return 'at-risk';
    } else {
      return 'breached';
    }
  }
}
