// lib/sla.ts (diff)
export interface Request {
  // ...
  slaTargetAt?: Date;
}

export function updateRequest(request: Request, slaTargetAt: Date): Request {
  return { ...request, slaTargetAt };
}

export function computeSlaStatus(request: Request): string {
  if (!request.slaTargetAt) {
    return 'Not Set';
  }

  const now = new Date();
  const slaStatus = now > request.slaTargetAt
    ? 'Breached'
    : now < request.slaTargetAt
    ? 'On Track'
    : 'At Risk';

  return slaStatus;
}
```
