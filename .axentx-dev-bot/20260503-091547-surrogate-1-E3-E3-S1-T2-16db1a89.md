+export type SlaStatus = 'on-track' | 'at-risk' | 'breached';
+
+export function computeSlaStatus(slaTargetAt: string | null, status: string): SlaStatus {
+  // If no SLA target, default to on-track
+  if (!slaTargetAt) {
+    return 'on-track';
+  }
+
+  const targetDate = new Date(slaTargetAt);
+  const now = new Date();
+  
+  // If target date is in the past, it's breached
+  if (targetDate < now) {
+    return 'breached';
+  }
+
+  // Calculate 20% buffer window
+  const buffer = 0.2;
+  const timeDiffMs = targetDate.getTime() - now.getTime();
+  const totalTimeMs = timeDiffMs / (1 - buffer);
+  const bufferTimeMs = totalTimeMs * buffer;
+  
+  // If we're within the buffer window, it's at risk
+  if (timeDiffMs <= bufferTimeMs) {
+    return 'at-risk';
+  }
+  
+  // Otherwise, it's on track
+  return 'on-track';
+}
