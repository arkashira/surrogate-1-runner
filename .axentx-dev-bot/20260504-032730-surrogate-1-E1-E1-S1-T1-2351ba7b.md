```python
#!/usr/bin/env python3
"""
battery_health_metrics.py
~~~~~~~~~~~~~~~~~~~~~~~~~

A lightweight, fully typed helper that reads battery health
information from the Linux sysfs tree.

Typical sysfs path: /sys/class/power_supply/BAT0/

The public API is a single function:

    get_battery_metrics() -> Optional[BatteryMetrics]

which returns a dataclass containing all available metrics, or
`None` if no battery is present.

The module also ships a tiny CLI that prints the metrics once per
second, making it useful for quick diagnostics or embedding in
monitoring dashboards.

Author: axentx-dev-bot
"""

from __future__ import annotations

import os
import time
from dataclasses import dataclass
from pathlib import Path
from typing import Optional, Dict, Any

# --------------------------------------------------------------------------- #
# 1.  Data container
# --------------------------------------------------------------------------- #
@dataclass(frozen=True)
class BatteryMetrics:
    """Container for all battery metrics that can be read from sysfs."""

    # Basic identification
    name: str
    present: bool

    # Capacity & charge
    capacity_percent: Optional[int]          # %
    charge_full: Optional[int]               # µAh
    charge_now: Optional[int]                # µAh
    voltage_now: Optional[int]               # µV
    charge_full_design: Optional[int]        # µAh

    # Wear / health
    cycle_count: Optional[int]
    health: Optional[str]                    # e.g. "Good", "Overheat", "Dead"

    # Temperature
    temp_deciKelvin: Optional[int]           # deciKelvin (10 K units)
    temp_celsius: Optional[float]            # °C (derived)

    # Current & power
    current_now: Optional[int]               # µA (negative = discharging)
    power_now: Optional[int]                 # µW

    # Status
    status: Optional[str]                    # "Charging", "Discharging", "Full", ...

    # Technology
    technology: Optional[str]                # "Li-ion", "Li-poly", ...

    # --------------------------------------------------------------------- #
    # Derived helpers
    # --------------------------------------------------------------------- #
    @property
    def is_charging(self) -> bool:
        return self.status == "Charging"

    @property
    def is_discharging(self) -> bool:
        return self.status == "Discharging"

    @property
    def is_full(self) -> bool:
        return self.status == "Full"

    @property
    def wear_percentage(self) -> Optional[float]:
        """
        Rough wear estimate: 1 - (charge_now / charge_full_design)
        Returns None if design capacity is unknown.
        """
        if self.charge_full_design and self.charge_now:
            return 100.0 * (1 - self.charge_now / self.charge_full_design)
        return None
