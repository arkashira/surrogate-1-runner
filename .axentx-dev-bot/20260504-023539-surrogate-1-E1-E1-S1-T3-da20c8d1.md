# ui/alerts.py
"""
FastAPI router for AlertForge consolidated alerts.

This module provides:
- Configuration via `AlertForgeSettings`.
- Asynchronous fetching of alerts from Prometheus and Grafana.
- Consolidation logic that groups alerts by `alertname` and label set.
- A single view endpoint `/alerts/consolidated` that returns the
  consolidated alerts as JSON and renders a simple HTML page.
"""

from __future__ import annotations

import asyncio
from collections import defaultdict
from typing import Any, Dict, Iterable, List

import httpx
from fastapi import APIRouter, Depends, FastAPI, Request, status
from fastapi.responses import HTMLResponse, JSONResponse
from pydantic import BaseModel, BaseSettings, Field

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #
class AlertForgeSettings(BaseSettings):
    prometheus_url: str = Field(
        "http://localhost:9090", env="PROMETHEUS_URL"
    )
    grafana_url: str = Field(
        "http://localhost:3000", env="GRAFANA_URL"
    )
    consolidation_threshold: int = Field(
        1, env="CONSOLIDATION_THRESHOLD"
    )  # minimum number of alerts to consider a group

    class Config:
        env_file = ".env"


def get_settings() -> AlertForgeSettings:
    return AlertForgeSettings()


# --------------------------------------------------------------------------- #
# Data models
# --------------------------------------------------------------------------- #
class Alert(BaseModel):
    status: str
    labels: Dict[str, str]
    annotations: Dict[str, str]
    startsAt: str
    endsAt: str | None = None
    generatorURL: str | None = None
    fingerprint: str | None = None


class ConsolidatedAlert(BaseModel):
    alertname: str
    labels: Dict[str, str]
    annotations: Dict[str, str]
    startsAt: str
    endsAt: str | None
    generatorURLs: List[str]
    fingerprints: List[str]
    count: int


# --------------------------------------------------------------------------- #
# Helper functions
# --------------------------------------------------------------------------- #
async def fetch_prometheus_alerts(client: httpx.AsyncClient, url: str) -> List[Alert]:
    """Fetch alerts from Prometheus API."""
    resp = await client.get(f"{url}/api/v1/alerts")
    resp.raise_for_status()
    data = resp.json()
    # Prometheus returns {"status":"success","data":{"alerts":[...]} }
    alerts = data.get("data", {}).get("alerts", [])
    return [Alert(**a) for a in alerts]


async def fetch_grafana_alerts(client: httpx.AsyncClient, url: str) -> List[Alert]:
    """Fetch alerts from Grafana API."""
    # Grafana's API for alerts is /api/alerts
    resp = await client.get(f"{url}/api/alerts")
    resp.raise_for_status()
    alerts = resp.json()
    # Grafana returns a list of alerts directly
    return [Alert(**a) for a in alerts]


def consolidate_alerts(alerts: Iterable[Alert], threshold: int) -> List[ConsolidatedAlert]:
    """
    Group alerts by alertname and label set. Only groups with size >= threshold
    are returned as ConsolidatedAlert. Others are returned as individual alerts.
    """
    groups: Dict[str, List[Alert]] = defaultdict(list)

    for alert in alerts:
        key = alert.labels.get("alertname", "unknown")
        # Use frozenset of label items to differentiate groups
        label_key = frozenset(alert.labels.items())
        groups[(key, label_key)].append(alert)

    consolidated: List[ConsolidatedAlert] = []

    for (alertname, _), group in groups.items():
        if len(group) >= threshold:
            # Merge annotations and generator URLs
            merged_annotations = {}
            merged_generator_urls = []
            merged_fingerprints = []
            for a in group:
                merged_annotations.update(a.annotations)
                if a.generatorURL:
                    merged_generator_urls.append(a.generatorURL)
