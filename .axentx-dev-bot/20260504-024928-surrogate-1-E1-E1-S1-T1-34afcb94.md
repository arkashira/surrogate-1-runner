// dynamic and not part of the API contract.

	Method  string            `json:"method"`

// Only headers listed in allowedHeaders are included.  Query
// parameters are taken from the first value of each key.
func ExtractSignature(r *http.Request, allowedHeaders []string) APISignature {
	// Normalise headers
	headers := make(map[string]string, len(allowedHeaders))
	for _, h := range allowedHeaders {
		if v := r.Header.Get(h); v != "" {
			headers[h] = v
		}
	}

	// Normalise query parameters
	params := make(map[string]string, len(r.URL.Query()))
	for k, vs := range r.URL.Query() {
		if len(vs) > 0 {
			params[k] = vs[0]
		}
	}

	return APISignature{
		Method:  r.Method,
		Path:    r.URL.Path,
		Headers: headers,
		Params:  params,
	}
}
