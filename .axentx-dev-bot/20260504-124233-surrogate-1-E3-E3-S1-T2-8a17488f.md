// /opt/axentx/surrogate-1/nanocoder/tests/health.test.ts
import { getHealthPayload } from '../src/health';
import { ParserRegistry } from '../src/parserRegistry';
import { ConflictTracker } from '../src/conflictTracker';

jest.mock('../src/parserRegistry');
jest.mock('../src/conflictTracker');

describe('getHealthPayload', () => {
  const mockRegistry = ParserRegistry as jest.Mocked<typeof ParserRegistry>;
  const mockTracker = ConflictTracker as jest.Mocked<typeof ConflictTracker>;

  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('returns correct structure with active parsers', async () => {
    mockRegistry.getActiveCount.mockResolvedValue(5);
    mockTracker.getLastConflict.mockResolvedValue(new Date('2026-05-01T12:00:00Z'));
    mockTracker.getCountLast24h.mockResolvedValue(3);

    const payload = await getHealthPayload();

    expect(payload).toEqual({
      active_parsers: 5,
      last_conflict_timestamp: '2026-05-01T12:00:00.000Z',
      conflict_count_last_24h: 3,
    });
  });

  test('handles no conflicts gracefully', async () => {
    mockRegistry.getActiveCount.mockResolvedValue(0);
    mockTracker.getLastConflict.mockResolvedValue(null);
    mockTracker.getCountLast24h.mockResolvedValue(0);

    const payload = await getHealthPayload();

    expect(payload).toEqual({
      active_parsers: 0,
      last_conflict_timestamp: null,
      conflict_count_last_24h: 0,
    });
  });

  test('throws if registry fails', async () => {
    mockRegistry.getActiveCount.mockRejectedValue(new Error('db'));
    await expect(getHealthPayload()).rejects.toThrow('db');
  });
});
