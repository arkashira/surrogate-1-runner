- **AC1**: The `/parser-health` endpoint returns a JSON object containing the keys `active_parsers`, `last_conflict_timestamp`, and `conflict_count_last_24h`.  
- **AC2**: `active_parsers` is an integer ≥ 0 representing the number of parser instances currently running.  
- **AC3**: `last_conflict_timestamp` is an ISO‑8601 UTC string or `null` if no conflicts have occurred.  
- **AC4**: `conflict_count_last_24h` is an integer ≥ 0 indicating the number of conflicts recorded in the past 24 hours.  
- **AC5**: The payload is generated fresh on every request (no stale data from caching layers).  
- **AC6**: The endpoint is guarded by the existing API authentication middleware; unauthenticated requests receive a 401 response.

## 2. Unit Tests (pseudo‑code, Jest)

```ts
// /opt/axentx/surrogate-1/nanocoder/tests/health.test.ts
import { getHealthPayload } from '../src/health'; // function that builds payload
import { ParserRegistry } from '../src/parserRegistry';
import { ConflictStore } from '../src/conflictStore';

jest.mock('../src/parserRegistry');
jest.mock('../src/conflictStore');

describe('Health Payload', () => {
  const mockRegistry = ParserRegistry as jest.Mocked<typeof ParserRegistry>;
  const mockStore = ConflictStore as jest.Mocked<typeof ConflictStore>;

  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('returns correct keys', () => {
    mockRegistry.getActiveCount.mockReturnValue(5);
    mockStore.getLastConflict.mockReturnValue(new Date('2024-05-01T12:00:00Z'));
    mockStore.getConflictCountLast24h.mockReturnValue(3);

    const payload = getHealthPayload();

    expect(payload).toHaveProperty('active_parsers');
    expect(payload).toHaveProperty('last_conflict_timestamp');
    expect(payload).toHaveProperty('conflict_count_last_24h');
  });

  test('active_parsers is non‑negative integer', () => {
    mockRegistry.getActiveCount.mockReturnValue(0);
    const payload = getHealthPayload();
    expect(typeof payload.active_parsers).toBe('number');
    expect(payload.active_parsers).toBeGreaterThanOrEqual(0);
  });

  test('last_conflict_timestamp is ISO string or null', () => {
    mockRegistry.getActiveCount.mockReturnValue(2);
    mockStore.getLastConflict.mockReturnValue(null);
    let payload = getHealthPayload();
    expect(payload.last_conflict_timestamp).toBeNull();

    mockStore.getLastConflict.mockReturnValue(new Date('2024-05-02T08:30:00Z'));
    payload = getHealthPayload();
    expect(typeof payload.last_conflict_timestamp).toBe('string');
    expect(new Date(payload.last_conflict_timestamp).toISOString()).toBe(
      '2024-05-02T08:30:00.000Z'
    );
  });

  test('conflict_count_last_24h is non‑negative integer', () => {
    mockRegistry.getActiveCount.mockReturnValue(1);
    mockStore.getConflictCountLast24h.mockReturnValue(0);
    const payload = getHealthPayload();
    expect(typeof payload.conflict_count_last_24h).toBe('number');
    expect(payload.conflict_count_last_24h).toBeGreaterThanOrEqual(0);
  });

  test('payload is fresh on each call', () => {
    mockRegistry.getActiveCount.mockReturnValueOnce(3).mockReturnValueOnce(4);
    const first = getHealthPayload();
    const second = getHealthPayload();
    expect(first.active_parsers).not.toBe(second.active_parsers);
  });
});
```
