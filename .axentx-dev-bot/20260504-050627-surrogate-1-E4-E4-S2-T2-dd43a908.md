```javascript
// file: email/templates/__tests__/monthly_summary.test.js
import { renderMonthlySummary } from '../monthly_summary';
import fs from 'fs';
import path from 'path';

describe('renderMonthlySummary', () => {
  const templatePath = path.resolve(__dirname, '../monthly_summary.html');
  const rawTemplate = fs.readFileSync(templatePath, 'utf8');

  const sampleData = {
    userName: 'Alice Founder',
    playbooksReceived: 12,
    onDemandRequests: 5,
    userId: 'usr_12345',
    email: 'alice@example.com',
    npsBaseUrl: 'https://survey.example.com/nps',
    unsubscribeBaseUrl: 'https://app.example.com/gdpr/unsubscribe',
  };

  test('replaces all placeholders with supplied data', () => {
    const html = renderMonthlySummary(rawTemplate, sampleData);
    expect(html).toContain(sampleData.userName);
    expect(html).toContain(`${sampleData.playbooksReceived}`);
    expect(html).toContain(`${sampleData.onDemandRequests}`);
    expect(html).toContain(
      `${sampleData.npsBaseUrl}?user_id=${encodeURIComponent(sampleData.userId)}`
    );
    expect(html).toContain(
      `${sampleData.unsubscribeBaseUrl}?email=${encodeURIComponent(sampleData.email)}`
    );
  });

  test('produces valid HTML (no unclosed tags)', () => {
    const html = renderMonthlySummary(rawTemplate, sampleData);
    // Simple sanity check using a DOM parser
    const parser = new DOMParser();
    const doc = parser.parseFromString(html, 'text/html');
    expect(doc.querySelectorAll('parsererror')).toHaveLength(0);
  });

  test('includes GDPR notice text in unsubscribe section', () => {
    const html = renderMonthlySummary(rawTemplate, sampleData);
    expect(html).toMatch(/We respect your privacy.*unsubscribe/i);
  });

  test('escapes HTML special characters in user‑provided fields', () => {
    const malicious = { ...sampleData, userName: '<script>alert(1)</script>' };
    const html = renderMonthlySummary(rawTemplate, malicious);
    expect(html).not.toContain('<script>');
    expect(html).toContain('&lt;script&gt;alert(1)&lt;/script&gt;');
  });
});
```
