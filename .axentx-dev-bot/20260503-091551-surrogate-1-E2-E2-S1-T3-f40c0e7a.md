        "title": "Help",
        "status": "open",
        "timeline": [],
        "sla": {"target_resolution_hours": 48, "status": "on_track", "due_at": "2026-05-04T12:00:00Z"},

        "internal_notes": "secret",
        "team_tags": ["ops"],
        "escalation_reason": "none",
        "created_by_user_id": "u-1",
    }
    result = filter_internal_fields(raw)
    assert all(k not in result for k in INTERNAL_KEYS)
    assert set(result.keys()) == {"slug", "title", "status", "timeline", "sla"}

def test_filter_internal_fields_preserves_nested_sla():
    raw = {"slug": "x", "title": "x", "status": "x", "timeline": [], "sla": {"target_resolution_hours": 1}}
