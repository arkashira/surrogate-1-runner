// feedback.test.js
const { submitFeedback, getFeedback, getMetrics } = require('../src/feedback/feedback');
const db = require('../src/db'); // mockable

describe('Feedback Service', () => {
  beforeEach(() => db.clear());

  test('submitFeedback stores record and returns id', async () => {
    const payload = { userId: 'u123', feedbackText: 'Great idea', rating: 4 };
    const result = await submitFeedback(payload);
    expect(result).toHaveProperty('id');
    const stored = await db.findById(result.id);
    expect(stored).toMatchObject(payload);
  });

  test('submitFeedback rejects missing fields', async () => {
    await expect(submitFeedback({ userId: 'u123' }))
      .rejects.toThrow('feedbackText is required');
  });

  test('submitFeedback rejects rating >5', async () => {
    await expect(submitFeedback({ userId: 'u123', feedbackText: 'x', rating: 6 }))
      .rejects.toThrow('rating must be between 0 and 5');
  });

  test('duplicate feedback within 24h returns error', async () => {
    const payload = { userId: 'u123', feedbackText: 'Same', rating: 3 };
    await submitFeedback(payload);
    await expect(submitFeedback(payload))
      .rejects.toThrow('Duplicate feedback within 24 hours');
  });

  test('getFeedback returns sorted array', async () => {
    await submitFeedback({ userId: 'u123', feedbackText: 'A', rating: 3 });
    await new Promise(r => setTimeout(r, 10)); // ensure different timestamps
    await submitFeedback({ userId: 'u123', feedbackText: 'B', rating: 4 });
    const list = await getFeedback('u123');
    expect(list.map(f => f.feedbackText)).toEqual(['B', 'A']);
  });

  test('getMetrics returns correct aggregates', async () => {
    await submitFeedback({ userId: 'u1', feedbackText: 'Nice', rating: 5 });
    await submitFeedback({ userId: 'u2', feedbackText: 'Nice', rating: 4 });
    await submitFeedback({ userId: 'u3', feedbackText: 'Good', rating: 3 });
    const metrics = await getMetrics();
    expect(metrics.totalFeedback).toBe(3);
    expect(metrics.averageRating).toBeCloseTo(4.0);
    expect(metrics.topSuggestions).toContain('Nice');
  });
});
