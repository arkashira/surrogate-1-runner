    sql = f"""
    CREATE TABLE IF NOT EXISTS {TABLE_NAME} (
        id              SERIAL PRIMARY KEY,
        file_id         UUID    NOT NULL UNIQUE,
        filename        TEXT    NOT NULL,
        file_size       BIGINT  NOT NULL,
        mime_type       TEXT    NOT NULL,
        sha256_checksum TEXT    NOT NULL,
        ingestion_timestamp TIMESTAMPTZ NOT NULL,
        created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
    );
    """
    with get_db_conn() as conn, conn.cursor() as cur:
        cur.execute(sql)
        conn.commit()


# --------------------------------------------------------------------------- #
# 3️⃣  Metadata extraction
# --------------------------------------------------------------------------- #
def calculate_sha256(file_obj) -> str:
    """Compute SHA‑256 in a streaming fashion."""
    sha = hashlib.sha256()
    for chunk in iter(lambda: file_obj.read(8192), b""):
        sha.update(chunk)
    # Reset file pointer for downstream use
    file_obj.seek(0)
    return sha.hexdigest()


def guess_mime_type(filename: str) -> str:
    """Return MIME type; fallback to application/octet-stream."""
    mime, _ = mimetypes.guess_type(filename)
    return mime or "application/octet-stream"


def extract_metadata(file_obj, filename: str, file_id: uuid.UUID | None = None) -> Dict:
    """
    Return a dict with all required metadata.
    If file_id is None, generate a new UUID.
    """
    file_obj.seek(0, os.SEEK_END)
    size = file_obj.tell()
    file_obj.seek(0)

    sha256 = calculate_sha256(file_obj)
    mime = guess_mime_type(filename)
    timestamp = datetime.now(timezone.utc)

    return {
        "file_id": file_id or uuid.uuid4(),
        "filename": filename,

        "mime_type": mime,
        "sha256_checksum": sha256,
        "ingestion_timestamp": timestamp,
    }
