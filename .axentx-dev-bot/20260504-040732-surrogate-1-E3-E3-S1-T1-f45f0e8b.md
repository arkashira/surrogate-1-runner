# /opt/axentx/surrogate-1/.github/actions/dependency-checker/action.yml
name: Dependency Checker
description: Check for dependency mismatches
inputs:
  token:
    description: GitHub token
    required: true
  repository:
    description: Repository to check
    required: true
  branch:
    description: Branch to check
    required: true
runs:
  using: composite
  steps:
    - name: Checkout code
      uses: actions/checkout@v3
    - name: Run dependency checker
      run: |
        ./entrypoint.sh --ci
      shell: bash
    - name: Create summary comment
      run: |
        echo "## Summary" >> summary.md
        echo "Missing imports:" >> summary.md
        ./entrypoint.sh --ci --output summary.md
      shell: bash
    - name: Upload summary comment
      uses: actions/upload-artifact@v3
      with:
        name: summary.md
        path: summary.md
    - name: Post summary comment
      uses: actions/github-script@v6
      with:
        github-token: ${{ inputs.token }}
        script: |
          github.issues.createComment({
            issue_number: github.context.pullRequest.number,
            owner: github.context.repo.owner,
            repo: github.context.repo.repo,
            body: github.actions.uploadArtifact('summary.md', { name: 'summary.md' })
          })
```

```bash
# /opt/axentx/surrogate-1/.github/actions/dependency-checker/entrypoint.sh
#!/bin/bash

while [ $# -gt 0 ]; do
  case $1 in
    --ci)
      CI_MODE=true
      shift
      ;;
    --output)
      OUTPUT_FILE=$2
      shift 2
      ;;
    *)
      echo "Unknown option: $1"
      exit 1
      ;;
  esac
done

if [ "$CI_MODE" = true ]; then
  # Run the CLI with --ci flag
  ./cli --ci
  if [ $? -ne 0 ]; then
    exit 1
  fi
fi

if [ -n "$OUTPUT_FILE" ]; then
  # Output a summary comment
  echo "## Summary" > $OUTPUT_FILE
  echo "Missing imports:" >> $OUTPUT_FILE
  ./cli --ci >> $OUTPUT_FILE
fi
```
