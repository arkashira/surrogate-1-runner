// /src/components/DashboardView.test.tsx
import { render, screen } from '@testing-library/react';
import DashboardView from './DashboardView';
import { setupMockServer } from '../mocks/server';

const server = setupMockServer();

describe('DashboardView Unit Tests', () => {
  beforeEach(() => {
    server.use(
      rest.get('/api/status', (req, res, ctx) => {
        return res(
          ctx.json({ status: 'Active', uptime: '99.9%' })
        );
      })
    );
  });

  it('renders the dashboard title', () => {
    render(<DashboardView />);
    expect(screen.getByText(/Surrogate-1 Status/i)).toBeInTheDocument();
  });

  it('displays status and uptime from mock API', async () => {
    render(<DashboardView />);
    expect(await screen.findByText(/Status: Active/i)).toBeInTheDocument();
    expect(await screen.findByText(/Uptime: 99.9%/i)).toBeInTheDocument();
  });

  it('renders a responsive grid layout', () => {
    render(<DashboardView />);
    const grid = screen.getByTestId('dashboard-grid');
    expect(grid).toHaveClass('grid', 'grid-cols-1', 'md:grid-cols-3');
  });

  it('shows a loading state initially', () => {
    render(<DashboardView />);
    expect(screen.getByRole('status')).toHaveTextContent(/Loading/i);
  });
});

// /src/types/status.test.ts
import { StatusResponse } from './status';

describe('Type Definitions', () => {
