```python
import json
import requests
from jinja2 import Template
import smtplib
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText

# Load configuration
with open('config.json') as f:
    config = json.load(f)

# Define SLA definitions
sla_definitions = config['sla_definitions']

# Define team channel and email
team_channel = config['team_channel']
team_email = config['team_email']

# Define the URL to fetch at-risk requests
url = 'https://surrogate-1-api.com/at-risk-requests'

# Fetch at-risk requests
response = requests.get(url)
at_risk_requests = response.json()

# Define the template for the digest email or message
template = Template('''
Subject: Daily Digest of At-Risk Requests

Dear Team,

The following requests are approaching or have breached their SLA:

{% for request in at_risk_requests %}
*   {{ request.name }} ({{ request.sla_definition }} days): {{ request.status }}
{% endfor %}

Best,
{{ team_name }}
''')

# Generate the digest email or message
digest = template.render(
    at_risk_requests=at_risk_requests,
    team_name='Surrogate-1 Team',
    sla_definition=sla_definitions['request_type_1']
)

# Send the digest to the team channel or email
if team_channel:
    # Send to team channel
    slack_token = 'YOUR_SLACK_TOKEN'
    slack_client = slack.WebClient(slack_token)
    message = 'At-risk requests: ' + ', '.join([request['name'] for request in at_risk_requests])
    slack_client.chat_postMessage(channel=team_channel, text=message)
elif team_email:
    # Send to team email
    sender_email = 'YOUR_EMAIL'
    receiver_email = team_email
    msg = MIMEMultipart()
    msg['From'] = sender_email
    msg['To'] = receiver_email
    msg['Subject'] = 'Daily Digest of At-Risk Requests'
    msg.attach(MIMEText(digest, 'plain'))
    server = smtplib.SMTP('YOUR_EMAIL_SERVER')
    server.sendmail(sender_email, receiver_email, msg.as_string())
    server.quit()
else:
    print('No team channel or email configured')
```
