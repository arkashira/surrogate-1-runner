from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from hashlib import sha256
import json
from typing import Dict, List, Optional, Set


class PipelineStage(str, Enum):
    RESEARCH = "research"
    SIGNAL = "signal"
    RISK = "risk"
    EXECUTION = "execution"


VALID_TRANSITIONS: Dict[PipelineStage, Set[PipelineStage]] = {
    PipelineStage.RESEARCH: {PipelineStage.SIGNAL},
    PipelineStage.SIGNAL: {PipelineStage.RISK},
    PipelineStage.RISK: {PipelineStage.EXECUTION},
    PipelineStage.EXECUTION: set(),
}


class InvalidPipelineError(ValueError):
    """Raised when a pipeline DAG or transition is invalid."""


@dataclass(frozen=True)
class PipelineDefinition:
    """
    Immutable pipeline definition validated as a strict linear DAG:
    research -> signal -> risk -> execution
    """
    name: str
    stages: List[PipelineStage] = field(default_factory=lambda: list(PipelineStage))

    def __post_init__(self) -> None:
        if not self.name or not self.name.strip():
            raise InvalidPipelineError("Pipeline name must be non-empty.")
        if not self.stages:
            raise InvalidPipelineError("Pipeline must contain at least one stage.")
        if len(set(self.stages)) != len(self.stages):
            raise InvalidPipelineError("Pipeline stages must be unique.")

        # Validate linear DAG ordering
        expected = list(PipelineStage)
        if self.stages != expected:
            raise InvalidPipelineError(
                f"Invalid DAG order. Expected stages {expected}, got {self.stages}."
            )

        # Validate transitions
        for i in range(len(self.stages) - 1):
            src = self.stages[i]
            dst = self.stages[i + 1]
            if dst not in VALID_TRANSITIONS.get(src, set()):
                raise InvalidPipelineError(
                    f"Invalid transition {src} -> {dst}."
