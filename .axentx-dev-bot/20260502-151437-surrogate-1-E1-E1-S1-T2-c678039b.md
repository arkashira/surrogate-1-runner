    from_stage: Optional[Stage]
    to_stage: Stage

    payload_hash: str
    previous_hash: Optional[str] = None
    event_hash: str

    @model_validator(mode="after")
    def _compute_hashes(self) -> "StageTransition":
        if not self.payload_hash:
            self.payload_hash = self._hash_payload()
        data = f"{self.previous_hash or ''}{self.payload_hash}{self.timestamp.isoformat()}{self.event_id}"
        self.event_hash = sha256(data.encode("utf-8")).hexdigest()
        return self
