from dataclasses import dataclass, field
from datetime import datetime, timezone
from hashlib import sha256
import json
from typing import Dict, List, Literal, Optional

Stage = Literal["research", "signal", "risk", "execution"]
VALID_STAGES: List[Stage] = ["research", "signal", "risk", "execution"]


@dataclass(frozen=True)
class PipelineDefinition:
    """
    Immutable DAG definition for a trading pipeline.
    Edges must progress forward in VALID_STAGES order and must not form cycles.
    """
    name: str
    stages: List[Stage]
    edges: List[tuple[Stage, Stage]]  # (from, to)
    metadata: Dict[str, str] = field(default_factory=dict)

    def validate(self) -> None:
        if not self.name or not self.name.strip():
            raise ValueError("PipelineDefinition.name is required and non-empty.")
        if len(self.stages) == 0:
            raise ValueError("PipelineDefinition.stages must not be empty.")
        if len(set(self.stages)) != len(self.stages):
            raise ValueError("PipelineDefinition.stages contains duplicates.")
        for s in self.stages:
            if s not in VALID_STAGES:
                raise ValueError(f"Invalid stage '{s}'. Must be one of {VALID_STAGES}.")

        # Build adjacency and detect cycles via Kahn's algorithm (deterministic).
        incoming: Dict[Stage, int] = {s: 0 for s in self.stages}
        adj: Dict[Stage, List[Stage]] = {s: [] for s in self.stages}

        for u, v in self.edges:
            if u not in self.stages or v not in self.stages:
                raise ValueError(f"Edge ({u} -> {v}) references unknown stage.")
            # Enforce forward-only semantics by VALID_STAGES index.
            if VALID_STAGES.index(u) >= VALID_STAGES.index(v):
                raise ValueError(
                    f"Invalid edge ({u} -> {v}): edges must progress forward in stage order."
                )
            adj[u].append(v)
            incoming[v] += 1

        q = [s for s in self.stages if incoming[s] == 0]

            # Deterministic ordering by VALID_STAGES index.
