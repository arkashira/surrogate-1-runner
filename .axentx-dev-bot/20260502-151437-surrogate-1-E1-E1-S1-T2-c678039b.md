Stage = Literal["research", "signal", "risk", "execution"]
VALID_STAGES: Tuple[Stage, ...] = ("research", "signal", "risk", "execution")


class DAGValidationError(ValueError):
    """Raised when a pipeline DAG is invalid."""


@dataclass(frozen=True)
class PipelineDefinition:
    """
    Immutable pipeline DAG definition.
    edges: list of (from_stage, to_stage)
    """
    pipeline_id: str
    edges: Tuple[Tuple[Stage, Stage], ...]
    nodes: Tuple[Stage, ...] = field(init=False)

    def __post_init__(self) -> None:
        nodes_set: Set[Stage] = set()
        normalized_edges: List[Tuple[Stage, Stage]] = []
        for u, v in self.edges:
            if u not in VALID_STAGES or v not in VALID_STAGES:
                raise DAGValidationError(
                    f"Invalid stage in edge ({u} -> {v}). Valid stages: {VALID_STAGES}"
                )
            nodes_set.add(u)
            nodes_set.add(v)
            normalized_edges.append((u, v))

        # Ensure all required stages are present
        expected = set(VALID_STAGES)
        if not expected.issubset(nodes_set):
            missing = expected - nodes_set
            raise DAGValidationError(f"Pipeline must include stages: {sorted(missing)}")

        # Deterministic node ordering
        nodes = tuple(sorted(nodes_set, key=lambda s: VALID_STAGES.index(s)))
        object.__setattr__(self, "nodes", nodes)
        object.__setattr__(self, "edges", tuple(normalized_edges))

        self._validate_dag()

    def _validate_dag(self) -> None:
        adj: Dict[Stage, List[Stage]] = {n: [] for n in self.nodes}
        indeg: Dict[Stage, int] = {n: 0 for n in self.nodes}
        for u, v in self.edges:
            adj[u].append(v)
            indeg[v] = indeg.get(v, 0) + 1

        # Kahn's algorithm for cycle detection
        q = [n for n in self.nodes if indeg[n] == 0]
        topo: List[Stage] = []
        while q:
            n = q.pop(0)
            topo.append(n)

                indeg[nb] -= 1
                if indeg[nb] == 0:
                    q.append(nb)
