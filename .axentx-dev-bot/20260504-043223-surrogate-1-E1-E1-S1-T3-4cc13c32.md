#!/bin/bash
# AWS Practice Environment Provisioning Script
# Self-service environment creation for AWS certification candidates
# Target: < 5 minutes, repeatable, self-service

set -euo pipefail

# Configuration
readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly LOG_FILE="${SCRIPT_DIR}/provision.log"
readonly PROVISION_TIMEOUT=300  # 5 minutes max

# AWS Credentials (use environment variables or IAM role)
readonly AWS_ACCESS_KEY="${AWS_ACCESS_KEY_ID:-}"
readonly AWS_SECRET_KEY="${AWS_SECRET_ACCESS_KEY:-}"
readonly AWS_REGION="${AWS_DEFAULT_REGION:-us-east-1}"

# Environment identifiers
readonly ENV_NAME="${ENV_NAME:-aws-practice-env-$(date +%Y%m%d-%H%M%S)}"
readonly VPC_CIDR="10.0.0.0/16"
readonly SUBNET_CIDR="10.0.1.0/24"

# Logging function
log() {
    local level="$1"
    shift
    local message="$*"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[${timestamp}] [${level}] ${message}" | tee -a "${LOG_FILE}"
}

# Check prerequisites
check_prerequisites() {
    log "INFO" "Checking prerequisites..."
    
    # Check AWS CLI
    if ! command -v aws &> /dev/null; then
        log "ERROR" "AWS CLI not found. Please install AWS CLI."
        exit 1
    fi
    
    # Check for required AWS credentials
    if [[ -z "${AWS_ACCESS_KEY}" && -z "${AWS_SECRET_KEY}" ]]; then
        log "WARN" "AWS credentials not set. Using IAM role or environment variables."
    fi
    
    # Check AWS CLI configuration
    if ! aws sts get-caller-identity &> /dev/null; then
        log "ERROR" "Cannot authenticate with AWS. Please configure AWS credentials."
        exit 1
    fi
    
    log "INFO" "Prerequisites check passed."
}

# Create VPC
create_vpc() {
    log "INFO" "Creating VPC: ${ENV_NAME}..."
    
    local vpc_id
    vpc_id=$(aws ec2 create-vpc \
        --region "${AWS_REGION}" \
        --cidr-block "${VPC_CIDR}" \
        --tag-specifications "ResourceType=vpc,Tags=[{Key=Name,Value=${ENV_NAME}}]" \
        --query 'Vpc.VpcId' \
        --output text)
    
    if [[ -z "${vpc_id}" ]]; then
        log "ERROR" "Failed to create VPC"
        exit 1
    fi
    
    log "INFO" "VPC created: ${vpc_id}"
    echo "${vpc_id}"
}

# Create Internet Gateway
create_internet_gateway() {
    log "INFO" "Creating Internet Gateway..."
    
    local igw_id
    igw_id=$(aws ec2 create-internet-gateway \
        --region "${AWS_REGION}" \
        --tag-specifications "ResourceType=internet-gateway,Tags=[{Key=Name,Value=${ENV_NAME}-igw}]" \
        --query 'InternetGateway.InternetGatewayId' \
        --output text)
    
    if [[ -z "${igw_id}" ]]; then
        log "ERROR" "Failed to create Internet Gateway"
        exit 1
    fi
    
    log "INFO" "Internet Gateway created: ${igw_id}"
    echo "${igw_id}"
}

# Attach Internet Gateway to VPC
attach_internet_gateway() {
    local vpc_id="$1"
    local igw_id="$2"
    
    log "INFO" "Attaching Internet Gateway to VPC..."
    
    aws ec2 attach-internet-gateway \
        --region "${AWS_REGION}" \
        --vpc-id "${vpc_id}" \
        --internet-gateway-id "${igw_id}"
    
    log "INFO" "Internet Gateway attached."
}
