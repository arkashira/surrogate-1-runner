package com.axentx.surrogate1.adaptation;

import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;

public class AdaptationSystem {
    private ExecutorService executorService;
    private HardwareMonitor hardwareMonitor;
    private PerformanceMonitor performanceMonitor;

    public AdaptationSystem() {
        this.executorService = Executors.newFixedThreadPool(4);
        this.hardwareMonitor = new HardwareMonitor();
        this.performanceMonitor = new PerformanceMonitor();
    }

    public void start() {
        executorService.submit(hardwareMonitor);
        executorService.submit(performanceMonitor);
    }

    public void stop() {
        executorService.shutdown();
    }

    private class HardwareMonitor implements Runnable {
        @Override
        public void run() {
            while (!Thread.currentThread().isInterrupted()) {
                // Simulate hardware configuration detection and adaptation
                System.out.println("Detected hardware configuration and adapting...");
                try {
                    Thread.sleep(5000); // Simulate monitoring interval
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }
        }
    }

    private class PerformanceMonitor implements Runnable {
        @Override
        public void run() {
            while (!Thread.currentThread().isInterrupted()) {
                // Simulate real-time performance monitoring and feedback

                    Thread.currentThread().interrupt();
                }
            }
        }
    }
}

describe('AdaptationSystem', () => {
  let system;
  let mockHardwareMonitor;
  let mockPerformanceMonitor;

  beforeEach(() => {
    mockHardwareMonitor = {
      getHardwareConfig: jest.fn().mockResolvedValue({ gpu: 'RTX3080', cpu: 'i9' }),
      adjustConfiguration: jest.fn().mockResolvedValue(true),
      onHardwareChange: jest.fn()
    };
    mockPerformanceMonitor = {
      start: jest.fn(),
      stop: jest.fn(),
      getLatestMetrics: jest.fn().mockReturnValue({ throughput: 0.9, latency: 120 })
    };
    system = new AdaptationSystem(mockHardwareMonitor, mockPerformanceMonitor);
  });

  test('initializes hardware config within 500ms', async () => {
    const start = Date.now();
    await system.start();
    const elapsed = Date.now() - start;
    expect(elapsed).toBeLessThanOrEqual(500);
    expect(mockHardwareMonitor.getHardwareConfig).toHaveBeenCalled();
  });

  test('provides latest metrics thread‑safe', async () => {
    await system.start();
    const metrics = system.getLatestMetrics();
    expect(metrics).toEqual({ throughput: 0.9, latency: 120 });
  });

  test('triggers adaptation on low throughput', async () => {
    mockPerformanceMonitor.getLatestMetrics.mockReturnValue({ throughput: 0.7, latency: 200 });
    await system.start();
    // wait for adaptation cycle
    await new Promise(r => setTimeout(r, 350));
    expect(mockHardwareMonitor.adjustConfiguration).toHaveBeenCalled();
  });

  test('restarts monitors on hardware change', async () => {
    await system.start();
    mockHardwareMonitor.onHardwareChange.mockImplementation(cb => cb({ gpu: 'RTX3090' }));
    // simulate hardware change event
    mockHardwareMonitor.onHardwareChange();
    await new Promise(r => setTimeout(r, 450));
    expect(mockPerformanceMonitor.stop).toHaveBeenCalled();
    expect(mockPerformanceMonitor.start).toHaveBeenCalled();
  });

  test('shuts down gracefully within 1s', async () => {
    await system.start();
    const shutdownFuture = system.shutdown();
    const elapsed = Date.now() - shutdownFuture.startTime;
    await shutdownFuture;
    expect(elapsed).toBeLessThanOrEqual(1000);
    expect(mockPerformanceMonitor.stop).toHaveBeenCalled();
  });
});
