import json
import csv
import os
from datetime import datetime
from typing import Dict, List, Any, Optional
import fpdf
from PyQt5.QtWidgets import (QDialog, QVBoxLayout, QHBoxLayout, QLabel, 
                            QComboBox, QPushButton, QFileDialog, QMessageBox,
                            QProgressBar, QGroupBox, QFormLayout, QLineEdit)
from PyQt5.QtCore import Qt, QThread, pyqtSignal

class ExportWorker(QThread):
    progress_updated = pyqtSignal(int)
    export_complete = pyqtSignal(bool, str)
    
    def __init__(self, data: List[Dict[str, Any]], export_format: str, output_path: str):
        super().__init__()
        self.data = data
        self.export_format = export_format
        self.output_path = output_path
        
    def run(self):
        try:
            if self.export_format == "csv":
                self.export_csv()
            elif self.export_format == "json":
                self.export_json()
            elif self.export_format == "pdf":
                self.export_pdf()
            self.export_complete.emit(True, f"Data exported successfully to {self.output_path}")
        except Exception as e:
            self.export_complete.emit(False, f"Export failed: {str(e)}")
    
    def export_csv(self):
        if not self.data:
            return
            
        fieldnames = self.data[0].keys()
        
        with open(self.output_path, 'w', newline='') as csvfile:
            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
            writer.writeheader()
            for i, row in enumerate(self.data):
                writer.writerow(row)
                self.progress_updated.emit(int((i + 1) / len(self.data) * 100))
    
    def export_json(self):
        with open(self.output_path, 'w') as jsonfile:
            json.dump(self.data, jsonfile, indent=4)
        self.progress_updated.emit(100)
    
    def export_pdf(self):
        pdf = fpdf.FPDF()
        pdf.add_page()
        pdf.set_font("Arial", size=12)
        
        # Add title
        pdf.cell(200, 10, txt="Disk Geometry Report", ln=True, align='C')
        pdf.ln(10)
        
        # Add timestamp
        pdf.cell(200, 10, txt=f"Generated: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}", ln=True)
        pdf.ln(10)
        
        # Add data
        for item in self.data:
            for key, value in item.items():
                pdf.cell(200, 10, txt=f"{key}: {value}", ln=True)
            pdf.ln(5)
            
            # Update progress
            progress = int((self.data.index(item) + 1) / len(self.data) * 100)
            self.progress_updated.emit(progress)
        
        pdf.output(self.output_path)
