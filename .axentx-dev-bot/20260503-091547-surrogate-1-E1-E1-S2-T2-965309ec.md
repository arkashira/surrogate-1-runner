            "resolved": {"closed"},

import pytest
from datetime import datetime, timezone
from models.request import Request
from models.status_history import StatusHistory
from services.transition_service import transition_request_status

def test_idempotent_same_status_creates_no_history(session):
    req = Request(id="req-1", status="submitted", updated_at=datetime.now(timezone.utc))
    session.add(req)

def test_valid_transition_creates_history(session):
    req = Request(id="req-2", status="submitted", updated_at=datetime.now(timezone.utc))
    session.add(req)
    session.commit()

    result = transition_request_status(session, "req-2", "in_progress")
    session.commit()

    assert result.status == "in_progress"
    assert result.updated_at > req.updated_at
    rows = session.query(StatusHistory).filter_by(request_id="req-2").all()
    assert len(rows) == 1
    h = rows[0]
    assert h.old_status == "submitted"
