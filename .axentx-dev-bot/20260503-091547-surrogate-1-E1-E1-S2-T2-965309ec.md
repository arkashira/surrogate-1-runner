CREATE TABLE IF NOT EXISTS status_history (
    id            BIGSERIAL PRIMARY KEY,
    request_id    BIGINT NOT NULL REFERENCES requests(id) ON DELETE CASCADE,
    from_status   TEXT NOT NULL,
    to_status     TEXT NOT NULL,
    created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Indexes for common access patterns
CREATE INDEX IF NOT EXISTS idx_status_history_request_id ON status_history(request_id);
CREATE INDEX IF NOT EXISTS idx_status_history_created_at ON status_history(created_at);

-- Safety: prevent duplicate transition rows for same request+from+to in short time window
-- (helpful for race protection; app-level idempotency is primary)
CREATE UNIQUE INDEX IF NOT EXISTS idx_status_history_request_from_to_unique
ON status_history(request_id, from_status, to_status)
WHERE created_at > (NOW() - INTERVAL '1 day');

    request = relationship("Request", back_populates="status_history")
