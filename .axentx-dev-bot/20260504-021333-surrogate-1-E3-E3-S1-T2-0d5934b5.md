import (
	"context"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"log"
	"net"
	"net/http"
	"os"
	"time"

	"github.com/jmoiron/sqlx"
)

// DecisionRequest is the JSON payload sent to /decision.
type DecisionRequest struct {
	FileName string `json:"file_name"`
	Approved bool   `json:"approved"`
}

// AuditLog represents a single audit record.
type AuditLog struct {
	ID           int64     `db:"id"`
	FileName     string    `db:"file_name"`
	Decision     string    `db:"decision"`
	TimestampUTC time.Time `db:"timestamp_utc"`
	ApproverIP   string    `db:"approver_ip"`
	HashPrev     string    `db:"hash_prev"`
	HashCurrent  string    `db:"hash_current"`
}

// DB is injected elsewhere in the application.
var DB *sqlx.DB

// emailTemplate holds the e‑mail body template, loaded from an env var.
var emailTemplate = os.Getenv("AUDIT_EMAIL_TEMPLATE")

// getLastHash retrieves the hash_current of the most recent audit record.
func getLastHash(ctx context.Context) (string, error) {
	var lastHash string
	err := DB.GetContext(ctx, &lastHash,
		`SELECT hash_current FROM audit_logs ORDER BY id DESC LIMIT 1`)
	if err != nil {
		if err == sqlx.ErrNotFound {
			return "", nil // no previous record
		}
		return "", err
	}
	return lastHash, nil
}

// computeHash creates a SHA‑256 hash of the audit record fields.
func computeHash(a AuditLog) string {
	h := sha256.New()
	h.Write([]byte(a.FileName))
	h.Write([]byte(a.Decision))
	h.Write([]byte(a.TimestampUTC.UTC().Format(time.RFC3339Nano)))
	h.Write([]byte(a.ApproverIP))
	h.Write([]byte(a.HashPrev))
	return hex.EncodeToString(h.Sum(nil))
}

// sendAuditEmail queues an e‑mail asynchronously. Errors are logged but not returned.
func sendAuditEmail(ctx context.Context, a AuditLog) {
	go func() {
		// In production replace this with a real e‑mail client.
		body := fmt.Sprintf(emailTemplate,
			a.FileName,
			a.Decision,
			a.TimestampUTC.UTC().Format(time.RFC3339),
			a.ApproverIP,
			fmt.Sprintf("https://audit.example.com/records/%d", a.ID),
		)
		log.Printf("[AUDIT EMAIL] To: compliance@example.com\nSubject: File %s %s\nBody:\n%s",
			a.FileName, a.Decision, body)
	}()
}
