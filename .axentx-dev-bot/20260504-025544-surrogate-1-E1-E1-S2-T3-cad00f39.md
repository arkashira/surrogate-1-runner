package api

import (
	"database/sql"
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"strconv"
	"time"

	"github.com/gorilla/mux"
)

// UpgradeRecord is the JSON representation of a row in the upgrade_history table.
type UpgradeRecord struct {
	ID          int64      `json:"id"`
	ClusterID   string     `json:"cluster_id"`
	CurrentVer  string     `json:"current_version"`
	TargetVer   string     `json:"target_version"`
	StartTime   time.Time  `json:"start_time"`
	Status      string     `json:"status"`
	CompletedAt *time.Time `json:"completed_at,omitempty"`
}

// getUpgradeHistory retrieves the requested page of upgrade history from the DB.
func getUpgradeHistory(db *sql.DB, page, pageSize int) ([]UpgradeRecord, error) {
	const query = `
		SELECT id, cluster_id, current_version, target_version,
		       start_time, status, completed_at
		FROM upgrade_history
		ORDER BY start_time DESC
		LIMIT ? OFFSET ?
	`
	rows, err := db.Query(query, pageSize, (page-1)*pageSize)
	if err != nil {
		return nil, fmt.Errorf("query: %w", err)
	}
	defer rows.Close()

	var history []UpgradeRecord
	for rows.Next() {
		var r UpgradeRecord
		var completed sql.NullTime
		if err := rows.Scan(&r.ID, &r.ClusterID, &r.CurrentVer, &r.TargetVer,
			&r.StartTime, &r.Status, &completed); err != nil {
			return nil, fmt.Errorf("scan: %w", err)
		}
		if completed.Valid {
			r.CompletedAt = &completed.Time
		}
		history = append(history, r)
	}
	if err := rows.Err(); err != nil {
		return nil, fmt.Errorf("rows: %w", err)
	}
	return history, nil
}

// upgradeHistoryHandler is the HTTP handler for `/api/upgrade/history`.
func upgradeHistoryHandler(db *sql.DB) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		// Default pagination
		page, pageSize := 1, 10

		// Optional query parameters
		if p := r.URL.Query().Get("page"); p != "" {
			if v, err := strconv.Atoi(p); err == nil && v > 0 {
				page = v
			}
		}
		if s := r.URL.Query().Get("size"); s != "" {
			if v, err := strconv.Atoi(s); err == nil && v > 0 && v <= 100 {
				pageSize = v
			}
		}

		history, err := getUpgradeHistory(db, page, pageSize)
		if err != nil {
			log.Printf("upgrade history query error: %v", err)
			http.Error(w, "internal server error", http.StatusInternalServerError)
			return
		}

		w.Header().Set("Content-Type", "application/json")
		if err := json.NewEncoder(w).Encode(history); err != nil {
			log.Printf("upgrade history encode error: %v", err)
			http.Error(w, "internal server error", http.StatusInternalServerError)
		}
	}
}

// registerUpgradeHistory registers the history endpoint on the provided router.
func registerUpgradeHistory(router *mux.Router, db *sql.DB) {
	router.HandleFunc("/api/upgrade/history", upgradeHistoryHandler(db)).
		Methods(http.MethodGet)
}
