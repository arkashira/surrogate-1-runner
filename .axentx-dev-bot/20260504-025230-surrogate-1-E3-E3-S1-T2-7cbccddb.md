const ReportGenerator = () => {

import cron from 'node-cron';

const schedules = new Map();

export const scheduleReport = (cronExpression, taskFn) => {
  const task = cron.schedule(cronExpression, taskFn, {
    scheduled: true,
  });
  const id = `${Date.now()}-${Math.random()}`;
  schedules.set(id, task);
  return id;
};

export const cancelSchedule = (id) => {
  const task = schedules.get(id);
  if (task) {
    task.stop();
    schedules.delete(id);
  }
};
