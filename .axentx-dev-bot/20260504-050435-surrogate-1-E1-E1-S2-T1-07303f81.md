# src/dashboard.py
"""
Marketing KPI Dashboard – production‑ready Flask app.

Features:
* `/dashboard` – renders the KPI page.
* `/api/kpis` – returns the latest KPI snapshot (JSON).
* `/api/kpis/stream` – SSE endpoint that pushes updates every 5 s.
* External KPI source is fetched via `requests` with retry logic.
* Optional Redis cache for last‑known values (fallback if API fails).
"""

from __future__ import annotations

import json
import time
from datetime import datetime, timezone
from typing import Dict

import requests
from flask import (
    Flask,
    Response,
    jsonify,
    render_template,
    stream_with_context,
    url_for,
)
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

# --------------------------------------------------------------------------- #
# Configuration – keep this in a .env file or environment variables
# --------------------------------------------------------------------------- #
API_ENDPOINT = "https://api.example.com/kpi"  # Replace with real endpoint
API_TIMEOUT = 3.0  # seconds
STREAM_INTERVAL = 5.0  # seconds
REDIS_URL = None  # e.g. "redis://localhost:6379/0" – optional

# --------------------------------------------------------------------------- #
# Flask app
# --------------------------------------------------------------------------- #
app = Flask(__name__, template_folder="templates")


# --------------------------------------------------------------------------- #
# HTTP client with retries
# --------------------------------------------------------------------------- #
session = requests.Session()
retry = Retry(
    total=3,
    backoff_factor=0.5,
    status_forcelist=[502, 503, 504],
    allowed_methods={"GET"},
)
adapter = HTTPAdapter(max_retries=retry)
session.mount("https://", adapter)
session.mount("http://", adapter)


def fetch_kpis_from_api() -> Dict[str, float]:
    """
    Call the external KPI API and return a dict.
    Raises requests.HTTPError on failure.
    """
    resp = session.get(API_ENDPOINT, timeout=API_TIMEOUT)
    resp.raise_for_status()
    data = resp.json()
    # Normalise keys to our internal names
    return {
        "acquisition_cost": float(data.get("cac", 0)),
        "lifetime_value": float(data.get("clv", 0)),
        "retention_rate": float(data.get("retention_rate", 0)),
    }


def get_latest_kpis() -> Dict[str, float]:
    """
    Return the latest KPI snapshot.
    Tries the API first; falls back to a cached value if available.
    """
    try:
        kpis = fetch_kpis_from_api()
    except Exception as exc:
        app.logger.warning("API fetch failed: %s", exc)

            kpis = {"acquisition_cost": 0.0, "lifetime_value": 0.0, "retention_rate": 0.0}
    # Persist to cache for next request
