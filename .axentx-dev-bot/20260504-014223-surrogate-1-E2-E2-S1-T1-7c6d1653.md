## 2. Unit Tests (Jest – TypeScript)

```ts
import { ContextExtractor } from './ContextExtractor';
import * as vscode from 'vscode';

describe('ContextExtractor', () => {
  let extractor: ContextExtractor;
  let mockEditor: vscode.TextEditor;

  beforeEach(() => {
    extractor = new ContextExtractor();
    mockEditor = {
      document: {
        uri: { fsPath: '/path/to/file.ts' },
        getText: jest.fn(),
        lineCount: 200,
        lineAt: jest.fn(),
      } as any,
      selection: { active: { line: 100, character: 5 } } as any,
    } as any;
  });

  test('extracts full file content', () => {
    const content = 'const a = 1;\nconst b = 2;';
    (mockEditor.document.getText as jest.Mock).mockReturnValue(content);

    const ctx = extractor.extract(mockEditor);
    expect(ctx.fileContent).toBe(content);
  });

  test('detects language from document languageId', () => {
    mockEditor.document.languageId = 'typescript';
    const ctx = extractor.extract(mockEditor);
    expect(ctx.language).toBe('typescript');
  });

  test('captures cursor line and character', () => {
    const ctx = extractor.extract(mockEditor);
    expect(ctx.cursor).toEqual({ line: 100, character: 5 });
  });

  test('returns 50 lines before and after cursor', () => {
    const lines = Array.from({ length: 200 }, (_, i) => `line ${i + 1}`);
    (mockEditor.document.getText as jest.Mock).mockImplementation((range) => {
      const start = range.start.line;
      const end = range.end.line;
      return lines.slice(start, end + 1).join('\n');
    });

    const ctx = extractor.extract(mockEditor);
    const contextLines = ctx.context.split('\n');
    expect(contextLines.length).toBeLessThanOrEqual(101); // 50 before + 1 cursor + 50 after
    expect(contextLines[0]).toBe('line 51'); // 50 lines before line 100
  });

  test('handles file start boundary', () => {
    mockEditor.selection.active.line = 10;
    const ctx = extractor.extract(mockEditor);
    const contextLines = ctx.context.split('\n');
    expect(contextLines[0]).toBe('line 1'); // starts at file start
  });

  test('handles file end boundary', () => {
    mockEditor.selection.active.line = 195;
    const ctx = extractor.extract(mockEditor);
    const contextLines = ctx.context.split('\n');
    expect(contextLines[contextLines.length - 1]).toBe('line 200'); // ends at file end
  });

  test('returns null when no active editor', () => {
    expect(() => extractor.extract(undefined as any)).toThrow('No active editor');
  });
});
```
