from .database import db

class Account(db.Model):
    """A user‑owned account that holds a credit balance."""
    __tablename__ = "account"

    id = db.Column(db.Integer, primary_key=True)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)

    # relationship back‑ref is defined on User (not needed for this endpoint)
    credit_balance = db.relationship(
        "CreditBalance", uselist=False, back_populates="account"
    )


class CreditBalance(db.Model):
    """Persisted credit counters for an account."""
    __tablename__ = "credit_balance"

    id = db.Column(db.Integer, primary_key=True)
    account_id = db.Column(
        db.Integer, db.ForeignKey('account.id'), nullable=False, unique=True
    )
    account = db.relationship("Account", back_populates="credit_balance")

    monthly_total = db.Column(db.Float, nullable=False)
    monthly_used  = db.Column(db.Float, nullable=False)
    bulk_total    = db.Column(db.Float, nullable=False)
    bulk_used     = db.Column(db.Float, nullable=False)
```
