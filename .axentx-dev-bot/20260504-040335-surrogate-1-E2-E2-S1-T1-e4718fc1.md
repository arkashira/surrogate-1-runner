describe('Dashboard component', () => {

  });

  test('renders loading spinner initially', () => {
    render(<Dashboard />);
    expect(screen.getByTestId('loading-spinner')).toBeInTheDocument();
  });

  test('displays workflows after fetch', async () => {
    const mockWorkflows = [{ id: 1, status: 'running', agents: [{ name: 'gpt-4', status: 'running' }] }];
    jest.spyOn(global, 'fetch').mockResolvedValue({
      json: () => Promise.resolve(mockWorkflows),
    });

    render(<Dashboard />);
    const row = await screen.findByText('gpt-4');
    expect(row).toBeInTheDocument();
  });

  test('updates status on WebSocket message', async () => {
    const ws = new WebSocketMock();
    jest.spyOn(window, 'WebSocket').mockImplementation(() => ws);

    render(<Dashboard />);
    await screen.findByText('running'); // initial status

    ws.emit('message', JSON.stringify({ workflowId: 1, agent: 'gpt-4', status: 'completed' }));
    const updated = await screen.findByText('completed');
    expect(updated).toBeInTheDocument();
  });

  test('filters workflows by status', async () => {
    // ...setup mock data with multiple statuses
    render(<Dashboard />);
    fireEvent.change(screen.getByLabelText('Status filter'), { target: { value: 'failed' } });
    const rows = await screen.findAllByTestId('workflow-row');
    rows.forEach(row => expect(row).toHaveTextContent('failed'));
  });

  test('redirects unauthenticated users', () => {
    jest.spyOn(require('../src/auth'), 'useAuth').mockReturnValue({ user: null });
    const { container } = render(<Dashboard />);

  });
});
