from __future__ import annotations

import os
import json
import time
import uuid
import sqlite3
from dataclasses import dataclass, asdict
from datetime import datetime, timedelta, timezone
from typing import Optional, List, Dict, Any

DB_PATH = os.getenv("AUDIT_DB_PATH", "/opt/axentx/surrogate-1/data/audit.db")
RETENTION_DAYS = int(os.getenv("AUDIT_RETENTION_DAYS", "90"))

@dataclass(frozen=True)
class RunLog:
    run_id: str
    pipeline_id: str
    node_id: str
    inputs_hash: str
    outputs_hash: str
    start_ts: float          # monotonic-ish epoch seconds with ms precision
    end_ts: Optional[float]
    node_version: str
    status: str              # "running" | "succeeded" | "failed" | "cancelled"
    metadata: Dict[str, Any]

    @staticmethod
    def create(
        pipeline_id: str,
        node_id: str,
        inputs_hash: str,
        outputs_hash: str,
        node_version: str,
        status: str = "running",
        run_id: Optional[str] = None,
        start_ts: Optional[float] = None,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> "RunLog":
        return RunLog(
            run_id=run_id or str(uuid.uuid4()),
            pipeline_id=pipeline_id,
            node_id=node_id,
            inputs_hash=inputs_hash,
            outputs_hash=outputs_hash,
            start_ts=start_ts or time.time(),
            end_ts=None,
            node_version=node_version,
            status=status,
            metadata=metadata or {},
        )

    def complete(self, status: str, end_ts: Optional[float] = None) -> "RunLog":
        # immutable update: produce new instance
        return RunLog(
            run_id=self.run_id,
            pipeline_id=self.pipeline_id,
            node_id=self.node_id,
            inputs_hash=self.inputs_hash,
            outputs_hash=self.outputs_hash,
            start_ts=self.start_ts,
            end_ts=end_ts or time.time(),
            node_version=self.node_version,
            status=status,
            metadata=self.metadata,
        )

    def to_record(self) -> Dict[str, Any]:
        d = asdict(self)
        d["metadata"] = json.dumps(self.metadata, separators=(",", ":"), sort_keys=True)
        return d

    @staticmethod
    def from_record(row: Dict[str, Any]) -> "RunLog":
        return RunLog(

            node_id=row["node_id"],
            inputs_hash=row["inputs_hash"],
            outputs_hash=row["outputs_hash"],
            start_ts=row["start_ts"],
            end_ts=row["end_ts"],
            node_version=row["node_version"],
            status=row["status"],
            metadata=json.loads(row["metadata"]) if row["metadata"] else {},
        )
