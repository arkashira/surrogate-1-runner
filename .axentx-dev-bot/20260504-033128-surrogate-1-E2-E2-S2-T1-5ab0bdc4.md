## 3. Export handler

```rust
// src/api/handlers/export.rs
use actix_web::{web, HttpResponse, Responder};
use chrono::{DateTime, Utc};
use csv::WriterBuilder;
use futures_util::stream::{self, StreamExt};
use sqlx::PgPool;
use std::pin::Pin;
use std::task::{Context, Poll};

use crate::models::{CsvEventRow, Event};

/// Query parameters for the export endpoint
#[derive(serde::Deserialize)]
pub struct ExportQuery {
    pub start: DateTime<Utc>,
    pub end: DateTime<Utc>,
    #[serde(default = "default_limit")]
    pub limit: i64,
    #[serde(default)]
    pub offset: i64,
}

fn default_limit() -> i64 { 100_000 }

/// Stream that yields CSV rows for a chunk of events.
struct CsvChunkStream {
    writer: csv::Writer<Vec<u8>>,
    finished: bool,
}

impl CsvChunkStream {
    fn new() -> Self {
        let mut writer = WriterBuilder::new()
            .has_headers(true)
            .from_writer(Vec::new());
        // Write header once
        writer
            .write_record(&["event_id", "timestamp", "event_type", "device_id", "app_version"])
            .unwrap();
        Self {
            writer,
            finished: false,
        }
    }
}

impl futures_util::stream::Stream for CsvChunkStream {
    type Item = Result<bytes::Bytes, actix_web::Error>;

    fn poll_next(mut self: Pin<&mut Self>, _cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
        if self.finished {
            return Poll::Ready(None);
        }
        let data = self.writer.into_inner().unwrap();
        self.finished = true;
        Poll::Ready(Some(Ok(bytes::Bytes::from(data))))
    }
}
