"""
LLM client wrapper that loads provider credentials from a JSON file,
validates them on start‑up, and offers a tiny façade for downstream code.

The configuration file location can be overridden with the environment
variable ``LLM_CONFIG_PATH``; otherwise ``config/llm_providers.json`` is used.

Typical ``llm_providers.json``:

{
    "openai":   { "api_key": "sk-...", "model_name": "gpt-4" },
    "anthropic":{ "api_key": "sk-ant...", "model_name": "claude-2" }
}
"""

import json
import os
from typing import Any, Dict

# ----------------------------------------------------------------------
# Configuration handling
# ----------------------------------------------------------------------
CONFIG_PATH = os.getenv("LLM_CONFIG_PATH", "config/llm_providers.json")


class ProviderConfigError(ValueError):
    """Raised when the LLM provider configuration is missing or malformed."""
    pass


def _load_config() -> Dict[str, Dict[str, str]]:
    """
    Load and validate the JSON configuration file.

    Returns
    -------
    dict
        Mapping ``provider_name -> {api_key: str, model_name: str}``.

    Raises
    ------
    FileNotFoundError
        If the file does not exist at ``CONFIG_PATH``.
    ProviderConfigError
        If the JSON is not a dict, or any provider entry lacks required fields.
    """
    if not os.path.exists(CONFIG_PATH):
        raise FileNotFoundError(f"LLM configuration file not found at {CONFIG_PATH}")

    with open(CONFIG_PATH, "r", encoding="utf-8") as f:
        raw = json.load(f)

    if not isinstance(raw, dict):
        raise ProviderConfigError(
            "Configuration root must be a JSON object mapping provider names to settings"
        )

    # Validate each provider entry
    for provider, settings in raw.items():
        if not isinstance(settings, dict):
            raise ProviderConfigError(
                f"Settings for provider '{provider}' must be a JSON object"
            )
        missing = [
            field
            for field in ("api_key", "model_name")
            if not settings.get(field)  # catches missing *or* empty strings
        ]
        if missing:
            raise ProviderConfigError(
                f"Provider '{provider}' is missing required fields: {', '.join(missing)}"
            )
    return raw
