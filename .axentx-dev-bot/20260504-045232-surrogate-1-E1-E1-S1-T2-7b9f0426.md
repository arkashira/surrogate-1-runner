    """Supported public data sources for catalog ingestion."""
    PC_PART_PICKER = "pcpartpicker"
    AMAZON = "amazon"
    NEWEGG = "newegg"


class IngestionStatus(str, Enum):
    """Status of an ingestion record."""
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    FAILED = "failed"
