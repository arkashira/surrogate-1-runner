const axios = require('axios');

class SecurityStatus {
  constructor() {
    this.status = 'unknown';
  }

  async fetch() {
    try {
      const response = await axios.get('https://api.example.com/security/status');
      this.status = response.data.status;
    } catch (error) {
      console.error('Error fetching security status:', error);
      this.status = 'error';
    }
  }

  getStatus() {
    return this.status;
  }
}

module.exports = SecurityStatus;

// tests/securityStatus.test.js
const request = require('supertest');
const express = require('express');
const axios = require('axios');
const securityRouter = require('../backend/trackers/securityStatus');

jest.mock('axios');

describe('GET /security', () => {
  let app;

  beforeAll(() => {
    app = express();
    app.use('/', securityRouter);
  });

  afterEach(() => jest.clearAllMocks());

  test('returns active status when external API 200', async () => {
    axios.get.mockResolvedValue({ status: 200, data: { result: 'ok' } });

    const res = await request(app).get('/security');

    expect(res.status).toBe(200);
    expect(res.body).toEqual({ status: 'active' });
    expect(axios.get).toHaveBeenCalledTimes(1);
  });

  test('returns error status when external API 500', async () => {
    axios.get.mockResolvedValue({ status: 500, data: {} });

    const res = await request(app).get('/security');

    expect(res.status).toBe(200);
    expect(res.body).toEqual({ status: 'error' });
  });

  test('returns error status when axios throws', async () => {
    axios.get.mockRejectedValue(new Error('Network error'));

    const res = await request(app).get('/security');

    expect(res.status).toBe(200);
    expect(res.body).toEqual({ status: 'error' });
  });

  test('each request triggers a new axios call', async () => {
    axios.get.mockResolvedValue({ status: 200, data: {} });

    await request(app).get('/security');
    await request(app).get('/security');

    expect(axios.get).toHaveBeenCalledTimes(2);
  });
});
