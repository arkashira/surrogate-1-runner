"""
Notification Service Integration with API Gateway
Handles real-time notifications for API errors and updates.
"""

import logging
import json
from datetime import datetime, timedelta
from typing import Dict, List, Optional
from dataclasses import dataclass, asdict
from enum import Enum

import requests
from apscheduler.schedulers.background import BackgroundScheduler
from apscheduler.triggers.interval import IntervalTrigger

logger = logging.getLogger(__name__)


class NotificationType(Enum):
    API_ERROR = "api_error"
    API_UPDATE = "api_update"
    SYSTEM_ALERT = "system_alert"


class NotificationPriority(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"


@dataclass
class Notification:
    id: str
    type: NotificationType
    priority: NotificationPriority
    message: str
    source: str
    timestamp: datetime
    user_id: Optional[str] = None
    metadata: Optional[Dict] = None

    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "type": self.type.value,
            "priority": self.priority.value,
            "message": self.message,
            "source": self.source,
            "timestamp": self.timestamp.isoformat(),
            "user_id": self.user_id,
            "metadata": self.metadata
        }


class NotificationService:
    def __init__(self, config: Dict):
        self.config = config
        self.scheduler = BackgroundScheduler()
        self.notification_queue: List[Notification] = []
        self._init_scheduler()

    def _init_scheduler(self):
        """Initialize background scheduler for periodic tasks."""
        # Check for stale notifications (older than 1 hour for updates)
        self.scheduler.add_job(
            self._cleanup_stale_notifications,
            IntervalTrigger(hours=1),
            id='cleanup_notifications',
            replace_existing=True
        )
        self.scheduler.start()

    def _cleanup_stale_notifications(self):
        """Remove notifications older than 1 hour."""
        cutoff = datetime.now() - timedelta(hours=1)
        self.notification_queue = [
            n for n in self.notification_queue
            if n.timestamp > cutoff
        ]
        logger.info(f"Cleaned up notifications older than 1 hour. Remaining: {len(self.notification_queue)}")

        user_id: Optional[str] = None
    ) -> Notification:
