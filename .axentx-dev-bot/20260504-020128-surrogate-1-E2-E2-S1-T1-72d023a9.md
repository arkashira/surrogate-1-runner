#!/usr/bin/env python3
"""
Run a saved workflow locally.

The workflow JSON must contain a top‑level ``steps`` key whose value is a list of
objects, each having a ``command`` field.  Example:

{
    "steps": [
        {"command": "echo 'Hello'"},
        {"command": "sleep 1"}
    ]
}
"""

from __future__ import annotations

import argparse
import json
import logging
import subprocess
import sys
import time
from pathlib import Path
from typing import Dict, Iterable, List, Tuple

# --------------------------------------------------------------------------- #
# 1. Helper: Load workflow definition
# --------------------------------------------------------------------------- #
def load_workflow_definition(file_path: str | Path) -> Dict:
    """Return the workflow definition from a JSON file."""
    with open(file_path, "r", encoding="utf-8") as fp:
        return json.load(fp)


# --------------------------------------------------------------------------- #
# 2. Core: WorkflowRunner
# --------------------------------------------------------------------------- #
class WorkflowRunner:
    """
    Execute a workflow defined in a JSON document.

    Parameters
    ----------
    workflow_definition : dict
        Parsed JSON containing a ``steps`` list.
    timeout : int, optional
        Seconds to allow a single step to run before killing it.
    """

    def __init__(self, workflow_definition: Dict, timeout: int = 1800) -> None:
        self.workflow_definition = workflow_definition
        self.timeout = timeout
        self.logger = logging.getLogger(self.__class__.__name__)

    # ----------------------------------------------------------------------- #
    # 2.1. Run the workflow (no output capture)
    # ----------------------------------------------------------------------- #
    def run(self) -> None:
        """
        Execute each step sequentially.

        If a step fails (non‑zero exit code or timeout) the runner stops
        immediately and logs the failure.  This mirrors the behaviour of
        many CI/CD pipelines where a single failure aborts the job.
        """
        for idx, step in enumerate(self._steps(), start=1):
            cmd = step["command"]
            self.logger.info("Step %s: %s", idx, cmd)
            try:
                subprocess.run(
                    cmd,
                    shell=True,
                    check=True,
                    timeout=self.timeout,
                    stdout=subprocess.PIPE,
                    stderr=subprocess.PIPE,
                )
            except subprocess.TimeoutExpired:
                self.logger.error("Step %s timed out after %s seconds", idx, self.timeout)
                raise
            except subprocess.CalledProcessError as exc:
                self.logger.error(
                    "Step %s failed with exit code %s: %s",
                    idx,
                    exc.returncode,
                    exc.cmd,
                )
                raise
