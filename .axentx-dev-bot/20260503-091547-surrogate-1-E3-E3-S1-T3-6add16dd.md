import { NotificationQueue } from './notificationQueue';

describe('NotificationQueue', () => {
  let queue;
  const baseJob = {
    requestId: 'req-1',
    title: 'Request title',
    newStatus: 'approved',
    publicUrl: 'https://app.example.com/requests/req-1',
    actor: 'user-42',
    channels: ['slack', 'email'],
    slackWebhookUrl: null,
  };

  beforeEach(() => {
    queue = new NotificationQueue();
    queue._resetDedupeCache();
  });

  test('enqueueNotification returns true on first call for requestId', () => {
    expect(queue.enqueueNotification(baseJob)).toBe(true);
  });

  test('enqueueNotification returns false within 5 minutes for same requestId', () => {
    queue.enqueueNotification(baseJob);
    expect(queue.enqueueNotification(baseJob)).toBe(false);
  });

  test('enqueueNotification returns true after dedupe window expires', () => {
    queue.enqueueNotification(baseJob);
    jest.advanceTimersByTime(5 * 60 * 1000 + 1);
    expect(queue.enqueueNotification(baseJob)).toBe(true);
  });

  test('enqueueNotification logs warning and continues when Slack webhook absent', () => {
    const warnSpy = jest.spyOn(console, 'warn').mockImplementation();

      expect.objectContaining({ requestId: 'req-1' })

  });

  test('_resetDedupeCache clears dedupe map and allows new enqueue', () => {
