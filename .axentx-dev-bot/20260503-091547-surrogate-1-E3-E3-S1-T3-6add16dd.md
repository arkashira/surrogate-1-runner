```ts
import express from 'express';
import { Request, Response } from 'express';
import { notificationQueue } from '../queue/notificationQueue';

const router = express.Router();

/**
 * PATCH /requests/:id/status
 * Body: { status: string }
 * Updates request status and enqueues a status-change notification.
 * Deduplication: at most one notification per request per 5 minutes.
 */
router.patch('/:id/status', async (req: Request, res: Response) => {
  const { status } = req.body;
  const actor = req.user?.email || req.user?.id || 'system';

  // Fetch current request
  const request = await db.requests.findUnique({
    where: { id: req.params.id },
    select: { title: true, publicUrl: true, status: true }
  });

  if (!request) {
    return res.status(404).json({ error: 'Request not found' });
  }

  // No-op if status unchanged
  if (request.status === status) {
    return res.status(200).json({ ok: true, message: 'Status unchanged' });
  }

  // Update status
  const updated = await db.requests.update({
    where: { id: req.params.id },
    data: { status }
  });

  // Enqueue notification (deduped per request for 5 minutes)
  await notificationQueue.add(
    'status-change-notification',
    {
      requestId: req.params.id,
      title: request.title,
      newStatus: status,
      publicUrl: request.publicUrl || `${process.env.PUBLIC_APP_URL}/requests/${req.params.id}`,
      actor
    },
    {
      // BullMQ deduplication: same deduplication.id within ttl prevents duplicates
      deduplication: {
        id: `notify-dedupe-${req.params.id}`,
        ttl: 300 // 5 minutes
      },
      // Best-effort cleanup
      removeOnComplete: true,
      removeOnFail: 100
    }
  );

  return res.json({ ok: true, request: updated });
});

export default router;
```
