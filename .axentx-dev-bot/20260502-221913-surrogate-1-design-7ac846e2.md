```python
#!/usr/bin/env python3
"""
list-files-once.py
Usage:
  HF_TOKEN=... python3 list-files-once.py \
    --owner axentx \
    --repo surrogate-1-training-pairs \
    --date-folder 2026-05-02 \
    --output file-list.json

Produces deterministic file list for a date folder (recursive).
"""
import argparse
import json
import os
import sys
from huggingface_hub import HfApi

def main() -> None:
    parser = argparse.ArgumentParser(description="List repo files under a date folder.")
    parser.add_argument("--owner", default="axentx")
    parser.add_argument("--repo", default="surrogate-1-training-pairs")
    parser.add_argument("--date-folder", required=True)
    parser.add_argument("--output", default="file-list.json")
    args = parser.parse_args()

    api = HfApi(token=os.environ.get("HF_TOKEN"))
    repo_id = f"{args.owner}/{args.repo}"

    files = []

    def collect(prefix: str) -> None:
        items = api.list_repo_tree(repo=repo_id, path=prefix, recursive=False)
        for item in items:
            if item.type == "directory":
                collect(item.path)
            else:
                files.append(item.path)

    collect(args.date_folder)
    files.sort()

    out = {
        "repo": repo_id,
        "date_folder": args.date_folder,
        "files": files,
    }
    with open(args.output, "w", encoding="utf-8") as f:
        json.dump(out, f, indent=2, ensure_ascii=False)
    print(f"Wrote {len(files)} files to {args.output}", file=sys.stderr)

if __name__ == "__main__":
    main()
```
