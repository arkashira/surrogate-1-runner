import requests
from typing import List, Dict, Optional
from abc import ABC, abstractmethod
import logging

class ToolDiscoveryBase(ABC):
    """Abstract base class for tool discovery integrations"""
    
    def __init__(self, base_url: str, api_key: Optional[str] = None):
        self.base_url = base_url
        self.api_key = api_key
        self.session = requests.Session()
        if api_key:
            self.session.headers.update({'Authorization': f'Bearer {api_key}'})
        
    @abstractmethod
    def search_tools(self, query: str) -> List[Dict]:
        pass
    
    @abstractmethod
    def filter_tools(self, category: str) -> List[Dict]:
        pass
    
    @abstractmethod
    def get_new_tools(self) -> List[Dict]:
        pass

class RedditIntegration(ToolDiscoveryBase):
    def __init__(self, api_key: Optional[str] = None):
        super().__init__("https://www.reddit.com/r/creators/api", api_key)
    
    def search_tools(self, query: str) -> List[Dict]:
        try:
            response = self.session.get(
                f"{self.base_url}/search",
                params={"q": query, "limit": 50}
            )
            response.raise_for_status()
            return response.json().get('data', [])
        except Exception as e:
            logging.error(f"Error searching Reddit tools: {e}")
            return []

            return response.json().get('data', [])

            logging.error(f"Error filtering Reddit tools: {e}")
