import hashlib
from datetime import datetime
from typing import List, Dict, Optional

class DiscrepancyDetector:
    def __init__(self):
        self.duplicate_invoices = set()
        self.unusual_amounts = []
        self.missing_information = []

    def detect_duplicates(self, invoice: Dict) -> bool:
        invoice_hash = hashlib.md5(str(invoice).encode()).hexdigest()
        if invoice_hash in self.duplicate_invoices:
            return True
        self.duplicate_invoices.add(invoice_hash)
        return False

    def detect_unusual_amounts(self, invoice: Dict, threshold: float) -> bool:
        if invoice.get('amount', 0) > threshold:
            self.unusual_amounts.append(invoice)
            return True
        return False

    def detect_missing_information(self, invoice: Dict, required_fields: List[str]) -> bool:
        missing_fields = [field for field in required_fields if field not in invoice]
        if missing_fields:
            self.missing_information.append({
                'invoice': invoice,
                'missing_fields': missing_fields
            })
            return True
        return False

    def detect_discrepancies(self, invoice: Dict, threshold: float, required_fields: List[str]) -> Dict:
        discrepancies = {
            'duplicate': self.detect_duplicates(invoice),
            'unusual_amount': self.detect_unusual_amounts(invoice, threshold),
            'missing_information': self.detect_missing_information(invoice, required_fields)
        }
        return discrepancies

## 1. Acceptance Criteria
- **Duplicate Detection**: The system identifies invoices with identical `invoice_id` or `customer_id` + `invoice_date` and flags them as duplicates.
- **Unusual Amount Detection**: Invoices whose `amount` deviates more than 30% from the average amount of the last 30 invoices for the same customer are flagged as unusual.
- **Missing Information Detection**: Invoices missing any of the required fields (`invoice_id`, `customer_id`, `invoice_date`, `amount`) are flagged.
- **Alert Delivery**: For each flagged invoice, an alert record is created with a unique `alert_id`, timestamp, and status `UNACKNOWLEDGED`.
- **Acknowledgement Flow**: Users can acknowledge an alert, changing its status to `ACKNOWLEDGED` and recording the `acknowledged_at` timestamp.
- **Resolution Tracking**: Once an alert is acknowledged, the system allows the user to mark it as resolved, updating status to `RESOLVED` and recording `resolved_at`.
- **No False Positives**: Invoices that meet none of the above conditions do not generate alerts.

    assert dup_ids == {"A1"}

    assert detect_duplicates([]) == set()

# ---------- Unusual Amount Detection ----------
def test_detect_unusual_amounts_flagging():
    # 30 previous invoices average 100
    invoices = [{"invoice_id": f"I{i}", "customer_id": "C1", "invoice_date": "2024-01-01", "amount": 100} for i in range(30)]
    invoices.append({"invoice_id": "I30", "customer_id": "C1", "invoice_date": "2024-01-02", "amount": 200})  # +100% -> flag
    unusual_ids = detect_unusual_amounts(invoices, threshold=0.3)
    assert "I30" in unusual_ids

def test_detect_unusual_amounts_no_flag():
    invoices = [{"invoice_id": f"I{i}", "customer_id": "C1", "invoice_date": "2024-01-01", "amount": 100} for i in range(30)]
    invoices.append({"invoice_id": "I30", "customer_id": "C1", "invoice_date": "2024-01-02", "amount": 120})  # +20% -> no flag
    unusual_ids = detect_unusual_amounts(invoices, threshold=0.3)
