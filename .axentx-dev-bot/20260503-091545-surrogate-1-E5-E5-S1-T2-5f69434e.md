import { db } from '../db'; // adjust import to your actual db client

export interface RequestFilters {
  status?: string;
  owner?: string;
  teamId?: string;
  page?: number;
  pageSize?: number;
}

export interface PaginatedRequests {
  items: any[]; // replace `any` with your Request type
  total: number;
  page: number;
  pageSize: number;
}

export interface StatusCounts {
  [status: string]: number;
}

/**
 * Find paginated requests with optional filters.
 * If teamId is provided, only requests owned by that team are returned.
 */
export async function findRequests(filters: RequestFilters): Promise<PaginatedRequests> {
  const { status, owner, teamId, page = 1, pageSize = 20 } = filters;
  const offset = (page - 1) * pageSize;

  const whereParts = [];
  const params: any[] = [];

  if (teamId) {
    whereParts.push(`team_id = $${params.length + 1}`);
    params.push(teamId);
  }
  if (status) {
    whereParts.push(`status = $${params.length + 1}`);
    params.push(status);
  }
  if (owner) {
    whereParts.push(`owner = $${params.length + 1}`);
    params.push(owner);
  }

  const whereClause = whereParts.length > 0 ? `WHERE ${whereParts.join(' AND ')}` : '';

  // Items
  const itemsQuery = `
    SELECT *
    FROM requests
    ${whereClause}
    ORDER BY created_at DESC
    LIMIT $${params.length + 1} OFFSET $${params.length + 2}
  `;
  const itemsParams = [...params, pageSize, offset];
  const { rows: items } = await db.query(itemsQuery, itemsParams);

  // Total count
  const countQuery = `
    SELECT COUNT(*) as total
    FROM requests
    ${whereClause}
  `;
  const { rows: countRows } = await db.query(countQuery, params);
  const total = parseInt(countRows[0].total, 10) || 0;

  return {
    items,
    total,
    page,
    pageSize,
  };
}
