// src/models/request.ts
export interface RequestData {
  id: string;
  requestType: string;
  team: string;
  createdAt: string; // ISO
  resolvedAt?: string | null; // ISO or null
  slaHours?: number; // optional SLA window; default 48h
}

export class Request {
  id: string;
  requestType: string;
  team: string;
  createdAt: Date;
  resolvedAt: Date | null;
  slaHours: number;

  constructor(data: RequestData) {
    this.id = data.id;
    this.requestType = data.requestType;
    this.team = data.team;
    this.createdAt = new Date(data.createdAt);
    this.resolvedAt = data.resolvedAt ? new Date(data.resolvedAt) : null;
    this.slaHours = data.slaHours ?? 48;
  }

  get timeToResolution(): number | null {
    if (!this.resolvedAt) return null;
    return this.resolvedAt.getTime() - this.createdAt.getTime(); // ms
  }

  get timeToResolutionHours(): number | null {
    const ms = this.timeToResolution;
    return ms !== null ? ms / (1000 * 60 * 60) : null;
  }

  get slaMet(): boolean | null {
    if (!this.resolvedAt) return null;
    const hours = this.timeToResolutionHours;
    return hours !== null ? hours <= this.slaHours : null;
  }

  toReportItem() {
    return {
      id: this.id,
      requestType: this.requestType,
      team: this.team,
      createdAt: this.createdAt.toISOString(),
      resolvedAt: this.resolvedAt?.toISOString() || null,
      timeToResolution: this.timeToResolution,
      timeToResolutionHours: this.timeToResolutionHours,
      slaMet: this.slaMet
    };
  }
}
