export interface RequestInput {
  id?: string;
  type: string;
  team: string;
  createdAt: string | Date;
  resolvedAt?: string | Date | null;
  title: string;
  description?: string;
}

export class Request {
  id: string;
  type: string;
  team: string;
  createdAt: Date;
  resolvedAt: Date | null;
  title: string;
  description: string;

  constructor(input: RequestInput) {
    this.id = input.id || crypto.randomUUID();
    this.type = input.type;
    this.team = input.team;
    this.createdAt = new Date(input.createdAt);
    this.resolvedAt = input.resolvedAt ? new Date(input.resolvedAt) : null;
    this.title = input.title;
    this.description = input.description || '';
  }

  /**
   * Time-to-resolution in milliseconds.
   * If not resolved, returns elapsed time since creation (for open requests).
   */
  get timeToResolution(): number | null {
    const end = this.resolvedAt || new Date();
    const ms = Math.max(0, end.getTime() - this.createdAt.getTime());
    return Number.isFinite(ms) ? ms : null;
  }

  /**
   * SLA met flag.
   * Uses SLA_MS threshold (default 72h). For unresolved requests, returns true
   * if still within SLA window from creation to now.
   */
  get slaMet(): boolean {
    const threshold = Number(process.env.SLA_MS) || 259200000; // 72h in ms
    const ttr = this.timeToResolution;
    if (ttr === null) return true;
    return ttr <= threshold;
  }

  toJSON() {
    return {
      id: this.id,
      type: this.type,
      team: this.team,
      createdAt: this.createdAt.toISOString(),
      resolvedAt: this.resolvedAt?.toISOString() || null,
      title: this.title,
      description: this.description,
      timeToResolution: this.timeToResolution,
      slaMet: this.slaMet,
    };
  }
}
