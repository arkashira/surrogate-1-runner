# tests/unit/test_remediation_loader.py
import pytest
import yaml
from alerts import RemediationCatalog, RemediationConfigError

@pytest.fixture
def valid_yaml(tmp_path):
    data = {
        "GuardDuty": {
            "steps": [
                "Open GuardDuty console",
                "Identify the affected resource",
                "Apply the recommended IAM policy"
            ],
            "link": "https://docs.aws.amazon.com/guardduty/latest/ug/remediation.html"
        },
        "Config": {"steps": ["Review Config rule", "Update resource configuration"]},
    }
    file = tmp_path / "alerts-remediation.yaml"
    yaml.safe_dump(data, open(file, "w"))
    return file

def test_load_valid_yaml(valid_yaml):
    catalog = RemediationCatalog(str(valid_yaml))
    assert catalog.get("GuardDuty")["steps"][0] == "Open GuardDuty console"
    assert catalog.get("Config")["steps"] == ["Review Config rule", "Update resource configuration"]

def test_missing_file_raises():
    with pytest.raises(FileNotFoundError):
        RemediationCatalog("/non/existent/file.yaml")

def test_malformed_yaml_raises(tmp_path):
    bad = tmp_path / "bad.yaml"
    bad.write_text("GuardDuty: steps: - not-a-list")   # invalid structure
    with pytest.raises(RemediationConfigError):
        RemediationCatalog(str(bad))

def test_unknown_alert_type_returns_none():
    catalog = RemediationCatalog(str(valid_yaml))
    assert catalog.get("NonExistent") is None
