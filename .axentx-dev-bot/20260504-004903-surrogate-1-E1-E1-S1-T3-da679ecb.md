"""
Audit trail implementation.

* Thread‑safe, JSON‑Lines file.
* Global singleton helper.
* Optional `logging.Logger` integration.
* Helper for checking cluster‑version skew.
"""

from __future__ import annotations

import json
import logging
import os
import threading
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Optional, Any

# --------------------------------------------------------------------------- #
# Core audit‑trail class
# --------------------------------------------------------------------------- #

class AuditTrail:
    """
    Thread‑safe audit trail that writes warnings to a JSON‑Lines file.

    Parameters
    ----------
    path : str | Path
        Path to the audit trail file. If the file does not exist it will be
        created. If it exists, new entries are appended.
    logger : logging.Logger | None, optional
        Optional logger that receives the same warning messages that are
        written to the file.  Useful when you want to keep the audit trail
        in sync with the normal application log.
    """

    def __init__(self, path: str | Path, logger: Optional[logging.Logger] = None):
        self._path = Path(path)
        self._path.parent.mkdir(parents=True, exist_ok=True)
        self._lock = threading.Lock()
        self._logger = logger

    # --------------------------------------------------------------------- #
    # Internal helpers
    # --------------------------------------------------------------------- #

    def _current_timestamp(self) -> str:
        """Return an ISO‑8601 UTC timestamp with a trailing Z."""
        return datetime.utcnow().isoformat(timespec="seconds") + "Z"

    # --------------------------------------------------------------------- #
    # Public API
    # --------------------------------------------------------------------- #

    def add_warning(self, message: str) -> None:
        """
        Append a warning entry to the audit trail.

        Parameters
        ----------
        message : str
            The warning message to record.
        """
        entry: Dict[str, Any] = {
            "timestamp": self._current_timestamp(),
            "level": "WARNING",
            "message": message,
        }
        line = json.dumps(entry, ensure_ascii=False)

        with self._lock, self._path.open("a", encoding="utf-8") as f:
            f.write(line + "\n")

        if self._logger:
            self._logger.warning(message)

    def get_warnings(self) -> List[Dict[str, Any]]:
        """
        Retrieve all warning entries from the audit trail.

        Returns
        -------
        List[Dict[str, Any]]
            A list of warning entries sorted by timestamp.
        """
        if not self._path.exists():
            return []

        warnings: List[Dict[str, Any]] = []
        with self._path.open("r", encoding="utf-8") as f:
            for line in f:
                try:
                    entry = json.loads(line)
                    if entry.get("level") == "WARNING":
                        warnings.append(entry)
