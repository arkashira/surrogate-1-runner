#!/usr/bin/env python3
"""
ci/lifecycle.py – Datadog sandbox lifecycle manager for CI/CD

Supported sub‑commands
----------------------
    provision  – spin‑up a sandbox (creates if missing, waits until ready)
    cleanup    – delete an existing sandbox
    report     – pull test results from the sandbox’s logs

All commands accept a mandatory ``--env`` argument that selects a JSON
configuration file (``ci/config/<env>.json``) or falls back to environment
variables.

The module is deliberately self‑contained:
* It validates required environment variables.
* It lazily imports the official ``datadog`` client, providing a helpful
  stub if the library is not installed.
* It retries transient failures and respects a configurable timeout.
* It prints human‑readable output while also returning structured data
  (useful for downstream automation).

Example usage
-------------
    # Provision a sandbox for the “staging” environment
    python ci/lifecycle.py provision --env staging

    # After tests have run, fetch the results
    python ci/lifecycle.py report --env staging --out results.json

    # Clean up the sandbox
    python ci/lifecycle.py cleanup --env staging
"""

# --------------------------------------------------------------------------- #
# Imports & optional stub for the `datadog` library
# --------------------------------------------------------------------------- #
import argparse
import json
import os
import subprocess
import sys
import time
from datetime import datetime
from pathlib import Path
from typing import Any, Dict, Optional

try:
    from datadog import initialize, api
except ImportError:  # pragma: no cover
    class _MissingDatadog:
        """Raise a clear error the first time any attribute is accessed."""
        def __getattr__(self, name):
            raise RuntimeError(
                "The 'datadog' Python package is required for sandbox lifecycle "
                "operations. Install it with:\n    pip install datadog"
            )
    initialize = _MissingDatadog()
    api = _MissingDatadog()
