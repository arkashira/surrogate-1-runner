import { ConnectorFn, NormalizedOutput, Signal } from '../types/connector';

export class ConnectorRegistry {
  private readonly connectors = new Map<string, ConnectorFn>();

  register(name: string, fn: ConnectorFn): void {
    if (!name || typeof name !== 'string' || !name.trim()) {
      throw new Error('Connector name must be a non-empty string.');
    }
    if (typeof fn !== 'function') {
      throw new Error(`Connector "${name}" must be a function.`);
    }
    if (this.connectors.has(name)) {
      throw new Error(`Connector "${name}" is already registered.`);
    }
    this.connectors.set(name, fn);
  }

  resolve(name: string): ConnectorFn {
    if (!this.connectors.has(name)) {
      throw new Error(`Connector "${name}" is not registered.`);
    }
    // existence checked above
    return this.connectors.get(name)!;
  }

  has(name: string): boolean {
    return this.connectors.has(name);
  }

  normalize(raw: any): NormalizedOutput {
    const signal = String(raw?.signal ?? 'hold').toLowerCase();
    if (!['buy', 'sell', 'hold'].includes(signal)) {
      throw new Error(`Invalid signal "${raw?.signal}". Expected one of: buy, sell, hold.`);
    }

    const price = Number(raw?.price);
    if (!Number.isFinite(price) || price <= 0) {
      throw new Error(`Invalid price "${raw?.price}". Expected finite positive number.`);
    }

    const qty = Number(raw?.qty);
    if (!Number.isFinite(qty) || qty <= 0) {
      throw new Error(`Invalid qty "${raw?.qty}". Expected finite positive number.`);
    }

    return {
      signal: signal as Signal,
      price,
      qty,
      meta: raw?.meta && typeof raw.meta === 'object' ? raw.meta : undefined,
    };
  }

  async invoke(name: string, ...args: any[]): Promise<NormalizedOutput> {
    const connector = this.resolve(name);
    const raw = connector(...args);
    const awaited = raw instanceof Promise ? await raw : raw;
    return this.normalize(awaited);
  }
}

export const registry = new ConnectorRegistry();
```
