// Types
export type Signal = 'buy' | 'sell' | 'hold';

export interface NormalizedOutput {
  signal: Signal;
  price: number;
  qty: number;
  meta?: Record<string, unknown>;
}

export type ConnectorFn = (inputs: Record<string, unknown>) => NormalizedOutput | Promise<NormalizedOutput>;

// Registry (singleton)
const registry = new Map<string, ConnectorFn>();

export function register(name: string, fn: ConnectorFn): void {
  if (!name || typeof name !== 'string') {
    throw new Error('Connector name must be a non-empty string');
  }
  if (typeof fn !== 'function') {
    throw new Error('Connector must be a function');
  }
  if (registry.has(name)) {
    throw new Error(`Connector already registered: "${name}"`);
  }
  registry.set(name, fn);
}

export function resolve(name: string): ConnectorFn {
  const connector = registry.get(name);
  if (!connector) {
    throw new Error(
      `Connector not found: "${name}". Available connectors: ${Array.from(registry.keys()).join(', ') || 'none'}`
    );
  }
  return connector;
}

// Runner (resolve + invoke + normalize + fail-fast)
export async function run(name: string, inputs: Record<string, unknown> = {}): Promise<NormalizedOutput> {
  const connector = resolve(name);
  try {
    const result = await connector(inputs);
    validateNormalizedOutput(result);
    return result;
  } catch (err) {
    throw new Error(
      `Connector "${name}" execution failed: ${err instanceof Error ? err.message : String(err)}`
    );
  }
}

// Validation (fail-fast, clear errors)
function validateNormalizedOutput(value: unknown): asserts value is NormalizedOutput {
  if (!value || typeof value !== 'object' || Array.isArray(value)) {
    throw new Error('Connector must return an object');
  }
  const v = value as Partial<NormalizedOutput>;

  if (!['buy', 'sell', 'hold'].includes(v.signal as string)) {
    throw new Error(`Invalid signal: ${String(v.signal)}. Must be one of: buy, sell, hold`);
  }

  if (typeof v.price !== 'number' || !Number.isFinite(v.price) || v.price <= 0) {
    throw new Error(`Invalid price: ${String(v.price)}. Must be a positive finite number`);
  }

  if (typeof v.qty !== 'number' || !Number.isFinite(v.qty) || v.qty <= 0) {
    throw new Error(`Invalid qty: ${String(v.qty)}. Must be a positive finite number`);
  }

  if (v.meta !== undefined) {
    if (typeof v.meta !== 'object' || v.meta === null || Array.isArray(v.meta)) {
      throw new Error('meta must be a plain object if provided');
    }
  }
}
