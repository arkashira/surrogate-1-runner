## 2) Connector registry (sync + async support, strict validation)

`src/connectors/registry.ts`
```ts
import { ConnectorFn, ConnectorError } from '../types/connector';

type RegistryRecord = {
  fn: ConnectorFn;
  registeredAt: Date;
};

export class ConnectorRegistry {
  private readonly connectors = new Map<string, RegistryRecord>();

  /**
   * Register a connector function (sync or async).
   * Throws ConnectorError for invalid name, non-function, or duplicate.
   */
  register(name: string, fn: ConnectorFn): void {
    if (!name || typeof name !== 'string' || !name.trim()) {
      throw new ConnectorError('Connector name must be a non-empty string');
    }
    if (typeof fn !== 'function') {
      throw new ConnectorError('Connector must be a function');
    }
    if (this.connectors.has(name)) {
      throw new ConnectorError(`Connector "${name}" is already registered`);
    }

    this.connectors.set(name, { fn, registeredAt: new Date() });
  }

  /**
   * Resolve a connector by name.
   * Throws ConnectorError if not found.
   */
  resolve(name: string): ConnectorFn {
    const record = this.connectors.get(name);
    if (!record) {
      throw new ConnectorError(`Connector "${name}" not found`, name);
    }
    return record.fn;
  }

  has(name: string): boolean {
    return this.connectors.has(name);
  }

  list(): Array<{ name: string; registeredAt: Date }> {
    return Array.from(this.connectors.entries()).map(([name, record]) => ({
      name,
      registeredAt: record.registeredAt,
    }));
  }
}
