// Connector registry + normalization.
// Supports sync and async connector functions.
// Normalized output:
// { signal: 'buy'|'sell'|'hold', price: number, qty: number, meta?: object }

type Signal = 'buy' | 'sell' | 'hold';

export interface NormalizedOutput {
  signal: Signal;
  price: number;
  qty: number;
  meta?: object;
}

type ConnectorFn = (...args: any[]) => NormalizedOutput | Promise<NormalizedOutput>;

const registry = new Map<string, ConnectorFn>();

function isPlainObject(value: unknown): boolean {
  return value !== null && typeof value === 'object' && !Array.isArray(value);
}

function normalize(raw: unknown): NormalizedOutput {
  // Already valid normalized output
  if (
    isPlainObject(raw) &&
    ['buy', 'sell', 'hold'].includes((raw as NormalizedOutput).signal) &&
    typeof (raw as NormalizedOutput).price === 'number' &&
    typeof (raw as NormalizedOutput).qty === 'number'
  ) {
    return raw as NormalizedOutput;
  }

  // Best-effort normalization for common shapes; otherwise fail fast.
  if (isPlainObject(raw)) {
    const obj = raw as Record<string, unknown>;
    const signalRaw = obj.signal;
    const signal = typeof signalRaw === 'string' ? signalRaw.toLowerCase() : '';
    if (!['buy', 'sell', 'hold'].includes(signal)) {
      throw new Error(`Connector returned invalid signal: ${String(signalRaw)}`);
    }

    const price = Number(obj.price ?? obj.last ?? NaN);
    if (!Number.isFinite(price) || price < 0) {
      throw new Error(`Connector returned invalid price: ${String(obj.price)}`);
    }

    const qty = Number(obj.qty ?? obj.size ?? NaN);
    if (!Number.isFinite(qty) || qty < 0) {
      throw new Error(`Connector returned invalid qty: ${String(obj.qty)}`);
    }

    return {
      signal: signal as Signal,
      price,
      qty,
      meta: isPlainObject(obj.meta) ? obj.meta : (obj.meta ?? {}),
    };
  }

  throw new Error(`Connector returned unrecognized output shape: ${JSON.stringify(raw)}`);
}

export function register(name: string, fn: ConnectorFn): void {
  if (!name || typeof name !== 'string') {
    throw new Error('Connector name must be a non-empty string');
  }
  if (typeof fn !== 'function') {
    throw new Error('Connector must be a function');
  }
  registry.set(name, fn);
}

export function resolve(name: string): ConnectorFn | undefined {
  return registry.get(name);
}

export async function run(name: string, inputs?: any): Promise<NormalizedOutput> {
  if (!name || typeof name !== 'string') {
    throw new Error('Connector name must be a non-empty string');
  }

  const connector = resolve(name);
  if (!connector) {
    throw new Error(
      `Connector not found: "${name}". Available connectors: ${Array.from(registry.keys()).join(', ') || 'none'}`
    );
  }

  try {
    const raw = connector(inputs);
    const resolved = raw instanceof Promise ? await raw : raw;
    return normalize(resolved);
  } catch (err: any) {
    // Preserve original cause while adding context.
    throw new Error(`Connector "${name}" failed: ${err?.message ?? String(err)}`);
  }
}
