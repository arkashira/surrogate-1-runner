from __future__ import annotations

import hashlib
import json
import time
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Dict, List


@dataclass(frozen=True)
class DeterministicContext:
    """
    Immutable execution context for deterministic pipeline runs.
    - clock: monotonic seconds since context creation (for durations/timeouts)
    - utc_iso: wall-clock timestamps for human audit readability
    - seed: deterministic RNG seed shared with stages
    - run_id: deterministic fingerprint of pipeline definition + inputs
    - audit_log: append-only, ordered event stream
    """
    run_id: str
    seed: int
    _clock_start: float = field(default_factory=time.monotonic)
    _audit_log: List[Dict[str, Any]] = field(default_factory=list)

    @staticmethod
    def create(deterministic_key: str, seed: int = 42) -> "DeterministicContext":
        run_id = hashlib.sha256(deterministic_key.encode("utf-8")).hexdigest()[:16]
        return DeterministicContext(run_id=run_id, seed=seed)

    def now(self) -> float:
        """Monotonic seconds since context creation (safe for durations/timeouts)."""
        return time.monotonic() - self._clock_start

    def utc_iso(self) -> str:
        """Wall-clock UTC for audit readability."""
        return datetime.now(timezone.utc).isoformat()

    def audit(self, stage_id: str, event: str, payload: Dict[str, Any] | None = None) -> None:
        entry = {
            "run_id": self.run_id,
            "stage_id": stage_id,
            "event": event,
            "monotonic_ts": round(self.now(), 6),
            "utc": self.utc_iso(),
            "payload": payload or {},
        }
        # Frozen dataclass: mutate internal list safely.
        object.__setattr__(self, "_audit_log", self._audit_log + [entry])

    def audit_log(self) -> List[Dict[str, Any]]:
        return list(self._audit_log)

    def snapshot(self) -> Dict[str, Any]:
        return {
            "run_id": self.run_id,
            "seed": self.seed,
            "audit_log": self.audit_log(),
        }
