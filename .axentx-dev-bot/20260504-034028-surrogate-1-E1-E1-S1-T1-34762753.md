import { HardwarePricingAPI } from './HardwarePricingAPI';
import axios from 'axios';
jest.mock('axios');
const mockedAxios = axios as jest.Mocked<typeof axios>;

describe('HardwarePricingAPI', () => {
  const api = new HardwarePricingAPI('test-key');

  afterEach(() => jest.clearAllMocks());

  test('fetches price for valid component', async () => {
    const mockResponse = {
      data: {
        componentId: 'gpu-rtx3080',
        priceUsd: 699.99,
        currency: 'USD',
        lastUpdated: '2026-05-19T12:34:56Z',
      },
    };
    mockedAxios.get.mockResolvedValue(mockResponse);

    const price = await api.getPrice('gpu-rtx3080');
    expect(price).toEqual(699.99);
    expect(mockedAxios.get).toHaveBeenCalledWith(
      'https://api.gamerprice.com/v1/prices',
      expect.objectContaining({
        params: { componentId: 'gpu-rtx3080' },
        headers: { Authorization: 'Bearer test-key' },
      })
    );
  });

  test('throws NotFoundError for 404', async () => {
    mockedAxios.get.mockRejectedValue({
      response: { status: 404 },
    });
    await expect(api.getPrice('unknown')).rejects.toThrow('Component not found');
  });

  test('throws RateLimitError for 429', async () => {
    mockedAxios.get.mockRejectedValue({
      response: { status: 429 },
    });
    await expect(api.getPrice('gpu-rtx3080')).rejects.toThrow('Rate limited');
  });

  test('caches responses for 5 minutes', async () => {
    const mockResponse = { data: { componentId: 'cpu-ryzen5', priceUsd: 199.99, currency: 'USD', lastUpdated: '2026-05-19T12:34:56Z' } };
    mockedAxios.get.mockResolvedValue(mockResponse);

    const price1 = await api.getPrice('cpu-ryzen5');
    const price2 = await api.getPrice('cpu-ryzen5');

    expect(mockedAxios.get).toHaveBeenCalledTimes(1);
    expect(price1).toEqual(price2);
  });

  test('expires cache after 5 minutes', async () => {
    jest.useFakeTimers();
    const mockResponse1 = { data: { componentId: 'cpu-ryzen5', priceUsd: 199.99, currency: 'USD', lastUpdated: '2026-05-19T12:34:56Z' } };
    const mockResponse2 = { data: { componentId: 'cpu-ryzen5', priceUsd: 189.99, currency: 'USD', lastUpdated: '2026-05-19T12:35:56Z' } };
    mockedAxios.get
      .mockResolvedValueOnce(mockResponse1)
      .mockResolvedValueOnce(mockResponse2);

    await api.getPrice('cpu-ryzen5');
    jest.advanceTimersByTime(5 * 60 * 1000 + 1); // 5 minutes + 1 ms
    const price = await api.getPrice('cpu-ryzen5');

    expect(mockedAxios.get).toHaveBeenCalledTimes(2);
    expect(price).toEqual(189.99);
    jest.useRealTimers();
  });
});
