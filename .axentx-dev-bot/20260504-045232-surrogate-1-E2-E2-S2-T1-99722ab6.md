"""
Price aggregation utilities.

This module provides an in‑memory store for price data coming from multiple
vendors.  It exposes helper functions that are used by the API layer to
persist new price entries, fetch the lowest current price for a component,
and retrieve the full price history.

The store keeps data in the following structure:

```python
{
    component_id: [
        {
            "vendor_id": str,
            "price": float,
            "timestamp": datetime.datetime,
        },
        ...
    ],
    ...
}
```

All timestamps are stored as UTC `datetime.datetime` objects.  The
``add_price`` function automatically records the current UTC time if a
timestamp is not supplied.

Thread safety is provided via a `threading.Lock` so that concurrent
updates from multiple workers do not corrupt the in‑memory state.
"""

from __future__ import annotations

import datetime
import threading
from typing import Dict, List, Tuple, Optional, Any

# Global in‑memory store and lock
_store_lock = threading.Lock()
_price_store: Dict[str, List[Dict[str, Any]]] = {}


def _current_utc() -> datetime.datetime:
    """Return the current UTC timestamp."""
    return datetime.datetime.utcnow().replace(tzinfo=datetime.timezone.utc)


def add_price(
    component_id: str,
    vendor_id: str,
    price: float,
    *,
    timestamp: Optional[datetime.datetime] = None,
) -> None:
    """
    Persist a new price entry for a component.

    Parameters
    ----------
    component_id : str
        Identifier of the component.
    vendor_id : str
        Identifier of the vendor providing the price.
    price : float
        Price value.
    timestamp : datetime.datetime, optional
        UTC timestamp of the price.  If omitted, the current UTC time is used.

    Raises
    ------
    ValueError
        If the price is negative.
    """
    if price < 0:
        raise ValueError("Price must be non‑negative")

    if timestamp is None:
        timestamp = _current_utc()
    elif timestamp.tzinfo is None:
        # Assume naive datetime is UTC
        timestamp = timestamp.replace(tzinfo=datetime.timezone.utc)

    entry = {
        "vendor_id": vendor_id,
        "price": float(price),
        "timestamp": timestamp,
    }

    with _store_lock:
        _price_store.setdefault(component_id, []).append(entry)


def get_lowest_price(component_id: str) -> Tuple[float, str, datetime.datetime]:
    """
    Return the lowest current price for a component.

    The "current" price is defined as the most recent entry for each vendor.
    Among those, the lowest price is returned.

    Returns
    -------
    price : float
        Lowest price value.
    vendor_id : str
        Vendor that offers the lowest price.
    timestamp : datetime.datetime
        Timestamp of the lowest price entry.

    Raises
    ------
    KeyError
        If the component has no price data.
    """
    with _store_lock:
        entries = _price_store.get(component_id, [])

    if not entries:
        raise KeyError(f"No price data for component {component_id}")

    # Determine the latest entry per vendor
    latest_per_vendor: Dict[str, Dict[str, Any]] = {}
    for entry in entries:
        vid = entry["vendor_id"]
        if vid not in latest_per_vendor or entry["timestamp"] > latest_per_vendor[vid]["timestamp"]:
            latest_per_vendor[vid] = entry

    # Find the lowest price among the latest entries
    lowest_entry = min(
        latest_per_vendor.values(),
        key=lambda e: e["price"],
    )
    return lowest_entry["price"], lowest_entry["vendor_id"], lowest_entry["timestamp"]

- Fix the race condition in `get_lowest_price` and `get_price_history`: both functions release the lock before processing the entries list, but they hold a reference to the actual in-memory list (not a copy). - The `get_price_history` function correctly returns new dict copies for each entry (good defensive coding), but the iteration still happens outside the lock - fix this alongside the above.
- Consider whether `get_lowest_price` should also return copies of entries rather than the originals, for consistency with `get_price_history`.
- The `_reset_store` helper is correctly placed and useful for testing.

The code is otherwise well-structured: proper timezone handling, good docstrings, correct validation, appropriate use of `setdefault`, and the "latest per vendor" logic in `get_lowest_price` is sound. Thread safety is partially implemented but needs the above fix to be complete.
