// src/ui/TerminalModal.jsx
import React, { useState, useEffect } from 'react';
import { Modal, Button } from 'react-bootstrap';

const TerminalModal = ({ show, onHide, onTimeout }) => {
  const [idleTime, setIdleTime] = useState(0);

  useEffect(() => {
    const idleInterval = setInterval(() => {
      setIdleTime(prevTime => prevTime + 1);
    }, 60000);

    return () => clearInterval(idleInterval);
  }, []);

  useEffect(() => {
    if (idleTime >= 24 * 60) {
      onTimeout();
    }
  }, [idleTime, onTimeout]);

  return (
    <Modal show={show} onHide={onHide} size="lg" aria-labelledby="contained-modal-title-vcenter" centered>
      <Modal.Header closeButton>
        <Modal.Title id="contained-modal-title-vcenter">
          Session Timeout
        </Modal.Title>
      </Modal.Header>
      <Modal.Body>
        <h4>Your session will be disconnected in {24 * 60 - idleTime} minutes due to inactivity.</h4>
      </Modal.Body>
      <Modal.Footer>
        <Button onClick={onHide}>Extend Session</Button>
      </Modal.Footer>
    </Modal>
  );
};

export default TerminalModal;

## 1. Acceptance Criteria
- **Configurable idle timeout** – The idle timeout value can be set via a prop or environment variable; default is exactly 24 hours (86 400 seconds). The value is read once on component mount and used for all calculations.
- **Modal appears only after threshold** – The idle‑timer modal is not rendered on initial mount; it becomes visible **only** when the elapsed idle time ≥ configured timeout.
- **Countdown display** – Once the modal is shown, it displays a live countdown (in minutes and seconds) that updates every second until termination.
- **Graceful termination** – When the countdown reaches zero, the component calls `terminateSession()` which sends a `POST /api/session/terminate`. On success the UI navigates away / closes; on failure an error toast is shown and the modal remains.
- **Cleanup on unmount** – All timers/intervals are cleared when the component unmounts to avoid memory leaks.
- **`onTimeout` prop memoization** – If a parent supplies an `onTimeout` callback, it must be memoized with `useCallback` (shallow equality) to prevent unnecessary re‑renders of the modal.
- **Error handling** – Network or server errors from the termination endpoint are caught, logged, and surface a user‑friendly message without crashing the app.
