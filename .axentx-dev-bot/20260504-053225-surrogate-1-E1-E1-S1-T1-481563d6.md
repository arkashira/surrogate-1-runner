class ProxyConfig:
    def __init__(self, ai_endpoints, proxy_url):
        self.ai_endpoints = ai_endpoints
        self.proxy_url = proxy_url

    def to_dict(self):
        return {
            "ai_endpoints": self.ai_endpoints,
            "proxy_url": self.proxy_url

    def from_dict(cls, data):
