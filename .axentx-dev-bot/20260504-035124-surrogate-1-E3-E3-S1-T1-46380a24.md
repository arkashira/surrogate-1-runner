"""
Session termination utilities for surrogate-1.

This module provides:
* Session representation with activity tracking.
* Automatic session termination after a configurable inactivity timeout.
* Manual termination of active sessions.
* Basic resource usage monitoring and logging.

The implementation is intentionally lightweight and self‑contained,
so it can be used in environments where external dependencies
(e.g., psutil) may or may not be available.
"""

import logging
import os
import signal
import threading
import time
from datetime import datetime, timedelta
from typing import Dict, Optional

# Optional dependency for detailed resource usage
try:
    import psutil  # type: ignore
except ImportError:  # pragma: no cover
    psutil = None

# Configure module level logger
logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)
handler = logging.StreamHandler()
handler.setFormatter(
    logging.Formatter("[%(asctime)s] %(levelname)s %(name)s: %(message)s")
)
logger.addHandler(handler)


class Session:
    """
    Represents an active shell session.

    Attributes
    ----------
    session_id : str
        Unique identifier for the session.
    process : os.ProcessHandle
        The underlying OS process running the shell.
    last_activity : datetime
        Timestamp of the last recorded activity.
    """

    def __init__(self, session_id: str, process: os.ProcessHandle) -> None:
        self.session_id = session_id
        self.process = process
        self.last_activity = datetime.utcnow()

    def touch(self) -> None:
        """Mark the session as active."""
        self.last_activity = datetime.utcnow()

    def terminate(self, force: bool = False) -> None:
        """Terminate the underlying process."""
        if self.process is None:
            logger.debug(f"Session {self.session_id} has no process to terminate.")
            return

        try:
            if force:
                self.process.kill()
            else:
                self.process.terminate()
            logger.info(f"Session {self.session_id} terminated (force={force}).")
        except Exception as exc:
            logger.warning(
                f"Failed to terminate session {self.session_id}: {exc}"
            )
