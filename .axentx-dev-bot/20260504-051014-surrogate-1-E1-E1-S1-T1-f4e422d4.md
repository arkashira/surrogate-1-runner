package com.axentx.surrogate1.integrations;

import java.io.IOException;
import java.net.URI;
import java.net.http.*;
import java.time.LocalDate;
import java.util.Map;
import com.fasterxml.jackson.core.type.TypeReference;
import com.fasterxml.jackson.databind.ObjectMapper;

/**
 * QuickBooks API connector supporting OAuth2 token refresh,
 * profit‑and‑loss report retrieval, margin calculation and
 * deviation alerting.
 *
 * Configuration is supplied via environment variables:
 *   QB_CLIENT_ID, QB_CLIENT_SECRET, QB_REFRESH_TOKEN, QB_REALM_ID
 *
 * The client is deliberately lightweight and does not depend on
 * any framework – it uses Java 11+ HttpClient and Jackson for JSON.
 */
public class QuickBooksClient {

    private static final String TOKEN_URL = "https://oauth.platform.intuit.com/oauth2/v1/tokens/bearer";
    private static final String REPORT_URL_TEMPLATE =
            "https://quickbooks.api.intuit.com/v3/company/%s/reports/ProfitAndLoss?start_date=%s&end_date=%s";

    private final String clientId;
    private final String clientSecret;
    private final String refreshToken;
    private final String realmId;
    private final HttpClient httpClient;
    private final ObjectMapper objectMapper;

    // mutable OAuth state
    private String accessToken;
    private long tokenExpiryEpochSec; // epoch seconds

    /**
     * Default constructor reads configuration from environment.
     */
    public QuickBooksClient() {
        this.clientId = System.getenv("QB_CLIENT_ID");
        this.clientSecret = System.getenv("QB_CLIENT_SECRET");
        this.refreshToken = System.getenv("QB_REFRESH_TOKEN");
        this.realmId = System.getenv("QB_REALM_ID");
        this.httpClient = HttpClient.newHttpClient();
        this.objectMapper = new ObjectMapper();
    }

    /**
     * Constructor used by tests to inject dependencies.
     */

        this.clientId = clientId;

        this.refreshToken = refreshToken;
        this.realmId = realmId;
        this.httpClient = httpClient;
        this.objectMapper = objectMapper;
    }
