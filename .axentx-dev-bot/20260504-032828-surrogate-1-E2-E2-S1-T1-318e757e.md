// src/visualization.rs
#[cfg(test)]
mod tests {
    use super::*;
    use std::sync::mpsc::{channel, Sender};

    // Helper to create a dummy interaction
    fn dummy_interaction(id: u64) -> AgentInteraction {
        AgentInteraction {
            id,
            source: "agent_a".into(),
            target: "agent_b".into(),
            payload: "msg".into(),
            timestamp: 0,
        }
    }

    #[test]
    fn test_visualization_state_initializes_empty() {
        let state = VisualizationState::new();
        assert!(state.interactions.is_empty());
    }

    #[test]
    fn test_receive_interaction_appends_to_state() {
        let mut state = VisualizationState::new();
        let inter = dummy_interaction(42);
        state.handle_event(inter.clone());
        assert_eq!(state.interactions.len(), 1);
        assert_eq!(state.interactions[0], inter);
    }

    #[test]
    fn test_pause_stops_rendering() {
        let mut state = VisualizationState::new();
        state.pause();
        assert!(state.is_paused);
        // Simulate an event while paused
        state.handle_event(dummy_interaction(1));
        // Rendering should not occur; we assert that the internal buffer still holds the event
        assert_eq!(state.interactions.len(), 1);
    }

    #[test]
    fn test_step_advances_one_interaction() {
        let mut state = VisualizationState::new();
        state.handle_event(dummy_interaction(1));
        state.handle_event(dummy_interaction(2));
        state.step(); // should render first interaction
        assert_eq!(state.rendered.len(), 1);
        assert_eq!(state.rendered[0].id, 1);
    }

    #[test]
    fn test_reset_clears_state() {
        let mut state = VisualizationState::new();
        state.handle_event(dummy_interaction(1));
        state.reset();
        assert!(state.interactions.is_empty());
        assert!(state.rendered.is_empty());
    }

    #[test]
    fn test_malformed_payload_is_logged_and_ignored() {
        let mut state = VisualizationState::new();
        let bad = AgentInteraction::malformed(); // assume this constructs an invalid payload
        let result = state.handle_event(bad);
        assert!(result.is_err());
        // Ensure state unchanged
        assert!(state.interactions.is_empty());
    }
}
