import json
import os
from typing import List, Dict, Any, Set, Optional


class HashStore:
    """
    A persistent store for MD5 hashes to detect duplicate data entries.
    
    Uses JSON for robust storage with automatic directory creation.
    """
    
    def __init__(self, filepath: str = "/tmp/hash_store.json"):
        self.filepath = filepath
        self._store: Set[str] = self._load()
    
    def _load(self) -> Set[str]:
        """Load existing hashes from disk."""
        if os.path.exists(self.filepath):
            try:
                with open(self.filepath, 'r', encoding='utf-8') as f:
                    return set(json.load(f))
            except (json.JSONDecodeError, IOError, ValueError):
                return set()
        return set()
    
    def _save(self) -> None:
        """Persist the current set of hashes to disk."""
        # Ensure directory exists before writing
        directory = os.path.dirname(self.filepath
        if directory:
            os.makedirs(directory, exist_ok=True)
        
        with open(self.filepath, 'w', encoding='utf-8') as f:
            json.dump(list(self._store), f, indent=2)

        """Check if a hash already exists in the store."""
