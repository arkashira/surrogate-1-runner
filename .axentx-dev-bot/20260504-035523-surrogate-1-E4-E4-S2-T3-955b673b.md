## 1. Acceptance Criteria  
- **AC1** – After every LLM action, the workflow engine receives a `cost` object containing `tokens` (integer) and `usd` (float) values.  
- **AC2** – The node inspector panel displays the `tokens` and `usd` for the most recent LLM step.

- **AC4** – If no LLM steps were executed, the Total Cost section displays `$0.00`.  
- **AC5** – The Total Cost section updates in real‑time as new LLM steps complete during a run.

## 2. Unit Tests (Jest – React Native)  

```ts
// __tests__/RunSummaryScreen.unit.test.tsx
import { render, screen } from '@testing-library/react-native';
import RunSummaryScreen from '../src/screens/RunSummaryScreen';
import { WorkflowContextProvider } from '../src/context/WorkflowContext';

describe('RunSummaryScreen – unit', () => {
  const mockWorkflow = {
    steps: [
      { id: '1', type: 'llm', cost: { tokens: 120, usd: 0.30 } },
      { id: '2', type: 'llm', cost: { tokens: 80, usd: 0.20 } },
    ],
  };

  const renderScreen = (workflow = mockWorkflow) =>
    render(
      <WorkflowContextProvider initialState={workflow}>
        <RunSummaryScreen />
      </WorkflowContextProvider>
    );

  test('renders total cost correctly', () => {
    renderScreen();
    expect(screen.getByText('Total Cost')).toBeTruthy();
    expect(screen.getByText('$0.50')).toBeTruthy(); // 0.30 + 0.20
  });

  test('shows $0.00 when no LLM steps', () => {
    renderScreen({ steps: [] });
    expect(screen.getByText('$0.00')).toBeTruthy();
  });

  test('updates total cost when new step is added', () => {
    const { rerender } = renderScreen();
    const newWorkflow = {
      steps: [
        ...mockWorkflow.steps,
        { id: '3', type: 'llm', cost: { tokens: 50, usd: 0.10 } },
      ],
    };
    rerender(
      <WorkflowContextProvider initialState={newWorkflow}>
        <RunSummaryScreen />
      </WorkflowContextProvider>
    );
    expect(screen.getByText('$0.60')).toBeTruthy(); // 0.30+0.20+0.10
  });

  test('formats usd to two decimal places', () => {
    renderScreen({
      steps: [
        { id: '1', type: 'llm', cost: { tokens: 100, usd: 0.333 } },
      ],
    });
    expect(screen.getByText('$0.33')).toBeTruthy();
  });
});
```
