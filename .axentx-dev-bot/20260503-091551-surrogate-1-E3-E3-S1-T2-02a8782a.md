    AT_RISK = "at_risk"
    OK = "ok"


def _calculate_sla_state(created_at: datetime, sla_hours: Optional[float], now: Optional[datetime] = None) -> str:
    """Return SLA state: breached, at_risk, or ok."""
    if not sla_hours or not created_at:
        return SLAState.OK

    now = now or datetime.now(timezone.utc)
    if created_at.tzinfo is None:
        created_at = created_at.replace(tzinfo=timezone.utc)

    elapsed = (now - created_at).total_seconds() / 3600.0
    if elapsed > sla_hours:
        return SLAState.BREACHED
    if elapsed > sla_hours * 0.8:
        return SLAState.AT_RISK
    return SLAState.OK
