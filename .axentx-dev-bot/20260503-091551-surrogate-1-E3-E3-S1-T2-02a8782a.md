# ── Slack webhook notifier ───────────────────────────────────────────────────

class SlackWebhookNotifier:
    """
    Send formatted messages to a Slack incoming webhook.
    Webhook URL can be provided at init or via NOTIFY_SLACK_WEBHOOK_URL.
    """

    _STATUS_EMOJI = {
        "open": ":large_blue_circle:",
        "in_progress": ":large_yellow_circle:",
        "resolved": ":large_green_circle:",
        "closed": ":white_check_mark:",
        "blocked": ":octagonal_sign:",
    }

    _SLA_ICON = {"ok": ":white_check_mark:", "warning": ":warning:", "breach": ":x:"}

    def __init__(self, webhook_url: Optional[str] = None):
        self.webhook_url = webhook_url or os.getenv("NOTIFY_SLACK_WEBHOOK_URL")
        if not self.webhook_url:
            raise ValueError(
                "Slack webhook URL required. Provide argument or set NOTIFY_SLACK_WEBHOOK_URL."
            )
        if requests is None:
            raise RuntimeError("Slack notifier requires 'requests'. pip install requests")

    def send(self, text: str, blocks: Optional[List[Dict[str, Any]]] = None) -> Dict[str, Any]:
        payload: Dict[str, Any] = {"text": text}
        if blocks:
            payload["blocks"] = blocks
        resp = requests.post(self.webhook_url, json=payload, timeout=10)
        resp.raise_for_status()
        return resp.json()

    def format_request_message(self, ctx: RequestContext) -> Dict[str, Any]:
        emoji = self._STATUS_EMOJI.get(ctx.current_status.lower(), ":black_circle:")
        title_link = f"*{ctx.title}* <{ctx.public_url}|{ctx.request_id}>"
        status_line = f"Status changed to {emoji} *{ctx.current_status}*"
        if ctx.previous_status:
            status_line += f" (was {ctx.previous_status})"

        timeline_line = "_No recent updates_"
        if ctx.last_timeline_entry:
            te = ctx.last_timeline_entry
            timeline_line = f"Last update by *{te.actor}*: {te.action}"
            if te.comment:
                timeline_line += f" — {te.comment}"

        if ctx.sla and ctx.sla.remaining_hours is not None:

                sla_line += f" (due {ctx.sla.due_at})"
        elif ctx.sla:

            sla_line = f"SLA {icon} {ctx.sla.status}"
