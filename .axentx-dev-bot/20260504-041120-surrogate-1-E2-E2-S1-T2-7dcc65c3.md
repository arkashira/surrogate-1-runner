from starlette.middleware.base import BaseHTTPMiddleware, RequestResponseEndpoint
from jose import jwt, JWTError

# ----------------------------------------------------------------------
# Configuration (can be moved to env / settings module)

_RATE_LIMIT_WINDOW = 60          # seconds
_RATE_LIMIT_MAX = 60             # requests per window
_JWT_ALGORITHM = "HS256"


class JWTAuthMiddleware(BaseHTTPMiddleware):
    """
    FastAPI middleware that:
      1. Extracts a Bearer JWT (using the same logic as a FastAPI dependency).
      2. Verifies the token signature with the supplied secret.
      3. Stores the decoded payload on ``request.state.user``.
      4. Checks that the ``account_id`` path‑parameter matches the ``account_id`` claim.
      5. Enforces a per‑account sliding‑window rate limit (60 req/min).
    """

    def __init__(self, app, secret_key: str):
        super().__init__(app)
        self.secret_key = secret_key
        self.scheme = HTTPBearer()
        # {account_id: deque([timestamp, ...])}
        self._rate_store: Dict[str, deque] = defaultdict(deque)

    # ------------------------------------------------------------------
    # Helper: decode JWT (mirrors the AuthGuard dependency)
    # ------------------------------------------------------------------

        credentials: Optional[HTTPAuthorizationCredentials] = await self.scheme(request)
        if credentials is None:
            raise HTTPException(
                status_code=status.HTTP_401_UNAUTHORIZED,
                detail="Bearer authentication required",
