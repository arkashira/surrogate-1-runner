import os
import datetime

CSV_PATH = "/opt/axentx/surrogate-1/data/startup_benchmarks.csv"

def read_csv(path):
    with open(path, newline='') as f:
        return list(csv.DictReader(f))

def test_file_exists():
    assert os.path.isfile(CSV_PATH), "CSV file missing"

def test_row_count():
    rows = read_csv(CSV_PATH)
    assert len(rows) == 30, f"Expected 30 rows, got {len(rows)}"

def test_required_columns():
    rows = read_csv(CSV_PATH)
    expected = {"company_name","sector","funding_stage","employees",
                "revenue_mrr","users","conversion_rate","retention_rate",
                "last_updated","region"}
    assert set(rows[0].keys()) == expected, "Missing or extra columns"

def test_numeric_fields_positive():
    rows = read_csv(CSV_PATH)
    for r in rows:
        for col in ["funding_stage","employees","revenue_mrr","users"]:
            val = float(r[col])
            assert val > 0, f"{col} must be > 0"

def test_percentage_bounds():
    rows = read_csv(CSV_PATH)
    for r in rows:
        for col in ["conversion_rate","retention_rate"]:
            val = float(r[col])
            assert 0 <= val <= 100, f"{col} out of bounds"

def test_date_format():
    rows = read_csv(CSV_PATH)
    for r in rows:
        try:
            datetime.datetime.strptime(r["last_updated"], "%Y-%m-%d")
        except ValueError:
            assert False, f"Invalid date format: {r['last_updated']}"

def test_region_nonempty():
    rows = read_csv(CSV_PATH)

        assert r["region"], "Region cannot be empty"
