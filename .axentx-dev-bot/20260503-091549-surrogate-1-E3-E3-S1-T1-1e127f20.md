const allowedTransitions: { [from: string]: string[] } = {

import { validateStatusTransition } from '../lib/statusTransitions';
import { Request } from '../models/Request';

// PATCH handler to update request status and append timeline event
async function patchRequestStatus(req: Request, res: any) {
  const { id, status, notes } = req.body;

  // Retrieve the request from the database
  const request = await Request.findById(id);

  if (!request) {
    return res.status(404).json({ error: 'Request not found' });
  }

  // Validate the status transition
  if (!validateStatusTransition(request.status, status)) {
    return res.status(400).json({ error: 'Invalid status transition' });
  }

  // Update the request status and append a timeline event
  request.status = status;
  request.updatedAt = new Date();
  request.timeline.push({
    actor: req.user.id,
    timestamp: new Date(),
    previousStatus: request.status,
    newStatus: status,
    notes,
  });

  await request.save();

  return res.json(request);
}

export { patchRequestStatus };
