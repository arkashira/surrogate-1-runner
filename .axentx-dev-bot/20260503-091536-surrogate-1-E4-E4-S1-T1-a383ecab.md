-- Requests being tracked (minimal)
CREATE TABLE IF NOT EXISTS requests (
  id              TEXT PRIMARY KEY,
  created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,
  current_status  TEXT NOT NULL DEFAULT 'pending'
);

-- Immutable audit log for status transitions
CREATE TABLE IF NOT EXISTS request_audit (
  id              INTEGER PRIMARY KEY AUTOINCREMENT,
  request_id      TEXT NOT NULL,
  actor           TEXT NOT NULL,           -- user/actor who caused transition
  previous_status TEXT NOT NULL,
  new_status      TEXT NOT NULL,
  created_at      DATETIME DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (request_id) REFERENCES requests(id)
);

-- Retention helper: index for fast lookup and cleanup

// audit.model.test.ts
import { recordStatusTransition } from './audit';
import { getDb } from './db';

beforeEach(() => {
  // fresh in-memory db per test
  getDb().exec(`DELETE FROM request_audit; DELETE FROM requests;`);
  getDb().exec(`INSERT INTO requests(id, current_status) VALUES (1, 'draft');`);
});

test('recordStatusTransition inserts immutable audit row and updates current_status', () => {
  const beforeCount = getDb().prepare(`SELECT COUNT(*) as c FROM request_audit`).get().c;
  recordStatusTransition({ requestId: 1, actor: 'u-123', from: 'draft', to: 'submitted', ts: new Date('2026-01-01T00:00:00Z') });
  const afterCount = getDb().prepare(`SELECT COUNT(*) as c FROM request_audit`).get().c;
  expect(afterCount).toBe(beforeCount + 1);

  const req = getDb().prepare(`SELECT current_status FROM requests WHERE id = 1`).get();
  expect(req.current_status).toBe('submitted');

  const audit = getDb().prepare(`SELECT * FROM request_audit ORDER BY id DESC LIMIT 1`).get();
  expect(audit.actor).toBe('u-123');
  expect(audit.previous_status).toBe('draft');
  expect(audit.new_status).toBe('submitted');
  expect(audit.created_at).toBe('2026-01-01T00:00:00.000Z');
});

test('duplicate transitions insert separate rows (append-only)', () => {
  const t = { requestId: 1, actor: 'u-123', from: 'draft', to: 'submitted', ts: new Date('2026-01-01T00:00:00Z') };
  recordStatusTransition(t);
  recordStatusTransition(t);
  const count = getDb().prepare(`SELECT COUNT(*) as c FROM request_audit`).get().c;
  expect(count).toBe(2);
});

test('audit row rejects null actor/previous_status/new_status', () => {
  expect(() => recordStatusTransition({ requestId: 1, actor: null, from: 'draft', to: 'submitted', ts: new Date() }))

    .toThrow();
  expect(() => recordStatusTransition({ requestId: 1, actor: 'u-1', from: 'draft', to: null, ts: new Date() }))
    .toThrow();
});
