-- Add immutable audit log for request status transitions
CREATE TABLE IF NOT EXISTS audit_log (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    request_id      UUID NOT NULL REFERENCES requests(id) ON DELETE CASCADE,
    actor_id        UUID NOT NULL,                     -- user who performed the transition
    previous_status TEXT NOT NULL,
    new_status      TEXT NOT NULL,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
    -- immutable: no updates allowed
    CONSTRAINT audit_log_no_updates CHECK (created_at <= now())
);

-- Index for fast lookup by request and time
CREATE INDEX IF NOT EXISTS idx_audit_log_request_created
    ON audit_log (request_id, created_at DESC);

-- Retention: automatically drop entries older than 90 days
-- Implemented via cron or PostgreSQL partitioning; here we add a helper function
CREATE OR REPLACE FUNCTION prune_audit_log_older_than(days INTEGER)
RETURNS INTEGER AS $$
DECLARE
    deleted_count INTEGER;
BEGIN
    DELETE FROM audit_log
    WHERE created_at < (now() - (days || ' days')::INTERVAL);
    GET DIAGNOSTICS deleted_count = ROW_COUNT;
    RETURN deleted_count;
END;
$$ LANGUAGE plpgsql;
