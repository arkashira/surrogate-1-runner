```python
# /opt/axentx/surrogate-1/notifications.py
"""
Unified notification system for Surrogate‑1.

Provides two ready‑to‑use notifiers:

* EmailNotifier – SMTP email (STARTTLS optional)
* SlackNotifier – Incoming webhook

Both read configuration from environment variables and expose a
`send` method that accepts a dict payload:

    {
        "title":   "High CPU usage",
        "message": "CPU usage exceeded 90% on host X",
        "remediation": "Check processes, consider scaling",
        "doc_url": "https://docs.example.com/alerts/cpu"
    }

The module also offers a small helper `send_alert` that sends the same
payload through all configured notifiers.
"""

from __future__ import annotations

import json
import os
import smtplib
import ssl
import urllib.request
from dataclasses import dataclass, field
from email.message import EmailMessage
from typing import Dict, Iterable, List, Optional, Protocol, runtime_checkable


# --------------------------------------------------------------------------- #
# 1.  Common alert body formatting
# --------------------------------------------------------------------------- #
def _format_body(alert: Dict[str, str]) -> str:
    """Return a human‑readable body from an alert dict."""
    parts = [
        alert.get("message", ""),
        "",
        f"Remediation: {alert.get('remediation', 'N/A')}",
        f"Documentation: {alert.get('doc_url', 'N/A')}",
    ]
    return "\n".join(parts)


# --------------------------------------------------------------------------- #
# 2.  Notifier protocol – useful for static type checking
# --------------------------------------------------------------------------- #
@runtime_checkable
class BaseNotifier(Protocol):
    def send(self, alert: Dict[str, str]) -> None: ...


# --------------------------------------------------------------------------- #
# 3.  Email notifier
# --------------------------------------------------------------------------- #
@dataclass
class EmailNotifier:
    smtp_server: str = field(default_factory=lambda: os.getenv("EMAIL_SMTP_SERVER", ""))
    smtp_port: int = field(default_factory=lambda: int(os.getenv("EMAIL_SMTP_PORT", "0")))
    username: str = field(default_factory=lambda: os.getenv("EMAIL_USERNAME", ""))
    password: str = field(default_factory=lambda: os.getenv("EMAIL_PASSWORD", ""))
    from_addr: str = field(default_factory=lambda: os.getenv("EMAIL_FROM", ""))
    to_addrs: List[str] = field(default_factory=lambda: [
        addr.strip()
        for addr in os.getenv("EMAIL_TO", "").split(",")
        if addr.strip()
    ])

    def __post_init__(self) -> None:
        missing = [
            var
            for var, val in [
                ("EMAIL_SMTP_SERVER", self.smtp_server),
                ("EMAIL_SMTP_PORT", self.smtp_port),
                ("EMAIL_FROM", self.from_addr),
                ("EMAIL_TO", self.to_addrs),
            ]
            if not val
        ]
        if missing:
            raise ValueError(f"Missing required email configuration: {', '.join(missing)}")

    def send(self, alert: Dict[str, str]) -> None:
        subject = f"[Alert] {alert.get('title', 'Untitled')}"
        body = _format_body(alert)
