import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

const JWT_SECRET = process.env.JWT_SECRET || 'dev-secret-change-me';
const PLATFORM_TEAM = 'platform';

export interface AuthRequest extends Request {
  user?: {
    sub: string;
    roles: string[];
    [key: string]: any;
  };
}

/**
 * Verify JWT and attach user to request.
 * Rejects if token missing/invalid or user lacks platform role.
 */
export function requirePlatformTeam(req: AuthRequest, res: Response, next: NextFunction) {
  const authHeader = req.headers.authorization;
  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Missing authorization token' });
  }

  const token = authHeader.split(' ')[1];
  try {
    const payload = jwt.verify(token, JWT_SECRET) as any;
    // Expected shape: { sub, roles: [...] }
    if (!payload?.roles || !Array.isArray(payload.roles)) {
      return res.status(403).json({ error: 'Invalid token claims' });
    }

    req.user = {
      sub: payload.sub || payload.userId || 'unknown',
      roles: payload.roles,
      ...payload,
    };

    if (!req.user.roles.includes(PLATFORM_TEAM)) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }

    return next();
  } catch (err) {
    return res.status(401).json({ error: 'Invalid or expired token' });
  }
}

/**
 * Optional authentication (does not require platform role).
 * Useful for other endpoints.
 */
export function optionalAuth(req: AuthRequest, res: Response, next: NextFunction) {
  const authHeader = req.headers.authorization;
  if (!authHeader?.startsWith('Bearer ')) return next();

  const token = authHeader.split(' ')[1];
  try {
    const payload = jwt.verify(token, JWT_SECRET) as any;
    req.user = {
      sub: payload.sub || payload.userId || 'unknown',
      roles: payload.roles || [],
      ...payload,
    };
  } catch {
    // ignore invalid token for optional auth
  }
  return next();
}
