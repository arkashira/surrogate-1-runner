import { NextFunction, Request, Response, Router } from 'express';
import { verify } from 'jsonwebtoken';
import { platformTeamMembers } from '../config/platform-team-members';

export interface AuthRequest extends Request {
  user?: {
    id: string;
    roles: string[];
    [key: string]: any;
  };
}

/**
 * Require authenticated user.
 */
export function requireAuth(req: AuthRequest, res: Response, next: NextFunction) {
  if (!req.user) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  next();
}

/**
 * Require user to be a platform team member.
 * Expects req.user.roles to include 'platform'.
 */
export function requirePlatformTeam(req: AuthRequest, res: Response, next: NextFunction) {
  if (!req.user) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  const roles = req.user.roles || [];
  if (!roles.includes('platform')) {
    return res.status(403).json({ error: 'Forbidden: platform team required' });
  }
  next();
}
