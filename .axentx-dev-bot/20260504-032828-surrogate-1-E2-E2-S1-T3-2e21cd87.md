## 1. High‑level goals

| Goal | Why it matters |
|------|----------------|
| **Session‑centric** | All collaboration is scoped to a session; participants can join/leave, and state is isolated per session. |
| **Fine‑grained presence** | Each participant has a cursor, selection, and last‑activity timestamp so the UI can show who is doing what. |
| **Shared state sync** | A single `SharedState` struct is broadcast on every change; clients can request a full sync if they fall behind. |
| **Activity log** | Every action is recorded for audit and replay. |
| **Real‑time messaging** | Use `tokio::sync::broadcast` for low‑latency, fan‑out communication. |
| **Extensibility** | New message types (e.g., chat, annotations) can be added without touching the core loop. |
| **Testability** | All public methods are `async` and return `Result`, making unit tests straightforward. |

use std::{
    collections::HashMap,
    sync::Arc,
};
use chrono::{DateTime, Utc};
use tokio::sync::{broadcast, RwLock};
use uuid::Uuid;

/// Unique identifiers ---------------------------------------------------------

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct SessionId(pub String);

impl SessionId {
    pub fn new() -> Self { Self(Uuid::new_v4().to_string()) }
}
impl Default for SessionId { fn default() -> Self { Self::new() } }

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
pub struct ParticipantId(pub String);

impl ParticipantId {
    pub fn new(name: &str) -> Self {
        Self(format!("{}_{}", name, Uuid::new_v4().to_simple().to_string()[..8]))
    }
}

/// Participant metadata -------------------------------------------------------

#[derive(Debug, Clone)]
pub struct Participant {
    pub id: ParticipantId,
    pub name: String,
    pub color: String,
    pub joined_at: DateTime<Utc>,
    pub last_activity: DateTime<Utc>,
    pub cursor: Option<CursorPosition>,
    pub selection: Option<SelectionRange>,
}

#[derive(Debug, Clone, PartialEq)]
pub struct CursorPosition {
    pub x: f64,
    pub y: f64,
    pub node_id: Option<String>,
}

#[derive(Debug, Clone, PartialEq)]
pub struct SelectionRange {
    pub start_node_id: String,
    pub end_node_id: String,
}

/// Shared state visible to all participants ----------------------------------

#[derive(Debug, Clone, Default)]
pub struct SharedState {
    pub highlighted_nodes: Vec<String>,
    pub expanded_groups: Vec<String>,
    pub zoom_level: f64,
    pub pan_offset: (f64, f64),
    pub breakpoints: Vec<String>,
    pub annotations: HashMap<String, String>,
}

/// Activity log --------------------------------------------------------------

#[derive(Debug, Clone)]
pub struct ActivityEntry {
    pub timestamp: DateTime<Utc>,
    pub participant_id: ParticipantId,
    pub action: ActivityAction,
}

#[derive(Debug, Clone)]
pub enum ActivityAction {
    Joined { participant_name: String },
    Left { participant_name: String },
    CursorMoved { node_id: Option<String> },
    SelectionChanged { start: String, end: String },
    NodeExpanded { node_id: String },
    NodeCollapsed { node_id: String },
    BreakpointAdded { node_id: String },
    BreakpointRemoved { node_id: String },
    AnnotationAdded { node_id: String, text: String },
    AnnotationRemoved { node_id: String },
    StateSync { state_type: String },
    CommentAdded { node_id: String, text: String },
}

/// Session --------------------------------------------------------------

#[derive(Debug, Clone)]
pub struct Session {
    pub id: SessionId,
    pub name: String,
    pub created_at: DateTime<Utc>,
    pub created_by: ParticipantId,
    pub participants: HashMap<ParticipantId, Participant>,
    pub shared_state: SharedState,
    pub activity_log: Vec<ActivityEntry>,
}
