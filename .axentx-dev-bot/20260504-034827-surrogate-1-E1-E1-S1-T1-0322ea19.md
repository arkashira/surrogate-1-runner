import java.util.Random;

public class Generator {
    private static final Random random = new Random();

    /**
     * Generates synthetic data based on the specified integration type.
     * 
     * @param integrationType The type of integration ("logs" or "metrics").
     * @return A map containing synthetic data.
     */
    public Map<String, Object> generateSyntheticData(String integrationType) {
        Map<String, Object> syntheticData = new HashMap<>();
        
        switch (integrationType) {

                syntheticData.put("level", getRandomLogLevel());

                syntheticData.put("metric_name", "example_metric");
                syntheticData.put("value", random.nextDouble() * 100);
                syntheticData.put("timestamp", System.currentTimeMillis());
                break;
            default:
                throw new IllegalArgumentException("Unsupported integration type: " + integrationType);
        }
        
        return syntheticData;
    }

    /**
     * Generates a random string of a specified length.
     * 
     * @param length The length of the string to generate.
     * @return A random string.
     */
    private String generateRandomString(int length) {
        String characters = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
        StringBuilder sb = new StringBuilder(length);
        for (int i = 0; i < length; i++) {
            int index = random.nextInt(characters.length());
            sb.append(characters.charAt(index));
        }
        return sb.toString();
    }

    /**
     * Returns a random log level from a predefined list.
     * 
     * @return A random log level.
     */
    private String getRandomLogLevel() {
        String[] logLevels = {"DEBUG", "INFO", "WARNING", "ERROR", "CRITICAL"};
        return logLevels[random.nextInt(logLevels.length)];
    }

    /**
     * Generates synthetic data using a provided generator function.
     * 
     * @param <T> The type of the generated data.
     * @param generator The function used to generate the data.
     * @param customization An integer representing customization parameters.
     * @return The generated synthetic data.
     */
    public static <T> T generateSyntheticData(Function<Integer, T> generator, int customization) {
        return generator.apply(customization);
    }
