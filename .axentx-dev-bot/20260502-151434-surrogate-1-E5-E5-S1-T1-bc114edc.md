## 4) Pipeline executor (src/executor.py)
```python
from __future__ import annotations

import json
import time
from pathlib import Path
from typing import Any, Dict, List


class PipelineExecutor:
    """
    Minimal, deterministic pipeline executor.
    Supports: noop, sleep, transform.
    """

    @staticmethod
    def run_step(step: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        step_type = step.get("type")
        config = step.get("config", {})

        if step_type == "noop":
            return {"status": "success", "output": context}

        if step_type == "sleep":
            ms = int(config.get("ms", 0))
            time.sleep(max(0, ms) / 1000.0)
            return {"status": "success", "output": context}

        if step_type == "transform":
            return {"status": "success", "output": config.get("output", {})}

        return {"status": "failed", "error": f"Unknown step type: {step_type}"}

    @classmethod
    def run_pipeline(cls, pipeline_path: str | Path) -> Dict[str, Any]:
        path = Path(pipeline_path)
        if not path.is_file():
            return {"status": "failed", "error": f"Pipeline file not found: {pipeline_path}"}

        payload = json.loads(path.read_text(encoding="utf-8"))
        steps: List[Dict[str, Any]] = payload.get("pipeline", {}).get("steps", [])
        context: Dict[str, Any] = {}

        for step in steps:
            result = cls.run_step(step, context)
            if result.get("status") != "success":
                return result
            context = result.get("output", context)

        return {"status": "success", "output": context}
```
