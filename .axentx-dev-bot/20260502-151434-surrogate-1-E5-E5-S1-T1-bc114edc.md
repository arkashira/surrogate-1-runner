import json
import time
from pathlib import Path
from typing import Any, Dict, List, Callable


Fn = Callable[..., Any]


def builtin_uppercase(s: str) -> str:
    return s.upper()


def builtin_length_lt(s: str, max: int) -> bool:
    return len(s) < max


FN_MAP: Dict[str, Fn] = {
    "uppercase": builtin_uppercase,
    "length_lt": builtin_length_lt,
}


class PipelineExecutor:
    def __init__(self, pipeline_path: str | Path) -> None:
        self.pipeline_path = Path(pipeline_path)
        with self.pipeline_path.open("r") as f:
            self.pipeline = json.load(f)
        self.steps: List[Dict[str, Any]] = self.pipeline.get("steps", [])
        self.context: Dict[str, Any] = {}

    def run(self) -> Dict[str, Any]:
        start = time.perf_counter()
        value: Any = None

        for step in self.steps:
            step_id = step.get("id")
            fn_name = step.get("fn")
            fn = FN_MAP.get(fn_name)
            if fn is None:
                raise ValueError(f"Unknown function: {fn_name}")

            if step_id == "step_1":
                value = step.get("input", "")
                value = fn(value)
                self.context[step_id] = value
            elif step_id == "step_2":
                args = step.get("args", {})
                ok = fn(value, **args)
                self.context[step_id] = ok
                if not ok:
                    raise AssertionError(f"Validation failed at {step_id}")
            else:
                raise ValueError(f"Unknown step id: {step_id}")

        elapsed_ms = (time.perf_counter() - start) * 1000.0
        return {
            "pipeline_id": self.pipeline.get("id"),
            "success": True,
            "elapsed_ms": elapsed_ms,
            "output": value,
            "context": self.context,
        }
