import time
from dataclasses import dataclass, asdict
from typing import Any, Dict, List


@dataclass
class StepResult:
    name: str
    status: str  # "ok" | "fail"
    latency_ms: float
    output: Any = None


class PipelineExecutor:
    """
    Minimal, safe pipeline executor.

    Pipeline schema:
    {
      "name": "sample",
      "steps": [
        {"name": "step1", "type": "noop", "delay_ms": 10},
        {"name": "step2", "type": "transform", "expr": "x * 2"}
      ]
    }

    Notes:
    - Uses a tiny safe evaluator (no builtins) for transform expressions.
    - Keeps context as a dict; transform steps must produce a dict output.
    """

    _SAFE_OPS = {
        "+": operator.add,
        "-": operator.sub,
        "*": operator.mul,
        "/": operator.truediv,
        "//": operator.floordiv,
        "%": operator.mod,
        "**": operator.pow,

        Supported forms (examples):
          - "x * 2"
          - "x + 1"
        Returns a dict with the same keys as ctx, each transformed by expr
        where variable names in ctx are available in the expression scope.
        """
        if not expr or not expr.strip():
            raise ValueError("expr is required for transform steps")
