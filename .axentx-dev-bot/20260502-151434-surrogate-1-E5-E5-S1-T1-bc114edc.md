import json
import time
from pathlib import Path


def execute_step(step, context):
    step_type = step.get("type")
    cfg = step.get("config", {})

    if step_type == "noop":
        delay_ms = cfg.get("delay_ms", 0)
        if delay_ms:
            time.sleep(delay_ms / 1000.0)
        return {"status": "ok", "output": context}

    if step_type == "transform":
        expr = cfg.get("expression", "")
        try:
            payload = context.get("payload", "hello world")
            if "x.upper()" in expr:
                output = payload.upper() if isinstance(payload, str) else payload
            else:
                output = payload
            context = {"payload": output}
        except Exception as exc:
            return {"status": "error", "error": str(exc)}
        return {"status": "ok", "output": context}

    return {"status": "error", "error": f"Unknown step type: {step_type}"}


def run_pipeline(pipeline):
    context = {"payload": "hello world"}
    for step in pipeline.get("steps", []):
        result = execute_step(step, context)
        if result["status"] != "ok":
            return {"success": False, "error": result.get("error"), "context": context}
        context = result.get("output", context)
    return {"success": True, "context": context}


def run_pipeline_from_config(config_path):
    p = Path(config_path)
    with open(p, "r", encoding="utf-8") as f:
        pipeline = json.load(f)

    start = time.perf_counter()
    result = run_pipeline(pipeline)
    elapsed_ms = (time.perf_counter() - start) * 1000.0
    result["latency_ms"] = elapsed_ms
    return result
```
