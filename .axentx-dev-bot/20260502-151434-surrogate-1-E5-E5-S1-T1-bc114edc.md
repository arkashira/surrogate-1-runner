from typing import Any, Dict


class PipelineExecutor:
    """
    Minimal deterministic executor for the sample pipeline.
    Deterministic behavior is required for tests and CI.
    """

    def run(self, config: Dict[str, Any]) -> Dict[str, Any]:
        steps = config.get("steps", [])
        payload = {"input": "hello world"}

        for step in steps:
            step_type = step.get("type")
            cfg = step.get("config", {})

            if step_type == "validator":
                required = cfg.get("required_fields", [])
                for field in required:
                    if field not in payload:
                        return {"ok": False, "error": f"missing {field}", "output": payload}

            elif step_type == "transformer":
                op = cfg.get("operation")
                field = cfg.get("field")
                if op == "uppercase" and field in payload:
                    payload[field] = str(payload[field]).upper()

            elif step_type == "emitter":
                out_key = cfg.get("output_key", "result")
                # Emit copies the transformed input (or final computed value) to output_key
                payload[out_key] = payload.get("input", "")

        return {"ok": True, "output": payload}
```

"""

from __future__ import annotations

import json
import os
from pathlib import Path
from typing import Iterable


class DedupStore:
    """
    Append-only md5 dedup store backed by a JSONL file.

    Format (one line per record):
        {"md5": "<hex>", "ts": "<iso8601>"}
    """

    def __init__(self, index_path: str | Path) -> None:
        self.index_path = Path(index_path)
        self._seen: set[str] = set()
        self._load()

    def _load(self) -> None:
        if not self.index_path.exists():
            self.index_path.parent.mkdir(parents=True, exist_ok=True)
            return
        try:
            with self.index_path.open("r", encoding="utf-8") as fh:
                for line in fh:
                    line = line.strip()
                    if not line:
                        continue
                    try:
                        rec = json.loads(line)
                        md5 = rec.get("md5")
                        if isinstance(md5, str) and md5:
                            self._seen.add(md5)
                    except json.JSONDecodeError:
                        continue
        except OSError:
            self._seen.clear()

    def exists(self, md5: str) -> bool:
        return md5 in self._seen

    def add(self, md5: str) -> bool:
        """Record md5 if not already seen. Returns True if newly added."""
        if self.exists(md5):
            return False
        self._seen.add(md5)

        return True
