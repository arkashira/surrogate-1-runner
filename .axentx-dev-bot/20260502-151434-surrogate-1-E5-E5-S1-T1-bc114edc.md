from typing import Any, Dict


class PipelineExecutor:
    """
    Minimal deterministic executor for the sample pipeline.
    Deterministic behavior is required for tests and CI.
    """

    def run(self, config: Dict[str, Any]) -> Dict[str, Any]:
        steps = config.get("steps", [])
        payload = {"input": "hello world"}

        for step in steps:
            step_type = step.get("type")
            cfg = step.get("config", {})

            if step_type == "validator":
                required = cfg.get("required_fields", [])
                for field in required:
                    if field not in payload:
                        return {"ok": False, "error": f"missing {field}", "output": payload}

            elif step_type == "transformer":
                op = cfg.get("operation")
                field = cfg.get("field")
                if op == "uppercase" and field in payload:
                    payload[field] = str(payload[field]).upper()

            elif step_type == "emitter":
                out_key = cfg.get("output_key", "result")
                # Emit copies the transformed input (or final computed value) to output_key
                payload[out_key] = payload.get("input", "")

        return {"ok": True, "output": payload}

#!/usr/bin/env python3
"""
Central MD5 deduplication store.

Provides idempotent duplicate checks and atomic mark-seen operations
using a local append-only log + fast index.
"""

import json
import os
import sys
from pathlib import Path
from typing import Set

DEDUP_DIR = Path(os.getenv("DEDUP_DIR", ".dedup"))
INDEX_FILE = DEDUP_DIR / "index.json"
LOG_FILE = DEDUP_DIR / "seen.log"


def _ensure_store() -> None:
    DEDUP_DIR.mkdir(exist_ok=True)
    if not INDEX_FILE.exists():
        INDEX_FILE.write_text("{}")
    if not LOG_FILE.exists():
        LOG_FILE.touch()


def _load_index() -> Set[str]:
    try:
        return set(json.loads(INDEX_FILE.read_text()))
    except (json.JSONDecodeError, FileNotFoundError):
        return set()


def _persist_index(index: Set[str]) -> None:
    tmp = INDEX_FILE.with_suffix(".tmp")
    tmp.write_text(json.dumps(sorted(index), separators=(",", ":")))
    tmp.replace(INDEX_FILE)


def _append_log(md5: str) -> None:
    with LOG_FILE.open("a") as f:
        f.write(f"{md5}\n")


def is_duplicate(md5: str) -> bool:
    """Return True if md5 was already marked seen."""
    _ensure_store()
    index = _load_index()
    return md5 in index


def mark_seen(md5: str) -> bool:
    """
    Mark md5 as seen if not already present.
    Returns True if newly marked, False if duplicate.
    """
    _ensure_store()
    index = _load_index()
    if md5 in index:
        return False
    index.add(md5)
    _append_log(md5)
    _persist_index(index)
    return True


def batch_mark_seen(md5s: list[str]) -> int:
    """Mark multiple hashes; return count of newly marked."""
    if not md5s:
        return 0
    _ensure_store()
    index = _load_index()
    new_hashes = [h for h in md5s if h not in index]
    if not new_hashes:
        return 0
    index.update(new_hashes)
    with LOG_FILE.open("a") as f:
        f.write("\n".join(new_hashes) + "\n")
    _persist_index(index)
    return len(new_hashes)


if __name__ == "__main__":
    # CLI: dedup.py <md5> [...]
    # Prints one line per arg: <md5> <is_duplicate>
    for arg in sys.argv[1:]:
        md5 = arg.strip().lower()
        dup = is_duplicate(md5)
        print(f"{md5} {dup}")
```
