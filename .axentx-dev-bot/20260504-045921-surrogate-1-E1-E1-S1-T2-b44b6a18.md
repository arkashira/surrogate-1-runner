First, we have the `ClaudeLLMAdapter` class that implements the `LLMAdapter` interface. This class handles the conversion between the common LLM request/response format and the Claude specific format. It uses the WebClient reactive API to make requests to the Anthropic API.

```java
@Component
public class ClaudeLLMAdapter implements LLMAdapter {

    private final WebClient webClient;
    private final String apiKey;
    private final String baseUrl;

    public ClaudeLLMAdapter(
            @Value("${claude.api-key}") String apiKey,
            @Value("${claude.base-url}") String baseUrl) {
        this.apiKey = apiKey;
        this.baseUrl = baseUrl;
        this.webClient = WebClient.builder()
                .baseUrl(baseUrl)
                .defaultHeader("x-api-key", apiKey)
                .defaultHeader("anthropic-version", "2023-06-01")
                .build();
    }

    @Override
    public Mono<LLMResponse> generate(LLMRequest request) {
        // Build the Claude request payload
        ClaudeRequestPayload payload = new ClaudeRequestPayload(
                request.getPrompt(),
                request.getMaxTokens(),
                request.getTemperature()
        );

        return webClient.post()
                .uri("/messages")
                .bodyValue(payload)
                .retrieve()
                .bodyToMono(ClaudeResponsePayload.class)
                .map(this::toLLMResponse);
    }

    private LLMResponse toLLMResponse(ClaudeResponsePayload payload) {
        // Claude returns a list of message objects; we take the first content block
        String content = payload.getContent().stream()
                .map(msg -> msg.getText())
                .findFirst()
                .orElse("");
        return new LLMResponse(content);
    }

    // DTOs for request/response mapping

    private static class ClaudeRequestPayload {
        private final String model = "claude-3-haiku-20240307";
        private final String max_tokens_to_sample;
        private final double temperature;
        private final String system;
        private final java.util.List<Message> messages;

        ClaudeRequestPayload(String prompt, int maxTokens, double temperature) {
            this.max_tokens_to_sample = String.valueOf(maxTokens);
            this.temperature = temperature;
            this.system = "You are a helpful assistant.";
            this.messages = java.util.List.of(new Message("user", prompt));
        }

        static class Message {
            private final String role;
            private final String content;

            Message(String role, String content) {
                this.role = role;
                this.content = content;
            }

            public String getRole() { return role; }
            public String getContent() { return content; }
        }
    }
}
```
