import logging
from datetime import datetime, timezone
from typing import Optional

from fastapi import APIRouter, Depends, HTTPException, Header, Response
from fastapi.responses import JSONResponse

router = APIRouter()
logger = logging.getLogger("document_open")

# Lightweight store abstraction — replace with real DB/object-store calls.
# Keep this layer thin to guarantee sub-1s responses.
_DOCUMENT_STORE = {
    "doc-001": {
        "id": "doc-001",
        "title": "Annual Report 2025",
        "page_count": 12,
        "pages": {
            1: {"text": "Executive Summary...", "image_url": "/static/doc-001/page-1.png"},
            2: {"text": "Financial Overview...", "image_url": "/static/doc-001/page-2.png"},
        },
    }
}

def get_current_user_id(x_user_id: Optional[str] = Header(None)) -> str:
    """
    Resolve user identity.
    Prefer header in this minimal example; replace with auth/session logic in prod.
    """
    return x_user_id or "anonymous"

def _log_open_event(document_id: str, user_id: str) -> None:
    """Emit analytics event. Keep non-blocking in production (e.g., background task / queue)."""
    logger.info(
        "document_open",
        extra={
            "document_id": document_id,
            "user_id": user_id,
            "timestamp": datetime.now(timezone.utc).isoformat(),
        },
    )

@router.get("/documents/{document_id}/open")
async def open_document(
    document_id: str,
    user_id: str = Depends(get_current_user_id),
    x_cache_ttl: Optional[int] = Header(30),
) -> JSONResponse:
    doc = _DOCUMENT_STORE.get(document_id)
    if not doc:
        raise HTTPException(status_code=404, detail="Document not found")

    # Required analytics
    _log_open_event(document_id=document_id, user_id=user_id)

    # Lightweight payload for sub-1s initial render
    payload = {
        "id": doc["id"],
        "title": doc["title"],
        "page_count": doc["page_count"],
        "current_page": 1,
        "page": doc["pages"].get(1, {"text": "", "image_url": None}),
    }

    # Encourage fast repeat loads with short-lived caching; omit for personalized responses.
    response = JSONResponse(content=payload)
    response.headers["Cache-Control"] = f"private, max-age={min(int(x_cache_ttl), 5)}"
    return response
