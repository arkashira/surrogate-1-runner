"""
Base integration classes for Surrogate-1.

Provides abstract base classes and common functionality
for all external service integrations.
"""

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Dict, List, Optional
from datetime import datetime
import logging
import hashlib
import secrets
import requests
from urllib.parse import urljoin

logger = logging.getLogger(__name__)


class IntegrationType(Enum):
    """Supported integration types."""
    GITHUB = "github"
    SLACK = "slack"
    JIRA = "jira"
    ZAPIER = "zapier"


class ActionType(Enum):
    """Common integration actions."""
    # GitHub actions
    CREATE_ISSUE = "create_issue"
    CREATE_PR = "create_pr"
    LIST_REPOS = "list_repos"
    GET_WORKFLOW = "get_workflow"
    TRIGGER_WORKFLOW = "trigger_workflow"
    
    # Slack actions
    SEND_MESSAGE = "send_message"
    CREATE_CHANNEL = "create_channel"
    INVITE_USER = "invite_user"
    
    # Jira actions
    CREATE_TICKET = "create_ticket"
    UPDATE_TICKET = "update_ticket"
    TRANSITION_TICKET = "transition_ticket"
    
    # Zapier actions
    TRIGGER_ZAP = "trigger_zap"
    GET_ZAPS = "get_zaps"


@dataclass
class IntegrationConfig:
    """Configuration for an integration."""
    integration_type: IntegrationType
    api_key: Optional[str] = None
    oauth_token: Optional[str] = None
    webhook_secret: Optional[str] = None
    base_url: Optional[str] = None
    extra_headers: Dict[str, str] = field(default_factory=dict)
    timeout: int = 30
    retry_count: int = 3
    
    def __post_init__(self):
        """Validate configuration after initialization."""
        if not self.api_key and not self.oauth_token:
            raise ValueError("Either api_key or oauth_token must be provided")


@dataclass
class IntegrationResult:
    """Result from an integration action."""
    success: bool
    data: Optional[Dict[str, Any]] = None
    error: Optional[str] = None
    action: ActionType
    timestamp: datetime = field(default_factory=datetime.utcnow)
    request_id: str = field(default_factory=lambda: secrets.token_hex(8))
    
    def to_dict(self) -> Dict[str, Any]:
        """Convert result to dictionary."""
        return {
            "success": self.success,
            "data": self.data,
            "error": self.error,
            "action": self.action.value,
            "timestamp": self.timestamp.isoformat(),
            "request_id": self.request_id,
        }


class IntegrationBase(ABC):
    """Abstract base class for all integrations."""
    
    def __init__(self, config: IntegrationConfig):
        """Initialize integration with configuration."""
        self.config = config
        self._validate_credentials()
        self.logger = logging.getLogger(f"{__name__}.{self.__class__.__name__}")
