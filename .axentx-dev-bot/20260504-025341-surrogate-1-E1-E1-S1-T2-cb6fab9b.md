  /** 2‑D array of data – each sub‑array is a row */
  data: T[][];
  /** Optional header row(s) – each sub‑array is a header row */
  headers?: string[][];
  /** Optional cell renderer – receives cell value, row index, col index */
  renderCell?: (

    rowIndex: number,
    colIndex: number

  renderCell,

        setSelected({ row, col });

      let newCol = col;
