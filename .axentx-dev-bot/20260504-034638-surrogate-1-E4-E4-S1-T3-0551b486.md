import { render, screen } from '@testing-library/react';
import { BrowserRouter as Router } from 'react-router-dom';
import { Provider } from 'react-redux';
import store from '../store';
import VulnerabilityDetails from '../src/main/webapp/dashboard/vulnerabilityDetails';

describe('VulnerabilityDetails component', () => {
  it('renders a "Vulnerability Details" page with a back link to the dashboard', () => {

      <Provider store={store}>
        <Router>
          <VulnerabilityDetails />
        </Router>
      </Provider>

    expect(screen.getByText('Back to dashboard')).toBeInTheDocument();
  });

  it('fetches vulnerability data from the /api/vulnerabilities/{id} endpoint', async () => {

    const fetchMock = jest.fn(() => Promise.resolve(response));
    global.fetch = fetchMock;
    render(
      <Provider store={store}>
        <Router>
          <VulnerabilityDetails id={id} />
        </Router>
      </Provider>
    );
    await new Promise(resolve => setTimeout(resolve, 100));
    expect(fetchMock).toHaveBeenCalledTimes(1);
    expect(fetchMock).toHaveBeenCalledWith(`/api/vulnerabilities/${encodeURIComponent(id)}`);

      <Provider store={store}>
        <Router>
          <VulnerabilityDetails />
        </Router>
      </Provider>
    );

  });
