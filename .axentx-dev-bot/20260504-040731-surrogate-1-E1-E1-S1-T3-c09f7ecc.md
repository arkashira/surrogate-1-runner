const fs = require('fs');
const path = require('path');
const glob = require('glob');
const { parse } = require('@babel/parser');
const traverse = require('@babel/traverse').default;

function readPackageJson() {
  const packageJsonPath = path.join(__dirname, '../package.json');
  return JSON.parse(fs.readFileSync(packageJsonPath, 'utf-8'));
}

function getDeclaredPackages(packageJson) {
  return new Set([
    ...Object.keys(packageJson.dependencies || {}),
    ...Object.keys(packageJson.devDependencies || {})
  ]);
}

function findImportsInFile(filePath) {
  const content = fs.readFileSync(filePath, 'utf-8');
  const ast = parse(content, { sourceType: 'module' });
  const imports = [];

  traverse(ast, {
    ImportDeclaration(path) {
      const source = path.node.source.value;
      imports.push(source);
    }
  });

  return imports;
}

function resolvePackageNames(imports) {
  const packageNames = new Set();
  imports.forEach(importPath => {
    try {
      const resolvedPath = require.resolve(importPath, { paths: [__dirname] });
      const packageName = require(resolvedPath).name;
      packageNames.add(packageName);
    } catch (error) {
      // Ignore errors, it might be a local file import
    }
  });
  return packageNames;
}

function checkMissingImports() {
  const packageJson = readPackageJson();
  const declaredPackages = getDeclaredPackages(packageJson);

  const jsTsFiles = glob.sync('src/**/*.@(js|ts)');
  const allImports = jsTsFiles.flatMap(findImportsInFile);
  const importedPackages = resolvePackageNames(allImports);

  const missingImports = Array.from(importedPackages).filter(pkg => !declaredPackages.has(pkg));
  const result = {
    missingImports,
    hasErrors: missingImports.length > 0
  };

  console.log(JSON.stringify(result, null, 2));

  if (result.hasErrors) {
    process.exit(1);
  }
}

checkMissingImports();

## 1. Acceptance Criteria  
- **A1** – The checker scans **all** `.js` and `.ts` files under the `src/` directory recursively.  
- **A2** – It extracts every **static** `import` statement (e.g., `import foo from 'bar'`, `import {baz} from 'qux'`).  
- **A3** – Import paths that resolve to **node_modules packages** (e.g., `'lodash'`, `'@org/pkg'`) are mapped to their package names.  
- **A4** – The set of resolved package names is compared against the union of `dependencies` and `devDependencies` in `package.json`.  
- **A5** – A JSON report is printed to stdout with the shape:  
  ```json
  {
    "missingImports": ["pkg1", "pkg2"]
  }
  ```  
  If no missing imports exist, the array is empty.  
- **A6** – The process exits with status code **0** when `missingImports` is empty; otherwise it exits with status code **1**.  
- **A7** – The checker ignores relative imports (paths starting with `./` or `../`) and built‑in Node modules (e.g., `fs`, `path`).

const { findMissingImports, resolvePackageName } = require('./checker');
const fs = require('fs');
const path = require('path');

jest.mock('fs');
jest.mock('path');

describe('resolvePackageName', () => {
  it('returns the package name for a simple import', () => {
    expect(resolvePackageName('lodash')).toBe('lodash');
  });

  it('returns the package name for scoped imports', () => {
    expect(resolvePackageName('@org/pkg')).toBe('@org/pkg');
  });

  it('returns null for relative paths', () => {
    expect(resolvePackageName('./utils')).toBeNull();
  });

  it('returns null for built‑in modules', () => {
    expect(resolvePackageName('fs')).toBeNull();
  });
});

describe('findMissingImports', () => {
  const mockPackageJson = {
    dependencies: { lodash: '^4.17.21' },
    devDependencies: { jest: '^27.0.0' }
  };

  beforeEach(() => {
    fs.readdirSync.mockReset();
    fs.readFileSync.mockReset();
  });

  it('returns empty array when all imports are declared', () => {
    fs.readdirSync.mockReturnValueOnce(['src'])
      .mockReturnValueOnce(['index.js']);
    fs.readFileSync.mockReturnValueOnce(`
      import _ from 'lodash';
      import { test } from 'jest';
    `);

    const missing = findMissingImports('src', mockPackageJson);
    expect(missing).toEqual([]);
  });

  it('detects a single missing import', () => {
    fs.readdirSync.mockReturnValueOnce(['src'])
      .mockReturnValueOnce(['index.js']);
    fs.readFileSync.mockReturnValueOnce(`
      import _ from 'lodash';
      import foo from 'missing-lib';
    `);

    const missing = findMissingImports('src', mockPackageJson);
    expect(missing).toEqual(['missing-lib']);
  });

  it('ignores relative imports', () => {
    fs.readdirSync.mockReturnValueOnce(['src'])

    const missing = findMissingImports('src', mockPackageJson);
    expect(missing).toEqual([]);
  });

  it('ignores built‑in modules', () => {
    fs.readdirSync.mockReturnValueOnce(['src'])

    fs.readFileSync.mockReturnValueOnce(`
      import fs from 'fs';
    `);

    const missing = findMissingImports('src', mockPackageJson);
    expect(missing).toEqual([]);
