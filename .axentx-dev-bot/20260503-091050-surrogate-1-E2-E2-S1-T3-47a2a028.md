// Session-persisted, reactive filter state for library search.
// Designed for <500ms search across ≤1k documents.

type DateRange = { from?: string; to?: string };

interface Stored {
  fileTypes: string[];
  dateAdded: DateRange;
}

const STORAGE_KEY = 'axentx:library:filters';
const listeners = new Set<() => void>();

function load(): Stored {
  try {
    const raw = sessionStorage.getItem(STORAGE_KEY);
    if (!raw) return { fileTypes: [], dateAdded: {} };
    return JSON.parse(raw) as Stored;
  } catch {
    return { fileTypes: [], dateAdded: {} };
  }
}

let state = load();

function save() {
  sessionStorage.setItem(STORAGE_KEY, JSON.stringify(state));
  listeners.forEach((l) => l());
}

export const filterState = {
  get fileTypes() {
    return state.fileTypes.slice();
  },
  get dateAdded() {
    return { ...state.dateAdded };
  },

  hasFileType(type: string) {
    return state.fileTypes.includes(type);
  },

  setFileTypes(types: string[]) {
    state.fileTypes = types.slice();
    save();
  },

  addFileType(type: string) {
    if (!state.fileTypes.includes(type)) {
      state.fileTypes.push(type);
      save();
    }
  },

  removeFileType(type: string) {
    const idx = state.fileTypes.indexOf(type);
    if (idx >= 0) {
      state.fileTypes.splice(idx, 1);
      save();
    }
  },

  setDateAdded(range: DateRange) {
    state.dateAdded = { ...range };
    save();
  },

  clearDateAdded() {
    state.dateAdded = {};
    save();
  },

  clearAll() {
    state.fileTypes = [];
    state.dateAdded = {};
    save();
  },

  subscribe(cb: () => void) {
    listeners.add(cb);
    return () => listeners.delete(cb);
  },

  snapshot() {
    return {
      fileTypes: state.fileTypes.slice(),
      dateAdded: { ...state.dateAdded },
    };
  },
} as const;
```
