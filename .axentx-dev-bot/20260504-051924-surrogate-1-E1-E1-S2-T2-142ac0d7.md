```javascript
const { renderHook } = require('@testing-library/react-hooks');
const { useValidation } = require('../hooks/useValidation');

describe('useValidation', () => {
  it('returns initial state with no errors', () => {
    const { result } = renderHook(() => useValidation('fieldName', 'initialValue'));
    expect(result.current.errors).toEqual({});
  });

  it('adds error when validation fails', () => {
    const { result, rerender } = renderHook(({ value }) => useValidation('fieldName', value), {
      initialProps: { value: 'initialValue' },
    });
    rerender({ value: 'invalidValue' });
    expect(result.current.errors).toHaveProperty('fieldName', 'Error message for invalidValue');
  });

  it('removes error when validation passes', () => {
    const { result, rerender } = renderHook(({ value }) => useValidation('fieldName', value), {
      initialProps: { value: 'invalidValue' },
    });
    rerender({ value: 'validValue' });
    expect(result.current.errors).not.toHaveProperty('fieldName');
  });
});
```
