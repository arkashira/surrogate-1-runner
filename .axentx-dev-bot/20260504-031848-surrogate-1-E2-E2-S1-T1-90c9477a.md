// trivy_parser_test.go
package validate

import (
    "testing"
    "encoding/json"
    "github.com/stretchr/testify/assert"
)

func TestParseTrivyJSON_SimpleVulnerability(t *testing.T) {
    jsonInput := `{
        "Results": [
            {
                "Vulnerabilities": [
                    {
                        "VulnerabilityID": "CVE-2023-1234",
                        "PkgName": "openssl",
                        "InstalledVersion": "1.1.1",
                        "FixedVersion": "1.1.1g",
                        "Severity": "HIGH",
                        "Title": "OpenSSL arbitrary code execution",
                        "Description": "Some description",
                        "PrimaryURL": "https://nvd.nist.gov/vuln/detail/CVE-2023-1234"
                    }
                ]
            }
        ]
    }`

    var trivyOut TrivyOutput
    err := json.Unmarshal([]byte(jsonInput), &trivyOut)
    assert.NoError(t, err)

    vulns := ParseTrivyOutput(&trivyOut)
    assert.Len(t, vulns, 1)
    v := vulns[0]
    assert.Equal(t, "CVE-2023-1234", v.ID)
    assert.Equal(t, "HIGH", v.Severity)
    assert.Contains(t, v.ReportLink, "CVE-2023-1234")
}

func TestParseTrivyJSON_SkipsLowSeverity(t *testing.T) {
    jsonInput := `{
        "Results": [
            {
                "Vulnerabilities": [
                    {
                        "VulnerabilityID": "CVE-2023-0001",
                        "Severity": "LOW",
                        "Title": "Low risk vulnerability"
                    }
                ]
            }
        ]
    }`

    var trivyOut TrivyOutput
    json.Unmarshal([]byte(jsonInput), &trivyOut)
    vulns := ParseTrivyOutput(&trivyOut)
    assert.Len(t, vulns, 0)
}

func TestParseTrivyJSON_MissingSeverity(t *testing.T) {
    jsonInput := `{
        "Results": [
            {
                "Vulnerabilities": [
                    {
                        "VulnerabilityID": "CVE-2023-9999",
                        "Title": "Missing severity"
                    }
                ]
            }
        ]
    }`

    var trivyOut TrivyOutput
    json.Unmarshal([]byte(jsonInput), &trivyOut)
    vulns := ParseTrivyOutput(&trivyOut)
    assert.Len(t, vulns, 0)
}

func TestParseTrivyJSON_MultipleResults(t *testing.T) {
    jsonInput := `{
        "Results": [
            {
                "Vulnerabilities": [
                    {"VulnerabilityID":"CVE-1","Severity":"CRITICAL","Title":"A"},
                    {"VulnerabilityID":"CVE-2","Severity":"LOW","Title":"B"}
                ]
            },
            {
                "Vulnerabilities": [
                    {"VulnerabilityID":"CVE-3","Severity":"HIGH","Title":"C"}
                ]
            }
        ]
    }`

    var trivyOut TrivyOutput
    json.Unmarshal([]byte(jsonInput), &trivyOut)
    vulns := ParseTrivyOutput(&trivyOut)
    assert.Len(t, vulns, 2)
    ids := []string{vulns[0].ID, vulns[1].ID}
    assert.Contains(t, ids, "CVE-1")
    assert.Contains(t, ids, "CVE-3")
}
