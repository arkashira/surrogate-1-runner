- **The function handles arrays of required fields of any length (1-10+)**

// Jest-style unit tests for /opt/axentx/surrogate-1/frontend/src/utils/validation.js

describe('validateForm', () => {
  const requiredFields = ['title', 'description', 'templateType'];

  test('returns isValid: true when all required fields are present', () => {
    const formData = {
      title: 'My AI Spec',
      description: 'A classification model',
      templateType: 'Classification'
    };
    
    const result = validateForm(formData, requiredFields);
    
    expect(result.isValid).toBe(true);
    expect(result.errors).toEqual([]);
  });

  test('returns isValid: false with single error for one missing field', () => {
    const formData = {
      title: 'My AI Spec',
      description: 'A classification model',
      templateType: ''  // empty string
    };
    
    const result = validateForm(formData, requiredFields);
    
    expect(result.isValid).toBe(false);
    expect(result.errors).toContain('templateType');
    expect(result.errors.length).toBe(1);
  });

  test('returns all missing fields in errors array', () => {
    const formData = {
      title: '',  // missing
      description: null,  // missing
      templateType: undefined  // missing
    };
    
    const result = validateForm(formData, requiredFields);
    
    expect(result.isValid).toBe(false);
    expect(result.errors).toHaveLength(3);
    expect(result.errors).toContain('title');
    expect(result.errors).toContain('description');
    expect(result.errors).toContain('templateType');
  });

  test('returns isValid: true for empty form with no required fields', () => {
    const result = validateForm({}, []);
    
    expect(result.isValid).toBe(true);
    expect(result.errors).toEqual([]);
  });

  test('ignores extra fields not in requiredFields array', () => {
    const formData = {
      title: 'My Spec',
      description: 'Desc',
      templateType: 'Retrieval',
      extraField: '',  // not required
      anotherExtra: null
    };
    
    const result = validateForm(formData, requiredFields);
    
    expect(result.isValid).toBe(true);
  });

  test('handles whitespace-only strings as missing', () => {
    const formData = {
      title: '   ',  // whitespace only
      description: 'Valid description',
      templateType: 'Generation'
    };
    
    const result = validateForm(formData, requiredFields);
    
    expect(result.isValid).toBe(false);
    expect(result.errors).toContain('title');
  });
