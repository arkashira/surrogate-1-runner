"""
Production-ready Billing API endpoints for subscription management.

Combines:
- Clean FastAPI router structure (Candidate 1)
- Database + service layer architecture (Candidate 2)
- Proper configuration management
- Complete Stripe integration with webhooks
"""

from fastapi import APIRouter, HTTPException, Request, Depends
from sqlalchemy.orm import Session
from pydantic import BaseModel, Field
from typing import Optional
import stripe
import os

from ..database import get_db
from ..models import User, Subscription
from ..services.stripe_service import (
    create_stripe_checkout_session,
    get_stripe_subscription,
    cancel_stripe_subscription
)
from ..services.email_service import send_subscription_email
from ..config import settings

# Configure Stripe
stripe.api_key = settings.STRIPE_API_KEY

router = APIRouter(prefix="/billing", tags=["billing"])


# ─────────────────────────────────────────────────────────────
# Pydantic Schemas (from Candidate 2's pattern)
# ─────────────────────────────────────────────────────────────

class CheckoutRequest(BaseModel):
    user_id: str = Field(..., description="User ID for the checkout session")
    price_id: str = Field(default="price_annual_4800", description="Stripe price ID")


class CheckoutResponse(BaseModel):
    checkout_url: str
    session_id: str


class SubscriptionStatusResponse(BaseModel):
    status: str
    plan_name: Optional[str] = None
    current_period_end: Optional[str] = None


class WebhookResponse(BaseModel):
    received: bool


# ─────────────────────────────────────────────────────────────
# API Endpoints (refined from Candidate 1)
# ─────────────────────────────────────────────────────────────

@router.post("/checkout", response_model=CheckoutResponse)
async def create_checkout_session(
    request: CheckoutRequest,
    db: Session = Depends(get_db)
):
    """
    Create a Stripe checkout session for the annual plan.
    
    Features:
    - 14-day free trial
    - Creates/retrieves Stripe customer
    - Stores subscription in database
    """
    # Verify user exists
    user = db.query(User).filter(User.id == request.user_id).first()
    if not user:
        raise HTTPException(status_code=404, detail="User not found")
    
    # Get or create Stripe customer
    if user.stripe_customer_id:
        customer_id = user.stripe_customer_id
    else:
        customer = stripe.Customer.create(
            email=user.email,
            metadata={"user_id": user.id}
        )
        customer_id = customer.id
        user.stripe_customer_id = customer_id
        db.commit()
    
    # Create checkout session with trial
    try:
        session = stripe.checkout.Session.create(
            payment_method_types=["card"],
            line_items=[{
                "price": request.price_id,
                "quantity": 1
            }],
            mode="subscription",
            trial_period_days=14,
            success_url=f"{settings.FRONTEND_URL}/success?session_id={{CHECKOUT_SESSION_ID}}",
            cancel_url=f"{settings.FRONTEND_URL}/cancel",
            customer=customer_id,
            metadata={"user_id": user.id}
        )
    except stripe.error.StripeError as e:
        raise HTTPException(status_code=400, detail=str(e))
    
    # Create pending subscription record
    subscription = Subscription(
        user_id=user.id,
        stripe_subscription_id=session.subscription,
        status="pending",
        plan="annual",
        trial_end=None  # Will be set via webhook
    )
    db.add(subscription)
    db.commit()
    
    return CheckoutResponse(
        checkout_url=session.url,
        session_id=session.id
    )
