import logging
from typing import Dict, Any, List, Tuple, Union

logger = logging.getLogger(__name__)

# Define the expected schema for discovered datasets
REQUIRED_FIELDS = {
    "title": str,
    "size": (int, str),  # Size can be an integer (bytes) or a string (e.g., "100MB")
    "task_category": list,
    "license": str,
    "source": str,  # e.g., "huggingface", "kaggle"
    "format": str,  # Additional field for file format
}

ALLOWED_FORMATS = ["csv", "json", "parquet", "txt", "xlsx"]  # Allowed file formats

class MetadataValidator:
    """
    Validates dataset metadata against a predefined schema.
    Ensures required fields exist, are of the correct type, and meet semantic rules.
    """

    def validate(self, dataset_info: Dict[str, Any]) -> Tuple[bool, List[str]]:
        """
        Validates a dataset dictionary against the schema.

        Args:
            dataset_info (Dict[str, Any]): The dataset metadata to validate.

        Returns:
            Tuple[bool, List[str]]: A tuple containing a boolean indicating validity and a list of error messages.
        """
        errors = []

        if not isinstance(dataset_info, dict):
            errors.append("Validation failed: dataset_info must be a dictionary.")
            return False, errors

        for field, expected_type in REQUIRED_FIELDS.items():
            if field not in dataset_info:
                errors.append(f"Validation failed: Missing required field '{field}' in dataset metadata.")
                continue

            value = dataset_info[field]
            if not isinstance(value, expected_type):
                errors.append(
                    f"Validation failed: Field '{field}' has invalid type. Expected {expected_type}, got {type(value)}."
                )
                continue

            if field == "title" and value.strip() == "":
                errors.append("Validation failed: 'title' field cannot be empty.")
                continue

            if field == "size":
                if isinstance(value, str) and not value.isdigit():
                    errors.append(f"Validation failed: 'size' string value '{value}' is not numeric.")
                    continue

            if field == "format" and value not in ALLOWED_FORMATS:
                errors.append(f"Unsupported 'format': {value!r}. Allowed: {ALLOWED_FORMATS}")

        if errors:
            logger.error("\n".join(errors))
            return False, errors

        logger.info("Dataset metadata validation passed.")
        return True, []
