from __future__ import annotations
from dataclasses import dataclass


class InsufficientCreditsError(RuntimeError):
    """
    Raised when a request for credits cannot be satisfied by either the
    monthly or bulk pool.

    The exception message contains the exact state of both pools so that
    callers (or logs) can see *why* the operation failed.
    """
    def __init__(self, *, monthly: int, bulk: int, requested: int):
        self.monthly = monthly
        self.bulk = bulk
        self.requested = requested
        super().__init__(self._make_message())

    def _make_message(self) -> str:
        parts = [
            f"Requested {self.requested} credit(s)",
            f"Monthly pool: {self.monthly}",
            f"Bulk pool: {self.bulk}",
        ]
        return " – ".join(parts) + " – insufficient credits."

    The class is deliberately thin – the four ``_get_*`` / ``_deduct_*`` helpers
    are *private* and meant to be overridden (or mocked) by the surrounding
