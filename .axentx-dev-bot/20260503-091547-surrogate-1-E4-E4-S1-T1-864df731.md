 export type RequestStatus = 'open' | 'in_progress' | 'resolved' | 'closed';

 export interface Request {
   id: string;
   teamId: string;
   title: string;
   description?: string;
   status: RequestStatus;
   createdAt: Date;
   updatedAt: Date;
   resolvedAt?: Date;
+  slaTargetHours?: number;
+}
+
+export interface RequestWithComputed extends Request {
+  slaElapsedHours: number;
+  slaState: SlaState;
+  slaBreachAt?: Date;
+}
+
+function hoursBetween(a: Date, b: Date): number {
+  return Math.abs(b.getTime() - a.getTime()) / (1000 * 60 * 60);
+}
+
+export function buildRequestWithComputed(req: Request): RequestWithComputed {
+  const now = new Date();
+  const elapsed = hoursBetween(req.createdAt, now);
+  const breachAt = req.slaTargetHours != null ? new Date(req.createdAt.getTime() + req.slaTargetHours * 60 * 60 * 1000) : undefined;
+
+  let state: SlaState = 'ok';
+  if (req.slaTargetHours != null && breachAt) {
+    if (now > breachAt) {
+      state = 'breach';
+    } else if (elapsed >= req.slaTargetHours * 0.8) {
+      state = 'warning';
+    }
+  }
+
+  return {
+    ...req,
+    slaElapsedHours: Number(elapsed.toFixed(2)),
+    slaState: state,
+    slaBreachAt: breachAt,
+  };
+}

    expect(req.slaBreachAt).toBeUndefined();

  });

  test('slaState ok when elapsed < target', () => {
