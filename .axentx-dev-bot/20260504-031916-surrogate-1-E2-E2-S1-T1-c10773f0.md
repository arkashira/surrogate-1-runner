import logging
import os
import json
from datetime import datetime
from typing import Any, Dict

class StructuredLogger:
    """Structured logger for surrogate-1 paste operations with configurable log levels."""
    
    def __init__(self, name: str = "surrogate-1-paste"):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(logging.INFO)
        
        # Remove existing handlers to avoid duplicates
        self.logger.handlers.clear()
        
        # Configure handler with JSON formatting for structured logs
        handler = logging.StreamHandler()
        formatter = logging.Formatter(
            json.dumps({
                "timestamp": datetime.utcnow().isoformat() + "Z",
                "level": logging.getLevelName(logging.INFO),
                "message": "{message}",
                "module": "{module}",
                "function": "{funcName}",
                "line": "{lineno}",
                "extra": "{extra}"
            }, default=str),
            "%(message)s"
        )
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)
        
        # Configure log level from environment variable
        self._configure_log_level()
    
    def _configure_log_level(self) -> None:
        """Configure log level from environment variable LOG_LEVEL."""
        env_level = os.getenv("LOG_LEVEL", "INFO").upper()
        level_map = {
            "DEBUG": logging.DEBUG,
            "INFO": logging.INFO,
            "WARNING": logging.WARNING,
            "ERROR": logging.ERROR,
            "CRITICAL": logging.CRITICAL

        self.logger.setLevel(level_map.get(env_level, logging.INFO))
    
    def log_paste_attempt(

import { StructuredLogger } from '../src/logger';
import { jest } from '@jest/globals';

describe('StructuredLogger', () => {
  let logger;
  beforeEach(() => {
    process.env.LOG_LEVEL = 'DEBUG';
    logger = new StructuredLogger();
  });

  test('log_paste_attempt outputs JSON with required fields', () => {
    const consoleSpy = jest.spyOn(console, 'log').mockImplementation(() => {});
    logger.log_paste_attempt('paste', 'op123');
    const logged = consoleSpy.mock.calls[0][0];
    const parsed = JSON.parse(logged);
    expect(parsed).toMatchObject({
      timestamp: expect.any(String),
      method: 'paste',
      operation_id: 'op123',
      level: 'INFO',
    });
    consoleSpy.mockRestore();
  });

  test('log_paste_failure includes failure_reason', () => {
    const consoleSpy = jest.spyOn(console, 'error').mockImplementation(() => {});
    logger.log_paste_failure('paste', 'op456', 'Clipboard empty');
    const logged = consoleSpy.mock.calls[0][0];
    const parsed = JSON.parse(logged);
    expect(parsed.failure_reason).toBe('Clipboard empty');
    consoleSpy.mockRestore();
  });

  test('log level respects LOG_LEVEL env var', () => {
    process.env.LOG_LEVEL = 'ERROR';
    const loggerErr = new StructuredLogger();
    const consoleSpy = jest.spyOn(console, 'error').mockImplementation(() => {});
    const consoleInfoSpy = jest.spyOn(console, 'log').mockImplementation(() => {});

    consoleSpy.mockRestore();

  });
