- Uses a **single API call** from the runner (after rate-limit window) to fetch `list_repo_tree(path=DATE_FOLDER, recursive=False)` and saves the file list to `manifest.json`.

  ```
  batches/public-merged/<DATE_FOLDER>/shard<N>-<HHMMSS>.jsonl
  ```

#!/usr/bin/env python3
"""
CDN-bypass ingestion worker for surrogate-1-training-pairs.
Usage (GitHub Actions matrix):
  SHARD_ID=0 SHARD_TOTAL=16 python bin/dataset-enrich.py

Optional env:
  DATE_FOLDER=YYYY-MM-DD   (default: today)
  HF_REPO=datasets/axentx/surrogate-1-training-pairs
  OUT_DIR=batches/public-merged
"""
import os
import sys
import json
import hashlib
import datetime
import time
from pathlib import Path
from typing import List, Dict, Any

import requests
import pyarrow.parquet as pq
import pyarrow as pa
from huggingface_hub import HfApi

# ---- config ----
HF_REPO = os.getenv("HF_REPO", "datasets/axentx/surrogate-1-training-pairs")
SHARD_ID = int(os.getenv("SHARD_ID", "0"))
SHARD_TOTAL = int(os.getenv("SHARD_TOTAL", "16"))
DATE_FOLDER = os.getenv("DATE_FOLDER", datetime.date.today().isoformat())
OUT_DIR = Path(os.getenv("OUT_DIR", f"batches/public-merged/{DATE_FOLDER}"))
MANIFEST_PATH = Path("manifest.json")
HF_TOKEN = os.getenv("HF_TOKEN", "")  # optional for API list; CDN downloads are public

# ---- helpers ----
def deterministic_shard(slug: str) -> int:
    h = int(hashlib.md5(slug.encode()).hexdigest(), 16)
    return h % SHARD_TOTAL

def cdn_url(repo: str, path: str) -> str:
    # repo format: datasets/owner/name
    _, owner, name = repo.split("/", 2)
    return f"https://huggingface.co/datasets/{owner}/{name}/resolve/main/{path}"

def list_date_files(date_folder: str) -> List[str]:
    """Single API call to list files in date folder (non-recursive)."""
    api = HfApi(token=HF_TOKEN or None)
    try:
        tree = api.list_repo_tree(repo_id=HF_REPO, path=date_folder, recursive=False)
    except Exception as e:
        print(f"[WARN] HF API list failed: {e}", file=sys.stderr)
        # fallback: try to read existing manifest
        if MANIFEST_PATH.exists():
            print("[INFO] Using existing manifest.json", file=sys.stderr)
            return json.loads(MANIFEST_PATH.read_text())
        raise
    files = [item.rfilename for item in tree if item.type == "file"]
    MANIFEST_PATH.write_text(json.dumps(files, indent=2))
    return files

def project_to_pair(file_path: Path) -> List[Dict[str, str]]:
    """
    Project file to {prompt, response} pairs.
    Supports .jsonl and .parquet; ignores schema mismatches by selecting
    only known columns and coercing to string.

    pairs = []
