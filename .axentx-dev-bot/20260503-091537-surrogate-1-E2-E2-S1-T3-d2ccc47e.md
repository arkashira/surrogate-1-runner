import { Hono } from 'hono';
import { cors } from 'hono/cors';

type Status = 'open' | 'in_progress' | 'resolved' | 'closed';

const validTransitions: Record<Status, Status[]> = {
  open: ['in_progress'],
  in_progress: ['resolved', 'closed'],
  resolved: ['closed'],
  closed: [],
};

interface Request {
  id: string;
  title: string;
  status: Status;
  createdAt: string;
}

interface AuditEntry {
  id: string;
  requestId: string;
  type: 'status_change' | 'comment';
  actor: string;
  timestamp: string;
  data: any;
}

// In-memory stores (replace with DB in production)
const requests = new Map<string, Request>();
const auditLog = new Map<string, AuditEntry[]>(); // requestId -> entries[]
const subscribers = new Map<string, ((entry: AuditEntry) => void)[]>();

function emitAudit(requestId: string, entry: AuditEntry) {
  const list = auditLog.get(requestId) || [];
  list.push(entry);
  // keep immutable and ordered by timestamp
  list.sort((a, b) => a.timestamp.localeCompare(b.timestamp));
  auditLog.set(requestId, list);

  // notify SSE subscribers
  (subscribers.get(requestId) || []).forEach((cb) => {
    try {
      cb(entry);
    } catch {}
  });
}

const app = new Hono();

app.use('*', cors());

// Helper to require actor (from header for now)
function getActor(c: any): string {
  return c.req.header('x-actor-id') || 'system';
}

// PATCH /requests/{id}/status
app.patch('/requests/:id/status', async (c) => {
  const id = c.req.param('id');
  const body = await c.req.json();
  const nextStatus = body.status as Status;
  const actor = getActor(c);

  if (!nextStatus) {
    return c.json({ error: 'status is required' }, 400);
  }

  const req = requests.get(id);
  if (!req) {
    return c.json({ error: 'request not found' }, 404);
  }

  const allowed = validTransitions[req.status] || [];
  if (!allowed.includes(nextStatus)) {
    return c.json(
      {
        error: 'invalid status transition',
        from: req.status,
        to: nextStatus,
        allowed,
      },
      400
    );
  }

  req.status = nextStatus;
  requests.set(id, req);

  const entry: AuditEntry = {
    id: crypto.randomUUID(),
    requestId: id,
    type: 'status_change',
    actor,
    timestamp: new Date().toISOString(),
    data: { from: req.status === nextStatus ? body.from || req.status : req.status, to: nextStatus },
  };
  // ensure we capture accurate previous state
  entry.data.from = body.from || req.status === nextStatus ? Object.keys(validTransitions).find((k) => validTransitions[k as Status].includes(nextStatus)) || 'open' : req.status;

  emitAudit(id, entry);
