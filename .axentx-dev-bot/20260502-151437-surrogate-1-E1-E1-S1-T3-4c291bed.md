class _Histogram:
    """Low-overhead, thread-safe histogram with linear interpolation per bucket."""
    def __init__(self, buckets: List[int]) -> None:
        self.buckets = sorted(buckets)
        self.counts = [0] * (len(self.buckets) + 1)  # +1 overflow
        self.total = 0.0
        self.count = 0
        self.min_v = float("inf")
        self.max_v = float("-inf")
        self._lock = threading.Lock()

    def observe(self, value_ms: float) -> None:
        with self._lock:
            self.count += 1
            self.total += value_ms
            self.min_v = min(self.min_v, value_ms)
            self.max_v = max(self.max_v, value_ms)
            idx = 0

                    break
                idx = i + 1
            else:

            self.counts[idx] += 1

        with self._lock:
            if self.count == 0:
                return None

            seen = 0
            prev_limit = 0.0
            for i, c in enumerate(self.counts):
                seen += c
