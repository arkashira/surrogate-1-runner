+    In-memory latency histogram for pipeline execution times.
+    Buckets: powers of 2 up to 2048ms, plus overflow.
+    Thread-unsafe by design; use per-process or external locking if needed.
+    """
+
+    def __init__(self, buckets: List[float] = None) -> None:
+        if buckets is None:
+            # 1,2,4,8,16,32,64,128,256,512,1024,2048 ms + overflow
+            buckets = [2 ** i for i in range(12)]  # 1..2048
+        self.buckets = sorted(buckets)
+        self.counts = [0] * (len(self.buckets) + 1)  # +1 for overflow
+        self.values: List[float] = []  # keep raw samples for exact percentiles
+        self._lock = False  # placeholder; replace with threading.Lock() if multithreaded
+
+    def record(self, duration_ms: float) -> None:
+        """Record a latency sample."""
+        if duration_ms < 0:
+            raise ValueError("duration_ms must be non-negative")
+        idx = bisect.bisect_left(self.buckets, duration_ms)
+        self.counts[idx] += 1
+        self.values.append(duration_ms)
+
+    def snapshot(self) -> Dict[str, Any]:
+        """Return percentile summary and bucket counts."""
+        if not self.values:
+            return {
+                "count": 0,
+                "p50": None,
+                "p95": None,
+                "p99": None,
+                "min": None,
+                "max": None,
+                "mean": None,

+                "buckets": dict(zip(self.bucket_labels(), self.counts)),
