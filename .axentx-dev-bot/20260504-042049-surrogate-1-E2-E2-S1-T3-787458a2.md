# --------------------------------------------------------------------------- #

class KnowledgeArticleDraft:
    """
    A lightweight, serialisable representation of a knowledge‑base article
    that has been generated from a single Git commit.
    """

    commit_hash: str
    author: str
    date: datetime
    jira_keys: List[str] = field(default_factory=list)
    title: str = ""
    body: str = ""

    # Convenience helpers
    # --------------------------------------------------------------------- #

            "title": self.title,
            "body": self.body,

        cls,
        commit_hash: str,

        jira_keys = _unique_jira_keys(message)
