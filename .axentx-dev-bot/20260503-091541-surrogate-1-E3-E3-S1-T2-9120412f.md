### 1) Add correct Slack webhook utility (bash/node hybrid approach)
We’ll use a tiny Node helper so the shell script can call it without complex curl escaping. This uses Slack Incoming Webhooks (env: `SLACK_WEBHOOK_URL`).

`bin/lib/slack.js`
```js
#!/usr/bin/env node
// Minimal, correct Slack webhook sender.
// Usage: SLACK_WEBHOOK_URL=... node bin/lib/slack.js "message text" [--channel "#alerts"]
// If SLACK_WEBHOOK_URL is not set, silently exits 0 (no-op).

const url = process.env.SLACK_WEBHOOK_URL;
if (!url) {
  process.exit(0);
}

const args = process.argv.slice(2);
if (args.length === 0) {
  console.error('slack.js: missing message');
  process.exit(1);
}

let text = args[0] || '';
let channel = null;
for (let i = 1; i < args.length; i++) {
  if (args[i] === '--channel' && args[i + 1]) {
    channel = args[i + 1];
    i++;
  }
}

const payload = { text };
if (channel) payload.channel = channel;

fetch(url, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify(payload),
})
  .then((res) => {
    if (!res.ok) {
      console.error(`slack.js: webhook responded ${res.status}`);
      process.exit(2);
    }
  })
  .catch((err) => {
    console.error('slack.js: failed to send', err.message);
    process.exit(3);
  });
```
