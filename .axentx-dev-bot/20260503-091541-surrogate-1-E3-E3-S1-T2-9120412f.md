import { Team } from '../models/team';
import { SlackService } from '../services/slackService';
import { logger } from '../utils/logger';

class TeamService {
  private slackService: SlackService;

  constructor(slackService: SlackService) {
    this.slackService = slackService;
  }

  async updateTeamStatus(team: Team, newStatus: string) {
    // Validate the Slack channel config
    if (!team.slack_channel_config) {
      logger.info(`No Slack channel config found for team: ${team.name}`);
      return;
    }

    // Send a formatted Slack message to the configured channel
    const message = `Request ${team.name} has been updated to ${newStatus}`;
    try {
      await this.slackService.postMessage(team.slack_channel_config, message);
    } catch (error) {
      logger.error('Error sending Slack message:', error);
    }
  }
}

export { TeamService };
