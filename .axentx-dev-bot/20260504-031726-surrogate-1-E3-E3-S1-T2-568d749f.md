// JiraConfig.vue.test.ts
import { mount } from '@vue/test-utils';
import JiraConfig from '@/ui/JiraConfig.vue';
import { describe, it, expect, vi } from 'vitest';
import { fetchJiraConfig, saveJiraConfig } from '@/api/jiraConfig';

vi.mock('@/api/jiraConfig');

describe('JiraConfig.vue', () => {
  it('renders configuration form with required fields', () => {
    const wrapper = mount(JiraConfig);
    expect(wrapper.find('input[name="jiraUrl"]').exists()).toBe(true);
    expect(wrapper.find('input[name="email"]').exists()).toBe(true);
    expect(wrapper.find('input[name="apiToken"]').exists()).toBe(true);
  });

  it('displays error for invalid Jira URL', async () => {
    const wrapper = mount(JiraConfig);
    const urlInput = wrapper.find('input[name="jiraUrl"]');
    await urlInput.setValue('invalid-url');
    await wrapper.find('form').trigger('submit');
    expect(wrapper.text()).toContain('Valid Jira URL is required');
  });

  it('calls save API with correct payload including Safety and Categories metadata', async () => {
    const mockSave = vi.mocked(saveJiraConfig);
    mockSave.mockResolvedValue({ success: true });

    const wrapper = mount(JiraConfig);
    await wrapper.find('input[name="jiraUrl"]').setValue('https://example.atlassian.net');
    await wrapper.find('input[name="email"]').setValue('user@example.com');
    await wrapper.find('input[name="apiToken"]').setValue('dummy-token');
    await wrapper.find('form').trigger('submit');

    expect(mockSave).toHaveBeenCalledWith(
      expect.objectContaining({
        jiraUrl: 'https://example.atlassian.net',
        email: 'user@example.com',
        apiToken: 'dummy-token',
        metadata: {
          Safety: 'ENCRYPTED',
          Categories: ['Integration', 'Jira']
        }
      })
    );
  });

  it('loads and displays existing config on mount', async () => {
    vi.mocked(fetchJiraConfig).mockResolvedValue({
      jiraUrl: 'https://saved.atlassian.net',
      email: 'saved@example.com',
      apiToken: '****',
      metadata: { Safety: 'ENCRYPTED', Categories: ['Integration', 'Jira'] }
    });

    const wrapper = mount(JiraConfig);
    await wrapper.vm.$nextTick();

    expect(wrapper.find('input[name="jiraUrl"]').element.value).toBe('https://saved.atlassian.net');
    expect(wrapper.find('input[name="email"]').element.value).toBe('saved@example.com');
  });

  it('does not expose API token in plain text when loaded', async () => {
    vi.mocked(fetchJiraConfig).mockResolvedValue({
      jiraUrl: 'https://example.atlassian.net',
      email: 'user@example.com',
      apiToken: 'hashed-or-encrypted-value',
      metadata: { Safety: 'ENCRYPTED', Categories: ['Integration', 'Jira'] }
    });

    const wrapper = mount(JiraConfig);
    await wrapper.vm.$nextTick();
    const tokenInput = wrapper.find('input[name="apiToken"]');
    expect(tokenInput.element.value).toBe(''); // Should be masked or empty
  });
});
