# test_workflow_controller.py
import pytest
from fastapi.testclient import TestClient
from surrogate_1.src.api.workflow_controller import app

client = TestClient(app)

def test_start_workflow_returns_id_and_running_state():
    resp = client.post("/workflow/start")
    assert resp.status_code == 200
    data = resp.json()
    assert "workflow_id" in data
    assert data["status"] == "running"

def test_pause_workflow_updates_status():
    start = client.post("/workflow/start").json()
    wid = start["workflow_id"]
    resp = client.post(f"/workflow/pause/{wid}")
    assert resp.status_code == 200
    assert resp.json()["status"] == "paused"

def test_resume_workflow_updates_status():
    start = client.post("/workflow/start").json()
    wid = start["workflow_id"]
    client.post(f"/workflow/pause/{wid}")  # pause first
    resp = client.post(f"/workflow/resume/{wid}")
    assert resp.status_code == 200
    assert resp.json()["status"] == "running"

def test_get_workflow_state():
    start = client.post("/workflow/start").json()
    wid = start["workflow_id"]
    resp = client.get(f"/workflow/{wid}")
    assert resp.status_code == 200
    state = resp.json()
    assert state["status"] in {"running", "paused", "completed"}
    assert "current_step" in state

def test_snapshot_is_deep_copy():
    start = client.post("/workflow/start").json()
    wid = start["workflow_id"]
    snap = client.get(f"/workflow/{wid}/snapshot").json()
    # mutate snapshot
    snap["current_step"] = "modified"
    # ensure live state unchanged
    live = client.get(f"/workflow/{wid}").json()
    assert live["current_step"] != "modified"

def test_override_updates_state():
    start = client.post("/workflow/start").json()
    wid = start["workflow_id"]
    override_payload = {"current_step": "overridden_step"}
    resp = client.post(f"/workflow/override/{wid}", json=override_payload)
    assert resp.status_code == 200
    assert resp.json()["current_step"] == "overridden_step"

def test_demo_counter_completes():
    resp = client.post("/workflow/start?demo=true")
    assert resp.status_code == 200
    data = resp.json()
    # wait until completion
    while data["status"] != "completed":
        data = client.get(f"/workflow/{data['workflow_id']}").json()
    assert data["result"] == 10
