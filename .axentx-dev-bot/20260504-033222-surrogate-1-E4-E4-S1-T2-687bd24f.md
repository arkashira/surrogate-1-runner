"""
Digest renderer – Markdown & plain‑text.

The module exposes:

* :class:`DigestItem` – a frozen dataclass that represents a single
  high‑impact item.
* :func:`render_markdown` – returns a Markdown string suitable for
  Slack/Teams.
* :func:`render_plain_text` – returns a plain‑text string suitable for
  email.

Both helpers accept a headline and a list of ``DigestItem`` objects.
They sort by ``confidence`` (descending) and truncate to the first
ten items.  Timestamps are normalised to UTC when possible.
"""

from __future__ import annotations

import datetime as _dt
from dataclasses import dataclass
from typing import List

__all__ = ["DigestItem", "render_markdown", "render_plain_text"]


# --------------------------------------------------------------------------- #
# 1️⃣  Data model
# --------------------------------------------------------------------------- #
@dataclass(frozen=True, slots=True)
class DigestItem:
    """A single high‑impact item to be displayed in the digest."""

    author: str
    """Original author of the message."""

    timestamp: _dt.datetime
    """Timestamp of the original message (timezone‑aware preferred)."""

    snippet: str
    """Translated snippet (already limited to ≤ 2 sentences)."""

    confidence: float
    """Confidence score used for ordering (higher = more important)."""


# --------------------------------------------------------------------------- #
# 2️⃣  Helpers
# --------------------------------------------------------------------------- #
def _format_timestamp(ts: _dt.datetime) -> str:
    """
    Return a human‑readable timestamp.

    * If ``ts`` is timezone‑aware, it is converted to UTC and suffixed
      with “UTC”.
    * Naïve datetimes are rendered as‑is.
    """
    if ts.tzinfo is not None:
        ts = ts.astimezone(_dt.timezone.utc)
        return ts.strftime("%Y-%m-%d %H:%M:%S UTC")
    return ts.strftime("%Y-%m-%d %H:%M:%S")
