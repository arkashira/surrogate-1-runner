# /opt/axentx/surrogate-1/src/lib/sla.ts
const MINUTE = 60;
const HOUR = 60 * MINUTE;
const DAY = 24 * HOUR;

export function computeSlaRemaining(
  createdAt: Date,
  slaDurationSeconds: number,
  now: Date = new Date()
): number {
  const elapsed = (now.getTime() - createdAt.getTime()) / 1000;
  return slaDurationSeconds - elapsed;
}

export function getSlaStatus(
  remainingSeconds: number,
  atRiskThresholdSeconds: number = 4 * HOUR
): 'on-track' | 'at-risk' | 'breached' {
  if (remainingSeconds < 0) return 'breached';
  if (remainingSeconds <= atRiskThresholdSeconds) return 'at-risk';
  return 'on-track';
}

export function formatSlaStatus(
  createdAt: Date,
  slaDurationSeconds: number,
  lastUpdated: Date,
  now: Date = new Date()
) {
  const remaining = computeSlaRemaining(createdAt, slaDurationSeconds, now);
  const status = getSlaStatus(remaining);

  const absRemaining = Math.abs(remaining);
  let human = '';
  if (absRemaining < MINUTE) human = `${Math.ceil(absRemaining)}s`;
  else if (absRemaining < HOUR) human = `${Math.ceil(absRemaining / MINUTE)}m`;
  else if (absRemaining < DAY) human = `${Math.ceil(absRemaining / HOUR)}h`;
  else human = `${Math.ceil(absRemaining / DAY)}d`;

  const prefix = remaining >= 0 ? '' : '-';
  const remainingHuman = remaining >= 0 ? `${prefix}${human} remaining` : `${prefix}${human} overdue`;

  return {
    status,
    remaining: remainingHuman,
    lastUpdated: lastUpdated.toISOString(),
  };
}
