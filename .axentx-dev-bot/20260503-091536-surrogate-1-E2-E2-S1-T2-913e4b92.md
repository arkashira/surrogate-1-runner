import { Request } from '../models/request';

const MS = 1000;

export function computeRemainingSeconds(request: Request): number | null {
  if (!request.slaDuration || request.slaDuration <= 0) return null;
  const elapsed = Date.now() - request.createdAt.getTime();
  const remaining = request.slaDuration - Math.floor(elapsed / MS);
  return Math.max(remaining, 0);
}

export function getSlaStatus(request: Request): 'on-track' | 'at-risk' | 'breached' {
  const remaining = computeRemainingSeconds(request);
  if (remaining === null) return 'on-track';
  if (remaining <= 0) return 'breached';
  const pct = remaining / request.slaDuration!;
  return pct <= 0.25 ? 'at-risk' : 'on-track';
}

export function formatRemaining(seconds: number | null): string {
  if (seconds === null) return 'No SLA';
  if (seconds <= 0) return 'Breached';
  const hours = Math.floor(seconds / 3600);
  const minutes = Math.floor((seconds % 3600) / 60);
  if (hours >= 24) {
    const days = Math.floor(hours / 24);
    const remHours = hours % 24;
    return remHours > 0 ? `${days}d ${remHours}h remaining` : `${days}d remaining`;
  }
  if (hours > 0) return `${hours}h ${minutes}m remaining`;
  return `${minutes}m remaining`;
}
