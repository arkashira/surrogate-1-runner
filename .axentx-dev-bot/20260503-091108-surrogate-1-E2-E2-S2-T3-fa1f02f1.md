def log_document_open_event(document_id, page_num):
    """
    Emit a structured, machine-readable document-open telemetry event.

    Args:
        document_id: Unique identifier for the opened document.
        page_num: Page number at time of open (1-based).

    event_data = {
        "event": "document_open",
        "document_id": document_id,

    }
    logger.info(json.dumps(event_data))
```
