// Compute SLA status for a request row (plain object from DB)
export function computeSlaStatus(row: {
  state: string;
  sla_target_hours?: number | null;
  created_at?: string | null;
  started_at?: string | null;
  resolved_at?: string | null;
  closed_at?: string | null;
}) {
  const targetHours = Number(row.sla_target_hours) || 48;
  const targetMs = targetHours * 60 * 60 * 1000;

  // Determine effective start
  let startMs: number | null = null;
  if (row.started_at) startMs = new Date(row.started_at).getTime();
  else if (row.created_at) startMs = new Date(row.created_at).getTime();

  if (startMs === null) {
    return { status: 'on_track' as const, elapsedMs: 0, targetMs };
  }

  // If terminal state, measure against actual closure time
  if (row.state === 'resolved' || row.state === 'closed') {
    const closedAt = row.resolved_at || row.closed_at;
    if (closedAt) {
      const closedMs = new Date(closedAt).getTime();
      const elapsedMs = closedMs - startMs;
      return {
        status: elapsedMs <= targetMs ? 'on_track' : 'breached',
        elapsedMs,
        targetMs
      };
    }
    // Fallback to now if terminal but no timestamp (shouldn't happen)
  }

  const nowMs = Date.now();
  const elapsedMs = nowMs - startMs;

  if (elapsedMs >= targetMs) return { status: 'breached' as const, elapsedMs, targetMs };
  if (elapsedMs >= targetMs * 0.8) return { status: 'at_risk' as const, elapsedMs, targetMs };
  return { status: 'on_track' as const, elapsedMs, targetMs };
}

// State machine rules
export const VALID_TRANSITIONS: Record<string, string[]> = {
  open: ['in_progress'],
  in_progress: ['resolved', 'closed'],
  resolved: ['closed'],
  closed: []
};
```
