// libraryStore.test.ts
describe('LibraryStore session pinning', () => {
  let store: LibraryStore;
  const docs = [
    { id: 'd1', name: 'A' },
    { id: 'd2', name: 'B' },
    { id: 'd3', name: 'C' },
  ];

  beforeEach(() => {
    store = new LibraryStore();
    store.setDocuments(docs);
  });

  test('getOrderedDocuments returns recently opened first', () => {
    store.openDocument('d2');
    const ordered = store.getOrderedDocuments();
    expect(ordered[0].id).toBe('d2');
    expect(ordered.map((d) => d.id)).toEqual(['d2', 'd1', 'd3']);
  });

  test('openDocument removes duplicates and respects MAX_RECENT=50', () => {
    for (let i = 0; i < 55; i++) {
      store.openDocument(`d${i % 3}`);
    }
    const recent = store.getRecentlyOpenedIds();
    expect(recent.length).toBeLessThanOrEqual(50);
    expect(new Set(recent).size).toBe(recent.length);
  });

  test('openDocument triggers active tracking event when window.axentxTrack exists', () => {
    const mockTrack = jest.fn();
    (window as any).axentxTrack = mockTrack;
    store.openDocument('d1');
    expect(mockTrack).toHaveBeenCalledWith('active', { documentId: 'd1', name: 'A' });
    delete (window as any).axentxTrack;
  });

  test('openDocument does not throw when window.axentxTrack missing', () => {
    delete (window as any).axentxTrack;
    expect(() => store.openDocument('d1')).not.toThrow();
  });

  test('session-only: recentlyOpenedIds not persisted to localStorage', () => {
    store.openDocument('d1');
    const persisted = localStorage.getItem('libraryStore.recentlyOpenedIds');
    expect(persisted).toBeNull();
  });

  test('getOrderedDocuments stable ordering for non-recent items', () => {
    store.openDocument('d3');
    const ordered = store.getOrderedDocuments().map((d) => d.id);
    // d3 recent first, then remaining preserve original relative order
    expect(ordered).toEqual(['d3', 'd1', 'd2']);
  });
});
