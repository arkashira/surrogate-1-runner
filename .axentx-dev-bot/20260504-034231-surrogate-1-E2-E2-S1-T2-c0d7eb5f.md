const streamsToRecord = document.getElementById('streams-to-record');
const recordedStreams = document.getElementById('recorded-streams');
const recordButton = document.getElementById('record-button');

// Fetch available streams and populate the select element
async function fetchStreams() {
  const response = await fetch('api/streams');
  const data = await response.json();
  data.forEach(stream => {
    const option = document.createElement('option');
    option.value = stream.id;
    option.textContent = stream.name;
    streamsToRecord.appendChild(option);
  });
}

fetchStreams();

// Handle recording a stream
recordButton.addEventListener('click', async () => {
  const streamId = streamsToRecord.value;
  const response = await fetch(`api/record/${streamId}`, { method: 'POST' });
  if (response.ok) {
    const stream = await response.json();
    const listItem = document.createElement('li');
    listItem.textContent = stream.name;
    recordedStreams.appendChild(listItem);
  }
});

## 1. Acceptance Criteria

2. **Start recording** – Clicking the **Record** button initiates a POST request to `/api/recordings` with the selected stream ID and shows a loading spinner; the request must succeed (HTTP 200) within 5 seconds.
3. **Recording list update** – After a successful POST, a new entry appears in the “Recorded Streams” list (`<ul id="recordedList">`) showing stream name, start time, and a **Play** button.
4. **Playback** – Clicking a **Play** button streams the recorded audio via an HTML5 `<audio>` element; playback starts within 1 second and the UI shows the correct title.
5. **Error handling** – If the POST fails (non‑200 or network error), an error toast appears and no new list item is added.
6. **Persistence** – On page reload, the recorded‑streams list is fetched from `/api/recordings` and displays all previously recorded items.
7. **Accessibility** – All interactive elements are focusable and have appropriate ARIA labels; screen‑reader tests confirm they are announced correctly.
