/**
 * ErrorLoggingService.js
 * Error logging and reporting service for automated workflows
 * Part of surrogate-1 workflow automation system
 * 
 * Usage:
 *   const errorService = new ErrorLoggingService({ logDirectory: './logs' });
 *   errorService.logError(new Error('Something failed'), { workflow: 'ingest' });
 * 
 * CLI Usage:
 *   node ErrorLoggingService.js log "error message" '{"workflow":"test"}'
 */

const fs = require('fs');
const path = require('path');

class ErrorLoggingService {
  constructor(options = {}) {
    this.logDirectory = options.logDirectory || process.env.ERROR_LOG_DIR || './logs';
    this.maxLogFiles = options.maxLogFiles || 30;
    this.includeStackTrace = options.includeStackTrace !== false;
    this.verbose = options.verbose || false;
    
    this.ensureLogDirectory();
  }

  ensureLogDirectory() {
    if (!fs.existsSync(this.logDirectory)) {
      fs.mkdirSync(this.logDirectory, { recursive: true });
    }
  }

  getLogFilePath(date = new Date()) {
    const dateStr = date.toISOString().split('T')[0];
    return path.join(this.logDirectory, `error-${dateStr}.log`);
  }

  formatErrorLog(error, context = {}) {
    const timestamp = new Date().toISOString();
    const errorObj = error instanceof Error ? error : new Error(String(error));
    
    const logEntry = {
      timestamp,
      error: {
        message: errorObj.message || String(error),
        code: errorObj.code || 'UNKNOWN_ERROR',
        name: errorObj.name,
        stack: this.includeStackTrace ? errorObj.stack : undefined
      },
      context: {
        workflow: context.workflow || process.env.GITHUB_WORKFLOW || 'unknown',
        runId: context.runId || process.env.GITHUB_RUN_ID || 'unknown',
        job: context.job || process.env.GITHUB_JOB || 'unknown',
        step: context.step || 'unknown',
        shardId: context.shardId || process.env.SHARD_ID || undefined,
        additional: context.additional || {}
      }
    };

    return JSON.stringify(logEntry, null, 2);
  }

  logError(error, context = {}) {
    const logEntry = this.formatErrorLog(error, context);
    const logFilePath = this.getLogFilePath();
    
    try {
      fs.appendFileSync(logFilePath, logEntry + '\n---\n');
      
      if (this.verbose) {
        console.error(`[ErrorLoggingService] Error logged to ${logFilePath}`);
        console.error(logEntry);
      }
      
      return {
        success: true,
        logFile: logFilePath,
        timestamp: new Date().toISOString()
      };
    } catch (writeError) {
      console.error(`[ErrorLoggingService] Failed to write error log: ${writeError.message}`);
      return {
        success: false,
        error: writeError.message
      };
    }
  }

  logWorkflowFailure(workflowName, step, error, runContext = {}) {
    const context = {
      workflow: workflowName,
      step: step,
      runId: runContext.runId || process.env.GITHUB_RUN_ID,
      runNumber: runContext.runNumber || process.env.GITHUB_RUN_NUMBER,
      additional: {
        failureType: 'WORKFLOW_ERROR',
        lastSuccessfulStep: runContext.lastSuccessfulStep,
        retryCount: runContext.retryCount || 0
      }
    };

    return this.logError(error, context);
  }

  logStepError(stepName, error, additionalContext = {}) {
    const context = {
      step: stepName,
      additional: additionalContext
    };

    return this.logError(error, context);
  }

  generateErrorReport(errorLogs, options = {}) {
    const report = {
      generatedAt: new Date().toISOString(),
      summary: {
        totalErrors: errorLogs.length,
        byWorkflow: {},
        byStep: {},
        byErrorCode: {}
      },
      errors: errorLogs
    };
