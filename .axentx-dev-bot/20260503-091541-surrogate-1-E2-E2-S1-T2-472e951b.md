// /opt/axentx/surrogate-1/src/lib/sla.ts
const TARGET_HOURS = 72;

export type SlaStatus = 'on_track' | 'at_risk' | 'breached';

export interface SlaInfo {
  status: SlaStatus;
  elapsedHours: number;
  targetHours: number;
  remainingHours: number;
  breached: boolean;
}

export function calculateSla(createdAt: Date, resolvedAt?: Date | null): SlaInfo {
  const now = resolvedAt && resolvedAt > createdAt ? resolvedAt : new Date();
  const elapsedMs = now.getTime() - createdAt.getTime();
  const elapsedHours = elapsedMs / (1000 * 60 * 60);
  const targetHours = TARGET_HOURS;
  const remainingHours = targetHours - elapsedHours;

  let status: SlaStatus = 'on_track';
  if (elapsedHours > targetHours) {
    status = 'breached';
  } else if (remainingHours <= 24) {
    status = 'at_risk';
  }

  return {
    status,
    elapsedHours: Number(elapsedHours.toFixed(2)),
    targetHours,
    remainingHours: Number(remainingHours.toFixed(2)),
    breached: elapsedHours > targetHours,
  };
}
```
