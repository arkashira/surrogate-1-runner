import { updateRequestStatus } from '../../src/routes/requests';
import { RequestAudit } from '../../src/db/schema';
import { db } from '../../src/db';

describe('updateRequestStatus', () => {
  beforeEach(async () => {
    await db.clear(RequestAudit);
    await db.clear(Request);
  });

  test('creates one audit row per status change with actor and timestamp', async () => {
    const req = await db.insert(Request, { status: 'open' });
    const result = await updateRequestStatus(req.id, 'alice', 'in_progress', 'started work');
    expect(result.request.status).toBe('in_progress');
    const audits = await db.find(RequestAudit, { request_id: req.id });
    expect(audits).toHaveLength(1);
    expect(audits[0].actor).toBe('alice');
    expect(audits[0].status).toBe('in_progress');
    expect(audits[0].note).toBe('started work');
    expect(audits[0].created_at).toBeDefined();
  });

  test('rejects invalid transition and inserts no audit row', async () => {
    const req = await db.insert(Request, { status: 'open' });
    await expect(updateRequestStatus(req.id, 'bob', 'closed')).rejects.toThrow();
    const audits = await db.find(RequestAudit, { request_id: req.id });
    expect(audits).toHaveLength(0);
  });

    await updateRequestStatus(req.id, 'u3', 'closed');
    const audits = await db.find(RequestAudit, { request_id: req.id });
    expect(audits).toHaveLength(3);
    audits.forEach(a => {
