// AutomatedSecurityComplianceCheckerTest.java
@SpringBootTest(classes = {AutomatedSecurityComplianceChecker.class, CheckProperties.class})
class AutomatedSecurityComplianceCheckerTest {

    @Autowired CheckProperties props;
    @Autowired AutomatedSecurityComplianceChecker checker;

    @Test
    void loadsPropertiesFromFile() {
        assertTrue(props.isEnabled());
        assertEquals(30000L, props.getIntervalMs());
    }

    @Test
    void envVarsOverrideProperties(@TempDir Path tmp) {
        System.setProperty("CHECK_ENABLED", "false");
        System.setProperty("CHECK_INTERVAL_MS", "60000");
        // reload context
        AnnotationConfigApplicationContext ctx = new AnnotationConfigApplicationContext();
        ctx.register(CheckProperties.class);
        ctx.refresh();
        CheckProperties p = ctx.getBean(CheckProperties.class);
        assertFalse(p.isEnabled());
        assertEquals(60000L, p.getIntervalMs());
        ctx.close();
    }

    @Test
    void loadsChecklistFromClasspath() throws Exception {
        List<CheckDefinition> defs = checker.loadCheckDefinitions();
        assertFalse(defs.isEmpty());
        assertTrue(defs.stream().allMatch(d -> d.getId() != null));
    }

    @Test
    void loadsChecklistFromExternalDir(@TempDir Path dir) throws Exception {
        Files.writeString(dir.resolve("sample.json"),
                "[{\"id\":\"C001\",\"description\":\"dummy\"}]");
        props.setChecklistLocation(dir.toString());
        List<CheckDefinition> defs = checker.loadCheckDefinitions();
        assertEquals(1, defs.size());
        assertEquals("C001", defs.get(0).getId());
    }

    @Test
    void executesChecksWhenEnabled() {
        props.setEnabled(true);
        when(checker.performChecks()).thenReturn(CheckResult.success());
        checker.runScheduledCheck();
        verify(checker).performChecks();
        // verify logging or metric emission could be added via a mock logger
    }

    @Test
    void skipsChecksWhenDisabled() {
        props.setEnabled(false);
        checker.runScheduledCheck();
        verify(checker, never()).performChecks();
    }
}
