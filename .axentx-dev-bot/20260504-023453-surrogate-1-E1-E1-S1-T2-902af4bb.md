from flask import Flask, request, jsonify
from werkzeug.security import check_password_hash

app = Flask(__name__)

@app.route('/api/v1/query', methods=['POST'])
def query_prometheus():
  prometheus_url = request.json['prometheusUrl']
  api_token = request.json['query']
  # Check if the API token is valid
  if check_password_hash(api_token, 'valid_api_token'):
    # Make a request to the Prometheus server
    response = requests.get(f'{prometheus_url}/api/v1/query', headers={'Authorization': 'Bearer valid_api_token'})
    return jsonify(response.json())
  else:
    return jsonify({'error': 'Invalid API token'}), 401
