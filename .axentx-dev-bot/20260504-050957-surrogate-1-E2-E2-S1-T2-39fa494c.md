## 2. Unit Tests (JUnit / Mockito style pseudo‑code)

```java
class DashboardControllerTest {

    @InjectMocks
    DashboardController controller;

    @Mock
    FinancialDataService dataService;          // service that parses & persists

    @Mock
    Logger logger;

    @Test
    void uploadCsv_validFile_success() throws Exception {
        MockMultipartFile file = new MockMultipartFile(
            "file", "data.csv", "text/csv", "col1,col2\n1,2".getBytes());

        when(dataService.importCsv(any(InputStream.class)))
            .thenReturn(new ImportResult(true, 1, Collections.emptyList()));

        ResponseEntity<?> resp = controller.uploadCsv(file);

        assertEquals(HttpStatus.OK, resp.getStatusCode());
        verify(dataService).importCsv(any(InputStream.class));
        verify(logger, never()).error(anyString(), any());
    }

    @Test
    void uploadCsv_invalidFormat_error() throws Exception {
        MockMultipartFile file = new MockMultipartFile(
            "file", "bad.csv", "text/csv", "invalid content".getBytes());

        when(dataService.importCsv(any(InputStream.class)))
            .thenReturn(new ImportResult(false, 0,
                List.of("Row 1: missing required column")));

        ResponseEntity<?> resp = controller.uploadCsv(file);

        assertEquals(HttpStatus.BAD_REQUEST, resp.getStatusCode());
        assertTrue(resp.getBody().toString().contains("missing required column"));
        verify(logger).error(contains("CSV import failed"));
    }

    @Test
    void uploadCsv_largeFile_rejected() throws Exception {
        byte[] largeContent = new byte[11 * 1024 * 1024]; // 11 MB
        MockMultipartFile file = new MockMultipartFile(
            "file", "large.csv", "text/csv", largeContent);

        ResponseEntity<?> resp = controller.uploadCsv(file);

        assertEquals(HttpStatus.PAYLOAD_TOO_LARGE, resp.getStatusCode());
        verify(logger).warn(contains("file size exceeds limit"));
    }
}
```
