def _utcnow() -> datetime:
    return datetime.now(timezone.utc)


def date_str(ts: Optional[datetime] = None) -> str:
    return (ts or _utcnow()).strftime("%Y-%m-%d")


def manifest_path(date_folder: str) -> Path:
    return MANIFEST_ROOT / date_folder / "files.json"


def _hash_manifest(entries: List[Dict]) -> str:
    payload = json.dumps(entries, sort_keys=True, separators=(",", ":")).encode()
    return hashlib.sha256(payload).hexdigest()


def build_manifest(date_folder: str, folders: Optional[List[str]] = None) -> Path:
    """
    Single authenticated list -> persisted manifest.
    Non-recursive per folder to avoid pagination explosion.
    """
    if HfApi is None:
        raise RuntimeError("huggingface_hub required to build manifest")
    api = HfApi()
    folders = folders or [""]
    entries: List[Dict] = []
    for folder in folders:
        tree = api.list_repo_tree(repo_id=HF_REPO, path=folder, recursive=False)
        for node in tree:
            if node.type == "file":
                entries.append(
                    {
                        "path": node.path,
                        "size": getattr(node, "size", None),
                        "etag": getattr(node, "etag", None),
                    }
                )
    out = manifest_path(date_folder)
    out.parent.mkdir(parents=True, exist_ok=True)
    with out.open("w", encoding="utf-8") as f:
        json.dump(entries, f, indent=2)
    # Store hash alongside manifest for deterministic checks.
    (out.parent / "hash.txt").write_text(_hash_manifest(entries))
    return out
