```javascript
// metrics.test.js
import { Registry, Counter } from 'prom-client';
import { initMetrics, incSuccessful, incBlocked } from '../src/metrics';
import config from '../config';

describe('Metrics module', () => {
  let registry;

  beforeEach(() => {
    registry = new Registry();
    initMetrics(registry); // registers counters on provided registry
  });

  test('creates successful_downloads_total counter with shard_id label', () => {
    const metric = registry.getSingleMetric('successful_downloads_total');
    expect(metric).toBeInstanceOf(Counter);
    expect(metric.labelNames).toContain('shard_id');
  });

  test('creates blocked_requests_total counter with shard_id label', () => {
    const metric = registry.getSingleMetric('blocked_requests_total');
    expect(metric).toBeInstanceOf(Counter);
    expect(metric.labelNames).toContain('shard_id');
  });

  test('incSuccessful increments counter by 1 for given shard', () => {
    incSuccessful('shard-7');
    const value = registry.getSingleMetric('successful_downloads_total')
                       .hashMap['shard_id:shard-7'].value;
    expect(value).toBe(1);
  });

  test('incBlocked increments counter by 1 for given shard', () => {
    incBlocked('shard-3');
    const value = registry.getSingleMetric('blocked_requests_total')
                       .hashMap['shard_id:shard-3'].value;
    expect(value).toBe(1);
  });

  test('multiple increments accumulate correctly', () => {
    incSuccessful('shard-1');
    incSuccessful('shard-1');
    incBlocked('shard-1');
    const succ = registry.getSingleMetric('successful_downloads_total')
                         .hashMap['shard_id:shard-1'].value;
    const blk  = registry.getSingleMetric('blocked_requests_total')
                         .hashMap['shard_id:shard-1'].value;
    expect(succ).toBe(2);
    expect(blk).toBe(1);
  });

  test('metrics are exported in Prometheus text format', async () => {
    incSuccessful('shard-A');
    const payload = await registry.metrics();
    expect(payload).toMatch(/successful_downloads_total{shard_id="shard-A"} 1/);
    expect(payload).toMatch(/blocked_requests_total{shard_id="shard-A"} 0/);
  });
});
```
