import { Request, Response } from 'express';
import { RequestModel } from '../models/request.model';

export async function getPublicRequest(req: Request, res: Response) {
  const requestId = req.params.id;
  const request = await RequestModel.findById(requestId);
  if (!request || !request.published) {
    return res.status(404).send({ message: 'Request not found or unpublished' });
  }
  const publicFields = {
    current_status: request.current_status,
    title: request.title,
    updated_at: request.updated_at,
    history: request.history.slice(-10), // get the latest 10 history entries
  };
  res.send(publicFields);
}

export async function getPublicRequestHistory(req: Request, res: Response) {

  }
  const publicFields = {
    history: request.history.slice(-10), // get the latest 10 history entries
  };
  res.send(publicFields);
}
