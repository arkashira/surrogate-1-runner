# -*- coding: utf-8 -*-
"""
TableProcessor
==============

Utility class used by the compliance service to:

* Sort compliance records by ``control_id`` (required for deterministic PDF output).
* Produce a data structure ready for the PDF renderer.
* Export the same data as a UTF‑8 CSV file (bytes) that can be streamed to the
  client or saved to disk.

The implementation is deliberately pure‑Python – no external dependencies
beyond the stdlib – so it can be reused in CLI scripts, FastAPI endpoints,
Celery workers, etc.
"""

from __future__ import annotations

import csv
import io
from typing import Any, Dict, List, Sequence

Record = Dict[str, Any]  # a single compliance row


class TableProcessor:
    """
    Parameters
    ----------
    records:
        A list of dictionaries, each containing at least the keys
        ``resource_type``, ``resource_name``, ``control_id``,
        ``control_description``, ``status`` and ``config_lines``.
    """

    _CSV_HEADER = [
        "Resource Type",
        "Resource Name",
        "Control ID",
        "Control Description",
        "Status",
        "Config Lines",
    ]

    def __init__(self, records: Sequence[Record]) -> None:
        # Defensive copy – callers can keep their original list untouched.
        self._records: List[Record] = list(records)

    # --------------------------------------------------------------------- #
    #  Private helpers
    # --------------------------------------------------------------------- #
    def _sorted(self) -> List[Record]:
        """
        Return a **new** list sorted lexicographically by ``control_id``.
        ``control_id`` may be missing; in that case we treat it as an empty
        string so the sort never crashes.
        """
        return sorted(self._records, key=lambda r: r.get("control_id", ""))

    # --------------------------------------------------------------------- #
    #  Public API – PDF side
    # --------------------------------------------------------------------- #
    def generate_table_data(self) -> List[Record]:
        """
        Return the sorted rows ready for the PDF renderer.
        The caller can iterate over the list and render each column as it sees
        fit.
        """
        return self._sorted()

    # --------------------------------------------------------------------- #
    #  Public API – CSV side
    # --------------------------------------------------------------------- #
    def generate_csv_bytes(self) -> bytes:
        """
        Produce a CSV representation (UTF‑8 encoded ``bytes``) of the sorted
        table.  The format matches the header defined in ``_CSV_HEADER`` and
        uses a semicolon‑separated list for the ``config_lines`` column
        (mirroring the original proposal).

        Returns
        -------
        bytes
            CSV content ready to be sent as ``application/csv`` or written to
            a file.

        for rec in self._sorted():

                    rec.get("resource_name", ""),
                    rec.get("control_id", ""),
                    rec.get("control_description", ""),
                    rec.get("status", ""),
                    ";".join(map(str, rec.get("config_lines", []))),
