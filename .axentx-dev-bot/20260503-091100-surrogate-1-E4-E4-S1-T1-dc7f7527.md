# Add to existing Cache class or create if not present
import json
import time
from typing import List, Dict, Any

class DocumentCache:
    def __init__(self, redis_client):
        self.redis = redis_client
        self.RECENT_DOCS_PREFIX = "recent_docs:user:"
        self.MAX_RECENT = 20

    def record_document_access(self, user_id: str, doc_info: Dict[str, Any]):
        """
        Record a document access in the user's recent list (LRU).
        doc_info must include: doc_id, title, created_at, size
        """
        key = f"{self.RECENT_DOCS_PREFIX}{user_id}"
        now = time.time()
        stored_data = {
            **doc_info,
            "last_accessed": now
        }

        # Use Redis ZSET sorted by timestamp (score = -timestamp for reverse chronological)
        pipe = self.redis.pipeline()
        # Remove existing if present
        pipe.zrem(key, doc_info["doc_id"])
        # Add with negative timestamp for descending sort
        pipe.zadd(key, {json.dumps(stored_data): -now})
        # Trim to latest 20
        pipe.ztrim(key, 0, self.MAX_RECENT - 1)
        # Optional: expire after 30 days
        pipe.expire(key, 30 * 24 * 60 * 60)
        pipe.execute()

    def get_recent_documents(self, user_id: str) -> List[Dict]:
        """
        Get up to 20 most recent documents for user.
        Returns list sorted by last_accessed descending.
        """
        key = f"{self.RECENT_DOCS_PREFIX}{user_id}"
        results = self.redis.zrange(key, 0, -1)
        docs = []
        for r in results:
            data = json.loads(r)
            # Convert timestamp back to ISO
            data["last_accessed"] = time.strftime(
                "%Y-%m-%dT%H:%M:%S.%fZ",
                time.gmtime(data["last_accessed"])
            )
            docs.append(data)
        return docs
```
