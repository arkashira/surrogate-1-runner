    # Optional – run once at startup

        "level": "DEBUG",
        "enabled": True,
        "json": True,               # keep JSON lines (default)
        "console_pretty": False,    # set True for indented console output
    })

    logger = get_logger()
    logger.log_metric("dd.api.requests", 42, tags={"env": "sandbox"})
    logger.log_event(
        title="Synthetic event generated",
        text="A test event for Datadog API integration.",
        tags={"env": "sandbox"},
    )
"""

from __future__ import annotations

import json
import logging
import os
import threading
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Dict, Mapping, Optional

# --------------------------------------------------------------------------- #
# Global singleton state (private to the module)
# --------------------------------------------------------------------------- #
_lock = threading.Lock()
_logger: Optional[logging.Logger] = None

    # Default values – callers can override via `configure()`

    "enabled": True,         # quick on/off switch
    "json": True,            # JSON‑lines file output (always true for file)
    "console_pretty": False, # pretty‑print JSON on console if True
    "filename_prefix": "synthetic_data",  # prefix for timestamped files
}

    """Create the log directory if it does not exist."""
