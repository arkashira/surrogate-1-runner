import { NotificationService } from './notificationService';
import { WebClient } from '@slack/web-api';

const slackWebClient = new WebClient(process.env.SLACK_TOKEN);

export class NotificationService {
  async sendSlackMessage(channel: string, text: string) {
    if (!channel) {
      console.log('No Slack channel configured. Skipping notification.');
      return;
    }

    try {
      await slackWebClient.chat.postMessage({
        channel,
        text,
      });
    } catch (error) {
      console.error('Error sending Slack message:', error);
    }
  }

  async sendNotification(request: Request, newStatus: string) {
    const formattedText = `Request ${request.title} has been updated to ${newStatus}. View request: ${request.publicUrl}`;
    await this.sendSlackMessage(request.config.slackChannel, formattedText);
  }
}
