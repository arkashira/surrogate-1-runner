// src/services/notificationService.ts
import { WebClient } from '@slack/web-api';
import { Request } from '../models/Request';

class NotificationService {
  private slackClient: WebClient | null;
  private slackChannel: string | null;

  constructor() {
    const token = process.env.SLACK_BOT_TOKEN;
    const channel = process.env.SLACK_CHANNEL;
    
    if (!token) {
      console.warn('SLACK_BOT_TOKEN not configured, Slack notifications disabled');
      this.slackClient = null;
      this.slackChannel = null;
      return;
    }
    
    if (!channel) {
      console.warn('SLACK_CHANNEL not configured, Slack notifications disabled');
      this.slackClient = null;
      this.slackChannel = null;
      return;
    }
    
    this.slackClient = new WebClient(token);
    this.slackChannel = channel;
  }

  async sendStatusUpdate(request: Request, oldStatus: string): Promise<void> {
    if (!this.slackClient || !this.slackChannel) {
      console.log('Slack notifications not configured, skipping');
      return;
    }

    try {
      const message = this.buildStatusUpdateMessage(request, oldStatus);
      
      await this.slackClient.chat.postMessage({
        channel: this.slackChannel,
        text: message,
        blocks: [
          {
            type: 'section',
            text: {
              type: 'mrkdwn',
              text: message
            }
          }
        ]
      });
    } catch (error) {
      console.error('Failed to send Slack notification:', error);
    }
  }

  private buildStatusUpdateMessage(request: Request, oldStatus: string): string {
    const url = `${process.env.APP_URL || 'https://app.axentx.com'}/requests/${request.id}`;
    
    return `*Status Update*\n` +
           `• *Title:* ${request.title}\n` +
           `• *From:* ${oldStatus} → ${request.status}\n` +
           `• *Owner:* ${request.owner}\n` +
           `• *Link:* <${url}|View Request>`;
  }
}

export default new NotificationService();
```
