import unittest
from datetime import datetime, timedelta
from unittest.mock import patch, MagicMock
from sqlalchemy import create_engine
from reports import get_metrics

class TestGetMetrics(unittest.TestCase):
    @patch('reports.get_db_engine')
    def test_get_metrics(self, mock_engine):
        mock_engine.return_value = create_engine('sqlite:///:memory:')
        mock_session = MagicMock()
        mock_session.query.return_value.all.return_value = [
            {'id': 1, 'created_at': datetime.now() - timedelta(hours=24)},
            {'id': 2, 'created_at': datetime.now() - timedelta(hours=24)},
            {'id': 3, 'created_at': datetime.now() - timedelta(hours=25)}
        ]
        metric_data = get_metrics(mock_session)
        self.assertIsInstance(metric_data, MetricData)
        self.assertEqual(metric_data.total_processed, 2)
        self.assertEqual(metric_data.unique_processed, 2)

    @patch('reports.get_db_engine')
    def test_get_metrics_empty_database(self, mock_engine):
        mock_engine.return_value = create_engine('sqlite:///:memory:')
        mock_session = MagicMock()
        mock_session.query.return_value.all.return_value = []
        metric_data = get_metrics(mock_session)
        self.assertIsInstance(metric_data, MetricData)
        self.assertEqual(metric_data.total_processed, 0)
        self.assertEqual(metric_data.unique_processed, 0)

    @patch('reports.get_db_engine')
    def test_get_metrics_invalid_database(self, mock_engine):
