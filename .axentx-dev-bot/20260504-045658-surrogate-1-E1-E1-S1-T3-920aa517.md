import { EventTracker, EventMap } from '../src/eventTracker';

    expect(tracker.config.environment).toBe('prod');
  });

    const tracker = new EventTracker();

    expect(tracker.eventMap.has(newEvent.name)).toBeTruthy();
  });

  test('should queue events and batch them', async () => {
    const tracker = new EventTracker({ batchSize: 2, batchInterval: 0 });

    // Fast‚Äëforward timers

    await Promise.resolve(); // allow microtasks to run
    expect(mockFetch).toHaveBeenCalledTimes(1);

    expect(payload.events).toHaveLength(2);
  });

  test('should retry on network failure with exponential back‚Äëoff', async () => {
    const tracker = new EventTracker({ maxRetries: 2, backoffFactor: 2 });

    mockFetch.mockResolvedValueOnce({ ok: true });

    tracker.track('step1', { step: 1 });

    jest.advanceTimersByTime(1000); // first retry
    await Promise.resolve();

  });

  });
