import axios, { AxiosInstance, AxiosError } from "axios";
import { EventEmitter } from "events";

/**
 * Azure Cost Estimator API client.
 * 
 * Provides real-time, line-item cost breakdowns via Azure's Consumption API.
 * Supports polling and push notifications for cost updates.
 * 
 * Environment variables:
 *   AZURE_TENANT_ID         – Azure AD tenant identifier
 *   AZURE_CLIENT_ID         – Service principal client ID
 *   AZURE_CLIENT_SECRET     – Service principal secret
 *   AZURE_SUBSCRIPTION_ID  – Subscription to query
 *   AZURE_API_TIMEOUT_MS   – Request timeout (default: 30000)
 */

interface AzureTokenResponse {
  token_type: string;
  expires_in: number;
  ext_expires_in: number;
  access_token: string;
}

export interface CostLineItem {
  meterId: string;
  meterName: string;
  usageStart: string;
  usageEnd: string;
  quantity: number;
  unit: string;
  pretaxCost: number;
  currency: string;
}

export interface CostQueryOptions {
  /** Start date for cost query (ISO 8601) */
  from?: string;
  /** End date for cost query (ISO 8601) */
  to?: string;
  /** Aggregation granularity */
  granularity?: 'Daily' | 'Monthly' | 'Hourly';
}

interface TokenCache {
  token: string;
  expiresAt: number;
}

/** Configuration validation */
function validateConfig(): void {
  const required = ['AZURE_TENANT_ID', 'AZURE_CLIENT_ID', 'AZURE_CLIENT_SECRET', 'AZURE_SUBSCRIPTION_ID'];
  const missing = required.filter(key => !process.env[key]);
  
  if (missing.length > 0) {
    throw new Error(`Missing required environment variables: ${missing.join(', ')}`);
  }
}

/** In-memory token cache with buffer */
let cachedToken: TokenCache | null = null;

/** Acquire Azure AD token using client credentials flow */
async function getAzureAccessToken(): Promise<string> {
  const now = Date.now() / 1000;
  // 60-second buffer before expiry
  if (cachedToken && cachedToken.expiresAt - 60 > now) {
    return cachedToken.token;
  }

  validateConfig();

  const { data } = await axios.post<AzureTokenResponse>(
    `https://login.microsoftonline.com/${process.env.AZURE_TENANT_ID}/oauth2/v2.0/token`,
    new URLSearchParams({
      client_id: process.env.AZURE_CLIENT_ID!,
      client_secret: process.env.AZURE_CLIENT_SECRET!,
      scope: 'https://management.azure.com/.default',
      grant_type: 'client_credentials',
    }),
    {
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      timeout: parseInt(process.env.AZURE_API_TIMEOUT_MS || '30000', 10),
    }
  );

  cachedToken = {
    token: data.access_token,
    expiresAt: now + data.expires_in,
  };
  
  return data.access_token;
}

/** Create configured Axios instance */
async function createApiClient(): Promise<AxiosInstance> {
  const token = await getAzureAccessToken();
  
  return axios.create({
    baseURL: 'https://management.azure.com',
    timeout: parseInt(process.env.AZURE_API_TIMEOUT_MS || '30000', 10),
    headers: {
      Authorization: `Bearer ${token}`,
      'Content-Type': 'application/json',
    },
  });
}
