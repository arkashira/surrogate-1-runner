## 1. Acceptance Criteria
- **All AWS accounts**: The system must register every AWS account listed in the configuration and enable CloudWatch/Config rules to forward alerts to surrogate‑1.
- **Delivery latency**: An alert generated by a misconfiguration must reach the compliance officer’s notification channel (e.g., email/SNS) within **5 minutes** of detection.
- **Actionable content**: Each alert payload must contain:
  1. A concise description of the misconfiguration.
  2. A unique identifier (e.g., rule ARN).
  3. Step‑by‑step remediation instructions.
- **Reliability**: The alerting pipeline must retry up to 3 times on transient failures and log failures for audit.
- **Security**: All communication between AWS services and surrogate‑1 must use TLS and IAM roles with least privilege.

// alerting-system.test.js
const { AlertingSystem } = require('../alerting-system');
const awsMock = require('aws-sdk-mock');

describe('AlertingSystem', () => {
  let system;

  beforeEach(() => {
    system = new AlertingSystem({ snsTopicArn: 'arn:aws:sns:us-east-1:123456789012:ComplianceAlerts' });
  });

  afterEach(() => {
    awsMock.restore();
  });

  test('registers all configured AWS accounts', async () => {
    const accounts = [{ id: '111111111111', name: 'Prod' }, { id: '222222222222', name: 'Dev' }];
    system.config.awsAccounts = accounts;
    awsMock.mock('ConfigService', 'putConfigurationRecorder', { status: 'Success' });

    await system.enableAlerting();

    expect(awsMock.calls('ConfigService', 'putConfigurationRecorder')).toHaveLength(accounts.length);
  });

  test('sends alert within 5 minutes', async () => {
    const start = Date.now();
    const alert = { ruleArn: 'arn:aws:config:us-east-1:123456789012:config-rule/ExampleRule', description: 'Public S3 bucket' };
    awsMock.mock('SNS', 'publish', { MessageId: 'msg-123' });

    await system.publishAlert(alert);
    const duration = Date.now() - start;

    expect(duration).toBeLessThanOrEqual(5 * 60 * 1000); // 5 minutes
  });

  test('includes remediation steps in alert payload', async () => {
    const alert = { ruleArn: 'arn:aws:config:us-east-1:123456789012:config-rule/ExampleRule', description: 'Public S3 bucket' };
    const payload = system.buildAlertPayload(alert);

    expect(payload).toHaveProperty('remediation');
    expect(payload.remediation).toMatch(/Disable public access/);
  });

  test('retries on transient SNS failure', async () => {
    const alert = { ruleArn: 'arn:aws:config:us-east-1:123456789012:config-rule/ExampleRule', description: 'Public S3 bucket' };
    const publishMock = jest.fn()
      .mockRejectedValueOnce(new Error('NetworkError'))
      .mockResolvedValue({ MessageId: 'msg-123' });

    awsMock.mock('SNS', 'publish', publishMock);

    await system.publishAlert(alert);

    expect(publishMock).toHaveBeenCalledTimes(2); // one retry
  });

  test('logs failure after max retries', async () => {
    const alert = { ruleArn: 'arn:aws:config:us-east-1:123456789012:config-rule/ExampleRule', description: 'Public S3 bucket' };
    const publishMock = jest.fn().mockRejectedValue(new Error('NetworkError'));

    awsMock.mock('SNS', 'publish', publishMock);
    const logSpy = jest.spyOn(system.logger, 'error');

    await expect(system.publishAlert(alert)).rejects.toThrow('NetworkError');
    expect(logSpy).toHaveBeenCalledWith(expect.stringContaining('Failed to publish alert after 3 attempts'));
  });
});
