    labelnames=["pipeline_id"],

ACTIVE_STAGES = Gauge(

    "Number of currently active pipeline stages",
    labelnames=["stage_name", "pipeline_id"],
)

    """
    Lightweight store for recent e2e samples to compute p95 over a time window.
    Thread-safe append/trim. Used by TimingMiddleware.
    """
    def __init__(self, window_seconds: int = 300):
        self.window_seconds = window_seconds
        self._samples: List[Tuple[float, float]] = []  # (timestamp_ns, value_ms)
        self._lock = threading.RLock()

    def add(self, value_ms: float) -> None:
        now = time.time_ns()
        with self._lock:
            self._samples.append((now, value_ms))
            self._trim(now)

    def p95(self) -> float:
        now = time.time_ns()
        with self._lock:
            self._trim(now)
            if not self._samples:
                return 0.0
            vals = sorted(v for _, v in self._samples)
            idx = int(len(vals) * 0.95)
            return vals[min(idx, len(vals) - 1)]

    def _trim(self, now_ns: int) -> None:
        cutoff = now_ns - int(self.window_seconds * 1_000_000_000)

import time
import pytest
from surrogate_1.pipeline.timing import TimingMiddleware, MonotonicClock, P95Store

def test_timing_middleware_records_stage_duration():

def test_timing_middleware_records_failure_status():
    clock = MonotonicClock()
    tm = TimingMiddleware(clock=clock)
    with pytest.raises(RuntimeError):
        with tm.measure("transform", pipeline_id="p-2", status_on_error="error"):
            raise RuntimeError("boom")
    samples = tm.stage_duration_samples("transform", "p-2")
    assert samples[0]["labels"]["status"] == "error"

def test_e2e_timing_from_first_stage_start_to_ack():
    clock = MonotonicClock()
    tm = TimingMiddleware(clock=clock)
    tm.record_pipeline_start("p-3", clock.now())
    time.sleep(0.002)
    ack_ts = clock.now()
    tm.record_pipeline_ack("p-3", ack_ts)
    e2e = tm.e2e_samples("p-3")
    assert len(e2e) == 1
    assert e2e[0]["labels"]["pipeline_id"] == "p-3"
    assert e2e[0]["duration_ms"] >= 2.0

def test_p95_store_updates_and_returns_p95():
    store = P95Store(window_seconds=300, clock=lambda: time.time())
    for i in range(100):
        store.add("p-4", float(i))  # 0..99ms
    p95 = store.p95("p-4")
    assert 94.0 <= p95 <= 96.0

def test_p95_store_eviction_after_window():
    clock = MonotonicClock()
    store = P95Store(window_seconds=1, clock=lambda: clock.now())
    t0 = clock.now()

    clock.advance(1.1)
    store.add("p-5", 10.0)  # forces cleanup of old sample
    p95 = store.p95("p-5")
    assert p95 < 100.0
