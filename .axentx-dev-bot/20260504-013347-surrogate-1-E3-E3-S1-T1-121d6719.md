        <tbody>
            <!-- Data will be populated dynamically via WebSocket -->
        </tbody>
    </table>

    <script>
        function filterTable(filterType) {
            const input = document.getElementById(`${filterType}Filter`);
            const filter = input.value.toUpperCase();
            const table = document.getElementById("locksTable");
            const rows = table.getElementsByTagName("tr");

            for (let i = 1; i < rows.length; i++) {
                const row = rows[i];
                const cell = row.getElementsByTagName("td")[filterType === 'document' ? 0 : 2];
                if (cell) {
                    const txtValue = cell.textContent || cell.innerText;
                    if (txtValue.toUpperCase().indexOf(filter) > -1) {
                        row.style.display = "";
                    } else {
                        row.style.display = "none";
                    }
                }
            }
        }

        // WebSocket setup for real-time data updates
        const socket = new WebSocket('ws://your-websocket-endpoint');

        socket.onmessage = function(event) {
            const data = JSON.parse(event.data);
            const tableBody = document.querySelector('#locksTable tbody');
            tableBody.innerHTML = ''; // Clear existing rows

            data.forEach(lock => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${lock.documentName}</td>
                    <td>${lock.sectionTitle}</td>
                    <td>${lock.owner}</td>
                    <td>${lock.remainingTTL}s</td>
                `;
                tableBody.appendChild(row);
            });
        };

        socket.onopen = function() {
            console.log('WebSocket connection established');
        };

        socket.onerror = function(error) {
            console.error('WebSocket error:', error);
        };
    </script>
</body>
</html>
