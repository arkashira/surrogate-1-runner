// Import the emitReadSessionEvent function
import { emitReadSessionEvent } from '../analytics/events';

// Optimize reader loading by using a cache
const documentCache = {};

const Reader = () => {
  const [document, setDocument] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const loadDocument = async () => {
      // Check if the document is cached
      if (documentCache[documentId]) {
        setDocument(documentCache[documentId]);
        setLoading(false);
      } else {
        // Load the document from the network
        const response = await fetch(`https://example.com/documents/${documentId}`);
        const data = await response.json();
        setDocument(data);
        setLoading(false);
        // Cache the document
        documentCache[documentId] = data;
      }
    };
    loadDocument();
  }, [documentId]);

  // Implement reading progress persistence per user per document
  const [readingProgress, setReadingProgress] = useState(() => {
    const storedProgress = localStorage.getItem(`readingProgress-${documentId}`);
    return storedProgress ? JSON.parse(storedProgress) : { currentPage: 1 };
  });

  const handlePageChange = (newPage) => {
    setReadingProgress({ currentPage: newPage });
    localStorage.setItem(`readingProgress-${documentId}`, JSON.stringify({ currentPage: newPage }));
  };

  // Emit read-session events
  useEffect(() => {
    emitReadSessionEvent('open', { documentId });
  }, [documentId]);

  useEffect(() => {
    const intervalId = setInterval(() => {
      emitReadSessionEvent('timeSpent', { timeSpent: 1000 });
    }, 1000);
    return () => clearInterval(intervalId);
  }, []);

  return (
    <div>
      {loading ? (
        <p>Loading...</p>
      ) : (
        <div>
          <p>Document content: {document.content}</p>
          <button onClick={() => handlePageChange(readingProgress.currentPage + 1)}>Next page</button>
        </div>
      )}
    </div>
  );
};
```
