import React, { useState, useEffect } from 'react';

/**
 * BootGuardDashboard
 *
 * Displays a dashboard of collision incidents for the last 30 days.
 * Features:
 *   - Filters by service name, severity, and resolution status.
 *   - CSV export of the current view.
 *   - Simple table view of incidents per device.
 *
 * Assumptions:
 *   - API endpoint: GET /api/bootguard/incidents
 *     Returns JSON array of incidents with fields:
 *       id, deviceId, timestamp, serviceName, severity, resolutionStatus
 *   - `timestamp` is ISO string.
 */
const BootGuardDashboard = () => {
  const [incidents, setIncidents] = useState([]);
  const [filtered, setFiltered] = useState([]);
  const [loading, setLoading] = useState(true);
  const [serviceFilter, setServiceFilter] = useState('');
  const [severityFilter, setSeverityFilter] = useState('');
  const [statusFilter, setStatusFilter] = useState('');

  // Fetch incidents on mount
  useEffect(() => {
    const fetchIncidents = async () => {
      setLoading(true);
      try {
        const res = await fetch('/api/bootguard/incidents');
        const data = await res.json();
        // Keep only last 30 days
        const thirtyDaysAgo = new Date();
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
        const recent = data.filter((inc) => new Date(inc.timestamp) >= thirtyDaysAgo);
        setIncidents(recent);
        setFiltered(recent);
      } catch (e) {
        console.error('Failed to fetch incidents', e);
        setIncidents([]);
        setFiltered([]);
      } finally {
        setLoading(false);
      }
    };
    fetchIncidents();
  }, []);

  // Apply filters whenever any filter changes or incidents change
  useEffect(() => {
    const applyFilters = () => {
      let result = [...incidents];
      if (serviceFilter) {
        result = result.filter((inc) => inc.serviceName === serviceFilter);
      }
      if (severityFilter) {
        result = result.filter((inc) => inc.severity === severityFilter);
      }
      if (statusFilter) {
        result = result.filter((inc) => inc.resolutionStatus === statusFilter);
      }
      setFiltered(result);
    };
    applyFilters();
  }, [serviceFilter, severityFilter, statusFilter, incidents]);

  // Unique values for filter dropdowns
  const serviceOptions = Array.from(new Set(incidents.map((i) => i.serviceName))).sort();
  const severityOptions = Array.from(new Set(incidents.map((i) => i.severity))).sort();
  const statusOptions = Array.from(new Set(incidents.map((i) => i.resolutionStatus))).sort();

  const exportCSV = () => {
    const headers = ['Device ID', 'Timestamp', 'Service', 'Severity', 'Resolution'];
    const rows = filtered.map((inc) => [
      inc.deviceId,
      inc.timestamp,
      inc.serviceName,
      inc.severity,
      inc.resolutionStatus,
    ]);
    const csvContent =
      [headers, ...rows]
        .map((row) => row.map((cell) => `"${cell}"`).join(','))
        .join('\n') + '\n';
    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const link = document.createElement('a');
    link.href = url;
    link.setAttribute('download', `bootguard_incidents_${new Date().toISOString()}.csv`);
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
  };

  return (
    <div className="bootguard-dashboard">
      <h2>BootGuard Collision Incidents</h2>

      <div className="filters" style={{ marginBottom: '1rem' }}>
