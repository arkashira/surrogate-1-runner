import secrets
import string
from typing import Optional
from datetime import datetime, timezone

# Keep this dependency local to avoid circular imports at module level.
# In production, replace with your real store/DB accessor.
def _get_db():
    from .models import Request  # type: ignore
    return Request

def generate_public_token(length: int = 32) -> str:
    """Generate an unguessable public token."""
    alphabet = string.ascii_letters + string.digits
    return "".join(secrets.choice(alphabet) for _ in range(length))

def public_url_for(token: str) -> str:
    """Return the canonical public URL for a token."""
    # In production, read base from app config (e.g., settings.PUBLIC_BASE_URL).
    return f"https://axentx.example.com/public/requests/{token}"

def safe_token_lookup(public_token: str) -> Optional[dict]:
    """
    404-safe token lookup.
    Returns request data dict for a valid token, or None for invalid/missing.
    Must NOT distinguish between 'not found' and 'unauthorized'.
    """
    if not public_token or not isinstance(public_token, str):
        return None

    try:
        Request = _get_db()
        # Use constant-ish query pattern; avoid branching on exception messages.
        row = Request.query.filter_by(public_token=public_token).first()
        if row is None:
            return None

        # Normalize timestamps to UTC ISO strings for consistent downstream handling.
        def _iso(dt):
            if dt is None:
                return datetime.now(timezone.utc).isoformat()
            if dt.tzinfo is None:
                dt = dt.replace(tzinfo=timezone.utc)
            return dt.astimezone(timezone.utc).isoformat()

        # Build a stable data contract for the route layer.
        return {
            "id": row.id,
            "public_token": row.public_token,
            "title": row.title,
            "status": row.status,
            "status_history": row.status_history or [],
            "sla": row.sla or {},
            "last_updated": _iso(row.last_updated),
            "created_at": _iso(row.created_at),
        }
    except Exception:
        # Do not leak any details; treat as invalid token.
        return None
```
