**Core diagnosis (merged, de-duplicated)**
- The training UI triggers repeated HF API calls (tree/list) instead of using CDN fetches, causing 429s.
- No persisted file-list cache; every session re-enumerates via the rate-limited API.
- No UI indication of CDN vs API paths and no “reuse running Lightning Studio” affordance, causing quota burn.
- No lightweight training manifest generator; Lightning jobs still hit API during data load.

**Chosen approach (correct + actionable)**
- Implement a single, framework-agnostic utility (TypeScript) that:
  - Calls HF API **once** (`list_repo_tree` non-recursive) per folder and caches the file list.
  - Converts results to CDN URLs (`resolve/main/...`) for zero-auth, zero-rate-limit fetches.
  - Exports a small React hook (`useCachedFileList`) and a manifest builder (`buildTrainingManifest`) so Lightning jobs can run with zero API calls.
- Add minimal UI changes:
  - Date-folder picker + “Generate file list (CDN)” button.
  - Visual CDN badge and “Reuse running Lightning Studio” toggle.
  - Clear rate-limit feedback with exponential backoff guidance.

// src/utils/hfCdn.ts
// CDN-bypass utilities for Hugging Face datasets.
// One API call (tree) per folder → cache → CDN-only fetches.

const HF_API = 'https://huggingface.co/api';
const HF_CDN = 'https://huggingface.co/datasets';

export interface RepoFile {
  path: string;
  type: 'file' | 'directory';
  size?: number;
}

export interface CachedFileList {
  list: string[]; // CDN URLs
  ts: number;
}

function cacheKey(repo: string, folderPath: string): string {
  const safeFolder = folderPath.replace(/\//g, '_') || 'root';
  return `hf-cdn-${repo}-${safeFolder}.json`;
}

export async function getRepoTree(
  repo: string,
  folderPath: string = '',
  token?: string
): Promise<RepoFile[]> {
  const url = `${HF_API}/datasets/${repo}/tree${folderPath ? `/${folderPath}` : ''}?recursive=false`;
  const res = await fetch(url, {
    headers: token ? { Authorization: `Bearer ${token}` } : {},
  });

  if (res.status === 429) {
    const retryAfter = Number(res.headers.get('retry-after') || 360);
    const err: any = new Error(`HF_RATE_LIMIT: Retry after ${retryAfter}s`);
    err.retryAfter = retryAfter;
    throw err;
  }

  if (!res.ok) {
    throw new Error(`HF_ERROR: ${res.status} ${res.statusText}`);
  }

  return res.json();
}

export function buildCdnUrl(repo: string, filePath: string): string {
  // Always encode path segments to avoid surprises with spaces/special chars
  const encoded = filePath.split('/').map(encodeURIComponent).join('/');
  return `${HF_CDN}/${repo}/resolve/main/${encoded}`;
}

export function buildCdnUrls(
  repo: string,
  folderPath: string,
  files: string[]
): string[] {
  return files.map((f) =>
    buildCdnUrl(repo, folderPath ? `${folderPath}/${f}` : f)
  );
}

  repo: string,
