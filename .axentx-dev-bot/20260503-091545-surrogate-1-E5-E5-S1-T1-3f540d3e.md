import { db } from '../db';

export type RequestStatus = 'pending' | 'processing' | 'completed' | 'failed' | 'cancelled';

export interface TrackedRequest {
  id: string;
  teamId: string;
  owner: string;
  status: RequestStatus;
  createdAt: Date;
  updatedAt: Date;
  payload: unknown;
}

export interface ListRequestsFilters {
  teamId: string;
  status?: RequestStatus;
  owner?: string;
  page?: number;
  limit?: number;
}

export interface ListRequestsResult {
  items: TrackedRequest[];
  total: number;
  statusCounts: Partial<Record<RequestStatus, number>>;
  page: number;
  limit: number;
}

function buildStatusCounts(rows: Array<{ status: RequestStatus; count: string }>): Partial<Record<RequestStatus, number>> {
  const counts: Partial<Record<RequestStatus, number>> = {};
  for (const row of rows) {
    counts[row.status] = Number(row.count);
  }
  return counts;
}

export async function listRequests(filters: ListRequestsFilters): Promise<ListRequestsResult> {
  const { teamId, status, owner } = filters;
  if (!teamId) {
    throw new Error('teamId is required');
  }

  const page = Math.max(1, Number.isFinite(filters.page) ? Math.floor(Number(filters.page)) : 1);
  const limit = Math.min(100, Math.max(1, Number.isFinite(filters.limit) ? Math.floor(Number(filters.limit)) : 50));
  const offset = (page - 1) * limit;

  const conditions: string[] = ['"teamId" = $1'];
  const params: (string | RequestStatus)[] = [teamId];

  if (status) {
    conditions.push(`status = $${params.length + 1}`);
    params.push(status);
  }
  if (owner) {
    conditions.push(`owner = $${params.length + 1}`);
    params.push(owner);
  }

  const where = conditions.length ? `WHERE ${conditions.join(' AND ')}` : '';

  // Items
  const itemsQuery = `
    SELECT id, status, owner, "teamId", "createdAt", "updatedAt", payload
    FROM requests
    ${where}
    ORDER BY "createdAt" DESC
    LIMIT $${params.length + 1} OFFSET $${params.length + 2}
  `;
  const { rows: items } = await db.query<TrackedRequest>(itemsQuery, [...params, limit, offset]);

  // Total
  const totalQuery = `SELECT COUNT(*) as total FROM requests ${where}`;
  const { rows: totalRows } = await db.query<{ total: string }>(totalQuery, params);
  const total = Number(totalRows[0]?.total || 0);

  // Status counts (within same filter set)
  const countsQuery = `
    SELECT status, COUNT(*) as count
    FROM requests
    ${where}
    GROUP BY status
  `;
  const { rows: countRows } = await db.query<{ status: RequestStatus; count: string }>(countsQuery, params);
  const statusCounts = buildStatusCounts(countRows);

  return {
    items,
    total,
    statusCounts,
    page,
    limit,
  };
}
