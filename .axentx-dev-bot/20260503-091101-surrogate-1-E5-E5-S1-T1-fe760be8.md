// src/frontend/onboarding/Telemetry.ts
/**
 * Lightweight onboarding telemetry.
 * Associates onboarding completion with higher open-rate measurement.
 */

const STORAGE_KEY = 'axentx_onboarding_v1';

export type OnboardingEvent =
  | { type: 'tour_start' }
  | { type: 'tour_step'; step: number; label: string }
  | { type: 'tour_complete' }
  | { type: 'first_document_upload'; docId: string; name: string }
  | { type: 'first_document_open'; docId: string }
  | { type: 'checklist_complete' };

export interface OnboardingState {
  completed: boolean;
  firstDocumentId: string | null;
  firstDocumentName: string | null;
  tourCompleted: boolean;
  checklistCompleted: boolean;
  lastEventAt: string;
}

function loadState(): OnboardingState {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? { ...initialState(), ...JSON.parse(raw) } : initialState();
  } catch {
    return initialState();
  }
}

function initialState(): OnboardingState {
  return {
    completed: false,
    firstDocumentId: null,
    firstDocumentName: null,
    tourCompleted: false,
    checklistCompleted: false,
    lastEventAt: new Date().toISOString(),
  };
}

function saveState(state: OnboardingState) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
  } catch {
    // ignore storage errors
  }
}

export function track(event: OnboardingEvent) {
  const state = loadState();
  state.lastEventAt = new Date().toISOString();

  switch (event.type) {
    case 'tour_start':
      break;
    case 'tour_step':
      break;
    case 'tour_complete':
      state.tourCompleted = true;
      break;
    case 'first_document_upload':
      state.firstDocumentId = event.docId;
      state.firstDocumentName = event.name;
      break;
    case 'first_document_open':
      // preserve firstDocumentId for attribution
      break;
    case 'checklist_complete':
      state.checklistCompleted = true;
      state.completed = true;
      break;
  }

  saveState(state);

  try {
    navigator.sendBeacon?.('/_/onboarding', JSON.stringify({ event, state }));
  } catch {
    // ignore
  }
}

export function getOnboardingState(): OnboardingState {
  return loadState();
}

export function isOnboardingCompleted(): boolean {
  return getOnboardingState().completed;
}
