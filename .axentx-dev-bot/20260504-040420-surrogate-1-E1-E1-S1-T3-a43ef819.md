```yaml
# .github/workflows/ci.yml
name: CI

on:
  pull_request:
    branches:
      - main
  workflow_dispatch:

jobs:
  compliance-scan:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Terraform
        uses: hashicorp/setup-terraform@v2

      - name: Set up Helm
        run: |
          curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/main/scripts/get-helm-3
          chmod 700 get_helm.sh
          ./get_helm.sh

      - name: Run Compliance-Scan
        id: compliance
        uses: axentx/compliance-scan-action@v1
        with:
          scan-targets: 'terraform/,helm/'
          output-format: 'github'
          auto-remediate: true

      - name: Upload findings as PR comment
        if: steps.compliance.outputs.findings_summary
        uses: marocchino/sticky-pull-request-comment@v2
        with:
          message: |
            ### 🔍 Compliance Scan Results
            ${{ steps.compliance.outputs.findings_summary }}
            
            🛠️ Auto-remediation has been applied for fixable issues.
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
```

- **AC5** – The pipeline fails (exits with non‑zero status) if any **high** severity findings remain unremediated after the auto‑remediation step.  
- **AC6** – All steps are idempotent: re‑running the workflow on the same commit produces no duplicate PR comments or redundant commits.
