```python
import os
import smtplib
import logging
from email.mime.multipart import MIMEMultipart
from email.mime.text import MIMEText
from typing import List
from jinja2 import Environment, FileSystemLoader
from dotenv import load_dotenv

load_dotenv()
logger = logging.getLogger(__name__)

class EmailSender:
    def __init__(self, template_dir: str = "src/templates"):
        self.smtp_host = os.getenv("SMTP_HOST")
        self.smtp_port = int(os.getenv("SMTP_PORT", "587"))
        self.smtp_user = os.getenv("SMTP_USER")
        self.smtp_password = os.getenv("SMTP_PASSWORD")
        self.smtp_from = os.getenv("SMTP_FROM", "digest@axentx.example")
        self.templates = Environment(loader=FileSystemLoader(template_dir))

    def _connect(self):
        server = smtplib.SMTP(self.smtp_host, self.smtp_port, timeout=30)
        server.starttls()
        if self.smtp_user and self.smtp_password:
            server.login(self.smtp_user, self.smtp_password)
        return server

    def send(self, to_addrs: List[str], subject: str, html_body: str, text_body: str):
        msg = MIMEMultipart("alternative")
        msg["Subject"] = subject
        msg["From"] = self.smtp_from
        msg["To"] = ", ".join(to_addrs)
        msg.attach(MIMEText(text_body, "plain"))
        msg.attach(MIMEText(html_body, "html"))

        max_retries = 3
        for attempt in range(1, max_retries + 1):
            try:
                with self._connect() as server:
                    server.sendmail(self.smtp_from, to_addrs, msg.as_string())
                logger.info("Digest sent to %s", to_addrs)
                return
            except Exception as exc:
                logger.warning("Attempt %s/%s failed sending to %s: %s", attempt, max_retries, to_addrs, exc)
                if attempt == max_retries:
                    raise
