# MapView Component for React Native

## Overview

The **MapView** component is a lightweight, high-performance map renderer for React Native applications. It supports:

- Multiple map providers: Google Maps, Mapbox, and OpenStreetMap
- Custom overlays (markers, polygons, heatmaps)
- Secure overlay handling with encrypted payloads
- Offline caching and vector tile support

---

## Installation

```bash
# Using npm
npm install @axentx/react-native-mapview

# Using yarn
yarn add @axentx/react-native-mapview
```

---

## Basic Usage

```tsx
import React from 'react';
import { View, StyleSheet } from 'react-native';
import MapView, { Marker, Overlay } from '@axentx/react-native-mapview';

export default function App() {
  return (
    <View style={styles.container}>
      <MapView
        provider="google"
        style={styles.map}
        initialRegion={{
          latitude: 37.78825,
          longitude: -122.4324,
          latitudeDelta: 0.0922,
          longitudeDelta: 0.0421,
        }}
      >
        <Marker
          coordinate={{ latitude: 37.78825, longitude: -122.4324 }}
          title="San Francisco"
          description="A beautiful city"
        />
        <Overlay
          id="secureHeatmap"
          type="heatmap"
          data={encryptedHeatmapData}
          options={{ radius: 50, opacity: 0.7 }}
        />
      </MapView>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1 },
  map: { flex: 1 },
});
```

## Advanced Usage: Secure Overlay Component

This example demonstrates secure overlay handling with location permissions and dynamic region handling:

```javascript
// src/components/SecureOverlay.js
import React, { useRef, useState } from 'react';
import MapView, { Marker, Callout, Polygon } from 'react-native-maps';
import * as Location from 'expo-location';

const SecureOverlay = () => {
  const mapRef = useRef(null);
  const [region, setRegion] = useState(null);
  const [overlayData, setOverlayData] = useState(null);

  const getLocationAsync = async () => {
    let { status } = await Location.requestPermissionsAsync();
    if (status !== 'granted') {
      console.log('Permission to access location was denied');
      return;
    }

    const location = await Location.getCurrentPositionAsync();
    setRegion({
      latitude: location.coords.latitude,
      longitude: location.coords.longitude,
      latitudeDelta: 0.0922,
      longitudeDelta: 0.0421,
    });
  };

  const handleRegionChange = (region) => {
    setRegion(region);
    // Fetch overlay data based on the new region
  };

  return (
    <MapView
      ref={mapRef}
      style={{ flex: 1 }}
      region={region}
      onRegionChange={handleRegionChange}
    >
      {overlayData && (
        <>
          <Marker
            coordinate={overlayData.coordinate}
            title={overlayData.title}
            description={overlayData.description}
          >
            <Callout onPress={() => console.log('Marker pressed')}>
              <View>
                <Text>{overlayData.calloutText}</Text>
              </View>
            </Callout>
          </Marker>
          <Polygon
            coordinates={overlayData.polygonCoordinates}
            fillColor="rgba(13, 110, 253, 0.2)"
            strokeColor="rgba(13, 110, 253, 1)"
            strokeWidth={1}
          />
        </>
      )}
    </MapView>
  );
};

export default SecureOverlay;
