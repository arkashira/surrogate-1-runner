Moderation and curation models for community forum.
Handles moderation actions, roles, and content review workflows.

from enum import Enum
from dataclasses import dataclass, field

from typing import Optional, List
import csv
import os
import logging
import threading
from pathlib import Path

    APPROVE = "approve"
    REJECT = "reject"
    FLAG = "flag"
    ESCALATE = "escalate"
    ARCHIVE = "archive"

    POST = "post"
    COMMENT = "comment"
    REPLY = "reply"

class ModeratorLevel(Enum):
    JUNIOR = "junior"
    SENIOR = "senior"
    LEAD = "lead"
    ADMIN = "admin"

class Moderator:
    id: str
    username: str
    level: ModeratorLevel

    joined_at: datetime = field(default_factory=datetime.now)
    reviews_count: int = 0
    specializations: List[str] = field(default_factory=list)

            self.notes or "",
            str(self.priority)

            rejection_reason=row[9] if row[9] else None,
            notes=row[10] if row[10] else None,
            priority=int(row[11])
        )
