CREATE TABLE IF NOT EXISTS notifications (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    request_id      UUID NOT NULL REFERENCES requests(id) ON DELETE CASCADE,
    event_type      TEXT NOT NULL,
    payload         JSONB NOT NULL,
    status          TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending','processing','sent','failed')),
    recipients      JSONB NOT NULL,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    processed_at    TIMESTAMPTZ,
    attempts        INT NOT NULL DEFAULT 0,
    last_error      TEXT
);

CREATE INDEX IF NOT EXISTS idx_notifications_request ON notifications (request_id);

from typing import Any, Dict, List, Optional
from uuid import UUID

import asyncpg

    event_type: str

    recipients: Dict[str, Any]  # {"slack_channel": "...", "emails": [...]}

    event_type: str

    status: str
    recipients: Dict[str, Any]
    created_at: datetime
    processed_at: Optional[datetime] = None
    attempts: int
    last_error: Optional[str] = None

    """
    Enqueue and dequeue notification jobs.
    Assumes asyncpg pool available via app state or dependency.
    """

    @staticmethod
    async def enqueue(
        pool: asyncpg.Pool,
        *,
        request_id: UUID,
        event_type: str,
        transition_details: Dict[str, Any],
        recipients: Dict[str, Any],
    ) -> UUID:
        payload = {
            "transition": transition_details,
            "timestamp": datetime.now(timezone.utc).isoformat(),
        }
        row = await pool.fetchrow(
            """
            INSERT INTO notifications (request_id, event_type, payload, recipients)
            VALUES ($1, $2, $3, $4)
            RETURNING id
            """,
            request_id,
            event_type,
            payload,
            recipients,
        )
        return row["id"]

    async def dequeue_batch(pool: asyncpg.Pool, limit: int = 10) -> List[Dict[str, Any]]:
        rows = await pool.fetch(

            FROM notifications
            WHERE status = 'pending'
            ORDER BY created_at
            FOR UPDATE SKIP LOCKED
            LIMIT $1
            """,
            limit,
        )

            return []

        await pool.execute(
