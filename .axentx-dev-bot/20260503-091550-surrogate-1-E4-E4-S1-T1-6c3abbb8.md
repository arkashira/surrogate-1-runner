CREATE TABLE IF NOT EXISTS notifications (
    id BIGSERIAL PRIMARY KEY,
    request_id BIGINT NOT NULL REFERENCES requests(id) ON DELETE CASCADE,
    transition_from VARCHAR(64),
    transition_to VARCHAR(64) NOT NULL,
    message TEXT,
    recipients TEXT[] NOT NULL DEFAULT '{}',
    channel VARCHAR(128),
    status VARCHAR(32) NOT NULL DEFAULT 'pending' CHECK (status IN ('pending','processing','sent','failed')),
    attempts INT NOT NULL DEFAULT 0,
    last_error TEXT,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    processed_at TIMESTAMPTZ
);

CREATE INDEX idx_notifications_pending ON notifications (status, created_at) WHERE status = 'pending';
CREATE INDEX idx_notifications_request ON notifications (request_id);
```
