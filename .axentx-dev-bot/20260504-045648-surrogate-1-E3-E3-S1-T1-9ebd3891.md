"""
Financial Reporting Module

Generates comprehensive financial reports for hardware startups,
including cloud cost breakdowns and cost savings summaries.
"""

import json
from datetime import datetime
from dataclasses import dataclass, field
from typing import List, Dict, Optional
from enum import Enum


class ReportFormat(Enum):
    """Supported output formats for financial reports."""
    HTML = "html"
    MARKDOWN = "markdown"
    JSON = "json"
    TEXT = "text"


@dataclass
class CloudService:
    """Represents a single cloud service with associated costs."""
    name: str
    category: str
    monthly_cost: float
    usage_hours: float = 0.0
    region: str = "us-east-1"
    metadata: Dict = field(default_factory=dict)

    @property
    def cost_per_hour(self) -> float:
        """Calculate cost per hour of usage."""
        if self.usage_hours > 0:
            return self.monthly_cost / self.usage_hours
        return 0.0


@dataclass
class CostSavingsItem:
    """Represents a cost savings entry."""
    initiative: str
    description: str
    monthly_savings: float
    annual_savings: float
    implementation_date: str
    status: str = "completed"


@dataclass
class CloudCostReport:
    """Detailed breakdown of cloud costs."""
    services: List[CloudService] = field(default_factory=list)
    reporting_period: str = ""
    total_monthly_cost: float = 0.0
    currency: str = "USD"

    def add_service(self, service: CloudService) -> None:
        """Add a cloud service to the report."""
        self.services.append(service)
        self.total_monthly_cost += service.monthly_cost

    def get_by_category(self) -> Dict[str, float]:
        """Group costs by category."""
        category_totals = {}
        for service in self.services:
            if service.category not in category_totals:
                category_totals[service.category] = 0.0
            category_totals[service.category] += service.monthly_cost
        return category_totals

    def get_top_services(self, n: int = 5) -> List[CloudService]:
        """Get the top N most expensive services."""
        return sorted(self.services, key=lambda s: s.monthly_cost, reverse=True)[:n]

                    "region": s.region,

                    "metadata": s.metadata
                }
                for s in self.services
