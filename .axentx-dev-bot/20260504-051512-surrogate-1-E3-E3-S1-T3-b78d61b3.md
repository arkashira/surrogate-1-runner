# /opt/axentx/surrogate-1/src/components/InstancePage.js
import React, { useEffect, useState } from 'react';
import { useParams } from 'react-router-dom';
import { getInstanceData } from '../services/api';

const InstancePage = () => {
  const { instanceId } = useParams();
  const [instanceData, setInstanceData] = useState(null);

  useEffect(() => {
    const fetchInstanceData = async () => {
      const data = await getInstanceData(instanceId);
      setInstanceData(data);
    };

    fetchInstanceData();
  }, [instanceId]);

  if (!instanceData) {
    return <div>Loading...</div>;
  }

  return (
    <div className="instance-page">
      <h1>{instanceData.name}</h1>
      <p>Total Data Size: {instanceData.totalDataSize}</p>
      {/* Other instance details */}
    </div>
  );
};

export default InstancePage;
```

```python
# /opt/axentx/surrogate-1/src/services/api.js
import axios from 'axios';

const API_BASE_URL = 'https://api.axentx.com';

export const getInstanceData = async (instanceId) => {
  try {
    const response = await axios.get(`${API_BASE_URL}/instances/${instanceId}`);
    return response.data;
  } catch (error) {
    console.error('Error fetching instance data:', error);
    throw error;
  }
};
```
