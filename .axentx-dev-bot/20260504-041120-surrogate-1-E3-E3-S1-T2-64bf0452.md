import React, { useEffect, useState, useRef } from "react";

/* --------------------------------------------------------------
   Configuration
   -------------------------------------------------------------- */
const API_ENDPOINT = "/api/v1/credits";
const REFRESH_INTERVAL_MS = 60_000; // 1 minute

/* --------------------------------------------------------------
   Helper – normalises the raw API payload
   -------------------------------------------------------------- */
function buildBarData({ total = 0, used = 0 } = {}) {
  const t = Number(total);
  const u = Number(used);
  const remaining = Math.max(t - u, 0);
  const percent = t > 0 ? Math.round((u / t) * 100) : 0;
  return { total: t, used: u, remaining, percent };
}

/* --------------------------------------------------------------
   Small utility – pretty‑print numbers (adds commas)
   -------------------------------------------------------------- */
function fmt(num) {
  return Number(num).toLocaleString();
}

/* --------------------------------------------------------------
   Reusable progress‑bar component
   -------------------------------------------------------------- */
function ProgressBar({ label, data, color, warning }) {
  const { total, used, remaining, percent } = data;

  // Inline styles – keep the component self‑contained
  const containerStyle = {
    width: "100%",
    backgroundColor: "#e0e0e0",
    borderRadius: "4px",
    overflow: "hidden",
    height: "24px",
    position: "relative",
    cursor: "pointer",
  };

  const fillStyle = {
    width: `${percent}%`,
    backgroundColor: warning ? "#ff4d4f" : color,
    height: "100%",
    transition: "width 0.3s ease",
  };

  const tooltip = `Total: ${fmt(total)} | Used: ${fmt(used)} | Remaining: ${fmt(
    remaining
  )}`;

  return (
    <div className="progress-bar-wrapper" style={{ marginBottom: "24px" }}>
      {/* Label */}
      <div className="progress-bar-label">{label}</div>

      {/* Bar + native tooltip */}
      <div
        style={containerStyle}
        title={tooltip}
        data-testid={`${label.toLowerCase().replace(/\s+/g, "-")}-bar`}
      >
        <div style={fillStyle} />
        {/* Percentage overlay */}
        <div
          style={{
            position: "absolute",
            left: "50%",
            top: "0",
            transform: "translateX(-50%)",
            lineHeight: "24px",
            fontSize: "12px",
