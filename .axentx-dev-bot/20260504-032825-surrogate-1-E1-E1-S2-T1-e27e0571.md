    };

    expect(contextManager.getContextValue('data')).toEqual(nestedData);
  });

  test('should merge contexts correctly', () => {
    contextManager.setContextValue('key1', 'value1');

    contextManager.setContextValue('undefinedValue', undefined);

  });
