// /opt/axentx/surrogate-1/src/db/schema.ts
import { pgTable, serial, text, timestamp, varchar } from 'drizzle-orm/pg-core';

export const requests = pgTable('requests', {
  id: serial('id').primaryKey(),
  publicId: varchar('public_id').notNull().unique(),
  title: text('title').notNull(),
  status: varchar('status').notNull(),
  slaState: varchar('sla_state'),
  createdAt: timestamp('created_at').notNull(),
  updatedAt: timestamp('updated_at').notNull(),
});

+ export const subscribers = pgTable('subscribers', {
+   id: serial('id').primaryKey(),
+   requestId: integer('request_id').notNull().references(() => requests.id, { onDelete: 'cascade' }),
+   channelType: varchar('channel_type', { enum: ['email', 'slack'] }).notNull(),
+   channelValue: text('channel_value').notNull(), // email addr or Slack webhook URL
+   createdAt: timestamp('created_at').notNotNull().defaultNow(),
+ });
+
+ // unique constraint: one subscription per channel per request (avoid duplicates)
+ export const subscribersUnique = uniqueIndex('subscribers_request_channel_unique').on(
+   subscribers.requestId,
+   subscribers.channelType,
+   subscribers.channelValue
+ );
```
