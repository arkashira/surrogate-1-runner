// /opt/axentx/surrogate-1/src/routes/public.ts
import { Request, Response } from 'express';
import { db } from '../db';
import { Subscriber } from '../db/schema';

// Add 'Subscribe' action to the public page
export const getPublicPage = async (req: Request, res: Response) => {
  const requestId = req.params.requestId;
  const request = await db.getRequest(requestId);
  const subscribers = await db.getSubscribers(requestId);

  res.render('status-page', {
    request,
    subscribers,
    subscribeAction: `/requests/${requestId}/subscribe`,
  });
};

// Handle 'Subscribe' action
export const postSubscribe = async (req: Request, res: Response) => {
  const requestId = req.params.requestId;
  const subscriber = req.body.subscriber;

  // Validate subscriber information
  if (!subscriber.email && !subscriber.slackChannel) {
    return res.status(400).send('Invalid subscriber information');
  }

  // Create a new subscriber record
  const newSubscriber = new Subscriber({
    requestId,
    email: subscriber.email,
    slackChannel: subscriber.slackChannel,
  });

  // Save the subscriber record to the database
  await db.saveSubscriber(newSubscriber);

  res.redirect(`/requests/${requestId}`);
};
```
