import { subscribers } from '../db/schema.js';

export function renderStatusPage(request: {
  id: string;
  title: string;
  status: string;
  publicUrl: string;
  slaState: 'ok' | 'warning' | 'breach';
  subscribers?: Array<{ id: number; channelType: string; channelValue: string }>;
}) {
  const subs = request.subscribers || [];
  return `
<!doctype html>
<html>
<head><title>${escapeHtml(request.title)} — Status</title></head>
<body>
  <h1>${escapeHtml(request.title)}</h1>
  <p>Status: <strong>${escapeHtml(request.status)}</strong></p>
  <p>SLA: <strong>${escapeHtml(request.slaState)}</strong></p>
  <p><a href="${escapeHtml(request.publicUrl)}">Public URL</a></p>

  <section aria-labelledby="subscribe-heading">
    <h2 id="subscribe-heading">Subscribe to updates</h2>
    <form action="/public/requests/${escapeHtml(request.id)}/subscribe" method="post">
      <label>
        Channel type:
        <select name="channelType" required>
          <option value="email">Email</option>
          <option value="slack">Slack webhook</option>
        </select>
      </label>
      <label>
        Value:
        <input name="channelValue" type="text" required placeholder="email or https://hooks.slack.com/...">
      </label>
      <button type="submit">Subscribe</button>
    </form>

    ${
      subs.length
        ? `
    <h3>Current subscribers</h3>
    <ul>
      ${subs
        .map(
          (s) =>
            `<li>${escapeHtml(s.channelType)}: ${escapeHtml(
              s.channelType === 'email'
                ? s.channelValue
                : new URL(s.channelValue).origin + '/...'
            )}</li>`
        )
        .join('')}
    </ul>`
        : '<p>No subscribers yet.</p>'
    }
  </section>
</body>
</html>
`;
}

function escapeHtml(s: string) {
  return String(s)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}
```
