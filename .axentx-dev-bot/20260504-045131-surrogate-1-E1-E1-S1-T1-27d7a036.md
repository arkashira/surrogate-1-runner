package com.axentx.surrogate;

import java.util.HashMap;
import java.util.Map;
import java.util.concurrent.CompletableFuture;

public interface LLMClient {
    /**
     * Generates text asynchronously for better performance
     * @param prompt The input prompt for text generation
     * @return CompletableFuture containing the generated text
     */
    CompletableFuture<String> generateTextAsync(String prompt);

    /**
     * Factory method to create LLM clients with proper error handling
     * @param provider The LLM provider name
     * @param apiKey The API key for authentication
     * @return Configured LLMClient instance
     * @throws IllegalArgumentException if provider is unsupported
     * @throws RuntimeException if client creation fails
     */
    static LLMClient create(String provider, String apiKey) {
        Map<String, Class<? extends LLMClient>> providers = new HashMap<>();
        providers.put("Claude", ClaudeClient.class);
        providers.put("Minimax", MinimaxClient.class);
        providers.put("OpenAI", OpenAIClient.class);
        providers.put("Anthropic", AnthropicClient.class);
        providers.put("Cohere", CohereClient.class);
        providers.put("Gemini", GeminiClient.class);

        Class<? extends LLMClient> clientClass = providers.get(provider);
        if (clientClass == null) {
            throw new IllegalArgumentException("Unsupported provider: " + provider);
        }

        try {
            return clientClass.getConstructor(String.class).newInstance(apiKey);
        } catch (Exception e) {
            throw new RuntimeException("Failed to create LLM client for provider: " + provider, e);
        }
    }
}

package com.axentx.surrogate;

import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.util.concurrent.CompletableFuture;

public class ClaudeClient implements LLMClient {
    private static final String API_ENDPOINT = "https://api.anthropic.com/v1/complete";
    private final String apiKey;
    private final HttpClient client;

    public ClaudeClient(String apiKey) {
        this.apiKey = apiKey;
        this.client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2)
            .build();
    }

    @Override
    public CompletableFuture<String> generateTextAsync(String prompt) {
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(API_ENDPOINT))
            .header("Content-Type", "application/json")
            .header("Authorization", "Bearer " + apiKey)
            .header("anthropic-version", "2023-06-01")
            .POST(HttpRequest.BodyPublishers.ofString(
                "{\"prompt\":\"" + prompt + "\",\"max_tokens\":50,\"model\":\"claude-2\"}"))
            .build();

        return client.sendAsync(request, HttpResponse.BodyHandlers.ofString())
            .thenApply(HttpResponse::body)
            .exceptionally(e -> {
                throw new RuntimeException("Failed to generate text: " + e.getMessage(), e);
            });
    }
}
