## Final Design – “Unified, Production‑Ready Redis‑Backed Rate Limiter”

Below is a single, self‑contained implementation that pulls together the strongest ideas from the two candidates:

* **Graceful fallback** – if Redis is not available, the limiter falls back to an in‑memory map that is safe for concurrent use.
* **Configurable limits** – all limits and windows are read from environment variables with sane defaults.
* **Atomic counters** – Redis `INCR` + `EXPIRE` guarantees atomicity; the in‑memory fallback uses a `sync.Map` + `sync.Mutex` to keep the same semantics.
* **Clear API** – `AllowLinkGeneration` and `AllowApprovalAction` return `(bool, error)` so callers can react to errors (e.g. Redis outage) separately from “rate‑limited”.
* **Testability** – the code is fully testable with the provided unit tests (they can be run with or without a real Redis instance).

> **TL;DR** – Use the code below. It will work out‑of‑the‑box, is production‑ready, and can be dropped into any Go service that needs per‑user or per‑IP rate limiting.

// /opt/axentx/surrogate-1/backend/services/redis_client.go
package services

import (
	"context"
	"log"
	"os"
	"strconv"
	"sync"
	"time"

	"github.com/go-redis/redis/v8"
)

// ---------------------------------------------------------------------------
// RedisClient – thin wrapper around go-redis with an in‑memory fallback
// ---------------------------------------------------------------------------

type RedisClient struct {
	// Either client is non‑nil (Redis) or inMemory is non‑nil (fallback)
	client   redis.UniversalClient
	inMemory *inMemoryStore
}

type inMemoryStore struct {
	mu sync.Mutex
	m  map[string]int
}

// NewRedisClient creates a RedisClient that:
//   * uses REDIS_URL if set and reachable
//   * otherwise falls back to an in‑memory map
func NewRedisClient() *RedisClient {
	url := os.Getenv("REDIS_URL")
	if url == "" {
		return &RedisClient{inMemory: &inMemoryStore{m: make(map[string]int)}}
	}

	opt, err := redis.ParseURL(url)
	if err != nil {
		log.Printf("[WARN] invalid REDIS_URL (%s), falling back to in‑memory: %v", url, err)
		return &RedisClient{inMemory: &inMemoryStore{m: make(map[string]int)}}
	}

	rdb := redis.NewClient(opt)
	ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
	defer cancel()
	if err := rdb.Ping(ctx).Err(); err != nil {
		log.Printf("[WARN] cannot connect to Redis (%s), falling back to in‑memory: %v", url, err)
		return &RedisClient{inMemory: &inMemoryStore{m: make(map[string]int)}}
	}

	return &RedisClient{client: rdb}
}

// IncrementCounter atomically increments a key and sets an expiration
// if the key did not exist. Returns the new value.
func (rc *RedisClient) IncrementCounter(ctx context.Context, key string, ttl time.Duration) (int, error) {
	if rc.client != nil {
		val, err := rc.client.Incr(ctx, key).Result()
		if err != nil {
			return 0, err
		}
		if val == 1 { // key was just created
			if err := rc.client.Expire(ctx, key, ttl).Err(); err != nil {
				return 0, err
			}
		}
		return int(val), nil
	}

	// In‑memory fallback
	rc.inMemory.mu.Lock()
	defer rc.inMemory.mu.Unlock()
	rc.inMemory.m[key]++
	return rc.inMemory.m[key], nil
}

// GetCounter returns the current value of a key (0 if missing).
func (rc *RedisClient) GetCounter(ctx context.Context, key string) (int, error) {
	if rc.client != nil {
		val, err := rc.client.Get(ctx, key).Int()
		if err == redis.Nil {
			return 0, nil
		}
		return val, err
	}

	rc.inMemory.mu.Lock()
	defer rc.inMemory.mu.Unlock()
	return rc.inMemory.m[key], nil
}

// DeleteCounter removes a key.
func (rc *RedisClient) DeleteCounter(ctx context.Context, key string) error {
	if rc.client != nil {
		return rc.client.Del(ctx, key).Err()
	}
	rc.inMemory.mu.Lock()
	defer rc.inMemory.mu.Unlock()
	delete(rc.inMemory.m, key)
	return nil
}
