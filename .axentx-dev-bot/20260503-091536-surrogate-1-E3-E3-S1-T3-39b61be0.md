// /opt/axentx/surrogate-1/src/lib/email.ts
import nodemailer from 'nodemailer';

const transporter = nodemailer.createTransport(process.env.SMTP_URL || '', {
  from: process.env.EMAIL_FROM || 'status@axentx.io',
  // Keep-alive and timeouts tuned for fast, reliable delivery
  connectionTimeout: 10_000,
  socketTimeout: 15_000,
});

export interface DigestPayload {
  requestTitle: string;
  newStatus: string;
  publicUrl: string;
  slaState: string;
  recipients: string[];
}

/**
 * Send a simple status digest email.
 * Non-throwing by design: queue job should handle retries/logging.
 */
export async function sendStatusDigest(payload: DigestPayload): Promise<void> {
  const { requestTitle, newStatus, publicUrl, slaState, recipients } = payload;

  if (!recipients?.length) return;

  const safeRecipients = recipients
    .map((r) => r.trim())
    .filter(Boolean);

  const subject = `Status update: ${requestTitle} → ${newStatus}`;
  const text = [
    `Request: ${requestTitle}`,
    `New status: ${newStatus}`,
    `SLA state: ${slaState}`,
    `Public URL: ${publicUrl}`,
    '',
    '— axentx status system',
  ].join('\n');

  const html = `
    <p><strong>Request:</strong> ${escapeHtml(requestTitle)}</p>
    <p><strong>New status:</strong> ${escapeHtml(newStatus)}</p>
    <p><strong>SLA state:</strong> ${escapeHtml(slaState)}</p>
    <p><strong>Public URL:</strong> <a href="${escapeAttr(publicUrl)}">${escapeHtml(publicUrl)}</a></p>
    <hr>
    <small>— axentx status system</small>
  `;

  await transporter.sendMail({
    to: safeRecipients.join(', '),
    subject,
    text,
    html,
  });
}

function escapeHtml(str: string): string {
  return String(str || '')
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}

function escapeAttr(str: string): string {
  return escapeHtml(str).replace(/"/g, '&quot;');
}
```
