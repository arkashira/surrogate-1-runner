```properties
# -------------------------------------------------------------------------
# Global notification switch
# -------------------------------------------------------------------------
# Turn the whole notification subsystem on or off.
notification.enabled=true

# -------------------------------------------------------------------------
# Timing & latency
# -------------------------------------------------------------------------
# Maximum time allowed between detection of a signature‑drift event and the
# moment the alert is sent.  The value is expressed in **seconds** – this is
# the most precise unit for the 60‑second SLA required by the SRE team.
notification.signatureDrift.maxLatencySeconds=60

# -------------------------------------------------------------------------
# Recipients
# -------------------------------------------------------------------------
# Primary on‑call list (comma‑separated).  Use the same address format for
# every team that must see the alert.
notification.signatureDrift.recipients=\
sre-oncall@example.com,\
ops-lead@example.com,\
oncall-sre@company.com

# -------------------------------------------------------------------------
# Message formatting
# -------------------------------------------------------------------------
# Subject line prefix – makes the alert easy to filter in mail clients.
notification.subject.prefix=[SURROGATE‑1] API Signature Drift Alert

# Body template – placeholders are replaced at runtime.
#   {jobId}   – internal identifier of the affected job
#   {jobName} – human‑readable name of the job (if available)
#   {signature} – the drifted request signature
notification.signatureDrift.body.template=\
API request signature has drifted for job **{jobName}** (ID: {jobId}).\n\
Drifted signature: `{signature}`

# -------------------------------------------------------------------------
# Transport – SMTP (example configuration)
# -------------------------------------------------------------------------
notification.smtp.host=smtp.example.com
notification.smtp.port=587
notification.smtp.username=alert-sender
# NEVER store a plain‑text password in source control.
# Replace the placeholder with a secret fetched from a vault or env‑var at runtime.
notification.smtp.password=${SMTP_ALERT_PASSWORD}
notification.smtp.from=alerts@axentx.com
notification.smtp.tls=true
```

```java
class NotificationSettingsTest {

    @Test
    void shouldLoadPropertiesWithoutHardCodedPassword() {
        Properties props = new Properties();
        props.load(NotificationSettingsTest.class
                .getResourceAsStream("/notification-settings.properties"));
        assertFalse(props.getProperty("smtp.password").contains("realPassword"));
    }

    @Test
    void shouldResolveSmtpPasswordFromEnv() {
        System.setProperty("SMTP_ALERT_PASSWORD", "secret");
        NotificationSettings settings = new NotificationSettings();
        assertEquals("secret", settings.getSmtpPassword());
    }

    @Test
    void shouldThrowIfSmtpPasswordMissing() {
        System.clearProperty("SMTP_ALERT_PASSWORD");
        Exception e = assertThrows(ConfigurationException.class,
                () -> new NotificationSettings());
        assertTrue(e.getMessage().contains("SMTP_ALERT_PASSWORD"));
    }

    @Test
    void shouldRenderSubjectTemplate() {
        NotificationTemplate tmpl = new NotificationTemplate("[SURROGATE-1] API Signature Drift Alert");
        assertEquals("[SURROGATE-1] API Signature Drift Alert", tmpl.render());
    }

    @Test
    void shouldReplacePlaceholdersInBody() {
        String bodyTemplate = "Job {jobName} ({jobId}) drifted signature {signature}";
        Map<String, String> vars = Map.of(
                "jobName", "DataSync",
                "jobId",   "1234",
                "signature","abcd1234");
        String rendered = NotificationTemplate.render(bodyTemplate, vars);
        assertEquals("Job DataSync (1234) drifted signature abcd1234", rendered);
```
