## 1.  The “one‑stop” CI‑integration script

```bash
#!/usr/bin/env bash
# ------------------------------------------------------------------
# /opt/axentx/surrogate-1/scripts/ci_integration.sh
#
# Reads a JSON audit report from STDIN and POSTs it to a configurable
# triage API endpoint.
#
# Environment variables
#   CI_API_URL   – (required)  Full URL to POST the findings to
#   CI_API_TOKEN – (optional) Bearer token for authentication
#   CI_TIMEOUT   – (optional) Request timeout in seconds (default 30)
#
# Usage examples
#   cat audit-report.json | CI_API_URL=https://api.example.com/triage \
#       CI_API_TOKEN=abc123 ./ci_integration.sh
#
# ------------------------------------------------------------------

set -euo pipefail

# ------------------------------------------------------------------
# 1️⃣  Configuration & validation
# ------------------------------------------------------------------
API_URL="${CI_API_URL:-}"
API_TOKEN="${CI_API_TOKEN:-}"
TIMEOUT="${CI_TIMEOUT:-30}"

if [[ -z "$API_URL" ]]; then
  echo "ERROR: CI_API_URL is required" >&2
  echo "Usage: cat audit.json | CI_API_URL=https://api.example.com/triage ./ci_integration.sh" >&2
  exit 1
fi

# ------------------------------------------------------------------
# 2️⃣  Read JSON from STDIN
# ------------------------------------------------------------------
if [[ -t 0 ]]; then
  echo "ERROR: No input on STDIN – pipe a JSON audit report to this script." >&2
  exit 1
fi

INPUT_JSON=$(cat -)

if [[ -z "$INPUT_JSON" ]]; then
  echo "ERROR: Empty input received on STDIN" >&2
  exit 1
fi

# ------------------------------------------------------------------
# 3️⃣  Build the curl command
# ------------------------------------------------------------------
CURL_CMD=(curl -sS -f -L \
  -X POST \
  -H "Content-Type: application/json" \
  --max-time "$TIMEOUT")

# Optional bearer‑token header
[[ -n "$API_TOKEN" ]] && CURL_CMD+=(-H "Authorization: Bearer $API_TOKEN")

# ------------------------------------------------------------------
# 4️⃣  POST the audit report
# ------------------------------------------------------------------
RESPONSE=$("${CURL_CMD[@]}" -d "$INPUT_JSON" "$API_URL" 2>&1) || {
  echo "ERROR: Failed to POST audit report to $API_URL" >&2
  echo "Response:" >&2
  echo "$RESPONSE" >&2
  exit 1
}

# ------------------------------------------------------------------
# 5️⃣  Success
# ------------------------------------------------------------------
echo "✅ Audit report successfully posted to $API_URL"
echo "$RESPONSE"
```
