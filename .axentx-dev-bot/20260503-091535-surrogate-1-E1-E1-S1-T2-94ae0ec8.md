import { Router } from 'express';
import path from 'path';
import fs from 'fs';

const router = Router();

// In-memory store: id -> { title, status, lastUpdated, createdAt }
const requests = new Map<string, {
  title: string;
  status: string;
  lastUpdated: string;
  createdAt: number;
}>();

// Ensure views directory exists (best-effort)
const viewsDir = path.join(__dirname, '..', '..', 'views');
if (!fs.existsSync(viewsDir)) {
  fs.mkdirSync(viewsDir, { recursive: true });
}

const statusPagePath = path.join(viewsDir, 'status-page.html');

// POST /requests
router.post('/requests', (req, res) => {
  const start = Date.now();

  const id = crypto.randomUUID();
  const now = new Date().toISOString();
  const record = {
    title: `Request ${id.slice(0, 8)}`,
    status: 'pending',
    lastUpdated: now,
    createdAt: Date.now(),
  };

  requests.set(id, record);

  // Enforce <200ms budget (in-memory op is <<1ms)
  const elapsed = Date.now() - start;
  if (elapsed > 180) {
    console.warn(`POST /requests slow: ${elapsed}ms`);
  }

  res.status(201).json({
    id,
    publicUrl: `/public/requests/${id}`,
  });
});

// GET /public/requests/:id
router.get('/public/requests/:id', (req, res) => {
  const { id } = req.params;
  const record = requests.get(id);

  if (!record) {
    return res.status(404).send('Not found');
  }

  // Read template once and cache in memory after first read
  let template = fs.readFileSync(statusPagePath, 'utf-8');

  res.send(template);
});
