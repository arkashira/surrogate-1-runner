#### `/opt/axentx/surrogate-1/src/routes/public/requests.ts`
```typescript
import express, { Request, Response } from 'express';
import { v4 as uuidv4 } from 'uuid';
import { RequestModel } from '../models/RequestModel';
import { renderStatusPage } from '../views/status-page';

const router = express.Router();

// Create a request and return public URL
router.post('/', async (req: Request, res: Response) => {
  const id = uuidv4();
  const now = new Date().toISOString();
  const request = new RequestModel({
    id,
    title: req.body.title || 'Support Request',
    status: 'pending',
    lastUpdated: now,
    slaTimer: 24, // hours remaining example; compute as needed
    createdAt: now,
  });
  await request.save();
  res.status(201).json({ publicUrl: `/public/requests/${id}` });
});

// Public status page (unlisted, no auth, 404 for invalid IDs)
router.get('/:id', async (req: Request, res: Response) => {
  const { id } = req.params;
  const request = await RequestModel.findOne({ id });
  if (!request) {
    return res.status(404).send('Not Found');
  }
  const html = renderStatusPage({
    title: request.title,
    status: request.status,
    lastUpdated: request.lastUpdated,
    slaTimer: request.slaTimer,
  });
  res.send(html);
});

export default router;
```

#### `/opt/axentx/surrogate-1/src/views/status-page.ts`
```typescript
export interface StatusPageData {
  title: string;
  status: string;
  lastUpdated: string;
  slaTimer: number | string;
}

export function renderStatusPage(data: StatusPageData): string {
  // Minimal, safe HTML template
  return `<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="utf-8" />
  <title>${escapeHtml(data.title)} - Status</title>
</head>
<body>
  <h1>Request Status</h1>
  <p>Title: ${escapeHtml(data.title)}</p>
  <p>Status: ${escapeHtml(data.status)}</p>
  <p>Last Updated: ${escapeHtml(data.lastUpdated)}</p>
  <p>SLA Timer: ${escapeHtml(String(data.slaTimer))}</p>
</body>
</html>`;
}

function escapeHtml(str: string): string {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#39;');
}
```
