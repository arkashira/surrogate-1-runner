### 2. src/lib/hashUtils.ts
```ts
export async function computeFileHash(file: File): Promise<string> {
  const buffer = await file.arrayBuffer();
  const hashBuffer = await crypto.subtle.digest('SHA-256', buffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  return hashArray.map((b) => b.toString(16).padStart(2, '0')).join('');
}

export function fileTypeFromName(name: string): 'pdf' | 'epub' | 'unknown' {
  const ext = name.split('.').pop()?.toLowerCase() || '';
  if (ext === 'pdf') return 'pdf';
  if (ext === 'epub') return 'epub';
  return 'unknown';
}
```

### 3. src/lib/libraryStore.ts
```ts
type LibraryItem = {
  id: string;       // file hash
  name: string;     // original filename
  title: string;
  author: string;
  type: 'pdf' | 'epub' | 'unknown';
  size: number;
  addedAt: number;
};

class LibraryStore {
  private items = new Map<string, LibraryItem>();

  has(hash: string): boolean {
    return this.items.has(hash);
  }

  add(item: LibraryItem): boolean {
    if (this.items.has(item.id)) return false;
    this.items.set(item.id, item);
    return true;
  }

  getAll(): LibraryItem[] {
    return Array.from(this.items.values()).sort((a, b) => b.addedAt - a.addedAt);
  }
}

export const libraryStore = new LibraryStore();
```
