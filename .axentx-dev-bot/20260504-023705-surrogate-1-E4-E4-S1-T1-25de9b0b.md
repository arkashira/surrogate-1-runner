import uuid
from typing import Optional

class WebhookManager:
    """Manages unique webhook URLs for users with test alert functionality."""

    def __init__(self):
        """Initialize the webhook manager with an empty URL dictionary."""
        self.webhook_urls = {}

    def generate_unique_webhook_url(self, user_id: str) -> str:
        """
        Generate a unique webhook URL for a given user ID.

        Args:
            user_id: The unique identifier for the user

        Returns:
            The generated webhook URL
        """
        unique_id = str(uuid.uuid4())
        webhook_url = f"https://api.example.com/webhooks/{user_id}/{unique_id}"
        self.webhook_urls[user_id] = webhook_url
        return webhook_url

    def get_webhook_url(self, user_id: str) -> Optional[str]:
        """
        Retrieve the webhook URL for a given user ID.

        Args:
            user_id: The unique identifier for the user

        Returns:
            The webhook URL if exists, None otherwise
        """
        return self.webhook_urls.get(user_id)

    def test_alert(self, user_id: str) -> str:
        """
        Simulate sending a test alert to the user's webhook URL.

        Args:
            user_id: The unique identifier for the user

        Returns:
            Confirmation message or error if URL not found
        """
        webhook_url = self.get_webhook_url(user_id)
        if webhook_url:
            # In a real implementation, this would actually send an HTTP request
            return f"Test alert sent to {webhook_url}"
        return "Webhook URL not found for this user."

# Example usage
