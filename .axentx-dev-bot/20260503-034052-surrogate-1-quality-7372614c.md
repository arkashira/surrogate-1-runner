def cdn_download(url: str, timeout: int = 30) -> bytes:
    for attempt in range(5):
        try:

            if resp.status_code == 429:
                wait = 360
                log.warning("CDN 429, waiting %ss", wait)
                time.sleep(wait)

            resp.raise_for_status()
            return resp.content
