### 1) Create Mac/Linux orchestrator snapshot script  
Path: `scripts/snapshot_hf_dataset.py`  
- Uses `huggingface_hub.HfApi` to list repo tree once (`recursive=False`).  
- Respects 429 with exponential backoff and a 360s ceiling.  
- Emits deterministic `file-manifest.json` (sorted paths).  
- Uses public CDN URLs (`/resolve/main/...`) — no auth required for public datasets.

```python
#!/usr/bin/env python3
"""
Create a deterministic CDN file manifest for axentx/surrogate-1-training-pairs.
Usage:
    HF_TOKEN=... python scripts/snapshot_hf_dataset.py \
        --repo axentx/surrogate-1-training-pairs \
        --folder batches/public-merged/2026-05-03 \
        --out file-manifest.json
"""
import argparse
import json
import os
import sys
import time
from datetime import datetime, timezone
from pathlib import Path

from huggingface_hub import HfApi, hf_hub_url

HF_API_RATE_LIMIT_RETRY = 360  # seconds

def exponential_backoff(attempt: int) -> int:
    return min(2 ** attempt, HF_API_RATE_LIMIT_RETRY)

def list_folder_safe(api, repo_id: str, folder: str, max_retries: int = 5):
    for attempt in range(max_retries):
        try:
            tree = api.list_repo_tree(repo_id=repo_id, path=folder, recursive=False)
            return sorted(t.path for t in tree if t.type == "file")
        except Exception as exc:
            status = getattr(exc, "status", None)
            if status == 429:
                wait = exponential_backoff(attempt)
                print(f"[snapshot] HF 429 hit, waiting {wait}s", file=sys.stderr)
                time.sleep(wait)
                continue
            raise
    raise RuntimeError(f"Failed to list {repo_id}/{folder} after {max_retries} retries")

def build_manifest(repo_id: str, folder: str, output_path: str):
    api = HfApi(token=os.getenv("HF_TOKEN"))
    print(f"[snapshot] Listing {repo_id}/{folder} ...")
    files = list_folder_safe(api, repo_id, folder)

    manifest = {
        "repo_id": repo_id,
        "folder": folder,
        "generated_at": datetime.now(timezone.utc).isoformat(),
        "files": [],
    }

    for f in files:
        cdn_url = hf_hub_url(repo_id=repo_id, filename=f, repo_type="dataset")
        cdn_url = cdn_url.replace("/api/", "/resolve/")
        manifest["files"].append(
            {
                "path": f,
                "cdn_url": cdn_url,
                "local_key": Path(f).name,
            }
        )

    out = Path(output_path)
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(manifest, indent=2))
    print(f"[snapshot] Wrote {len(files)} files -> {out}")

def main():
    parser = argparse.ArgumentParser(description="Create CDN file manifest.")
    parser.add_argument("--repo", default="axentx/surrogate-1-training-pairs")
    parser.add_argument("--folder", required=True, help="Folder to snapshot, e.g. batches/public-merged/2026-05-03")
    parser.add_argument("--out", default="file-manifest.json")
    args = parser.parse_args()

    build_manifest(args.repo, args.folder, args.out)

if __name__ == "__main__":
    main()
```
