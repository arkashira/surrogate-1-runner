import logging
from datetime import datetime, timedelta
from email.message import EmailMessage
import smtplib
from apscheduler.schedulers.background import BackgroundScheduler
from axentx.models import Recording

class CaptureWorker:
    def __init__(self, recording):
        self.recording = recording

    def send_email(self):
        if self.recording.user.opted_in:
            try:
                msg = EmailMessage()
                msg.set_content(
                    f"Recording '{self.recording.title}' is ready to listen. Duration: {self.recording.duration} seconds. Playback link: {self.recording.playback_link}"
                )
                msg['Subject'] = f"Recording '{self.recording.title}' is ready"
                msg['From'] = 'axentx@example.com'
                msg['To'] = self.recording.user.email

                with smtplib.SMTP('smtp.example.com', 587) as smtp:
                    smtp.starttls()
                    smtp.login('axentx@example.com', 'password')
                    smtp.send_message(msg)
            except Exception as e:
                logging.error(f"Failed to send email: {e}")

    def run(self):
        if self.recording.status == 'completed' and datetime.now() - self.recording.updated_at < timedelta(minutes=5):
            self.send_email()

def check_and_send_emails():
    completed_recordings = Recording.objects.filter(status="completed").order_by('-created_at')
    for recording in completed_recordings:
        CaptureWorker(recording).run()

def run_scheduler():
    scheduler = BackgroundScheduler()
    scheduler.add_job(check_and_send_emails, 'interval', minutes=5)
    scheduler.start()

    try:
        while True:
            pass
    except KeyboardInterrupt:
        scheduler.shutdown()
```
