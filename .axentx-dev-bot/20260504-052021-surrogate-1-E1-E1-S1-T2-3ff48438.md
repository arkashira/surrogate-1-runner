    os.chdir(cwd)

def run_init(args=None):
    cmd = ["surrogate1", "init"]
    if args:
        cmd.extend(args)
    result = subprocess.run(
        cmd,
        capture_output=True,
        text=True,
    )
    return result

def test_init_creates_file(temp_dir):
    result = run_init()
    assert result.returncode == 0
    assert (temp_dir / "surrogate.yaml").exists()

def test_generated_file_matches_template(temp_dir):
    run_init()
    generated = yaml.safe_load((temp_dir / "surrogate.yaml").read_text())
    template = yaml.safe_load(TEMPLATE_PATH.read_text())
    assert generated == template

def test_required_fields_present(temp_dir):
    run_init()
    cfg = yaml.safe_load((temp_dir / "surrogate.yaml").read_text())
    required = CONFIG_SCHEMA.schema["required"]
    missing = [k for k in required if k not in cfg or cfg[k] in (None, "", [])]
    assert not missing, f"Missing required fields: {missing}"

def test_schema_validation_passes(temp_dir):
    run_init()
    cfg = yaml.safe_load((temp_dir / "surrogate.yaml").read_text())
    # will raise ValidationError if invalid

def test_success_message_contains_launch_command(temp_dir, capsys):

    assert "surrogate1 start --config surrogate.yaml" in result.stdout

def test_exit_code_on_missing_required_field(monkeypatch, temp_dir):
