                endpoint=alert.endpoint,
                usage=alert.usage,
                budget=alert.budget,
                timestamp=alert.timestamp,
                message=alert.message
            )
            self.db.session.add(db_alert)
            self.db.session.commit()

    def all(self) -> List[Alert]:
        """Return all stored alerts."""

            db_alerts = self.DBAlert.query.all()
