import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta

@dataclass
class UserProgress:
    """Represents user's learning progress and preferences"""
    user_id: str
    completed_modules: List[str]
    preferred_topics: List[str]
    learning_style: str  # 'video' or 'article' or 'mixed'
    last_active: datetime
    difficulty_level: str  # 'beginner', 'intermediate', 'advanced'

@dataclass
class ContentItem:
    """Represents learning content item"""
    content_id: str
    title: str
    type: str  # 'video' or 'article'
    topic: str
    difficulty: str  # 'beginner', 'intermediate', 'advanced'
    duration: int  # seconds
    views: int
    rating: float
    published_date: datetime

class ContentRecommendationEngine:
    """
    Content recommendation engine that provides personalized recommendations
    based on user progress and preferences.
    """
    
    def __init__(self, user_data: Dict[str, UserProgress], 
                 content_data: Dict[str, List[ContentItem]]):
        """
        Initialize the recommendation engine.
        
        Args:
            user_data: Dictionary mapping user_id to UserProgress objects
            content_data: Dictionary mapping content type to list of ContentItems
        """
        self.user_data = user_data
        self.content_data = content_data
        self.user_profiles = {}
        self.content_matrix = {}
        
        # Preprocess data
        self._build_user_profiles()
        self._build_content_matrix()
    
    def _build_user_profiles(self):
        """Build user profiles from user data"""
        for user_id, user in self.user_data.items():
            # Normalize preferences
            normalized_topics = {topic: 1.0 for topic in user.preferred_topics}
            # Weight based on recency of activity
            recency_weight = (datetime.now() - user.last_active).days
            recency_factor = max(0.1, 1.0 / (1 + recency_weight))
            
            self.user_profiles[user_id] = {
                'topics': normalized_topics,
                'style': user.learning_style,
                'level': user.difficulty_level,
                'recency': recency_factor,
                'completed': set(user.completed_modules)
            }
    
    def _build_content_matrix(self):
        """Build content matrix for similarity calculations"""
        self.content_matrix = {}
        
        for content_type, items in self.content_data.items():
            self.content_matrix[content_type] = []
            for item in items:
                # Normalize content features
                normalized_item = {
                    'topic': item.topic,
                    'level': item.difficulty,
                    'views': item.views,
                    'rating': item.rating,
                    'duration': item.duration,
                    'type': item.type
                }
                self.content_matrix[content_type].append(normalized_item)

        """Calculate similarity between two content items"""
        # Simple cosine similarity based on features
