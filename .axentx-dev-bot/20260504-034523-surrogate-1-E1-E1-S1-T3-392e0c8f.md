import static org.mockito.Mockito.*;

import com.axentx.complianceguardian.api.model.Model;
import com.axentx.complianceguardian.api.model.ModelRepository;
import com.axentx.complianceguardian.api.model.ModelService;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

public class ModelServiceTest {
    @Mock
    private ModelRepository modelRepository;
    private ModelService modelService;

    @BeforeEach
    public void setup() {
        MockitoAnnotations.initMocks(this);
        modelService = new ModelService(modelRepository);
    }

    @Test
    public void testCreateModel() {
        Model model = new Model("ai-model-1", "description", "status");
        when(modelRepository.save(model)).thenReturn(model);
        Model savedModel = modelService.createModel(model);
        assertEquals(model, savedModel);
        verify(modelRepository, times(1)).save(model);
    }

    @Test
    public void testGetModelById() {
        Model model = new Model("ai-model-1", "description", "status");
        when(modelRepository.findById("123")).thenReturn(java.util.Optional.of(model));
        Model retrievedModel = modelService.getModelById("123");
        assertEquals(model, retrievedModel);
        verify(modelRepository, times(1)).findById("123");
    }

    @Test
    public void testRemoveModel() {
        Model model = new Model("ai-model-1", "description", "status");
        when(modelRepository.findById("123")).thenReturn(java.util.Optional.of(model));
        modelService.removeModel("123");
        verify(modelRepository, times(1)).deleteById("123");
    }

    @Test
    public void testRemoveNonExistentModel() {
        when(modelRepository.findById("123")).thenReturn(java.util.Optional.empty());
        Exception exception = assertThrows(ModelNotFoundException.class, () -> modelService.removeModel("123"));
        assertEquals("Model not found", exception.getMessage());
    }

    @Test
    public void testRemoveModelInUse() {
        Model model = new Model("ai-model-1", "description", "status");
        when(modelRepository.findById("123")).thenReturn(java.util.Optional.of(model));
        Exception exception = assertThrows(ModelInUseException.class, () -> modelService.removeModel("123"));
        assertEquals("Model is currently in use", exception.getMessage());
    }
}
