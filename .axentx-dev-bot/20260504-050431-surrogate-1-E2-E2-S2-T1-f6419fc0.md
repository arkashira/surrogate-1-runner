```python
# src/main/python/pnl_aggregator.py
"""
pnl_aggregator

Utility for turning daily P&L JSON files into weekly and monthly summaries.

Daily file format (required keys):
    {
        "date":      "YYYY-MM-DD",
        "revenue":   float,
        "expense":   float
    }

Aggregated file format:
    {
        "period":            "YYYY-WW" or "YYYYMM",
        "total_revenue":     float,
        "total_expense":     float,
        "net_profit":        float,
        "percent_change":    float | None   # None if no prior period
    }

The module is idempotent – an aggregate file is rewritten only when its
content differs from the current calculation.

CLI usage:
    python -m pnl_aggregator <directory> [--weekly] [--monthly]
"""

from __future__ import annotations

import json
import logging
import sys
from datetime import datetime, timedelta
from pathlib import Path
from typing import Dict, Iterable, List, Optional, Tuple

# --------------------------------------------------------------------------- #
# Logging
# --------------------------------------------------------------------------- #
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s",
    stream=sys.stderr,
)
log = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Types
# --------------------------------------------------------------------------- #
DailyRecord = Dict[str, float]
AggregateRecord = Dict[str, Optional[float]]  # percent_change may be None

# --------------------------------------------------------------------------- #
# Helpers
# --------------------------------------------------------------------------- #
def _load_daily_file(path: Path) -> Optional[DailyRecord]:
    """Return a parsed daily record or None if the file is unreadable or invalid."""
    try:
        with path.open("r", encoding="utf-8") as f:
            data = json.load(f)
        # Basic validation
        if not all(k in data for k in ("date", "revenue", "expense")):
            log.warning("Skipping %s – missing required keys", path.name)
            return None
        # Ensure numeric values
        revenue = float(data["revenue"])
        expense = float(data["expense"])
        return {"date": data["date"], "revenue": revenue, "expense": expense}
    except Exception as exc:
        log.warning("Failed to read %s: %s", path.name, exc)
        return None


def _load_all_daily_records(dir_path: Path) -> List[Tuple[Path, DailyRecord]]:
    """Return a list of (file_path, record) tuples sorted by date."""
    records: List[Tuple[Path, DailyRecord]] = []
    for file in dir_path.glob("*.json"):
        if file.name.startswith(("weekly_", "monthly_")):
            continue
        rec = _load_daily_file(file)
        if rec is None:
            continue
        records.append((file, rec))
    # Sort by the date field inside the JSON, not by filename
    records.sort(key=lambda x: x[1]["date"])
    return records


def _iso_week_key(date_str: str) -> str:
    """Return ISO week key 'YYYY-WWW' for a YYYY-MM-DD string."""
    dt = datetime.strptime(date_str, "%Y-%m-%d").date()
    iso_year, iso_week, _ = dt.isocalendar()
    return f"{iso_year}-W{iso_week:02d}"
