/**
 * Drill-down service for navigating type analysis reports
 */

import { TypeIssue, TypeAnalysisReport, DrillDownContext, DrillDownResult, DrillDownFilters, ReportSummary, OptimizationRecommendation } from './types';

export class DrillDownService {
  private reports: Map<string, TypeAnalysisReport> = new Map();

  /**
   * Load a type analysis report into the system
   */
  loadReport(report: TypeAnalysisReport): void {
    this.reports.set(report.id, report);
  }

  /**
   * Get a specific report by ID
   */
  getReport(reportId: string): TypeAnalysisReport | undefined {
    return this.reports.get(reportId);
  }

  /**
   * Get all loaded reports
   */
  getAllReports(): TypeAnalysisReport[] {
    return Array.from(this.reports.values());
  }

  /**
   * Perform drill-down query with filters
   */
  drillDown(context: DrillDownContext): DrillDownResult {
    const report = this.reports.get(context.reportId);
    if (!report) {
      return {
        issues: [],
        groupedBy: 'none',
        totalCount: 0,
        page: 1,
        pageSize: 20
      };
    }

    let filteredIssues = this.applyFilters(report.issues, context.filters);

    // Apply specific issue selection
    if (context.selectedIssueId) {
      filteredIssues = filteredIssues.filter(i => i.id === context.selectedIssueId);
    }

    // Apply file filter
    if (context.selectedFile) {
      filteredIssues = filteredIssues.filter(i => i.filePath.includes(context.selectedFile!));
    }

    const page = 1;
    const pageSize = 20;
    const startIndex = (page - 1) * pageSize;
    const paginatedIssues = filteredIssues.slice(startIndex, startIndex + pageSize);

    return {
      issues: paginatedIssues,
      groupedBy: this.determineGrouping(context),
      totalCount: filteredIssues.length,
      page,
      pageSize
    };
  }

  /**
   * Get issues grouped by file for drill-down

      existing.push(issue);
      grouped.set(issue.filePath, existing);
    }
    return grouped;
  }
