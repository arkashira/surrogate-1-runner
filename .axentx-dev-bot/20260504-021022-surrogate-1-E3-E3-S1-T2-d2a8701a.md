## Final Synthesized Solution

```yaml
# .github/workflows/doc_sync.yml
name: DocSync

on:
  push:
    branches:
      - '**'

permissions:
  contents: read

jobs:
  docsync:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Run DocSync in Docker
        env:
          DOC_SYNC_TOKEN: ${{ secrets.DOC_SYNC_TOKEN }}
        run: |
          docker run --rm \
            -e DOC_SYNC_TOKEN="$DOC_SYNC_TOKEN" \
            -v "${{ github.workspace }}:/workspace" \
            -w /workspace \
            axentx/docsync:latest \
            ./scripts/doc_sync.sh
        shell: bash

      - name: Upload preview artifact
        if: success()
        uses: actions/upload-artifact@v4
        with:
          name: doc-preview
          path: doc-preview.html
          retention-days: 7

      - name: Report failure
        if: failure()
        run: |
          echo "::error::DocSync failed. Check logs above for details."
          exit 1
```

```bash
#!/bin/bash
# scripts/doc_sync.sh
# DocSync wrapper script for CI execution

set -euo pipefail

# Configuration
DOC_SYNC_TOKEN="${DOC_SYNC_TOKEN:-}"
OUTPUT_FILE="doc-preview.html"
DOC_SYNC_IMAGE="axentx/docsync:latest"

# Validate required environment variables
if [[ -z "$DOC_SYNC_TOKEN" ]]; then
    echo "ERROR: DOC_SYNC_TOKEN is not set" >&2
    exit 1
fi

# Validate token format (basic sanity check)
if [[ ${#DOC_SYNC_TOKEN} -lt 10 ]]; then
    echo "ERROR: DOC_SYNC_TOKEN appears invalid (too short)" >&2
    exit 1
fi

echo "Starting DocSync process..."

# Run DocSync inside Docker container
if ! docker run --rm \
    -e DOC_SYNC_TOKEN="$DOC_SYNC_TOKEN" \
    -v "$PWD:/workspace" \
    -w /workspace \
    "$DOC_SYNC_IMAGE" \
    --output "$OUTPUT_FILE"; then
    echo "ERROR: DocSync command failed with exit code $?" >&2
    exit 1
fi

echo "DocSync completed successfully"

# Verify output file exists
if [[ -f "$OUTPUT_FILE" ]]; then
    echo "Preview artifact created: $OUTPUT_FILE"
    ls -la "$OUTPUT_FILE"
else
    echo "ERROR: Output file $OUTPUT_FILE was not created" >&2
    exit 1
fi
```
