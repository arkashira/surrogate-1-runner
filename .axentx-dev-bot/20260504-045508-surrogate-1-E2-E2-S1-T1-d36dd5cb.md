## 1. Acceptance Criteria  
- **AC1**: When a user submits *needs* and *budget* via the UI, the backend returns a non‑empty array of build recommendations.  
- **AC2**: Each recommendation contains the fields `id`, `name`, `price`, `specs`, and `rating`.  
- **AC3**: The recommendations array is sorted in descending order of `rating`.  
- **AC4**: If the user changes the budget, the recommendations list updates within 200 ms of the input change.  
- **AC5**: If the budget is lower than the cheapest available build, the API returns an empty array and a message `"No builds fit your budget"`.  
- **AC6**: The API endpoint `/api/recommendations` accepts a JSON payload `{needs: string[], budget: number}` and responds with status 200 and the recommendations JSON.

const { generateRecommendations } = require('../src/recommendations');

      expect(r).toHaveProperty('name');
      expect(r).toHaveProperty('price');
      expect(r).toHaveProperty('specs');
      expect(r).toHaveProperty('rating');
    });
  });

  test('recommendations are sorted by rating descending', () => {
    const recs = generateRecommendations(['gaming'], 2000);
    for (let i = 1; i < recs.length; i++) {
      expect(recs[i-1].rating).toBeGreaterThanOrEqual(recs[i].rating);
    }
  });

  test('returns empty array when budget too low', () => {
    const recs = generateRecommendations(['gaming'], 100);
    expect(recs).toEqual([]);
  });

```python
# tests/integration/test_api.py
import time
import json
import requests

BASE_URL = "http://localhost:3000/api"

def test_recommendations_endpoint_success():
    payload = {"needs": ["gaming", "budget"], "budget": 1200}
    resp = requests.post(f"{BASE_URL}/recommendations", json=payload)
    assert resp.status_code == 200
    data = resp.json()
    assert isinstance(data, list)
    assert len(data) > 0
    for rec in data:
        assert "id" in rec
        assert "name" in rec
        assert "price" in rec
        assert "specs" in rec
        assert "rating" in rec

def test_recommendations_sorted_by_rating():
    payload = {"needs": ["gaming"], "budget": 2000}
    resp = requests.post(f"{BASE_URL}/recommendations", json=payload)
    recs = resp.json()
    ratings = [r["rating"] for r in recs]
    assert ratings == sorted(ratings, reverse=True)

def test_recommendations_no_match():
    payload = {"needs": ["gaming"], "budget": 50}
    resp = requests.post(f"{BASE_URL}/recommendations", json=payload)
    assert resp.status_code == 200
    assert resp.json() == []

def test_realtime_update():
    # Simulate rapid budget changes
    budgets = [800, 900, 1000]
    times = []
    for b in budgets:
        start = time.time()
        requests.post(f"{BASE_URL}/recommendations", json={"needs": ["gaming"], "budget": b})
        times.append(time.time() - start)
    # All responses should be < 0.2s
    assert all(t < 0.2 for t in times)

def test_invalid_payload():
    resp = requests.post(f"{BASE_URL}/recommendations", json={"needs": "gaming", "budget": "cheap"})
    assert resp.status_code == 400
```
