  try {
    const requestInput = new RequestInput(req.body);
    await requestInput.validate();
    const request = await createRequest(requestInput);
    res.json({
      id: request.id,
      publicUrl: `https://example.com/public/requests/${request.id}`,
      status: 'submitted',
    });
  } catch (error) {
    res.status(400).json({ message: error.message });
  }
});

export default router;
