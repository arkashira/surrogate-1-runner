# /opt/axentx/surrogate-1/routes.py
from flask import Blueprint, jsonify, render_template
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime, timedelta
import os

db = SQLAlchemy()

# === DATA MODELS (if not already defined) ===
class Request(db.Model):
    __tablename__ = 'requests'
    id = db.Column(db.Integer, primary_key=True)
    team = db.Column(db.String(120), nullable=False, index=True)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

class RequestEvent(db.Model):
    __tablename__ = 'request_events'
    id = db.Column(db.Integer, primary_key=True)
    request_id = db.Column(db.Integer, db.ForeignKey('requests.id'), nullable=False, index=True)
    event_type = db.Column(db.String(40), nullable=False)  # 'status_update', 'view'
    created_at = db.Column(db.DateTime, default=datetime.utcnow)


# === METRICS BLUEPRINT ===
metrics_bp = Blueprint('metrics', __name__)

@metrics_bp.route('/metrics/teams', methods=['GET'])
def teams_wau():
    """
    Returns WAU (Weekly Active Tracked Requests) per team.
    Definition: Requests with at least one 'status_update' or 'view' event in the last 7 days.
    Output: JSON for automation.
    """
    week_ago = datetime.utcnow() - timedelta(days=7)

    # Subquery: request_ids with qualifying events in last 7 days
    from sqlalchemy import distinct
    active_request_ids = db.session.query(distinct(RequestEvent.request_id)).filter(
        RequestEvent.created_at >= week_ago,
        RequestEvent.event_type.in_(['status_update', 'view'])
    ).subquery()

    # Count per team
    from sqlalchemy import func
    results = db.session.query(
        Request.team,
        func.count(distinct(Request.id)).label('wau_requests')
    ).filter(
        Request.id.in_(active_request_ids)
    ).group_by(Request.team).all()

    teams = [{'team': r.team, 'wau_requests': r.wau_requests} for r in results]
    total = sum(t['wau_requests'] for t in teams)

    return jsonify({
        'window_days': 7,
        'as_of_utc': datetime.utcnow().isoformat() + 'Z',
        'total_wau': total,
        'teams': sorted(teams, key=lambda x: x['wau_requests'], reverse=True)
    })


@metrics_bp.route('/admin/metrics', methods=['GET'])
def admin_metrics():
    """
    Renders the admin dashboard for WAU metrics.
    """
    return render_template('metrics.html')
```
