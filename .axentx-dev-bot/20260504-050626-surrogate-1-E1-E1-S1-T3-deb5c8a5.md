import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useForm, SubmitHandler } from 'react-hook-form';
import { yupResolver } from '@hookform/resolvers/yup';
import * as yup from 'yup';
import axios from 'axios';

interface IFormInput {
  stage: string;
  targetMarket: string;
  pricingModel: string;
  currentFunnel: string;
  biggestGrowthPain: string;
  additionalInfo?: string;
  email: string;
  password: string;
}

const schema = yup.object().shape({
  stage: yup.string().required('Stage is required'),
  targetMarket: yup.string().required('Target market is required'),
  pricingModel: yup.string().required('Pricing model is required'),
  currentFunnel: yup.string().required('Current funnel is required'),
  biggestGrowthPain: yup.string().required('Biggest growth pain is required').max(500, 'Maximum 500 characters'),
  additionalInfo: yup.string().max(1000, 'Maximum 1000 characters'),

  password: yup.string().min(8, 'Password must be at least 8 characters').required('Password is required'),
});

const Questionnaire: React.FC = () => {
  const { register, handleSubmit, formState: { errors }, reset } = useForm<IFormInput>({
    resolver: yupResolver(schema),
  });
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [submitError, setSubmitError] = useState<string | null>(null);
  const navigate = useNavigate();

  const onSubmit: SubmitHandler<IFormInput> = async (data) => {
