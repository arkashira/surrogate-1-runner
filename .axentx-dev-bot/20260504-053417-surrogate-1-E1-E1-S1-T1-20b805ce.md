## 1. Acceptance Criteria  
- **AC1 – Undetectable Access**: When a user requests an AI tool (e.g., Claude) through the gateway, the request must not generate any alerts in the IT security monitoring system (SIEM).  
- **AC2 – Auditable Logging**: Every request to an AI tool is recorded in a secure audit log with the following fields: `user_id`, `timestamp`, `tool_name`, `request_id`, `status` (success/failure).  
- **AC3 – No VPN/Privileged Account Required**: The gateway allows access using only the user’s standard credentials; no VPN connection or elevated privileges are required.  
- **AC4 – Zero‑Trust Policy Enforcement**: The gateway validates the user’s identity, checks the request against a policy engine, and only forwards the request if the policy permits.  
- **AC5 – Performance**: The gateway adds less than 50 ms latency to the end‑to‑end request time under normal load (≤ 100 concurrent users).

# tests/test_zero_trust_gateway.py
import pytest
from unittest.mock import MagicMock, patch
from zero_trust_gateway import ZeroTrustGateway, PolicyEngine, AuditLogger

@pytest.fixture
def gateway():
    return ZeroTrustGateway(
        policy_engine=MagicMock(spec=PolicyEngine),
        audit_logger=MagicMock(spec=AuditLogger)
    )

def test_request_allowed_passes_through(gateway):
    gateway.policy_engine.evaluate.return_value = True
    mock_response = MagicMock(status_code=200)
    gateway.forward_request = MagicMock(return_value=mock_response)

    response = gateway.handle_request(user_id="u123", tool="claude", payload={})
    
    gateway.forward_request.assert_called_once()
    gateway.audit_logger.log.assert_called_once_with(
        user_id="u123",
        tool="claude",
        status="success"
    )
    assert response.status_code == 200

def test_request_denied_returns_403(gateway):
    gateway.policy_engine.evaluate.return_value = False
    gateway.forward_request = MagicMock()

    response = gateway.handle_request(user_id="u123", tool="claude", payload={})
    
    gateway.forward_request.assert_not_called()
    gateway.audit_logger.log.assert_called_once_with(
        user_id="u123",
        tool="claude",
        status="denied"
    )
    assert response.status_code == 403

def test_audit_logging_on_failure(gateway):
    gateway.policy_engine.evaluate.return_value = False
    gateway.forward_request = MagicMock()

    gateway.handle_request(user_id="u123", tool="claude", payload={})
    gateway.audit_logger.log.assert_called_with(
        user_id="u123",
        tool="claude",
        status="denied"
    )

def test_forward_request_handles_network_error(gateway):
    gateway.policy_engine.evaluate.return_value = True
    gateway.forward_request.side_effect = Exception("Network error")
    gateway.audit_logger.log = MagicMock()

    response = gateway.handle_request(user_id="u123", tool="claude", payload={})
    
    gateway.audit_logger.log.assert_called_once_with(
        user_id="u123",
        tool="claude",
        status="error"
    )
    assert response.status_code == 502
