import { promises as fs } from "fs";
import { join } from "path";

const LOG_DIR = join(process.cwd(), "logs");
const LOG_PATH = join(LOG_DIR, "jobRuns.jsonl");

export type JobRunStatus = "pending" | "running" | "completed" | "failed";

export interface JobRun {
  id: string;
  jobName: string;
  startedAt: string;
  finishedAt?: string;
  status: JobRunStatus;
  details?: Record<string, unknown>;
  createdAt: string;
}

async function ensureLogFile() {
  await fs.mkdir(LOG_DIR, { recursive: true });
  try {
    await fs.access(LOG_PATH);
  } catch {
    await fs.writeFile(LOG_PATH, "");
  }
}

/**
 * Append-only start record.
 * Uses crypto-random suffix to avoid collisions under high-frequency retries.
 */
export async function startRun(
  jobName: string,
  details: Record<string, unknown> = {}
): Promise<JobRun> {
  await ensureLogFile();
  const run: JobRun = {
    id: `${jobName}-${Date.now()}-${crypto.randomUUID().slice(0, 8)}`,
    jobName,
    startedAt: new Date().toISOString(),
    status: "running",
    details,
    createdAt: new Date().toISOString(),
  };
  await fs.appendFile(LOG_PATH, JSON.stringify(run) + "\n");
  return run;
}

/**
 * Rewrite log file to update terminal fields for a run.
 * For higher throughput, swap to a DB or append-only + compaction.
 */
export async function updateRun(
  id: string,
  update: Partial<Pick<JobRun, "status" | "finishedAt" | "details">>
): Promise<void> {
  const content = await fs.readFile(LOG_PATH, "utf8").catch(() => "");
  const lines = content.trim().split("\n").filter(Boolean);
  const updated = lines.map((line) => {
    const run = JSON.parse(line) as JobRun;
    return run.id === id ? JSON.stringify({ ...run, ...update }) : line;
  });
  await fs.writeFile(LOG_PATH, updated.join("\n") + "\n");
}

export async function completeRun(
  id: string,
  details: Record<string, unknown> = {}
): Promise<void> {
  await updateRun(id, {
    status: "completed",
    finishedAt: new Date().toISOString(),
    details,
  });
}

export async function failRun(
  id: string,
  error: unknown,
  details: Record<string, unknown> = {}
): Promise<void> {
  await updateRun(id, {
    status: "failed",
    finishedAt: new Date().toISOString(),
    details: { error: String(error), ...details },
  });
}

/**
 * Find most recent successful run for idempotency checks.
 */
export async function findLatestSuccess(jobName: string): Promise<JobRun | null> {
  try {
    const content = await fs.readFile(LOG_PATH, "utf8").catch(() => "");
    const runs = content
      .trim()
      .split("\n")
      .filter(Boolean)
      .map((line) => JSON.parse(line) as JobRun)
      .filter((r) => r.jobName === jobName && r.status === "completed" && r.finishedAt)
      .sort((a, b) => new Date(b.finishedAt!).getTime() - new Date(a.finishedAt!).getTime());
    return runs[0] || null;
  } catch {
    return null;
  }
}
