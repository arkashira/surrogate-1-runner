import { JobRunLog } from '../types';

// Lightweight in-memory store for runner context; replace with DB client in real system.
const RUNS: JobRunLog[] = [];

function now(): Date {
  return new Date();
}

function isoDate(d: Date): string {
  return d.toISOString().split('T')[0];
}

function sha256Hex(value: string): string {
  // Node built-in crypto; no external deps.
  const crypto = require('crypto');
  return crypto.createHash('sha256').update(value).digest('hex');
}

export class JobRunRepository {
  /**
   * Determine if a successful run already exists for the given job and date.
   * Uses checksum = sha256(jobName:YYYY-MM-DD) for idempotency key.
   */
  async hasSuccessfulRunForDate(jobName: string, date: Date): Promise<boolean> {
    const checksum = sha256Hex(`${jobName}:${isoDate(date)}`);
    return RUNS.some((r) => r.checksum === checksum && r.status === 'completed');
  }

  async create(run: Omit<JobRunLog, 'id' | 'createdAt'>): Promise<JobRunLog> {
    const record: JobRunLog = {
      ...run,
      id: `jr_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`,
      createdAt: now(),
    };
    RUNS.push(record);
    return record;
  }
