import { PrismaClient, JobRun as PrismaJobRun } from '@prisma/client';

const prisma = new PrismaClient();

export type JobRunStatus = 'running' | 'completed' | 'failed';

export interface JobRun {
  id: string;
  jobName: string;
  runDate: string;      // YYYY-MM-DD in requester timezone
  startedAt: Date;
  finishedAt: Date | null;
  status: JobRunStatus;
  metadata: Record<string, unknown>;
  createdAt: Date;
}

export interface JobRunStore {
  create(jobRun: Omit<JobRun, 'id' | 'createdAt'>): Promise<JobRun>;
  update(id: string, fields: Partial<Pick<JobRun, 'finishedAt' | 'status' | 'metadata'>>): Promise<JobRun>;
  findByJobAndDate(jobName: string, runDate: string): Promise<JobRun | null>;
}

// Prisma implementation (recommended)
export class PrismaJobRunStore implements JobRunStore {
  async create(input: Omit<JobRun, 'id' | 'createdAt'>): Promise<JobRun> {
    const record = await prisma.jobRun.create({
      data: {
        id: crypto.randomUUID(),
        jobName: input.jobName,
        runDate: input.runDate,
        startedAt: input.startedAt,
        finishedAt: input.finishedAt,
        status: input.status,
        metadata: input.metadata,
      },
    });
    return toJobRun(record);
  }

  async update(id: string, fields: Parameters<JobRunStore['update']>[1]): Promise<JobRun> {
    const record = await prisma.jobRun.update({
      where: { id },
      data: {
        finishedAt: fields.finishedAt,
        status: fields.status,
        metadata: fields.metadata,
      },
    });
    return toJobRun(record);
  }

  async findByJobAndDate(jobName: string, runDate: string): Promise<JobRun | null> {
    const record = await prisma.jobRun.findUnique({
      where: { jobName_runDate: { jobName, runDate } },
    });
    return record ? toJobRun(record) : null;
  }
}

function toJobRun(r: PrismaJobRun): JobRun {
  return {
    id: r.id,
    jobName: r.jobName,
    runDate: r.runDate,
    startedAt: r.startedAt,
    finishedAt: r.finishedAt,
    status: r.status as JobRunStatus,
    metadata: r.metadata as Record<string, unknown>,
    createdAt: r.createdAt,
  };
}

export const jobRunStore = new PrismaJobRunStore();
