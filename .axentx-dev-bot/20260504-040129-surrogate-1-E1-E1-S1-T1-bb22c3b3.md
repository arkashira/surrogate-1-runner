describe('WorkflowEngine', () => {
  describe('executeWorkflow', () => {
    test('executes workflow with 5 agents under 200ms', async () => {
      const engine = new WorkflowEngine();
      const workflow = createTestWorkflow(5);
      
      const startTime = performance.now();
      const result = await engine.executeWorkflow(workflow);
      const endTime = performance.now();
      
      expect(endTime - startTime).toBeLessThan(200);
      expect(result).toBeDefined();
    });

    test('returns deterministic results for same inputs', async () => {
      const engine = new WorkflowEngine();
      const workflow = createTestWorkflow(3);
      
      const results = [];
      for (let i = 0; i < 1000; i++) {
        const result = await engine.executeWorkflow(workflow);
        results.push(JSON.stringify(result));
      }
      
      const uniqueResults = new Set(results);
      expect(uniqueResults.size).toBe(1);
    });

    test('handles agent failures gracefully', async () => {
      const engine = new WorkflowEngine();
      const workflow = createFailingWorkflow();
      
      await expect(engine.executeWorkflow(workflow)).rejects.toThrow();
    });
  });

  describe('monitoring', () => {
    test('exposes real-time metrics within 10ms', async () => {
      const engine = new WorkflowEngine();
      const workflow = createTestWorkflow(2);
      
      const startTime = performance.now();
      await engine.executeWorkflow(workflow);
      const endTime = performance.now();
      
      const metrics = engine.getMetrics();
      expect(endTime - startTime).toBeLessThan(200);
      expect(metrics.executionTime).toBeLessThan(10);
    });
  });

  describe('state management', () => {
    test('supports rollback to previous state', async () => {
      const engine = new WorkflowEngine();
      const workflow = createTestWorkflow(2);
      
      const initialResult = await engine.executeWorkflow(workflow);
      const stateSnapshot = engine.saveState();
      
      // Simulate failure
      await engine.executeWorkflow(createFailingWorkflow());
      
      const rollbackResult = await engine.rollbackToState(stateSnapshot);
      expect(rollbackResult).toEqual(initialResult);
    });
  });
});

function createTestWorkflow(agentCount: number) {
  return {
    agents: Array.from({ length: agentCount }, (_, i) => ({
      id: `agent-${i}`,
      type: 'llm',
      config: { model: 'gpt-4', temperature: 0.7 }
    })),
    steps: [
      { agentId: 'agent-0', input: 'Hello' },
      { agentId: 'agent-1', input: 'World' }
    ]
  };
}

function createFailingWorkflow() {
  return {
    agents: [{ id: 'failing-agent', type: 'llm', config: {} }],
    steps: [{ agentId: 'failing-agent', input: 'invalid' }]
  };
}
