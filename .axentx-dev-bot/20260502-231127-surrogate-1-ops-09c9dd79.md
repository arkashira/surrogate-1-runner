repo_id, date_folder, out_path = sys.argv[1], sys.argv[2], sys.argv[3]
api = HfApi()

# Non-recursive listing for the date folder
entries = api.list_repo_tree(repo_id, path=date_folder, recursive=False)
files = sorted([e.path for e in entries if not e.path.endswith("/")])

snapshot = {
    "date": date_folder,
    "files": files,
    "snapshot_ts": int(time.time())
}

os.makedirs(os.path.dirname(out_path), exist_ok=True)
with open(out_path, "w") as f:
    json.dump(snapshot, f, indent=2)

print(f"Snapshot saved to {out_path} ({len(files)} files)")
PY

cdn_fetch() {
  local url=$1
  local out=$2
  local max_retries=5
  local attempt=0
  local wait=2

  while (( attempt < max_retries )); do
    # No Authorization header; use CDN resolve URL
    if curl -L --fail --retry 3 --retry-delay 1 -s -o "$out" "$url"; then
      return 0
    fi

      echo "ERROR: CDN fetch failed after ${max_retries} attempts: $url" >&2
      return 1

    sleep "$wait"
    wait=$(( wait * 2 ))
  done
