## 1. Acceptance Criteria  
- **Non‑zero exit**: The pre‑commit hook returns an exit code > 0 whenever any PEP8, MarkdownLint, or JSON syntax violation is detected.  
- **Line‑number reporting**: Each violation message includes the exact file name and line number where the issue occurs.  
- **File‑pattern exclusion**: Files or directories specified in `.pre‑commit‑config.yaml` under `exclude` are ignored by the hook.  
- **Multi‑format support**: The hook correctly validates Python (`.py`), Markdown (`.md`), and JSON (`.json`) files.  
- **Test isolation**: Each test creates a fresh temporary repository tree (`tmp_path`) so that tests do not interfere with one another.  
- **Deterministic output**: Test results are repeatable and do not depend on external state (e.g., environment variables, network).

// tests/unit/hook_runner.test.js
const { runHook, parseOutput } = require('../../src/hook_runner');

describe('runHook', () => {
  test('returns non‑zero exit when Python PEP8 violation exists', async () => {
    const result = await runHook(['bad_pep8.py']);
    expect(result.exitCode).toBeGreaterThan(0);
    expect(parseOutput(result.stdout)).toContain('E302 expected 2 blank lines, found 1');
  });

  test('returns zero exit when no violations', async () => {
    const result = await runHook(['clean.py']);
    expect(result.exitCode).toBe(0);
  });

  test('parses MarkdownLint errors with line numbers', async () => {
    const result = await runHook(['bad_markdown.md']);
    const errors = parseOutput(result.stdout);
    expect(errors).toMatch(/bad_markdown\.md:3:1: MD001/);
  });

  test('parses JSON syntax errors', async () => {
    const result = await runHook(['bad.json']);
    const errors = parseOutput(result.stdout);
    expect(errors).toMatch(/bad\.json:4:12: Unexpected token/);
  });
});
