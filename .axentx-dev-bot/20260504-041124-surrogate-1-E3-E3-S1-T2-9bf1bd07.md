### 3. Idempotent cron‑registration script (`bin/register_cron_job.py`)

```python
#!/usr/bin/env python3
"""
register_cron_job.py
--------------------
Reads config/cron_jobs.yaml and ensures the current user's crontab
contains exactly the jobs marked `enabled: true`.  The script is safe
to run repeatedly (CI/CD, provisioning, manual execution) because it
updates only when a change is detected.
"""

import os
import sys
import yaml
from crontab import CronTab
from pathlib import Path

CONFIG_PATH = Path(__file__).resolve().parents[2] / "config" / "cron_jobs.yaml"


def load_jobs():
    with open(CONFIG_PATH, "r") as f:
        data = yaml.safe_load(f)
    return [job for job in data.get("jobs", []) if job.get("enabled", False)]


def normalize_job(job):
    """Return a deterministic string that uniquely identifies a job."""
    return f"{job['schedule']} {job['command']}"


def register():
    user = os.getenv("USER")
    if not user:
        sys.exit("ERROR: $USER not set – cannot determine crontab owner.")

        key = f"{existing.slices} {existing.command}"
        if key not in desired:
