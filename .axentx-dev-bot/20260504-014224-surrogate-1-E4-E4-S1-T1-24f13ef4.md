 ```typescript
// src/views/SuggestionsView.tsx

import React, { useState, useEffect } from 'react';
import { useEditor } from '@codemuse/editor';

export const SuggestionsView = () => {
  const { editorView } = useEditor();
  const [suggestions, setSuggestions] = useState([]);

  useEffect(() => {
    // Fetch AI suggestions from server
    fetch('https://api.codemuse.ai/suggestions')
      .then((res) => res.json())
      .then((data) => setSuggestions(data));
  }, []);

  const handleSelect = (suggestion) => {
    const { start, end } = editorView.state.selection;
    const newCode = editorView.state.doc.text.slice(0, start) + suggestion + editorView.state.doc.text.slice(end);
    editorView.dispatch({ changes: { from: start, to: start + suggestion.length, insert: suggestion } });
  };

  return (
    <div>
      <h2>AI Suggestions</h2>
      <ul>
        {suggestions.map((suggestion, index) => (
          <li key={index} onClick={() => handleSelect(suggestion)}>
            {suggestion.slice(0, 3)} ...
          </li>
        ))}
      </ul>
    </div>
  );
};
```

```typescript
// src/hooks/useEditor.ts

import { Editor, EditorState, RichUtils, Transforms } from 'draft-js';
import { EditorView } from '@codemuse/editor';

export const useEditor = () => {
  const editorViewRef = React.useRef<EditorView | null>(null);

  return {
    editorView: editorViewRef.current,
    setEditorView: (view: EditorView) => editorViewRef.current = view,
    editorState: editorViewRef.current?.state,
    setEditorState: (state: EditorState) => editorViewRef.current?.onChange(state),
  };
};
```

```javascript
// SuggestionsView.test.js
import React from 'react';
import { render, screen, fireEvent } from '@testing-library/react';
import SuggestionsView from './SuggestionsView';

describe('SuggestionsView', () => {
  const mockSuggestions = [
    { id: 1, preview: 'Line 1\nLine 2\nLine 3', fullCode: 'Full Code 1' },
    { id: 2, preview: 'Line 4\nLine 5\nLine 6', fullCode: 'Full Code 2' },
  ];

  const mockUseEditor = {
    getEditorView: jest.fn(),
    setEditorContent: jest.fn(),
  };

  beforeEach(() => {
    jest.clearAllMocks();
  });

  it('renders a scrollable list of suggestions', () => {
    render(<SuggestionsView suggestions={mockSuggestions} useEditor={mockUseEditor} />);
    const suggestionList = screen.getByRole('list');
    expect(suggestionList).toBeInTheDocument();
    expect(suggestionList).toHaveStyle('overflow-y: auto');
  });

  it('displays each suggestion with a preview of the first 3 lines', () => {
    render(<SuggestionsView suggestions={mockSuggestions} useEditor={mockUseEditor} />);
    mockSuggestions.forEach((suggestion) => {
      expect(screen.getByText(suggestion.preview)).toBeInTheDocument();
    });
  });

  it('inserts full code at cursor position when a suggestion is clicked', () => {
    render(<SuggestionsView suggestions={mockSuggestions} useEditor={mockUseEditor} />);
    const suggestionItem = screen.getByText(mockSuggestions[0].preview);
    fireEvent.click(suggestionItem);
    expect(mockUseEditor.setEditorContent).toHaveBeenCalledWith(mockSuggestions[0].fullCode);
  });

  it('inserts selected suggestion when Ctrl+Enter is pressed', () => {
    render(<SuggestionsView suggestions={mockSuggestions} useEditor={mockUseEditor} />);
    const suggestionItem = screen.getByText(mockSuggestions[0].preview);
    fireEvent.keyDown(suggestionItem, { key: 'Enter', ctrlKey: true });
    expect(mockUseEditor.setEditorContent).toHaveBeenCalledWith(mockSuggestions[0].fullCode);
  });
});
```
