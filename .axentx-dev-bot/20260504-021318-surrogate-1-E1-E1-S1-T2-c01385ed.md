import boto3
from botocore.exceptions import ClientError, NoCredentialsError, EndpointConnectionError
from typing import Optional

class AWSClient:
    def __init__(self, role_arn: str, session_name: str = "surrogate-1-session", external_id: Optional[str] = None):
        self.role_arn = role_arn
        self.session_name = session_name
        self.external_id = external_id
        self.credentials = self._assume_role()

    def _assume_role(self) -> Optional[dict]:
        sts = boto3.client("sts")
        try:
            response = sts.assume_role(
                RoleArn=self.role_arn,
                RoleSessionName=self.session_name,
                ExternalId=self.external_id
            )
            return response["Credentials"]
        except (ClientError, NoCredentialsError, EndpointConnectionError) as e:
            raise RuntimeError(f"Failed to assume role {self.role_arn}: {e}") from e

    def get_client(self, service_name: str, region_name: Optional[str] = None):
        if not self.credentials:
            raise RuntimeError("No credentials available for client creation.")
        return boto3.client(
            service_name,
            region_name=region_name,
            aws_access_key_id=self.credentials["AccessKeyId"],
            aws_secret_access_key=self.credentials["SecretAccessKey"],
            aws_session_token=self.credentials["SessionToken"],
        )
```

import boto3
from botocore.exceptions import ClientError

REQUIRED_PERMISSIONS = [
    "sts:GetCallerIdentity",
    "ec2:DescribeInstances",
    "s3:ListAllMyBuckets",
]

class AWSValidator:
    def __init__(self, role_arn: str, external_id: str):

        self.external_id = external_id
        self.client = AWSClient(role_arn, external_id=external_id)

    def _has_permission(self, action: str, service: str) -> bool:
        try:
            if service == "sts":
                self.client.get_client("sts").get_caller_identity()
            elif service == "ec2":
                self.client.get_client("ec2").describe_instances(InstanceIds=['i-1234567890abcdef0'], DryRun=True)
            elif service == "s3":
                self.client.get_client("s3").list_buckets()
            else:
                return False
            return True
        except ClientError as e:
            error_code = e.response["Error"]["Code"]
            if error_code in ("AccessDenied", "UnauthorizedOperation"):
                return False
            return True

    def validate(self) -> List[str]:
        missing = [action for action in REQUIRED_PERMISSIONS if not self._has_permission(action.split(":")[0], action.split(":")[1])]
        return missing

    def is_valid(self) -> bool:
        return len(self.validate()) == 0
