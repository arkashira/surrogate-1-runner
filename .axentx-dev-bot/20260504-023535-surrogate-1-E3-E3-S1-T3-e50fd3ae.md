"""
CI/CD Status Reporter for Surrogate-1 Validator

Provides standardized pass/fail status reporting compatible with:
- GitHub Actions
- GitLab CI
- Jenkins

Exit codes:
  0 = PASS
  1 = FAIL
  2 = ERROR
"""

import json
import sys
import os
from typing import Optional, Dict, Any, List
from dataclasses import dataclass, asdict
from datetime import datetime


@dataclass
class ValidationResult:
    """Represents a single validation result."""
    name: str
    passed: bool
    message: str
    details: Optional[Dict[str, Any]] = None


@dataclass
class ValidationResultReport:
    """Complete validation report for CI/CD integration."""
    timestamp: str
    overall_status: str  # "PASS", "FAIL", "ERROR"
    total_checks: int
    passed_checks: int
    failed_checks: int
    results: List[Dict[str, Any]]
    metadata: Optional[Dict[str, str]] = None

    def to_dict(self) -> Dict[str, Any]:
        return asdict(self)

    def to_json(self, indent: int = 2) -> str:
        return json.dumps(self.to_dict(), indent=indent)


class StatusReporter:
    """
    CI/CD status reporter that provides standardized output formats.
    
    Supports:
    - Exit codes for pipeline gating
    - JSON output for parsing
    - JUnit XML for Jenkins/GitLab
    - Plain text for human readability
    """

    def __init__(self, metadata: Optional[Dict[str, str]] = None):
        self.metadata = metadata or {}
        self.results: List[ValidationResult] = []

    def add_result(self, name: str, passed: bool, message: str,
                   details: Optional[Dict[str, Any]] = None) -> None:
        """Add a validation result to the report."""
        self.results.append(ValidationResult(
            name=name,
            passed=passed,
            message=message,
            details=details
        ))

    def _determine_overall_status(self) -> str:
        """Determine overall status based on individual results."""
        if not self.results:
            return "PASS"

        failed = [r for r in self.results if not r.passed]
        if failed:
            return "FAIL"
        return "PASS"

    def generate_report(self) -> ValidationResultReport:
        """Generate the complete validation report."""
        return ValidationResultReport(
            timestamp=datetime.utcnow().isoformat() + "Z",
            overall_status=self._determine_overall_status(),
            total_checks=len(self.results),
            passed_checks=sum(1 for r in self.results if r.passed),
            failed_checks=sum(1 for r in self.results if not r.passed),
            results=[asdict(r) for r in self.results],
            metadata=self.metadata
        )

    def output_json(self) -> None:
        """Output report as JSON (parseable by CI/CD tools)."""
        report = self.generate_report()
        print(report.to_json())

    def output_junit_xml(self) -> None:
