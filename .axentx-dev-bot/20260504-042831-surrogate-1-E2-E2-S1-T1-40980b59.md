  <script>
    // Configuration – adjust endpoint or polling interval as needed
    const METRICS_ENDPOINT = '/metrics'; // Expected to return JSON
    const POLL_INTERVAL_MS = 5000; // 5 seconds

    async function fetchMetrics() {
      try {
        const resp = await fetch(METRICS_ENDPOINT, { cache: 'no-store' });
        if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
        const data = await resp.json();

        // Expected shape:
        // { latency_ms: number, cost_usd: number, error_rate: number }
        updateCard('latency', data.latency_ms?.toFixed(2) ?? '—');
        updateCard('cost', data.cost_usd?.toFixed(4) ?? '—');
        updateCard('error', data.error_rate?.toFixed(2) ?? '—');
      } catch (e) {
        console.error('Failed to fetch metrics:', e);
        // Show error state
        setErrorState('latency');
        setErrorState('cost');
        setErrorState('error');
      }
    }

    function updateCard(key, value) {
      const valueEl = document.getElementById(`${key}-value`);
      if (valueEl) {
        valueEl.textContent = value;
        valueEl.classList.remove('loading', 'error');
      }
    }

    function setErrorState(key) {
      const valueEl = document.getElementById(`${key}-value`);
      if (valueEl) {
        valueEl.textContent = 'ERR';
        valueEl.classList.add('error');
        valueEl.classList.remove('loading');
      }
    }

    // Initial fetch + periodic polling
    fetchMetrics();
    setInterval(fetchMetrics, POLL_INTERVAL_MS);
  </script>
</body>
</html>
