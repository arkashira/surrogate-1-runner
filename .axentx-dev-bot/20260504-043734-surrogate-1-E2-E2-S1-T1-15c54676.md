"""
Document embedding and vector store implementation.

This module provides a simple vector store backed by FAISS and
sentence-transformer embeddings. It supports:

* Adding documents (text + optional id) to the store.
* Querying the store for the top‑k most similar documents.
* Persisting the index to disk (optional).
"""

import os
import json
from pathlib import Path
from typing import Iterable, List, Optional, Tuple

import faiss
import numpy as np
from sentence_transformers import SentenceTransformer


class VectorStore:
    """
    A simple FAISS vector store for text embeddings.

    Parameters
    ----------
    model_name : str
        Name of the sentence transformer model to use for embeddings.
    dim : int, optional
        Dimensionality of the embeddings. If None, inferred from the model.
    nlist : int, optional
        Number of clusters for IVF index. If None, uses an exact flat index.
    index_path : str | Path, optional
        Path to load/save the FAISS index. If None, the index is kept in memory.
    """

    def __init__(
        self,
        model_name: str = "all-MiniLM-L6-v2",
        dim: Optional[int] = None,
        nlist: Optional[int] = None,
        index_path: Optional[Path] = None,
    ):
        self.model = SentenceTransformer(model_name)
        self.dim = dim or self.model.get_sentence_embedding_dimension()
        self.nlist = nlist
        self.index_path = Path(index_path) if index_path else None

        # Build index
        if self.nlist is None:
            self.index = faiss.IndexFlatL2(self.dim)
        else:
            quantizer = faiss.IndexFlatL2(self.dim)
            self.index = faiss.IndexIVFFlat(quantizer, self.dim, self.nlist, faiss.METRIC_L2)
            # Train with random vectors if index is empty
            if not self.index.is_trained:
                dummy = np.random.random((self.nlist, self.dim)).astype("float32")
                self.index.train(dummy)

        # Mapping from internal id to original id
        self.id_map: dict[int, str] = {}
        self.next_id = 0

        if self.index_path and self.index_path.exists():
            self.load(self.index_path)

    # ------------------------------------------------------------------
    # Embedding helpers
    # ------------------------------------------------------------------
    def _embed(self, texts: Iterable[str]) -> np.ndarray:
        """Return embeddings for a list of texts."""
        return self.model.encode(list(texts), convert_to_numpy=True, normalize_embeddings=True)

    # ------------------------------------------------------------------
    # CRUD operations
    # ------------------------------------------------------------------
    def add_documents(
        self,
        texts: Iterable[str],
        ids: Optional[Iterable[str]] = None,
    ) -> List[str]:
        """
        Add documents to the vector store.

        Parameters

        texts : Iterable[str]
            Text content of the documents.
        ids : Optional[Iterable[str]]
            Optional unique identifiers for each document. If omitted,
            integer ids are generated.

        Returns
        -------
        List[str]
            The ids assigned to the added documents.

                raise ValueError("Length of ids must match number of texts")

        for ext_id, int_id in zip(external_ids, internal_ids):
