// src/main/java/com/axentx/freerouter/KiCadFreerouterPlugin.java
package com.axentx.freerouter;

import java.awt.Desktop;
import java.io.*;
import java.nio.file.*;
import java.util.concurrent.TimeUnit;
import javax.swing.*;

/**
 * KiCAD 9 Action Plugin for Freerouter integration.
 * Handles DSN export, Freerouter launch, and DSN import workflow.
 */
public class KiCadFreerouterPlugin {

    private static final String PLUGIN_NAME = "Freerouter Integration";
    private static final String PLUGIN_VERSION = "1.0.0";
    private static final String FREEROUTER_VERSION = "1.8.0";
    private static final String FREEROUTER_URL = 
        "https://github.com/freerouter/freerouter/releases/download/v" + 
        FREEROUTER_VERSION + "/freerouter-" + FREEROUTER_VERSION + ".jar";

    private Path pluginDataDir;
    private Path freerouterJar;
    private Path currentDsnFile;

    public KiCadFreerouterPlugin() {
        initializePaths();
    }

    private void initializePaths() {
        try {
            String userHome = System.getProperty("user.home");
            pluginDataDir = Paths.get(userHome, ".config", "kicad", "freerouter-plugin");
            freerouterJar = pluginDataDir.resolve("freerouter-" + FREEROUTER_VERSION + ".jar");
            Files.createDirectories(pluginDataDir);
        } catch (IOException e) {
            showError("Initialization Error", "Failed to create plugin directories: " + e.getMessage());
        }
    }

    /**
     * Main entry point - called from KiCAD menu
     */
    public void run() {
        SwingUtilities.invokeLater(this::executePlugin);
    }

    private void executePlugin() {
        try {
            // Step 1: Ensure Freerouter is available
            if (!ensureFreerouterAvailable()) {
                return;
            }

            // Step 2: Get current board file from KiCAD
            String boardPath = getActiveBoardPath();
            if (boardPath == null || boardPath.isEmpty()) {
                showError("No Board Open", "Please open a PCB file in KiCAD first.");
                return;
            }

            // Step 3: Export DSN from KiCAD (user must do this manually or we guide them)
            String dsnPath = exportDsnFile(boardPath);
            if (dsnPath == null) {
                showUserInstructions("To export to Freerouter:\n" +
                    "1. In KiCAD: File → Export → Export Specctra DSN\n" +
                    "2. Save as: " + pluginDataDir.resolve("board.dsn") + "\n" +
                    "3. Click OK here to continue");
                dsnPath = pluginDataDir.resolve("board.dsn").toString();
            }

            // Step 4: Launch Freerouter with the DSN file
            launchFreerouter(dsnPath);

            // Step 5: Prompt for import after routing
            int result = JOptionPane.showConfirmDialog(null,
                "After routing in Freerouter:\n" +
                "1. In Freerouter: File → Export Specctra DSN\n" +
                "2. Save the file back to the original location\n" +
                "3. Return here and click 'Import'\n\n" +
                "Click 'Yes' to import the routed board now.",
                "Import Routed Board",
                JOptionPane.YES_NO_OPTION);
