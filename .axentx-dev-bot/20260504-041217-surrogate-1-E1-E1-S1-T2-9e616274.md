def analyze_project_structure(base_path="/opt/axentx/surrogate-1"):
    """
    Comprehensive analysis of project structure with actionable insights
    """
    # Verify base path exists
    if not os.path.exists(base_path):
        print(f"ERROR: Base path {base_path} does not exist")
        return

    # Initialize analysis results
    analysis = {
        "structure": {},
        "files": {
            "python": [],
            "config": [],
            "tests": [],
            "models": []
        },
        "missing": []
    }

    # 1. Analyze directory structure recursively
    print("\n=== Project Structure Analysis ===")
    for root, dirs, files in os.walk(base_path):
        rel_path = os.path.relpath(root, base_path)
        analysis["structure"][rel_path] = {
            "directories": sorted(dirs),
            "files": sorted(files)
        }

        # Print structure with indentation
        depth = rel_path.count(os.sep)
        print(f"{'  ' * depth}{os.path.basename(root)}/")

        # Collect file types
        for file in files:
            full_path = os.path.join(root, file)
            if file.endswith('.py'):
                analysis["files"]["python"].append(full_path)
            elif file.endswith(('.json', '.yaml', '.yml')):
                analysis["files"]["config"].append(full_path)
            elif 'test' in file.lower() or 'tests' in rel_path.lower():
                analysis["files"]["tests"].append(full_path)
            elif 'model' in file.lower() or 'schema' in file.lower():
                analysis["files"]["models"].append(full_path)

    # 2. Check for critical directories
    critical_dirs = [
        "src",
        "tests",
        "config",
        "models",
        "data"
    ]

    print("\n=== Critical Directory Check ===")
    for dir_name in critical_dirs:
        dir_path = os.path.join(base_path, dir_name)
        if os.path.exists(dir_path):
            print(f"✓ EXISTS: {dir_path}")
            analysis["structure"][dir_name] = {
                "exists": True,
                "contents": sorted(os.listdir(dir_path))
            }
        else:
            print(f"✗ MISSING: {dir_path}")
            analysis["missing"].append(dir_name)

    # 3. Summarize findings
    print("\n=== Summary ===")
    print(f"Total Python files: {len(analysis['files']['python'])}")
    print(f"Total config files: {len(analysis['files']['config'])}")
    print(f"Total test files: {len(analysis['files']['tests'])}")
    print(f"Total model files: {len(analysis['files']['models'])}")
    print(f"Missing critical directories: {len(analysis['missing'])}")

    # 4. Provide actionable recommendations
    print("\n=== Actionable Recommendations ===")
    if analysis["missing"]:
        print("Create missing critical directories:")
        for missing_dir in analysis["missing"]:
            print(f"- mkdir -p {os.path.join(base_path, missing_dir)}")

    if not analysis["files"]["tests"]:
        print("\nNo test files found. Recommendations:")
        print("- Create a 'tests' directory")
        print("- Add unit tests for existing functionality")
        print("- Implement CI/CD pipeline for testing")

    if not analysis["files"]["models"]:
        print("\nNo model/schema files found. Recommendations:")
        print("- Define data models in 'models' directory")
        print("- Consider using Pydantic or similar for schema validation")

    return analysis
