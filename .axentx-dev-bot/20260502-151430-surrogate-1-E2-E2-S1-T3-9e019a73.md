from __future__ import annotations

from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from enum import Enum
from typing import Any, Dict, List, Optional
import threading
import time


class ExecutionStatus(str, Enum):
    PENDING = "pending"
    FILLED = "filled"
    CANCELED = "canceled"
    REJECTED = "rejected"
    PARTIAL = "partial"


class ExecutionSide(str, Enum):
    BUY = "buy"
    SELL = "sell"


@dataclass
class ExecutionEvent:
    event_id: str
    connector: str
    action: str  # place_order, cancel_order, get_quote
    symbol: str
    side: Optional[ExecutionSide]
    order_id: Optional[str]
    client_order_id: Optional[str]
    status: ExecutionStatus
    price: Optional[float]
    size: Optional[float]
    filled_size: Optional[float]
    remaining_size: Optional[float]
    timestamp: datetime
    latency_ms: Optional[float]
    raw_response: Optional[Dict[str, Any]]
    error: Optional[str]

    def to_dict(self) -> Dict[str, Any]:
        d = asdict(self)
        d["timestamp"] = self.timestamp.isoformat()
        d["side"] = self.side.value if self.side else None
        d["status"] = self.status.value
        return d


class Journal:
    """Thread-safe execution journal for pipeline audit."""

    def __init__(self) -> None:
        self._lock = threading.RLock()
        self._events: List[ExecutionEvent] = []

    def record_execution(self, event: ExecutionEvent) -> None:
        with self._lock:
            self._events.append(event)

    def get_events(
        self,
        *,
        connector: Optional[str] = None,
        action: Optional[str] = None,
        symbol: Optional[str] = None,
        since: Optional[datetime] = None,
    ) -> List[Dict[str, Any]]:
        with self._lock:
            events = self._events
            if connector is not None:
                events = [e for e in events if e.connector == connector]
            if action is not None:
                events = [e for e in events if e.action == action]
            if symbol is not None:
                events = [e for e in events if e.symbol == symbol]
            if since is not None:
                events = [e for e in events if e.timestamp >= since]
            return [e.to_dict() for e in events]


# Global singleton used by connectors
journal = Journal()

def test_journal_record_execution_thread_safe():
    journal = Journal()
    n = 100
    barrier = threading.Barrier(n)
    def emit_one(i):
        barrier.wait()
        ev = ExecutionEvent(
            status="quoted", side="sell", symbol="TSLA", quantity=10,
            price=170.0 + i, ts_utc="2026-05-02T15:11:34.000Z", latency_ms=10 + i,
            connector="ws", action="get_quote", order_id=None, error=None
        )
        journal.record_execution(ev)
    threads = [threading.Thread(target=emit_one, args=(i,)) for i in range(n)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    assert len(journal.get_events()) == n
