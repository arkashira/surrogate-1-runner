def _tokenize(text: str, model: str = "gpt-3.5-turbo") -> List[int]:
    """
    Tokenize text using the specified model's tokenizer.
    Returns a list of token IDs.
    """
    try:
        encoding = tiktoken.get_encoding(model)
        return encoding.encode(text)
    except Exception:
        # Fallback to simple whitespace tokenization if tiktoken fails
        return text.split()

def _split_into_chunks(text: str, chunk_size: int, overlap: int, token_based: bool = True) -> List[str]:
    """
    Split text into chunks of size up to `chunk_size`.
    If `token_based` is True, uses token counts; otherwise, uses character counts.
    """
    if not text:
        return []

    tokens = _tokenize(text, "gpt-3.5-turbo") if token_based else text

    chunks = []
    current_chunk = []
    current_tokens = 0

    for token in tokens:
        current_chunk.append(token)
        current_tokens += 1

        if current_tokens > chunk_size:
            # Trim to fit within chunk size
            current_chunk = current_chunk[:chunk_size]
            current_tokens = chunk_size
