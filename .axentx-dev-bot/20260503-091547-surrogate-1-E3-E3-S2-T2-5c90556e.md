import os
import logging
import time
from datetime import timezone

import schedule

from src.jobs.daily_digest import run_daily_digest

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s — %(message)s",
    datefmt="%Y-%m-%dT%H:%M:%SZ",
)
logger = logging.getLogger("scheduler")


def run_and_log(job_func, name: str):
    try:
        logger.info("Starting scheduled job: %s", name)
        job_func()
        logger.info("Finished scheduled job: %s", name)
    except Exception as exc:
        logger.exception("Scheduled job %s failed: %s", name, exc)


def configure_jobs() -> None:
    # Run daily at 08:00 UTC
    schedule.every().day.at("08:00", "UTC").do(
        run_and_log, run_daily_digest, "daily_digest"
    )
    logger.info("Scheduler configured: daily_digest at 08:00 UTC")


def main() -> None:
    configure_jobs()
    logger.info("Scheduler started (PID=%s)", os.getpid())
    # Run immediately on startup for dev/testing (idempotent)
    run_and_log(run_daily_digest, "daily_digest (startup)")

    while True:
        schedule.run_pending()
        time.sleep(30)


if __name__ == "__main__":
    main()
