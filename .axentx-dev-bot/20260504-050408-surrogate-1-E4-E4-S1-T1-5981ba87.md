import os
import sqlite3
from datetime import datetime
from pathlib import Path
from typing import Iterable, Dict

from flask import (
    Blueprint,
    Flask,
    g,
    render_template,
    request,
    redirect,
    url_for,
    flash,
)

# ----------------------------------------------------------------------
# Configuration
# ----------------------------------------------------------------------
BASE_DIR = Path(__file__).resolve().parents[1]          # /opt/axentx/surrogate-1
DB_PATH = BASE_DIR / "analytics" / "performance.db"

# Ensure the analytics directory exists
DB_PATH.parent.mkdir(parents=True, exist_ok=True)

# ----------------------------------------------------------------------
# Database helpers
# ----------------------------------------------------------------------
def get_db() -> sqlite3.Connection:
    """Return a thread‑local SQLite connection."""
    if "db" not in g:
        g.db = sqlite3.connect(str(DB_PATH), check_same_thread=False)
        g.db.row_factory = sqlite3.Row
    return g.db


def close_db(exception=None) -> None:
    """Close the SQLite connection at request teardown."""
    db = g.pop("db", None)
    if db is not None:
        db.close()


def init_db() -> None:
    """Create the performance table if it does not exist."""
    with sqlite3.connect(str(DB_PATH)) as db:
        db.execute(
            """
            CREATE TABLE IF NOT EXISTS performance (
                id          INTEGER PRIMARY KEY AUTOINCREMENT,
                strategy    TEXT NOT NULL,
                date        TEXT NOT NULL,
                traffic     INTEGER NOT NULL,
                conversions INTEGER NOT NULL,
                mrr         REAL NOT NULL
            )
            """
        )
        db.commit()


# Initialise DB on import – safe for dev & production
init_db()

    url_prefix="/analytics",
)

# Request hooks

def before_request() -> None:
    """Ensure a DB connection is available for each request."""
    get_db()

def teardown_request(exception=None) -> None:
