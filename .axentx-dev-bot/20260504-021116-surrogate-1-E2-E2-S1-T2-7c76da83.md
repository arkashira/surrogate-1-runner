// services/spend_tracker.go
package services

import (
	"encoding/json"
	"errors"
	"os"
	"path/filepath"
	"sync"
	"time"
)

// SandboxSpend holds the spend data for a sandbox.
type SandboxSpend struct {
	SandboxID   string    `json:"sandbox_id"`
	Limit       float64   `json:"limit"`
	Spent       float64   `json:"spent"`
	UpdatedAt   time.Time `json:"updated_at"`
}

// SpendTracker manages spend limits and tracking per sandbox.
type SpendTracker struct {
	mu          sync.RWMutex
	sandboxes   map[string]*SandboxSpend
	storagePath string
}

// NewSpendTracker creates a new tracker with persistence at the given path.
// If the file does not exist, the tracker starts empty.
func NewSpendTracker(storagePath string) (*SpendTracker, error) {
	st := &SpendTracker{
		sandboxes:   make(map[string]*SandboxSpend),
		storagePath: storagePath,
	}
	if err := st.load(); err != nil {
		return nil, err
	}
	return st, nil
}

// load reads the JSON file into memory.
func (st *SpendTracker) load() error {
	st.mu.Lock()
	defer st.mu.Unlock()

	if _, err := os.Stat(st.storagePath); errors.Is(err, os.ErrNotExist) {
		// No file yet, nothing to load.
		return nil
	}
	data, err := os.ReadFile(st.storagePath)
	if err != nil {
		return err
	}
	var spends []*SandboxSpend
	if err := json.Unmarshal(data, &spends); err != nil {
		return err
	}
	for _, s := range spends {
		st.sandboxes[s.SandboxID] = s
	}
	return nil
}

// persist writes the current state to disk.
func (st *SpendTracker) persist() error {
	st.mu.RLock()
	defer st.mu.RUnlock()

	var spends []*SandboxSpend
	for _, s := range st.sandboxes {
		spends = append(spends, s)
	}
	data, err := json.MarshalIndent(spends, "", "  ")
	if err != nil {
		return err
	}
	dir := filepath.Dir(st.storagePath)
	if err := os.MkdirAll(dir, 0o755); err != nil {
		return err
	}
	return os.WriteFile(st.storagePath, data, 0o644)
}

// SetLimit sets or updates the spend limit for a sandbox.
// If the sandbox does not exist, it is created.
func (st *SpendTracker) SetLimit(sandboxID string, limit float64) error {
	st.mu.Lock()
	defer st.mu.Unlock()

	if sb, ok := st.sandboxes[sandboxID]; ok {
		sb.Limit = limit
		sb.UpdatedAt = time.Now()
	} else {
		st.sandboxes[sandboxID] = &SandboxSpend{
			SandboxID: sandboxID,
			Limit:     limit,
			Spent:     0,
			UpdatedAt: time.Now(),
		}
	}
	return st.persist()
}

	if !ok {
		return errors.New("sandbox not found")
	}
	sb.Spent += amount
	sb.UpdatedAt = time.Now()
	return st.persist()
}
