from __future__ import annotations

from datetime import datetime, timezone, timedelta
from typing import Any, Dict, Optional


def _ensure_utc(value: Any) -> datetime:
    """
    Normalize input to a timezone-aware UTC datetime.

    Accepts:
      - datetime (naive -> UTC, aware -> converted to UTC)
      - ISO 8601 string (with/without Z, with/without offset)
    Raises ValueError for unsupported types/values.
    """
    if isinstance(value, datetime):
        if value.tzinfo is None:
            return value.replace(tzinfo=timezone.utc)
        return value.astimezone(timezone.utc)

    if isinstance(value, str):
        v = value.strip()
        # Common suffix
        if v.endswith("Z"):
            v = v[:-1] + "+00:00"

        # Try fromisoformat-friendly formats first
        for fmt in (
            "%Y-%m-%dT%H:%M:%S.%f%z",
            "%Y-%m-%dT%H:%M:%S%z",
            "%Y-%m-%d %H:%M:%S.%f%z",
            "%Y-%m-%d %H:%M:%S%z",
        ):
            try:
                dt = datetime.strptime(v, fmt)
                return dt.astimezone(timezone.utc)
            except ValueError:
                continue

        # Fallback to fromisoformat (Python 3.7+)
        try:
            dt = datetime.fromisoformat(v)
            if dt.tzinfo is None:
                dt = dt.replace(tzinfo=timezone.utc)
            else:
                dt = dt.astimezone(timezone.utc)
            return dt
        except Exception as exc:
            raise ValueError(f"Cannot parse datetime from {value!r}") from exc

    raise ValueError(f"Unsupported datetime type: {type(value).__name__}")


def calculate_sla_status(
    target_seconds: float,
    last_transition_timestamp: Any,
    now: Optional[datetime] = None,
) -> Dict[str, Any]:
    """
    Calculate SLA status.

    Args:
        target_seconds: SLA target in seconds (e.g. 3600).
        last_transition_timestamp: UTC datetime or ISO string of last status transition.
        now: Optional datetime for deterministic testing.

    Returns:
        {
            "remaining_seconds": float,   # negative if breached
            "status": "ok" | "breached",
            "human": str,                 # e.g. "01:23:45" or "overdue"
            "breached_at_utc": str | None
        }
    """
    if target_seconds < 0:
        raise ValueError("target_seconds must be >= 0")

    utc_now = _ensure_utc(now) if now is not None else datetime.now(timezone.utc)
    transition = _ensure_utc(last_transition_timestamp)

    elapsed = (utc_now - transition).total_seconds()
    remaining = target_seconds - elapsed

    status = "ok" if remaining > 0 else "breached"
    if remaining > 0:
        human = str(timedelta(seconds=int(remaining)))
    else:
        human = "overdue"

    breached_at = None
    if remaining <= 0:
        breached_at = (transition + timedelta(seconds=target_seconds)).isoformat()

    return {
        "remaining_seconds": remaining,
        "status": status,
        "human": human,
        "breached_at_utc": breached_at,
    }
```
