#### sla.py (domain/sla.py)

```diff
--- a/domain/sla.py
+++ b/domain/sla.py
@@ -1,6 +1,11 @@
 from datetime import datetime, timedelta

 class SLA:
     def __init__(self, sla_target, last_status_transition_timestamp):
         self.sla_target = sla_target
         self.last_status_transition_timestamp = last_status_transition_timestamp

     def calculate_time_remaining(self):
         time_remaining = self.sla_target - self.last_status_transition_timestamp
         if time_remaining.total_seconds() <= 0:
             return "overdue"
         else:
             return time_remaining
```

#### public_view_service.py (services/public_view_service.py)

```diff

+++ b/services/public_view_service.py
@@ -1,6 +1,11 @@
 from flask import Blueprint
 from . import sla

 public_view = Blueprint('public_view', __name__)

 @public_view.route('/sla', methods=['GET'])
 def get_sla():
     sla_target = datetime.now() + timedelta(hours=1)
     last_status_transition_timestamp = datetime.now() - timedelta(minutes=30)
     sla_instance = sla.SLA(sla_target, last_status_transition_timestamp)
     return {'time_remaining': sla_instance.calculate_time_remaining()}
```
