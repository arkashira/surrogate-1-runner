// src/models/requestModel.ts
import { v4 as uuidv4 } from 'uuid';

export type RequestStatus = 'OPEN' | 'IN_PROGRESS' | 'RESOLVED' | 'CLOSED';

export interface AuditEntry {
  timestamp: string; // ISO
  by: string;
  action: string;
  details?: Record<string, unknown>;
}

export interface RequestRecord {
  id: string;
  title: string;
  description: string;
  owner: string;
  status: RequestStatus;
  createdAt: string;
  updatedAt: string;
  publicUrl: string;
  audit: AuditEntry[];
}

export function createRequestRecord(
  title: string,
  description: string,
  owner: string,
  initialStatus: RequestStatus,
  basePublicUrl: string
): RequestRecord {
  const id = uuidv4();
  const now = new Date().toISOString();
  const publicUrl = `${basePublicUrl.replace(/\/$/, '')}/public/requests/${id}`;
  const audit: AuditEntry[] = [
    {
      timestamp: now,
      by: owner,
      action: 'CREATED',
      details: { title, description, status: initialStatus },
    },
  ];

  return {
    id,
    title,
    description,
    owner,
    status: initialStatus,
    createdAt: now,
    updatedAt: now,
    publicUrl,
    audit,
  };
}

### 2) `src/services/requestService.ts`

```ts
// src/services/requestService.ts
import { RequestRecord, createRequestRecord, RequestStatus, AuditEntry } from '../models/requestModel';

const BASE_PUBLIC_URL = process.env.PUBLIC_BASE_URL || 'http://localhost:3000';
const store = new Map<string, RequestRecord>();

export function createRequest(
  title: string,
  description: string,
  owner: string,
  initialStatus: RequestStatus
): RequestRecord {
  if (!title?.trim() || !description?.trim() || !owner?.trim() || !initialStatus) {
    throw new Error('title, description, owner, and initialStatus are required');
  }

  const record = createRequestRecord(title.trim(), description.trim(), owner.trim(), initialStatus, BASE_PUBLIC_URL);
  store.set(record.id, record);
  return record;
}

export function getRequest(id: string): RequestRecord | undefined {
  return store.get(id);
}

export function updateRequestStatus(
  id: string,
  newStatus: RequestStatus,
  by: string,
  details?: Record<string, unknown>
): RequestRecord | undefined {
  const record = store.get(id);
  if (!record) return undefined;

  const previousStatus = record.status;
  const now = new Date().toISOString();

  record.status = newStatus;
  record.updatedAt = now;

  const auditEntry: AuditEntry = {
    timestamp: now,
    by,
    action: 'STATUS_UPDATED',
    details: { previousStatus, newStatus, ...details },
  };
  record.audit.push(auditEntry);

  store.set(id, record);
  return record;
}

// For tests/debug: clear store
export function _clearStore() {
  store.clear();
}
```
