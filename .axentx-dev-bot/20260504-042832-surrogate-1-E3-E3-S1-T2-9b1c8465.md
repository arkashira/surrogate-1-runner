import random
from typing import List

class Provider:
    def __init__(self, name: str):
        self.name = name
        self.available = True

    def process_request(self, request: str) -> str:
        if not self.available:
            raise Exception(f"Provider {self.name} is unavailable.")
        return f"Processed by {self.name}: {request}"

class Orchestrator:
    def __init__(self, providers: List[Provider]):
        self.providers = providers

    def _select_provider(self) -> Provider:
        available_providers = [p for p in self.providers if p.available]
        if not available_providers:
            raise Exception("No available providers.")
        return random.choice(available_providers)

    def process_request(self, request: str) -> str:
        try:
            provider = self._select_provider()
            return provider.process_request(request)
        except Exception as e:
            print(f"Error processing request: {e}")
            # Failover logic: Mark the current provider as unavailable and retry
            provider.available = False
            return self.process_request(request)

    def load_balance_requests(self, requests: List[str]) -> List[str]:
        results = []
        for request in requests:
            result = self.process_request(request)
            results.append(result)
        return results

# Example usage
if __name__ == "__main__":
    providers = [
        Provider("ProviderA"),
        Provider("ProviderB"),
        Provider("ProviderC")
    ]
    orchestrator = Orchestrator(providers)
    requests = ["Request1", "Request2", "Request3"]
    responses = orchestrator.load_balance_requests(requests)
    for response in responses:
        print(response)

import pytest
from unittest.mock import MagicMock, patch
from orchestrator import Orchestrator, Provider

@pytest.fixture
def providers():
    p1 = Provider(name="A", endpoint="http://a.com")
    p2 = Provider(name="B", endpoint="http://b.com")
    return [p1, p2]

def test_round_robin_distribution(providers):
    orch = Orchestrator(providers)
    # Mock successful responses
    for p in providers:
        p.send_request = MagicMock(return_value={"status": 200})
    # Send 4 requests
    responses = [orch.load_balance_requests({"data": i}) for i in range(4)]
    # Expect alternating providers: A, B, A, B
    assert [r["provider"] for r in responses] == ["A", "B", "A", "B"]

def test_failover_on_exception(providers):
    orch = Orchestrator(providers)
    # First provider fails, second succeeds
    providers[0].send_request.side_effect = Exception("down")
    providers[1].send_request.return_value = {"status": 200}
    resp = orch.load_balance_requests({"data": "test"})
    assert resp["provider"] == "B"
    assert providers[0].available is False

def test_failover_on_5xx(providers):
    orch = Orchestrator(providers)
    providers[0].send_request.return_value = {"status": 502}
    providers[1].send_request.return_value = {"status": 200}
    resp = orch.load_balance_requests({"data": "test"})
    assert resp["provider"] == "B"
    assert providers[0].available is False

def test_no_available_providers(providers):
    orch = Orchestrator(providers)
    for p in providers:
        p.send_request.side_effect = Exception("down")
    with pytest.raises(RuntimeError, match="No healthy providers"):
        orch.load_balance_requests({"data": "test"})

def test_get_status(providers):

    status = orch.get_status()
    assert status == {"A": True, "B": True}

    status = orch.get_status()
