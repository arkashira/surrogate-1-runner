# policy.py
from __future__ import annotations

from dataclasses import dataclass, asdict
from datetime import datetime, timezone
from enum import Enum
from typing import Dict, List, Optional, Any
import json
import os

class PolicyVersion(str, Enum):
    V1 = "v1"

class LimitType(str, Enum):
    MAX_NOTIONAL = "max_notional"
    MAX_SIZE = "max_size"
    PRICE_BAND = "price_band"

@dataclass(frozen=True)
class PriceBandRule:
    pct_above: float
    pct_below: float
    reference: str = "last"

@dataclass(frozen=True)
class RiskRule:
    rule_type: LimitType
    value: float
    band: Optional[PriceBandRule] = None

@dataclass(frozen=True)
class RiskPolicy:
    policy_id: str
    version: PolicyVersion
    pipeline_id: str
    enabled: bool
    rules: List[RiskRule]
    created_at: str
    updated_at: str

    def to_dict(self) -> Dict[str, Any]:
        d = asdict(self)
        d["rules"] = [asdict(r) for r in self.rules]
        return d

    @staticmethod
    def from_dict(data: Dict[str, Any]) -> "RiskPolicy":
        rules: List[RiskRule] = []
        for r in data.get("rules", []):
            band = None
            if r.get("band"):
                band = PriceBandRule(**r["band"])
            rules.append(
                RiskRule(
                    rule_type=LimitType(r["rule_type"]),
                    value=float(r["value"]),
                    band=band,
                )
            )
        return RiskPolicy(
            policy_id=data["policy_id"],
            version=PolicyVersion(data["version"]),
            pipeline_id=data["pipeline_id"],
            enabled=bool(data.get("enabled", True)),
            rules=rules,
            created_at=data.get("created_at", datetime.now(timezone.utc).isoformat()),
            updated_at=data.get("updated_at", datetime.now(timezone.utc).isoformat()),
        )

class PolicyStore:
    """
    File-based policy store.
    Policies: <base_path>/<pipeline_id>.json
    """
    def __init__(self, base_path: str = "./policies"):
        self.base_path = base_path
        os.makedirs(self.base_path, exist_ok=True)

    def get_policy(self, pipeline_id: str) -> Optional[RiskPolicy]:
        path = os.path.join(self.base_path, f"{pipeline_id}.json")
        if not os.path.exists(path):
            return None
        with open(path, "r", encoding="utf-8") as f:
            data = json.load(f)
        return RiskPolicy.from_dict(data)

    def save_policy(self, policy: RiskPolicy) -> None:
        path = os.path.join(self.base_path, f"{policy.pipeline_id}.json")
        with open(path, "w", encoding="utf-8") as f:
            json.dump(policy.to_dict(), f, indent=2)
