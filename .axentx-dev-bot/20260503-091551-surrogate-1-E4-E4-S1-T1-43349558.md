### db/schema.sql
```sql
-- Add materialized view for weekly active requests per team
-- WAU = distinct requests with at least one status_update or view in last 7 days
CREATE MATERIALIZED VIEW IF NOT EXISTS mv_weekly_active_requests AS
SELECT
  team_id,
  COUNT(DISTINCT request_id) AS wau_requests
FROM request_events
WHERE
  event_type IN ('status_update', 'view')
  AND occurred_at >= (now() AT TIME ZONE 'utc') - interval '7 days'
GROUP BY team_id;

-- Index for fast refresh/query
CREATE INDEX IF NOT EXISTS idx_mv_war_team ON mv_weekly_active_requests (team_id);

-- Optional: helper to refresh the view (call from API)
CREATE OR REPLACE FUNCTION refresh_mv_weekly_active_requests()
RETURNS void AS $$
BEGIN
  REFRESH MATERIALIZED VIEW CONCURRENTLY mv_weekly_active_requests;
EXCEPTION
  WHEN undefined_table THEN
    -- If CONCURRENTLY not possible (no unique index), do non-concurrent refresh
    REFRESH MATERIALIZED VIEW mv_weekly_active_requests;
END;
$$ LANGUAGE plpgsql;
```
