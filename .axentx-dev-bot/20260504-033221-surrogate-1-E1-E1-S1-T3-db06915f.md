/**
 * A robust context management system for hierarchical agent structures
 * with stack-based inheritance and snapshot capabilities.
 */
import { EventEmitter } from 'events';

export interface ContextData {
  [key: string]: any;
}

export interface ContextSnapshot {
  id: string;
  timestamp: number;
  data: Map<string, unknown>;
  parentId: string | null;
  childIds: string[];
}

export interface ContextConfig {
  maxHistorySize: number;
  snapshotInterval: number;
  enableAutoSnapshot: boolean;
}

const DEFAULT_CONFIG: ContextConfig = {
  maxHistorySize: 50,
  snapshotInterval: 30000,
  enableAutoSnapshot: true,
};

export class Context extends EventEmitter {
  private stack: ContextData[] = [];
  private history: ContextSnapshot[] = [];
  private config: ContextConfig;
  private snapshotTimer: NodeJS.Timeout | null = null;

  constructor(
    private id: string,
    private parentId: string | null = null,
    config: Partial<ContextConfig> = {}
  ) {
    super();
    this.config = { ...DEFAULT_CONFIG, ...config };
    if (this.config.enableAutoSnapshot) {
      this.startAutoSnapshot();
    }
  }

  // Stack-based context management
  push(data: ContextData) {
    this.stack.push(data);
    this.emit('context-push', { contextId: this.id, data });
  }

  pop(): ContextData | undefined {
    if (this.stack.length > 1) {
      const popped = this.stack.pop();
      this.emit('context-pop', { contextId: this.id });
      return popped;
    }
    return undefined;
  }

  get current(): ContextData {
    return this.stack.reduce((acc, layer) => ({ ...acc, ...layer }), {});
  }

  set(key: string, value: any) {
    if (this.stack.length === 0) {
      this.push({});
    }
    this.stack[this.stack.length - 1][key] = value;
    this.emit('context-set', { key, value, contextId: this.id });
  }

  get(key: string): any {
    return this.current[key];
  }

  // Snapshot capabilities
  private startAutoSnapshot() {
    this.snapshotTimer = setInterval(() => {
      this.createSnapshot();
    }, this.config.snapshotInterval);
  }

  createSnapshot(): ContextSnapshot {
    const snapshot: ContextSnapshot = {
      id: `${this.id}-snapshot-${Date.now()}`,
      timestamp: Date.now(),
      data: new Map(Object.entries(this.current)),
      parentId: this.parentId,
      childIds: [],
    };

    this.history.push(snapshot);
    if (this.history.length > this.config.maxHistorySize) {
      this.history.shift();
    }

    this.emit('snapshot-created', { snapshotId: snapshot.id, contextId: this.id });
    return snapshot;
  }

  restoreSnapshot(snapshot: ContextSnapshot) {
    this.stack = [Object.fromEntries(snapshot.data)];
    this.emit('snapshot-restored', { snapshotId: snapshot.id, contextId: this.id });
  }

  // Cleanup
  destroy() {
    if (this.snapshotTimer) {
      clearInterval(this.snapshotTimer);
    }
    this.removeAllListeners();
  }
}
