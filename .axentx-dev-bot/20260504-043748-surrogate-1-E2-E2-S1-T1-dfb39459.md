 *   - Session creation, joining, and leaving.
 *   - Owner can grant/revoke participant permissions.
 *   - Broadcasts messages to all participants with <200ms latency.
 *
 * Protocol (JSON over WebSocket):
 *   { type: 'create', sessionId: string, userId: string }
 *   { type: 'join',   sessionId: string, userId: string }
 *   { type: 'leave',  sessionId: string, userId: string }
 *   { type: 'message', sessionId: string, userId: string, data: any }
 *   { type: 'grant',   sessionId: string, ownerId: string, targetId: string }
 *   { type: 'revoke',  sessionId: string, ownerId: string, targetId: string }
 *
 * All responses are echoed back with a `status` field.
 */

const http = require('http');
const WebSocket = require('ws');
const { v4: uuidv4 } = require('uuid');

// In‑memory session store
const sessions = new Map(); // sessionId -> { owner, participants: Set, grants: Map }

function createSession(sessionId, ownerId) {
  if (sessions.has(sessionId)) {
    return false;
  }
  sessions.set(sessionId, {
    owner: ownerId,
    participants: new Set([ownerId]),
    grants: new Map([[ownerId, new Set(['owner'])]]),
  });
  return true;
}

function joinSession(sessionId, userId) {
  const session = sessions.get(sessionId);
  if (!session) return false;
  if (!session.grants.has(userId)) {
    // Default permission to 'viewer' if not granted
    session.grants.set(userId, new Set(['viewer']));
  }
  session.participants.add(userId);
  return true;
}

function leaveSession(sessionId, userId) {
  const session = sessions.get(sessionId);
  if (!session) return false;
  session.participants.delete(userId);
  if (session.participants.size === 0) {
    sessions.delete(sessionId);
  }
  return true;
}

function grantPermission(sessionId, ownerId, targetId) {
  const session = sessions.get(sessionId);
  if (!session) return false;
  if (session.owner !== ownerId) return false;
  const perms = session.grants.get(targetId) || new Set();
  perms.add('viewer');
  session.grants.set(targetId, perms);
  return true;
}

function revokePermission(sessionId, ownerId, targetId) {
  const session = sessions.get(sessionId);
  if (!session) return false;
  if (session.owner !== ownerId) return false;
  session.grants.delete(targetId);
  session.participants.delete(targetId);
  return true;
}
