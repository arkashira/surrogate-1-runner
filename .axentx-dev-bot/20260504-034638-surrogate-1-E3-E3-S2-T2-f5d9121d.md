import { fetchRemediations, renderRemediations, escapeHtml } from './ciRemediations';
import '@testing-library/jest-dom';
import { screen, render, fireEvent } from '@testing-library/react';

// Mock fetch
global.fetch = jest.fn();

describe('fetchRemediations()', () => {
  it('calls the correct endpoint with the current build ID', async () => {
    fetch.mockResolvedValueOnce({
      ok: true,
      json: async () => [{ id: 'r1', title: 'Update OpenSSL', severity: 'high', steps: [] }],
    });
    const data = await fetchRemediations('build-123');
    expect(fetch).toHaveBeenCalledWith('/api/ci/remediations?buildId=build-123');
    expect(data).toHaveLength(1);
  });

  it('throws an error when response is not ok', async () => {
    fetch.mockResolvedValueOnce({ ok: false, status: 500 });
    await expect(fetchRemediations('build-123')).rejects.toThrow('Failed to fetch remediations');
  });

  it('returns empty array when API returns []', async () => {
    fetch.mockResolvedValueOnce({ ok: true, json: async () => [] });
    const data = await fetchRemediations('build-123');
    expect(data).toEqual([]);
  });
});

describe('escapeHtml()', () => {
  it('escapes <, >, &, ", and \' characters', () => {
    const raw = `<script>alert('xss')</script>`;
    const escaped = escapeHtml(raw);
    expect(escaped).toBe('&lt;script&gt;alert(&#039;xss&#039;)&lt;/script&gt;');
  });
});

describe('renderRemediations()', () => {
  const sample = [
    {
      id: 'r1',
      title: 'Upgrade OpenSSL',
      severity: 'high',
      steps: ['Run apt-get update', 'Install libssl-dev >= 1.1.1'],
    },
  ];

  beforeEach(() => {
    document.body.innerHTML = `<div id="remediation-root"></div>`;
  });

  it('creates a card for each remediation', () => {
    renderRemediations(sample);
    const cards = screen.getAllByTestId('remediation-card');
    expect(cards).toHaveLength(sample.length);
    expect(cards[0]).toHaveTextContent('Upgrade OpenSSL');
  });

  it('displays severity badge with correct color class', () => {
    renderRemediations(sample);
    const badge = screen.getByText('high');
    expect(badge).toHaveClass('badge--high');
  });

  it('expands steps list when “View steps” button is clicked', () => {
    renderRemediations(sample);

  });
