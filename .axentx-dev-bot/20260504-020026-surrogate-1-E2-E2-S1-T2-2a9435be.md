"""
Pearson correlation calculator for revenue spike analysis.

Runs nightly after all connectors finish, computes Pearson correlation
between revenue and candidate variables, returns top-5 significant
correlations (p < 0.05) stored in /data/correlation/<date>.json
"""

import json
import sys
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any

import numpy as np
from scipy import stats


def compute_pearson(x: list[float], y: list[float]) -> dict[str, float]:
    """
    Compute Pearson correlation coefficient and p-value using scipy.
    
    Args:
        x: Independent variable values
        y: Dependent variable values (revenue)
    
    Returns:
        Dictionary with 'r' (correlation coefficient), 'p' (p-value), and 'n' (sample size)
    """
    x_arr = np.asarray(x, dtype=float)
    y_arr = np.asarray(y, dtype=float)
    
    # Remove any NaN/Inf pairs
    mask = ~(np.isnan(x_arr) | np.isnan(y_arr) | np.isinf(x_arr) | np.isinf(y_arr))
    x_clean = x_arr[mask]
    y_clean = y_arr[mask]
    
    if len(x_clean) < 3:
        return {"r": 0.0, "p": 1.0, "n": len(x_clean)}
    
    r, p = stats.pearsonr(x_clean, y_clean)
    return {"r": float(r), "p": float(p), "n": int(len(x_clean))}

    revenue_series: list[float],
    variable_matrix: dict[str, list[float]],
    top_n: int = 5,
    p_threshold: float = 0.05
) -> list[dict[str, Any]]:
    """
    Find top N variables most correlated with revenue.
    
    Args:
        revenue_series: Revenue time series data
        variable_matrix: Dict of variable name -> values
        top_n: Number of top correlations to return
        p_threshold: Maximum p-value for significance
    
    Returns:
        List of top correlations sorted by absolute correlation value
    """
    if len(revenue_series) < 3:
        return []
    
    results = []
    
    for var_name, var_values in variable_matrix.items():
        if not var_values or len(var_values) != len(revenue_series):
            continue

        results.append({
            "variable": var_name,
            "r": corr["r"],
            "p": corr["p"],
            "n": corr["n"]
        })
    
    # Filter significant only for ranking
