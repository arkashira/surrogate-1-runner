"""
GitHub Actions integration for Surrogate-1.

Handles workflow orchestration, shard dispatch, and result aggregation
via GitHub Actions matrix and artifact APIs.
"""
import os
import json
import hashlib
import subprocess
from typing import List, Dict, Any
from dataclasses import dataclass
from datetime import datetime


@dataclass
class WorkflowConfig:
    """Configuration for Surrogate-1 GitHub Actions workflow."""
    dataset_list_path: str = "bin/dataset-enrich.sh"
    num_shards: int = 16
    workflow_file: str = ".github/workflows/ingest.yml"
    artifact_name: str = "shard-output"
    timeout_minutes: int = 45


class GitHubActionsIntegration:
    """Manages Surrogate-1 integration with GitHub Actions."""

    def __init__(self, config: WorkflowConfig = None):
        self.config = config or WorkflowConfig()
        self.repo = os.getenv("GITHUB_REPOSITORY", "axentx/surrogate-1-training-pairs")

    def generate_shard_id(self, dataset_entry: str, total_shards: int) -> int:
        """Deterministically assign a dataset entry to a shard using hash."""
        h = hashlib.md5(dataset_entry.encode()).hexdigest()
        return int(h, 16) % total_shards

    def read_dataset_entries(self) -> List[str]:
        """Extract dataset identifiers from dataset-enrich.sh."""
        entries = []
        try:
            with open(self.config.dataset_list_path, 'r') as f:
                lines = f.readlines()

            for line in lines:
                line = line.strip()
                if line.startswith("python scripts/"):
                    # Extract dataset name from command like:
                    # python scripts/enrich.py --dataset my_dataset --source web
                    if "--dataset" in line:
                        parts = line.split("--dataset")
                        if len(parts) > 1:
                            dataset = parts[1].strip().split()[0]
                            entries.append(dataset)
        except Exception as e:
            raise RuntimeError(f"Failed to read dataset entries: {e}")
        return entries

    def build_matrix_strategy(self) -> Dict[str, Any]:
        """Generate GitHub Actions matrix config for parallel shards."""
        entries = self.read_dataset_entries()
        shards: Dict[int, List[str]] = {i: [] for i in range(self.config.num_shards)}

        for entry in entries:
            shard_id = self.generate_shard_id(entry, self.config.num_shards)
            shards[shard_id].append(entry)

        matrix = {
            "shard-id": list(range(self.config.num_shards)),
            "include": []
        }

        for shard_id in range(self.config.num_shards):
            matrix["include"].append({
                "shard-id": shard_id,
                "datasets": shards[shard_id],
                "slug": f"shard-{shard_id:02d}"
            })

        return matrix

    def generate_workflow_yaml(self) -> str:
        """Generate GitHub Actions workflow YAML content."""
        matrix = self.build_matrix_strategy()
        include_block = "\n".join([
            f"        - shard-id: {item['shard-id']}\n"
            f"          datasets: {json.dumps(item['datasets'])}\n"
            f"          slug: {item['slug']}"
            for item in matrix["include"]
        ])

        yaml_content = f"""# THIS FILE IS GENERATED — DO NOT EDIT MANUALLY
# See: surrogate-1/github_actions/integration.py

name: Ingest Public Datasets (Surrogate-1)

on:
  schedule:
    - cron: '*/30 * * * *'  # Every 30 minutes

jobs:

{include_block}
