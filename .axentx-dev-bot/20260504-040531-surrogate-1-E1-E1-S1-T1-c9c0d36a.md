    type: string
    description: Human-readable name of the policy
  description:
    type: string
    description: Description of the policy purpose
  rules:
    type: array

      type: object

        parameters:
          type: object

        - severity

    format: duration
    description: How often the policy should be evaluated
required:
  - policy_id
  - name
  - rules
