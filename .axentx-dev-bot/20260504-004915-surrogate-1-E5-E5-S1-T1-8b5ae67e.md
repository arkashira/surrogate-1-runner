describe('buildQueryString', () => {
  test('should return empty string when no filters applied', () => {
    const filters = {};
    expect(buildQueryString(filters)).toBe('');
  });

  test('should build query string with single filter', () => {
    const filters = { error_category: 'network' };
    expect(buildQueryString(filters)).toBe('?error_category=network');
  });

  test('should build query string with multiple filters', () => {
    const filters = { 
      error_category: 'network',
      resolution_status: 'resolved',
      start_date: '2023-01-01'
    };
    expect(buildQueryString(filters)).toBe(
      '?error_category=network&resolution_status=resolved&start_date=2023-01-01'
    );
  });
});
