## 1. Acceptance Criteria
- **AC‑1**: The route **GET /diagnostics** returns an HTML page that renders a table containing up to the 30 most recent diagnostic sessions, ordered by `created_at` descending.
- **AC‑2**: Each row displays at minimum: **Session ID**, **Date/Time**, **Error Category**, **Resolution Status**, and a **“View Details”** button.
- **AC‑3**: Clicking the “View Details” button opens a modal (or navigates to a detail view) that shows the full JSON resolution payload for that session.
- **AC‑4**: The page includes filter controls for **date range**, **error_category**, and **resolution_status**; applying any filter updates the table to show only matching sessions.
- **AC‑5**: Filters are sent to the backend via query parameters (`start`, `end`, `category`, `status`) and the server returns the correctly filtered list (max 30 rows).
- **AC‑6**: The dashboard is only accessible when a valid **API‑Key** header (`X-API-Key`) is present; otherwise the server returns **401 Unauthorized** and the page does not render.
- **AC‑7**: The UI is responsive: on viewports ≤ 768 px the table collapses to a stacked card layout and the modal remains usable.

// diagnostics.controller.test.js
import request from 'supertest';
import app from '../../src/app'; // Express instance
import { getRecentSessions } from '../../src/services/diagnosticsService';

jest.mock('../../src/services/diagnosticsService');

describe('GET /diagnostics (HTML view)', () => {
  const VALID_API_KEY = 'valid-key';

  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('returns 401 when API key missing', async () => {
    const res = await request(app).get('/diagnostics');
    expect(res.status).toBe(401);
    expect(res.text).toContain('Unauthorized');
  });

  test('returns 401 when API key invalid', async () => {
    const res = await request(app)
      .get('/diagnostics')
      .set('X-API-Key', 'bad-key');
    expect(res.status).toBe(401);
  });

  test('renders HTML with up to 30 sessions when key valid', async () => {
    const mockSessions = Array.from({ length: 30 }, (_, i) => ({
      id: `sess-${i}`,
      created_at: new Date(),
      error_category: 'network',
      resolution_status: 'resolved',
    }));
    getRecentSessions.mockResolvedValue(mockSessions);

    const res = await request(app)
      .get('/diagnostics')
      .set('X-API-Key', VALID_API_KEY);

    expect(res.status).toBe(200);
    expect(res.type).toBe('text/html');
    // Verify that each mock session appears in the HTML
    mockSessions.forEach(s => {
      expect(res.text).toContain(s.id);
    });
  });

  test('applies query filters correctly', async () => {
    const filtered = [{ id: 'sess-99', error_category: 'db', resolution_status: 'failed' }];
    getRecentSessions.mockResolvedValue(filtered);

    const res = await request(app)
      .get('/diagnostics')
      .query({ category: 'db', status: 'failed' })
      .set('X-API-Key', VALID_API_KEY);

    expect(getRecentSessions).toHaveBeenCalledWith(
      expect.objectContaining({
        category: 'db',
        status: 'failed',
        limit: 30,
      })
    );
    expect(res.text).toContain('sess-99');
  });
});
