"""
Invoice data models for storing parsed and categorized invoice information.

This module provides SQLAlchemy models for storing invoice data extracted
from document parsing, including vendor info, amounts, dates, and categories.
"""

from datetime import datetime
from decimal import Decimal
from enum import Enum
from typing import Optional

from sqlalchemy import (
    Column,
    Date,
    DateTime,
    Enum as SQLEnum,
    Float,
    ForeignKey,
    Index,
    Integer,
    Numeric,
    String,
    Text,
    UniqueConstraint,
)
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship


class InvoiceCategory(str, Enum):
    """Predefined categories for invoice classification."""
    UTILITIES = "utilities"
    SERVICES = "services"
    SUPPLIES = "supplies"
    RENT = "rent"
    INSURANCE = "insurance"
    SOFTWARE = "software"
    MARKETING = "marketing"
    TRAVEL = "travel"
    PROFESSIONAL_SERVICES = "professional_services"
    OTHER = "other"


class InvoiceStatus(str, Enum):
    """Processing status for invoices."""
    PENDING = "pending"
    PROCESSED = "processed"
    FAILED = "failed"
    FLAGGED = "flagged"


class Base(DeclarativeBase):
    """Base class for all models."""
    pass


class Vendor(Base):
    """
    Vendor/Supplier information extracted from invoices.
    
    Stores normalized vendor data to enable deduplication and
    consistent reporting across multiple invoices from the same vendor.
    """
    __tablename__ = "vendors"
    
    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    name: Mapped[str] = mapped_column(String(255), nullable=False, index=True)
    normalized_name: Mapped[str] = mapped_column(String(255), nullable=False, unique=True, index=True)
    default_category: Mapped[Optional[str]] = mapped_column(
        SQLEnum(InvoiceCategory),
        nullable=True
    )
    website: Mapped[Optional[str]] = mapped_column(String(512), nullable=True)
    contact_email: Mapped[Optional[str]] = mapped_column(String(255), nullable=True)
    created_at: Mapped[datetime] = mapped_column(
        DateTime,
        default=datetime.utcnow,
        nullable=False
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime,
        default=datetime.utcnow,
        onupdate=datetime.utcnow,
        nullable=False
    )
    
    # Relationships
    invoices: Mapped[list["Invoice"]] = relationship(
        "Invoice",
        back_populates="vendor",
        cascade="all, delete-orphan"
    )
    
    __table_args__ = (
        Index("ix_vendors_name_lower", "normalized_name"),
    )


class Invoice(Base):
    """
    Main invoice record containing parsed data from invoice documents.
    
    Stores extracted information including vendor, dates, amounts,
    and categorization for quick retrieval and analysis.
    """
    __tablename__ = "invoices"
    
    id: Mapped[int] = mapped_column(Integer, primary_key=True, autoincrement=True)
    
    # Vendor reference
    vendor_id: Mapped[Optional[int]] = mapped_column(
        Integer,
        ForeignKey("vendors.id", ondelete="SET NULL"),
        nullable=True,
        index=True
    )
    vendor_name_raw: Mapped[str] = mapped_column(String(255), nullable=False)
    
    # Invoice identification
    invoice_number: Mapped[Optional[str]] = mapped_column(String(100), nullable=True, index=True)
    
    # Date fields
    invoice_date: Mapped[Optional[datetime]] = mapped_column(Date, nullable=True, index=True)
    due_date: Mapped[Optional[datetime]] = mapped_column(Date, nullable=True, index=True)
    received_date: Mapped[Optional[datetime]] = mapped_column(DateTime, nullable=True)
    
    # Financial amounts (stored as Numeric for precision)
    total_amount: Mapped[Optional[Decimal]] = mapped_column(Numeric(12, 2), nullable=True, index=True)
    tax_amount: Mapped[Optional[Decimal]] = mapped_column(Numeric(12, 2), nullable=True)
