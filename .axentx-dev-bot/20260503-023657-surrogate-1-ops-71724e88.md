def test_deterministic_sharding():
    slugs = ["a/1.jsonl", "b/2.jsonl", "c/3.jsonl"]
    total = 16
    assignments = {}
    for s in slugs:
        idx = hash(s) % total
        assignments.setdefault(idx, []).append(s)
    # each slug assigned to exactly one shard
    assigned = sum(len(v) for v in assignments.values())
    assert assigned == len(slugs)

    shard_id = 3
