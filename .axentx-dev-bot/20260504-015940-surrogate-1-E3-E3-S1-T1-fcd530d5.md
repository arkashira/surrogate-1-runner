"""
AWS‑compatible sandbox API.

Provides a minimal subset of AWS‑style REST endpoints that allow
creation, retrieval, listing and deletion of generic “sandbox
resources”.  The implementation is deliberately lightweight and
in‑memory – it is intended for testing tools that expect an AWS‑like
interface (e.g. boto3, AWS CLI) without requiring a real AWS backend.

All responses follow the typical AWS JSON envelope pattern:

    {
        "OperationNameResponse": {
            ...payload...
        }
    }

The router is exported as ``router`` and can be included in the main
FastAPI application.
"""

from __future__ import annotations

import threading
import uuid
from typing import Dict, List

from fastapi import APIRouter, HTTPException, status
from pydantic import BaseModel, Field

# ----------------------------------------------------------------------
# In‑memory store (thread‑safe)
# ----------------------------------------------------------------------
class _ResourceStore:
    """Thread‑safe in‑memory store for sandbox resources."""
    def __init__(self) -> None:
        self._lock = threading.Lock()
        self._data: Dict[str, Dict] = {}

    def create(self, name: str, tags: Dict[str, str] | None = None) -> str:
        with self._lock:
            resource_id = str(uuid.uuid4())
            self._data[resource_id] = {
                "ResourceId": resource_id,
                "Name": name,
                "Tags": tags or {},
                "Status": "ACTIVE",
            }
            return resource_id

    def get(self, resource_id: str) -> Dict:
        with self._lock:
            if resource_id not in self._data:
                raise KeyError(resource_id)
            return self._data[resource_id]

    def delete(self, resource_id: str) -> None:
        with self._lock:
            if resource_id not in self._data:
                raise KeyError(resource_id)
            del self._data[resource_id]

    def list(self) -> List[Dict]:
        with self._lock:
            return list(self._data.values())


_store = _ResourceStore()

    Status: str = "CREATED"

    Name: str
    Tags: Dict[str, str]
