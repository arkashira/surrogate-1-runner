### 1) Model (`src/models/request.ts`)

```ts
import { Knex } from 'knex';

export interface Request {
  id: string;
  status: string;
  // ... other fields
}

export interface RequestAudit {
  id: string;
  request_id: string;
  actor: string;
  previous_status: string;
  new_status: string;
  created_at: Date;
}

export class RequestModel {
  constructor(private db: Knex) {}

  /**
   * Atomically transition request status and record immutable audit entry.
   * If status is unchanged, still records an audit row for visibility/traceability.
   */
  async transitionStatus(id: string, newStatus: string, actor: string): Promise<RequestAudit> {
    const trx = await this.db.transaction();
    try {
      const row = await trx<Request>('requests').where({ id }).first();
      if (!row) {
        await trx.rollback();
        throw new Error('Request not found');
      }

      const previousStatus = row.status;

      // Update status only when it changes
      if (previousStatus !== newStatus) {
        await trx('requests').where({ id }).update({ status: newStatus, updated_at: new Date() });
      }

      // Always create immutable audit row
      const [audit] = await trx<RequestAudit>('request_audits').insert(
        {
          id: trx.raw('uuid_generate_v4()') as any,
          request_id: id,
          actor,
          previous_status: previousStatus,
          new_status: newStatus,
          created_at: new Date(),
        },
        ['*']
      );

      await trx.commit();
      return audit;
    } catch (err) {
      await trx.rollback();
      throw err;
    }
  }

  async getAuditTrail(id: string, limit?: number): Promise<RequestAudit[]> {
    let q = this.db<RequestAudit>('request_audits')
      .where({ request_id: id })
      .orderBy('created_at', 'asc');
    if (limit != null) q = q.limit(Math.max(0, limit));
    return q;
  }
}
```

Key choices
- Single `transitionStatus` method couples status update + audit creation in one transaction (prevents drift).
- Idempotent: unchanged status still produces an audit row (improves traceability).
- `created_at` is set by DB where possible (`now()`) but also provided by app for portability.
- `getAuditTrail` defaults to ascending chronological order (oldest → newest).
