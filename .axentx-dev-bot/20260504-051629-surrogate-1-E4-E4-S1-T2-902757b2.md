// tests/App.test.jsx
import { render, screen, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import App from '../../dashboard/static/js/App';

// Mock WebSocket for metric updates
jest.mock('websocket', () => ({
  __esModule: true,
  default: class {
    onmessage = () => {};
    send = jest.fn();
    close = jest.fn();
  },
}));

describe('App Component - Unit Tests', () => {
  test('renders real-time charts updating at least once per second', async () => {
    jest.useFakeTimers();
    render(<App />);
    
    const chart = screen.getByTestId('gpu-chart');
    let updateCount = 0;
    const originalInnerHTML = chart.innerHTML;

    // Simulate time passing with frequent checks
    for (let i = 0; i < 10; i++) {
      jest.advanceTimersByTime(100);
      if (chart.innerHTML !== originalInnerHTML) updateCount++;
    }

    expect(updateCount).toBeGreaterThanOrEqual(5); // At least 5 updates in 1s (conservative)
  });

  test('displays visual alert after GPU >90% for >5 seconds', async () => {
    render(<App />);
    const alert = screen.queryByTestId('gpu-alert');

    expect(alert).not.toBeInTheDocument();

    // Simulate incoming high GPU usage over time
    for (let sec = 0; sec < 6; sec++) {
      act(() => {
        // Simulate message from backend
        App.handleMetricUpdate({ gpuUtil: 95 });
      });
      jest.advanceTimersByTime(1000);
    }

    await waitFor(() => screen.getByTestId('gpu-alert'));
    expect(screen.getByTestId('gpu-alert')).toBeVisible();
    expect(screen.getByText(/High GPU Utilization/i)).toBeInTheDocument();
  });

  test('enables CSV export of last 5 minutes of logs', async () => {
    const user = userEvent.setup();
    render(<App />);

    // Simulate receiving log data over time
    act(() => {
      for (let i = 0; i < 300; i++) { // 5 mins @ 1Hz
        App.handleMetricUpdate({
          timestamp: Date.now() - (300 - i) * 1000,
          gpuUtil: 70 + (i % 30),
          bandwidth: 40 + (i % 20),
          frameLatency: 16 + (i % 10),
        });
      }
    });

    await user.click(screen.getByRole('button', { name: /Export to CSV/i }));

    const csv = localStorage.getItem('lastExport');
    expect(csv).toBeDefined();

    const rows = csv.split('\n');
    expect(rows.length).toBe(301); // header + 300 entries
    expect(rows[0]).toContain('timestamp,gpuUtil,bandwidth,frameLatency');
  });
});
