from fastapi import APIRouter, Depends, HTTPException
from fastapi.responses import JSONResponse
from pydantic import BaseModel
from typing import List
from uuid import UUID
from axentx.db import get_db
from axentx.schemas import TagCreate, TagUpdate

router = APIRouter()

@router.post("/tags/")
async def create_tag(tag: TagCreate, db=Depends(get_db)):
    db_tag = await db.tags.create(tag)
    return JSONResponse(status_code=201, content={"id": db_tag.id})

@router.get("/tags/")
async def read_tags(db=Depends(get_db)):
    tags = await db.tags.all()
    return JSONResponse(status_code=200, content=[tag.dict() for tag in tags])

@router.get("/tags/{tag_id}")
async def read_tag(tag_id: UUID, db=Depends(get_db)):
    tag = await db.tags.get(tag_id)
    if not tag:
        raise HTTPException(status_code=404, detail="Tag not found")
    return JSONResponse(status_code=200, content=tag.dict())

@router.put("/tags/{tag_id}")
async def update_tag(tag_id: UUID, tag: TagUpdate, db=Depends(get_db)):
    db_tag = await db.tags.get(tag_id)
    if not db_tag:
        raise HTTPException(status_code=404, detail="Tag not found")
    await db.tags.update(db_tag, tag)
    return JSONResponse(status_code=200, content=db_tag.dict())

@router.delete("/tags/{tag_id}")
async def delete_tag(tag_id: UUID, db=Depends(get_db)):
    await db.tags.delete(tag_id)
    return JSONResponse(status_code=204, content={})
```
