describe('LlamaGatewayMonitor', () => {
  let monitor;

  beforeEach(() => {
    monitor = new LlamaGatewayMonitor({ windowSeconds: 5, healthPolicy: defaultPolicy });
  });

  test('records request count correctly', () => {
    monitor.record('endpointA', 120);
    monitor.record('endpointA', 80);
    expect(monitor.getSnapshot().endpoints['endpointA'].count).toBe(2);
  });

  test('calculates average latency', () => {
    monitor.record('ep', 100);
    monitor.record('ep', 200);
    expect(monitor.getSnapshot().endpoints['ep'].avgLatency).toBeCloseTo(150);
  });

  test('updates error rate when status code >= 400', () => {
    monitor.record('ep', 100, 200);
    monitor.record('ep', 150, 500);
    expect(monitor.getSnapshot().endpoints['ep'].errorRate).toBeCloseTo(0.5);
  });

  test('health status reflects policy', () => {
    monitor.record('ep', 100, 200); // healthy
    expect(monitor.isHealthy()).toBe(true);
    monitor.record('ep', 300, 500); // error
    expect(monitor.isHealthy()).toBe(false);
  });

  test('listener receives snapshot on update', done => {
    const listener = jest.fn();
    monitor.addListener(listener);
    monitor.record('ep', 100);
    setTimeout(() => {
      expect(listener).toHaveBeenCalledTimes(1);
      const snap = listener.mock.calls[0][0];
      expect(snap.endpoints['ep'].count).toBe(1);
      done();
    }, 10);
  });

  test('listener removal stops callbacks', done => {
    const listener = jest.fn();
    monitor.addListener(listener);
    monitor.removeListener(listener);
    monitor.record('ep', 100);
    setTimeout(() => {
      expect(listener).not.toHaveBeenCalled();
      done();
    }, 10);
  });

  test('snapshot is a deep copy', () => {
    monitor.record('ep', 100);
    const snap1 = monitor.getSnapshot();
    snap1.endpoints['ep'].count = 999;
    const snap2 = monitor.getSnapshot();
    expect(snap2.endpoints['ep'].count).toBe(1);
  });

  test('concurrent recording does not corrupt data', async () => {
    const promises = [];
    for (let i = 0; i < 1000; i++) {
      promises.push(Promise.resolve().then(() => monitor.record('ep', 100)));
    }
    await Promise.all(promises);
    expect(monitor.getSnapshot().endpoints['ep'].count).toBe(1000);
  });

  test('handles zero requests gracefully', () => {
    const snap = monitor.getSnapshot();
    expect(snap.endpoints).toEqual({});
  });

  test('handles extremely high latency values', () => {
    monitor.record('ep', Number.MAX_SAFE_INTEGER);
    const snap = monitor.getSnapshot();
    expect(snap.endpoints['ep'].maxLatency).toBe(Number.MAX_SAFE_INTEGER);
  });
