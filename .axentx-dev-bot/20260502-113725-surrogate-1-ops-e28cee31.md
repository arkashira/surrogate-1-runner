```bash
#!/usr/bin/env bash
# Generate a pinned CDN manifest for one date-folder.
# Run on Mac/CI (once per folder) after HF API window clears.
# Usage: HF_TOKEN=... bash ops/training_manifest.sh 2026-05-02

set -euo pipefail

REPO="${HF_REPO:-datasets/axentx/surrogate-1-ingest}"
DATE_FOLDER="${1:-$(date +%Y-%m-%d)}"
OUT_DIR="$(cd "$(dirname "$0")/../manifests" && pwd)"
OUT_FILE="${OUT_DIR}/manifest_${DATE_FOLDER}.json"
HF_TOKEN="${HF_TOKEN:-}"

mkdir -p "$OUT_DIR"

python3 - "$REPO" "$DATE_FOLDER" "$OUT_FILE" "$HF_TOKEN" <<'PY'
import os, json, sys
from huggingface_hub import HfApi

repo, date_folder, out_file, token = sys.argv[1], sys.argv[2], sys.argv[3], sys.argv[4] or None
api = HfApi(token=token)

# Non-recursive: list only direct children in the date folder
entries = api.list_repo_tree(repo=repo, path=date_folder, recursive=False)
files = sorted([e.path for e in entries if e.type == "file"])

manifest = {
    "repo": repo,
    "date_folder": date_folder,
    "files": files,
    "cdn_base": f"https://huggingface.co/datasets/{repo}/resolve/main"
}

os.makedirs(os.path.dirname(out_file), exist_ok=True)
with open(out_file, "w") as f:
    json.dump(manifest, f, indent=2)

print(f"Wrote {len(files)} files to {out_file}")
PY

echo "Manifest ready: $OUT_FILE"
```

Make executable:

```bash
chmod +x /opt/axentx/surrogate-1/ops/training_manifest.py
```
