package agent

import (
	"fmt"
	"log"
	"os"
	"path/filepath"

	"github.com/natefinch/lumberjack"
)

const (
	defaultLogRotateDaily = true
	defaultLogRotateSize  = 100 * 1024 * 1024 // 100MB
	defaultLogRotateCount  = 30
)

type loggingConfig struct {
	rotateDaily bool
	rotateSize  int
	rotateCount int
}

func (c loggingConfig) getLogRotateConfig() lumberjack.Logger {
	return lumberjack.Logger{
		Filename:   filepath.Join(os.Getenv("PROGRAMDATA"), "Surrogate", "agent.log"),
		MaxSize:    c.rotateSize,
		MaxBackups: c.rotateCount,
		MaxAge:     28, // 4 weeks
		Compress:   false,
		LocalTime:  true,
	}
}

func (c loggingConfig) getLogRotateFunc() func() {
	if c.rotateDaily {
		return func() {
			log.Println("Rotating log file daily...")
			if err := os.Rename(filepath.Join(os.Getenv("PROGRAMDATA"), "Surrogate", "agent.log"), filepath.Join(os.Getenv("PROGRAMDATA"), "Surrogate", "agent.log.1")); err != nil {
				log.Printf("Error rotating log file: %v", err)
			}
		}
	}
	return func() {}
}

func (c loggingConfig) getLogFunc() func(string, ...interface{}) {
	return func(format string, a ...interface{}) {
		log.Printf(format, a...)
	}
}

func (c loggingConfig) getLogFuncErr() func(error) {
	return func(err error) {
		if err != nil {
			log.Printf("Error: %v", err)
		}
	}
}

func NewLoggingConfig() loggingConfig {
	return loggingConfig{
		rotateDaily: defaultLogRotateDaily,
		rotateSize:  defaultLogRotateSize,
		rotateCount: defaultLogRotateCount,
	}
}

func init() {
	log.SetFlags(0)
	log.SetOutput(&log.Logger{Out: os.Stdout, Flags: log.Ldate | log.Ltime | log.Lshortfile})
}

## 1. Acceptance Criteria
- **Service Naming**: The Windows Service must be registered under the name `SurrogateWinAgent`.  
- **Polling Interval**: Default polling interval is 5 minutes; it can be overridden via the registry key `HKLM\Software\SurrogateWinAgent\PollingInterval`.  
- **Log File Path**: Logs are written to `%ProgramData%\Surrogate\agent.log`. If `%ProgramData%` is undefined, the fallback path `C:\Surrogate\agent.log` is used.  
- **Log Rotation**:  
  - File size rotation: when the log file exceeds **100 MB**, a new file is created.  
  - Time rotation: a new log file is created **daily** at midnight.  
- **Rotation Retention**: Old log files are retained for **28 days**; older files are automatically deleted.  
- **Logging Library**: All log output is routed through `lumberjack.Logger` via `log.SetOutput()`. No duplicate `log.SetOutput()` calls exist.  
- **Service Startup**: On service start, the logger is correctly initialized and the first log entry is written to the expected file.

func TestGetLogRotateConfig_Defaults(t *testing.T) {

    assert.Equal(t, 100*1024*1024, cfg.MaxSize) // 100 MB
    assert.Equal(t, 28, cfg.MaxBackups)         // 28 days retention
    assert.True(t, cfg.LocalTime)

func TestLogSetOutput_UsesLumberjack(t *testing.T) {
    // Capture the current output
    oldOut := log.Writer()
    defer log.SetOutput(oldOut)

    // Initialize logger
    initLogger()

    // Verify that output is now a lumberjack.Logger
    _, ok := log.Writer().(*lumberjack.Logger)
    assert.True(t, ok, "log output should be lumberjack.Logger")
}

func TestProgramData_Fallback(t *testing.T) {
    // Unset PROGRAMDATA
    os.Unsetenv("ProgramData")
    defer os.Setenv("ProgramData", filepath.Join(os.Getenv("ProgramData")))

    cfg := getLogRotateConfig()
    assert.Contains(t, cfg.Filename, "C:\\Surrogate\\agent.log")
}

func TestRotation_TriggersAtSize(t *testing.T) {
    // Create a temporary log file
    tmpDir := t.TempDir()
    cfg := &lumberjack.Logger{
        Filename:   filepath.Join(tmpDir, "agent.log"),
        MaxSize:    1, // 1 MB for test
        MaxBackups: 1,
        MaxAge:     1,
    }
    log.SetOutput(cfg)

    // Write >1 MB of data
    data := make([]byte, 2*1024*1024) // 2 MB
    for i := range data {
        data[i] = 'a'
    }
    _, err := log.Output(2, string(data))
    assert.NoError(t, err)

    // Wait a moment for rotation
    time.Sleep(100 * time.Millisecond)

    // Verify that a new file exists
    files, _ := filepath.Glob(filepath.Join(tmpDir, "agent.log.*"))
    assert.NotEmpty(t, files, "rotated log file should exist")
}
