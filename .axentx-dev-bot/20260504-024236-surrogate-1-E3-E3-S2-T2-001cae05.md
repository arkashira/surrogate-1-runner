	if err := os.MkdirAll(root, CacheDirPerm); err != nil {

	}
	return root, nil

// path.  We hash the **absolute** path (no need to include mtime – the
// modification time is already used for LRU tracking).  The result is a
// hex‑encoded SHA‑256 string.
