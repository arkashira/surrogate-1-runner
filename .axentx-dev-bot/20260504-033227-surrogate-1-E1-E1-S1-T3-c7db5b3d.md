"""
Battery health metrics refresh module for surrogate-1.

Combines cross-platform psutil support with Linux sysfs precision,
providing temperature, charge cycles, capacity, and power state.

Author: Synthesized from Candidate 1 & 2
"""

import asyncio
import argparse
import os
import subprocess
import sys
from datetime import datetime
from pathlib import Path
from typing import Optional, Callable, Dict, Any, List

# Cross-platform support
try:
    import psutil
    PSUTIL_AVAILABLE = True
except ImportError:
    PSUTIL_AVAILABLE = False


class BatteryMetrics:
    """Container for battery health metrics with serialization support."""
    
    def __init__(self):
        self.percentage: Optional[int] = None          # Current charge %
        self.secsleft: Optional[int] = None             # Seconds until full/empty
        self.power_plugged: Optional[bool] = None       # Charger connected
        self.temperature: Optional[float] = None        # Celsius
        self.cycle_count: Optional[int] = None          # Charge cycles
        self.capacity: Optional[int] = None             # Full capacity mAh
        self.voltage: Optional[float] = None            # Current voltage
        self.timestamp: datetime = datetime.now()
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            "percentage": self.percentage,
            "secsleft": self.secsleft,
            "power_plugged": self.power_plugged,
            "temperature_celsius": self.temperature,
            "cycle_count": self.cycle_count,
            "capacity_mah": self.capacity,
            "voltage": self.voltage,
            "timestamp": self.timestamp.isoformat()
        }
    
    def __repr__(self) -> str:
        return f"<BatteryMetrics {self.percentage}% charge>"


class BatteryReader:
    """
    Multi-source battery data reader with prioritized fallbacks.
    
    Priority order:
    1. psutil (cross-platform, always try first)
    2. Linux sysfs (precise hardware data)
    3. upower CLI (last resort fallback)
    """
    
    def __init__(self):
        self._sysfs_battery_path: Optional[str] = None
        self._discover_sysfs_path()
    
    def _discover_sysfs_path(self) -> None:
        """Locate the battery sysfs directory once at initialization."""
        try:
            base = "/sys/class/power_supply"
            if os.path.exists(base):
                for entry in os.listdir(base):
                    if entry.startswith("BAT"):
                        self._sysfs_battery_path = os.path.join(base, entry)
                        return
        except OSError:
            pass
    
    def read_sysfs(self, filename: str) -> Optional[str]:
        """Read a single value from sysfs battery path."""
        if not self._sysfs_battery_path:
            return None
        
        path = os.path.join(self._sysfs_battery_path, filename)
        try:
            if os.path.exists(path):
                with open(path, 'r') as f:
                    return f
