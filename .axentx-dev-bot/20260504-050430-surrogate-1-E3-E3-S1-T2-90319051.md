"""
Swarm Recommendation Engine

* Loads configuration from /opt/axentx/surrogate-1/config/swarm.yml
* Aggregates (dummy) usage data for each peer org
* Computes a simple instance‑type / reservation mix recommendation
* Persists the recommendation in a SQLite database
* Exposes a small CLI for ad‑hoc runs
"""

from __future__ import annotations

import logging
import sys
from datetime import datetime
from pathlib import Path
from typing import Dict, List, Tuple

import yaml
from sqlalchemy import (
    Column,
    Integer,
    String,
    Table,
    MetaData,
    create_engine,
    insert,
    select,
)
from sqlalchemy.engine import Engine
from sqlalchemy.exc import SQLAlchemyError

# --------------------------------------------------------------------------- #
# Logging
# --------------------------------------------------------------------------- #
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s [%(module)s] %(message)s",
)
log = logging.getLogger(__name__)

# --------------------------------------------------------------------------- #
# Configuration
# --------------------------------------------------------------------------- #
CONFIG_PATH = Path("/opt/axentx/surrogate-1/config/swarm.yml")

def load_config(path: Path) -> Dict:
    if not path.exists():
        raise FileNotFoundError(f"Missing config: {path}")
    with path.open("r", encoding="utf-8") as f:
        cfg = yaml.safe_load(f) or {}
    # Basic validation
    if "peer_orgs" not in cfg or not isinstance(cfg["peer_orgs"], list):
        raise ValueError("Config must contain a list of peer_orgs")
    if len(cfg["peer_orgs"]) < 10:
        log.warning(
            "Only %d peer_orgs configured – recommendations may be less accurate",
            len(cfg["peer_orgs"]),
        )
    cfg.setdefault("database_url", "sqlite:///swarm.db")
    return cfg

CONFIG = load_config(CONFIG_PATH)

# --------------------------------------------------------------------------- #
# Database
# --------------------------------------------------------------------------- #
engine: Engine = create_engine(
    CONFIG["database_url"], echo=False, future=True
)
metadata = MetaData()

recommendations_table = Table(
    "recommendations",
    metadata,
    Column("id", Integer, primary_key=True, autoincrement=True),
    Column("org_id", String, nullable=False),
    Column("instance_type", String, nullable=False),
    Column("size", String, nullable=False),
    Column("reservation_mix", String, nullable=False),
    Column("projected_savings", Integer, nullable=False),
    Column("created_at", String, nullable=False),
)

# Create table if missing
metadata.create_all(engine)

# --------------------------------------------------------------------------- #
# Core logic
# --------------------------------------------------------------------------- #
class SwarmRecommendationEngine:
    """Encapsulates recommendation generation and persistence."""

    def __init__(self, engine: Engine, peer_orgs: List[str]) -> Non
