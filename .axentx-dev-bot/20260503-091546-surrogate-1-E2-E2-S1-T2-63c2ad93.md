### 2. Service: `/opt/axentx/surrogate-1/src/services/requestTimelineService.ts`
```ts
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { RequestTimeline } from '../models/requestTimeline';

@Injectable()
export class RequestTimelineService {
  constructor(
    @InjectRepository(RequestTimeline)
    private readonly timelineRepo: Repository<RequestTimeline>,
  ) {}

  /**
   * Append a timeline entry for a status change.
   */
  async createTimelineEntry(
    requestId: string,
    status: string,
    actorId?: string,
  ): Promise<RequestTimeline> {
    const entry = this.timelineRepo.create({
      requestId,
      status,
      actorId: actorId || null,
      occurredAt: new Date(),
    });
    return this.timelineRepo.save(entry);
  }

  /**
   * Get timeline for a request, ordered by occurredAt ascending.
   */
  async getTimelineByRequestId(requestId: string): Promise<RequestTimeline[]> {
    return this.timelineRepo.find({
      where: { requestId },
      order: { occurredAt: 'ASC' },
    });
  }

  /**
   * Convenience: append entry and return full timeline (for PATCH response).
   */
  async appendAndReturnTimeline(
    requestId: string,
    status: string,
    actorId?: string,
  ): Promise<RequestTimeline[]> {
    await this.createTimelineEntry(requestId, status, actorId);
    return this.getTimelineByRequestId(requestId);
  }
}
```
