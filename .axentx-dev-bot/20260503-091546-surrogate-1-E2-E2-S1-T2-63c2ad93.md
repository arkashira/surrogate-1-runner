@Table({ tableName: 'request_timelines', timestamps: true })

  status!: string;

  occurredAt!: Date;

  @BelongsTo(() => Request)
  request!: Request;
}

import { RequestTimeline } from '../models/requestTimeline';
import { Request } from '../models/request';

export async function createStatusTimelineEntry(
  requestId: string,
  newStatus: string,
  actorId: string,
  note?: string
) {
  const entry = await RequestTimeline.create({
    requestId,
    status: newStatus,
    actorId,
    note,
    occurredAt: new Date(),
  });

  return entry;
}

export async function getTimelineByRequestId(requestId: string) {
  return RequestTimeline.findAll({

  });
}
