from surrogate_1.mappers.resource_mapper import (

def test_map_resource_known_type():
    controls = map_resource("terraform", "aws_s3_bucket")
    assert isinstance(controls, list)
    assert len(controls) > 0
    for c in controls:
        assert isinstance(c, ComplianceControl)
        assert c.framework in {"PDPA", "SOC 2"}
        assert c.category in {"Data Protection", "Access Control"}

def test_map_resource_unknown_type():
    controls = map_resource("unknown", "foo")
    assert controls == []

def test_compliance_control_to_dict():
    ctrl = ComplianceControl(
        framework="PDPA",
        category="Data Protection",
        title="Encryption at Rest",
        description="Data must be encrypted when stored.",
        resource="aws_s3_bucket",
    )
    d = ctrl.to_dict()
    assert d == {
        "framework": "PDPA",
        "category": "Data Protection",
        "title": "Encryption at Rest",
        "description": "Data must be encrypted when stored.",
        "resource": "aws_s3_bucket",
    }

def test_case_insensitive_mapping():
    controls_upper = map_resource("TERRAFORM", "aws_s3_bucket")
    controls_lower = map_resource("terraform", "aws_s3_bucket")
    assert controls_upper == controls_lower

def test_control_definitions_completeness():
    # Ensure every control has required keys
    for ctrl in CONTROL_DEFINITIONS.values():
        for key in ("framework", "category", "title", "description"):
            assert key in ctrl
