#### GitHub Actions Workflow
```yaml
# .github/workflows/compose_guard.yml
name: Compose Guard

on:
  pull_request:
    branches:
      - main

jobs:
  health-check:
    runs-on: ubuntu-latest
    if: ${{ secrets.COMPOSE_GUARD_DISABLED != 'true' }}
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v2

      - name: Start Docker Compose
        run: docker compose up -d

      - name: Run Health Wait Script
        run: |
          ./health_wait.sh
          if [ $? -ne 0 ]; then
            echo "Health check failed for some services."
            exit 1
          fi

      - name: Print Health Summary
        run: |
          echo "Health check summary:"
          docker ps --format "{{.Names}}: {{.Status}}"
```

#### Health Wait Script
```bash
# health_wait.sh
#!/bin/bash

set -e

TIMEOUT=60
UNHEALTHY_CONTAINERS=()

while [[ $TIMEOUT -gt 0 ]]; do
  HEALTHY=true
  for CONTAINER in $(docker ps --format "{{.Names}}" --filter health=unhealthy)
