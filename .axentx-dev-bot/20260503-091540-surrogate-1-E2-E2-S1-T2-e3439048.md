@@ -0,0 +1,50 @@
+import { Router } from 'express';
+import { Request, Response } from 'express';
+import { getPublicRequestById } from '../services/requestService';
+
+const router = Router();
+
+/**
+ * GET /public/requests/:id
+ * Returns public request details for sharing status with stakeholders
+ */
+router.get('/:id', async (req: Request, res: Response) => {
+  try {
+    const requestId = req.params.id;
+    
+    // Fetch public request data
+    const request = await getPublicRequestById(requestId);
+    
+    if (!request) {
+      return res.status(404).json({
+        error: 'Request not found',
+        message: 'The requested public request does not exist or is not accessible'
+      });
+    }
+
+    // Format response according to acceptance criteria
+    const responseData = {
+      id: request.id,
+      title: request.title,
+      type: request.type,
+      slaTarget: request.slaTarget,
+      status: request.status,
+      humanReadableStatus: getHumanReadableStatus(request.status),
+      timeline: request.timeline || [],
+      lastUpdated: request.lastUpdated || new Date().toISOString()
+    };
+
+    res.json(responseData);
+  } catch (error) {
+    console.error('Error fetching public request:', error);
+    res.status(500).json({
+      error: 'Internal server error',
+      message: 'Failed to retrieve request details'
+    });
+  }
+});
+
+/**
+ * Helper function to convert status codes to human-readable format
+ */
+function getHumanReadableStatus(status: string): string {
+  const statusMap: Record<string, string> = {
+    'draft': 'Draft',
+    'submitted': 'Submitted',
+    'in_progress': 'In Progress',
+    'completed': 'Completed',
+    'cancelled': 'Cancelled',
+    'pending_review': 'Pending Review'
+  };
+  
+  return statusMap[status] || status.charAt(0).toUpperCase() + status.slice(1);
+}
+
+export default router;
