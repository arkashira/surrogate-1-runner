// publicRequests.unit.test.ts
import { handler } from './routes/publicRequests';
import { findPublicRequestById } from '../services/requestService';

jest.mock('../services/requestService');

describe('GET /public/requests/:id', () => {
  const mockRequest = (id: string, isPublic = true) => ({
    id,
    status: 'in_progress',
    title: 'Onboarding',
    type: 'onboarding',
    slaTarget: '2026-06-01T12:00:00.000Z',
    timeline: [
      { status: 'created', changedAt: '2026-05-01T08:00:00.000Z', actorType: 'system' },
      { status: 'in_progress', changedAt: '2026-05-02T09:00:00.000Z', actorType: 'agent' }
    ],
    lastUpdated: '2026-05-02T09:00:00.000Z',
    humanReadableStatus: 'In progress',
    isPublic
  });

  test('returns 200 with required fields for public request', async () => {
    (findPublicRequestById as jest.Mock).mockResolvedValue(mockRequest('req-123', true));
    const res = await handler({ params: { id: 'req-123' } } as any);
    expect(res.statusCode).toBe(200);
    const body = JSON.parse(res.body);
    expect(body).toMatchObject({
      status: 'in_progress',
      title: 'Onboarding',
      type: 'onboarding',
      slaTarget: '2026-06-01T12:00:00.000Z',
      timeline: expect.arrayContaining([
        expect.objectContaining({ status: expect.any(String), changedAt: expect.any(String), actorType: expect.any(String) })
      ]),
      lastUpdated: '2026-05-02T09:00:00.000Z',
      humanReadableStatus: 'In progress'
    });
  });

  test('returns 404 for non-existent request', async () => {
    (findPublicRequestById as jest.Mock).mockResolvedValue(null);
    const res = await handler({ params: { id: 'req-none' } } as any);
    expect(res.statusCode).toBe(404);
    expect(res.body).toBe('');
  });

  test('returns 404 for private request (no leakage)', async () => {
    (findPublicRequestById as jest.Mock).mockResolvedValue(mockRequest('req-private', false));
    const res = await handler({ params: { id: 'req-private' } } as any);
    expect(res.statusCode).toBe(404);
    const body = JSON.parse(res.body || '{}');
    expect(body.title).toBeUndefined();
    expect(body.timeline).toBeUndefined();
  });

  test('handles missing timeline gracefully (returns empty array)', async () => {
    const req = mockRequest('req-bad');
    delete req.timeline;
    (findPublicRequestById as jest.Mock).mockResolvedValue(req);
    const res = await handler({ params: { id: 'req-bad' } } as any);
    expect(res.statusCode).toBe(200);
    const body = JSON.parse(res.body);
    expect(body.timeline).toEqual([]);
  });

  test('includes CORS headers', async () => {
    (findPublicRequestById as jest.Mock).mockResolvedValue(mockRequest('req-cors', true));
    const res = await handler({ params: { id: 'req-cors' } } as any);
    expect(res.headers['Access-Control-Allow-Origin']).toBe('*');
    expect(res.headers['Access-Control-Allow-Headers']).toContain('Content-Type');
  });
});
