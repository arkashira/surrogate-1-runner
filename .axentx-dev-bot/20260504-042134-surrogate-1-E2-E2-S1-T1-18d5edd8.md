/**
 * AWS provider – collects EC2, S3 and RDS metadata.
 *
 * The module is intentionally small so that the Surrogate runner stays
 * fast.  All heavy lifting is delegated to the official AWS SDK v3.
 *
 * Public API
 *  - collectResources(config?): Promise<{ ec2, s3, rds }>
 *
 * The returned object matches the unified dashboard schema:
 * {
 *   ec2: [ { instanceId, type, state, launchTime, tags } ],
 *   s3:  [ { name, creationDate, tags } ],
 *   rds: [ { dbInstanceIdentifier, dbInstanceClass, engine, status, endpoint, tags } ]
 * }
 *
 * Errors that prevent a single resource type from being fetched are
 * logged but do not stop the other types from being returned.
 */

 * Validate that the required AWS environment variables are present.
 * Throws if any are missing.
 */

  const { AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, AWS_REGION } = process.env;
  if (!AWS_ACCESS_KEY_ID || !AWS_SECRET_ACCESS_KEY || !AWS_REGION) {
    throw new Error(
      "Missing AWS credentials. Set AWS_ACCESS_KEY_ID, AWS_SECRET_ACCESS_KEY, and AWS_REGION."
    );
  }
}
