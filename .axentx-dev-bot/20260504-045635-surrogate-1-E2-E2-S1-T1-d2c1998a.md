// compliance-checks.test.js
const { runChecks, generateReport, persistReport, sendEmail } = require('./compliance-checks');
const aws = require('aws-sdk-mock');

describe('Compliance Engine', () => {
  beforeAll(() => {
    // Mock AWS SDK responses
    aws.mock('S3', 'getBucketAcl', (params, cb) => cb(null, { Grants: [] }));
    aws.mock('IAM', 'getAccountPasswordPolicy', (params, cb) => cb(null, { MinimumPasswordLength: 8 }));
    aws.mock('EC2', 'describeInstances', (params, cb) => cb(null, { Reservations: [] }));
  });

  afterAll(() => aws.restore());

  test('runChecks returns array of CheckResult objects', async () => {
    const results = await runChecks();
    expect(Array.isArray(results)).toBe(true);
    results.forEach(r => {
      expect(r).toHaveProperty('CheckID');
      expect(['PASS', 'FAIL']).toContain(r.Status);
      expect(r).toHaveProperty('Description');
    });
  });

  test('generateReport creates JSON serializable object', () => {
    const results = [{ CheckID: 'S3Public', Status: 'FAIL', Description: 'Public bucket', Remediation: 'Set ACL' }];
    const report = generateReport(results);
    expect(report).toHaveProperty('Timestamp');
    expect(report).toHaveProperty('Results');
    expect(report.Results).toEqual(results);
  });

  test('persistReport writes file with correct naming', async () => {
    const report = { Timestamp: new Date().toISOString(), Results: [] };
    const path = await persistReport(report);
    expect(path).toMatch(/report-\d{8}\.json$/);
    const content = require(path);
    expect(content).toEqual(report);
  });

  test('sendEmail attaches report and returns success flag', async () => {
    const mockSend = jest.fn().mockResolvedValue({ MessageId: '123' });
    const originalSend = require('nodemailer').createTransport().sendMail;
    require('nodemailer').createTransport().sendMail = mockSend;

    const result = await sendEmail('/tmp/report.json');
    expect(mockSend).toHaveBeenCalled();
    expect(result).toBe(true);

    // Restore original
    require('nodemailer').createTransport().sendMail = originalSend;
  });
});
