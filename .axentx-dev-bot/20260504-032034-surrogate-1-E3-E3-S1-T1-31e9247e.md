# test_dockerfile.py
import os
import subprocess
import pytest

DOCKERFILE_PATH = "/opt/axentx/surrogate-1/Dockerfile"
IMAGE_TAG = "internal-registry/surrogate-1:latest"

def test_dockerfile_exists():
    assert os.path.isfile(DOCKERFILE_PATH), "Dockerfile missing"

def test_dockerfile_builds():
    result = subprocess.run(
        ["docker", "build", "-t", IMAGE_TAG, "-f", DOCKERFILE_PATH, "."],
        capture_output=True, text=True
    )
    assert result.returncode == 0, f"Build failed: {result.stderr}"

def test_image_size():
    result = subprocess.run(
        ["docker", "image", "inspect", IMAGE_TAG, "--format", "{{.Size}}"],
        capture_output=True, text=True
    )
    size_bytes = int(result.stdout.strip())
    assert size_bytes <= 200 * 1024 * 1024, f"Image size {size_bytes} exceeds 200MB"

def test_health_endpoint():
    # Run container
    container = subprocess.run(
        ["docker", "run", "-d", "-p", "8080:8080", IMAGE_TAG],
        capture_output=True, text=True
    )
    cid = container.stdout.strip()
    try:
        resp = subprocess.run(
            ["curl", "-s", "-o", "/dev/null", "-w", "%{http_code}", "http://localhost:8080/health"],
            capture_output=True, text=True
        )
        assert resp.stdout.strip() == "200", "Health endpoint did not return 200"
    finally:
        subprocess.run(["docker", "rm", "-f", cid])

def test_registry_push():
    # Assume CI environment variable PUSHED=true when push occurs
    assert os.getenv("PUSHED") == "true", "Image not pushed to registry"
