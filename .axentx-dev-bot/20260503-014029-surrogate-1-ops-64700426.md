### 2) Replace `bin/dataset-enrich.sh` with CDN-bypass, manifest-driven sharding

`bin/dataset-enrich.sh`
```bash
#!/usr/bin/env bash
set -euo pipefail
export SHELL=/bin/bash

REPO="axentx/surrogate-1-training-pairs"
DATE_PATH="${DATE_PATH:-$(date +%Y-%m-%d)}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"
MANIFEST_FILE="${MANIFEST_FILE:-manifest-${DATE_PATH}.json}"
OUTDIR="batches/public-merged/${DATE_PATH}"
mkdir -p "${OUTDIR}"
TIMESTAMP=$(date +%H%M%S)
OUTFILE="${OUTDIR}/shard${SHARD_ID}-${TIMESTAMP}.jsonl"

echo "[$(date)] Shard ${SHARD_ID}/${TOTAL_SHARDS} starting"

# Load manifest (preferred) or fallback to API (rate-limited)
if [[ -f "${MANIFEST_FILE}" ]]; then
  mapfile -t ALL_FILES < <(jq -r '.[]' "${MANIFEST_FILE}")
else
  echo "Warning: manifest not found, falling back to API (rate-limited)"
  mapfile -t ALL_FILES < <(python3 -c "
import os, json
from huggingface_hub import list_repo_tree
files = list_repo_tree('${REPO}', path='public-merged/${DATE_PATH}', recursive=False)
print('\n'.join([f['path'] for f in files if f['type']=='file']))
")
fi

# Deterministic shard assignment by path hash
mapfile -t MY_FILES < <(
  for f in "${ALL_FILES[@]}"; do
    hash=$(echo -n "$f" | md5sum | awk '{print $1}')
    bucket=$(( 0x${hash:0:8} % TOTAL_SHARDS ))
    if [[ $bucket -eq $SHARD_ID ]]; then
      echo "$f"
    fi
  done
)

echo "Processing ${#MY_FILES[@]} files for shard ${SHARD_ID}"

# Process files via CDN download + schema projection
python3 bin/process_shard.py "${MY_FILES[@]}" --outfile "${OUTFILE}" --repo "${REPO}"

echo "[$(date)] Shard ${SHARD_ID} wrote ${OUTFILE}"
```
