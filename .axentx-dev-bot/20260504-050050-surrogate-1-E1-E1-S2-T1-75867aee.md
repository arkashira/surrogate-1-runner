import java.util.List;
import java.util.Random;

 * REST controller that provides cost trend data for the dashboard.

 * Endpoints:
 * - GET /dashboard/cost-trend/historical - Returns past cost data (last 30 days)
 * - GET /dashboard/cost-trend/forecast   - Returns projected future costs (next 15 days)
 * - GET /dashboard/cost-trend           - Returns both historical and forecast data
 *
 * Real-time updates are pushed to WebSocket subscribers on /topic/cost-trend
 * at the configured interval (default: 30 seconds).
 *
 * In production, data would be sourced from a persistent store or forecasting service.
 * This implementation uses mock data to satisfy acceptance criteria.
 */

@RequestMapping("/dashboard/cost-trend")

    private static final int DEFAULT_SEED = 42;

    private final SimpMessagingTemplate messagingTemplate;
    private final Random random;
    
    @Value("${dashboard.cost-trend.update-interval-ms:30000}")
    private long updateIntervalMs;

    private final List<DataPoint> historicalData = new ArrayList<>();
    private final List<DataPoint> forecastData = new ArrayList<>();

    public CostTrendDashboard(SimpMessagingTemplate messagingTemplate) {
        this.messagingTemplate = messagingTemplate;
        this.random = new Random(DEFAULT_SEED);
        initializeData(); // Generate initial data on startup
    }

     * Returns historical cost data for the past 30 days.

        return new ArrayList<>(historicalData);
    }

    @GetMapping("/forecast")
