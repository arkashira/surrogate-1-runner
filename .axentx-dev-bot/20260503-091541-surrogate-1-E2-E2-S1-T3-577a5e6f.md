// /opt/axentx/sorrogate-1/src/lib/sla.ts

export type SLAStatus = 'on track' | 'at risk' | 'breached';

export interface SLAResult {
  requestId: string;
  status: string;
  createdAt: string;
  updatedAt: string;
  elapsedHours: number;
  targetHours: number;
  statusLabel: SLAStatus;
  isBreached: boolean;
  percentComplete: number;
}

function getTargetHours(status: string): number {
  switch (status) {
    case 'closed':
      return 0;
    case 'resolved':
      return 12;
    case 'in_progress':
    case 'open':
    default:
      return 24;
  }
}

export function calculateSLA(request: {
  id: string;
  status: string;
  createdAt: string | Date;
  updatedAt?: string | Date;
}): SLAResult {
  const now = new Date();
  const created = new Date(request.createdAt);
  const updated = request.updatedAt ? new Date(request.updatedAt) : now;

  // Canonical start is createdAt; updatedAt is for record-keeping only
  const start = created;
  const elapsedMs = Math.max(0, now.getTime() - start.getTime());
  const elapsedHours = elapsedMs / (1000 * 60 * 60);

  const targetHours = getTargetHours(request.status);
  const isBreached = targetHours > 0 && elapsedHours > targetHours;
  const atRiskThreshold = targetHours * 0.8;
  const isAtRisk = !isBreached && targetHours > 0 && elapsedHours >= atRiskThreshold;

  let statusLabel: SLAStatus = 'on track';
  if (isBreached) {
    statusLabel = 'breached';
  } else if (isAtRisk) {
    statusLabel = 'at risk';
  }

  const percentComplete =
    targetHours > 0 ? Math.min(100, (elapsedHours / targetHours) * 100) : 100;

  return {
    requestId: request.id,
    status: request.status,
    createdAt: created.toISOString(),
    updatedAt: updated.toISOString(),
    elapsedHours: Number(elapsedHours.toFixed(2)),
    targetHours,
    statusLabel,
    isBreached,
    percentComplete: Number(percentComplete.toFixed(2)),
  };
}
