// /opt/axentx/surrogate-1/src/lib/sla.ts
export type RequestStatus = 'open' | 'in_progress' | 'resolved' | 'closed';

export interface SlaConfig {
  openHours: number;
  inProgressTotalHours: number;   // total from createdAt
  resolvedTotalHours: number;     // total from createdAt
}

export const DEFAULT_SLA_CONFIG: SlaConfig = {
  openHours: 24,
  inProgressTotalHours: 48,
  resolvedTotalHours: 72,
};

export type SlaStatus = 'on track' | 'at risk' | 'breached' | 'completed';

export interface SlaState {
  elapsedHours: number;
  targetHours: number;
  remainingHours: number;
  status: SlaStatus;
  isBreached: boolean;
  isAtRisk: boolean;
}

export function calculateSla(
  createdAt: string | Date,
  status: RequestStatus,
  config: SlaConfig = DEFAULT_SLA_CONFIG
): SlaState {
  const now = new Date();
  const created = new Date(createdAt);
  const elapsedHours = Math.max(0, (now.getTime() - created.getTime()) / (1000 * 60 * 60));

  // Closed: SLA no longer active
  if (status === 'closed') {
    return {
      elapsedHours,
      targetHours: 0,
      remainingHours: 0,
      status: 'completed',
      isBreached: false,
      isAtRisk: false,
    };
  }

  // Determine active target based on lifecycle stage
  let targetHours: number;
  switch (status) {
    case 'open':
      targetHours = config.openHours;
      break;
    case 'in_progress':
      targetHours = config.inProgressTotalHours;
      break;
    case 'resolved':
      targetHours = config.resolvedTotalHours;
      break;
    default:
      targetHours = config.openHours;
  }

  const remainingHours = Math.max(0, targetHours - elapsedHours);

  // Thresholds (favor correctness + simple policy)
  let statusLabel: SlaStatus;
  if (elapsedHours <= 0.8 * targetHours) {
    statusLabel = 'on track';
  } else if (elapsedHours <= targetHours) {
    statusLabel = 'at risk';
  } else {
    statusLabel = 'breached';
  }

  return {
    elapsedHours,
    targetHours,
    remainingHours,
    status: statusLabel,
    isBreached: statusLabel === 'breached',
    isAtRisk: statusLabel === 'at risk',
  };
}

export function formatHours(value: number): string {
  if (Number.isNaN(value) || !Number.isFinite(value)) return '—';
  if (value >= 24) {
    const days = Math.floor(value / 24);
    const hours = Math.round(value % 24);
    return hours > 0 ? `${days}d ${hours}h` : `${days}d`;
  }
  return `${Math.round(value * 10) / 10}h`;
}
