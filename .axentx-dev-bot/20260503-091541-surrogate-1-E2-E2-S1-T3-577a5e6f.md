// src/lib/sla.ts
export type SlaStatus = 'on track' | 'at risk' | 'breached';

export interface SlaResult {
  elapsedHours: number;
  targetHours: number;
  timeRemainingHours: number;
  progress: number; // 0..1+ (1 = exactly at target)
  status: SlaStatus;
  isBreached: boolean;
}

/**
 * Calculate SLA status for a request.
 *
 * Rules:
 * - targetHours must be > 0 (defaults to 72)
 * - elapsed = max(0, now - createdAt) in hours
 * - progress = elapsed / targetHours
 * - status:
 *     progress <= 0.8  => 'on track'
 *     progress <= 1.0  => 'at risk'
 *     else             => 'breached'
 * - isBreached = progress > 1
 */
export function getSlaStatus(
  createdAt: string | Date,
  targetHours: number = 72
): SlaResult {
  const target = Number(targetHours);
  if (!Number.isFinite(target) || target <= 0) {
    throw new Error('targetHours must be a positive number');
  }

  const created = new Date(createdAt);
  if (Number.isNaN(created.getTime())) {
    throw new Error('Invalid createdAt date');
  }

  const now = new Date();
  const elapsedMs = Math.max(0, now.getTime() - created.getTime());
  const elapsedHours = elapsedMs / (1000 * 60 * 60);

  const progress = elapsedHours / target;
  const timeRemainingHours = Math.max(0, target - elapsedHours);

  let status: SlaStatus;
  if (progress <= 0.8) {
    status = 'on track';
  } else if (progress <= 1) {
    status = 'at risk';
  } else {
    status = 'breached';
  }

  return {
    elapsedHours: Number(elapsedHours.toFixed(2)),
    targetHours: Number(target.toFixed(2)),
    timeRemainingHours: Number(timeRemainingHours.toFixed(2)),
    progress: Number(progress.toFixed(3)),
    status,
    isBreached: progress > 1,
  };
}
