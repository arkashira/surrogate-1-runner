import { Request } from '../models/request';
import { SlaStatus } from '../lib/sla';

const getRecentRequests = async (limit: number = 10): Promise<Request[]> => {
    const recentRequests = await Request.find().sort({ updatedAt: -1 }).limit(limit);
    return recentRequests;
};

const getCounts = async (): Promise<{ total: number; onTrack: number; atRisk: number; breached: number }> => {
    const recentRequests = await getRecentRequests();
    const counts: { total: number; onTrack: number; atRisk: number; breached: number } = {
        total: recentRequests.length,
        onTrack: 0,
        atRisk: 0,
        breached: 0,
    };

    recentRequests.forEach((request) => {
        const slaStatus = SlaStatus.getStatus(request.slaStatus);
        if (slaStatus === 'onTrack') {
            counts.onTrack++;
        } else if (slaStatus === 'atRisk') {
            counts.atRisk++;
        } else if (slaStatus === 'breached') {
            counts.breached++;
        }
    });

    return counts;
};

export { getCounts };
```
