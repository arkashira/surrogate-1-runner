import { Schema, model, models } from 'mongoose';

export type RequestStatus =
  | 'pending'
  | 'processing'
  | 'completed'
  | 'failed'
  | 'cancelled';

export interface IRequestStatusHistory {
  status: RequestStatus;
  note?: string;
  actor: string; // userId or 'system'
  changedAt: Date;
}

export interface IRequest {
  _id: string;
  slug: string;
  status: RequestStatus;
  requestUpdatedAt: Date;
  statusHistory: IRequestStatusHistory[];
  // preserve existing timestamps if needed
  createdAt: Date;
  updatedAt: Date;
}

const RequestStatusHistorySchema = new Schema<IRequestStatusHistory>({
  status: {
    type: String,
    required: true,
    enum: ['pending', 'processing', 'completed', 'failed', 'cancelled']
  },
  note: { type: String, required: false },
  actor: { type: String, required: true },
  changedAt: { type: Date, required: true, default: () => new Date() }
});

const RequestSchema = new Schema<IRequest>(
  {
    slug: { type: String, required: true, unique: true },
    status: {
      type: String,
      required: true,
      enum: ['pending', 'processing', 'completed', 'failed', 'cancelled'],
      default: 'pending'
    },
    requestUpdatedAt: { type: Date, required: true, default: () => new Date() },
    statusHistory: { type: [RequestStatusHistorySchema], default: [] }
  },
  { timestamps: true }
);

// Index for fast chronological audit reads
RequestSchema.index({ 'statusHistory.changedAt': 1 });

export const RequestModel = models.Request || model<IRequest>('Request', RequestSchema);
