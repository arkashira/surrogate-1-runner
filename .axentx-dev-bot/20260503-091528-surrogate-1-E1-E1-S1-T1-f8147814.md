// /opt/axentx/surrogate-1/src/search/indexer.ts
import * as fs from 'fs';
import * as path from 'path';
import { SearchSchema } from './schema';

interface SearchIndex {
  [key: string]: SearchSchema;
}

class SearchIndexer {
  private index: SearchIndex = {};

  constructor(private localPaths: string[]) {}

  crawl(): void {
    this.localPaths.forEach((localPath) => {
      const files = fs.readdirSync(localPath);
      files.forEach((file) => {
        const filePath = path.join(localPath, file);
        const fileStat = fs.statSync(filePath);
        if (fileStat.isFile()) {
          const fileContent = fs.readFileSync(filePath, 'utf8');
          const searchSchema: SearchSchema = {
            title: file,
            source: 'local',
            snippet: fileContent.substring(0, 100),
            lastModified: fileStat.mtime,
          };
          this.index[file] = searchSchema;
        }
      });
    });
  }

  search(query: string): SearchSchema[] {
    const results: SearchSchema[] = [];
    Object.keys(this.index).forEach((key) => {
      if (this.index[key].title.includes(query)) {
        results.push(this.index[key]);
      }
    });
    return results;
  }
}

export { SearchIndexer };
```
