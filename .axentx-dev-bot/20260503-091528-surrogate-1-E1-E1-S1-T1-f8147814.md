// /opt/axentx/surrogate-1/src/search/indexer.ts
import * as fs from 'fs';
import * as path from 'path';
import { LibraryConfig } from '../config/library';

interface SearchIndex {
  [key: string]: {
    title: string;
    source: string;
    snippet: string;
    lastModified: Date;
  };
}

class SearchIndexer {
  private index: SearchIndex = {};
  private libraryConfig: LibraryConfig;

  constructor(libraryConfig: LibraryConfig) {
    this.libraryConfig = libraryConfig;
  }

  async crawlLocalPaths(): Promise<void> {
    const localPaths = this.libraryConfig.localPaths;
    for (const localPath of localPaths) {
      const files = await fs.promises.readdir(localPath);
      for (const file of files) {
        const filePath = path.join(localPath, file);
        const fileStat = await fs.promises.stat(filePath);
        if (fileStat.isFile()) {
          const fileContent = await fs.promises.readFile(filePath, 'utf8');
          const title = file;
          const source = 'local';
          const snippet = fileContent.substring(0, 100);
          const lastModified = fileStat.mtime;
          this.index[filePath] = { title, source, snippet, lastModified };
        }
      }
    }
  }

  async search(query: string): Promise<any[]> {
    const results = [];
    for (const filePath in this.index) {
      const file = this.index[filePath];
      if (file.title.includes(query) || file.snippet.includes(query)) {
        results.push({
          title: file.title,
          source: file.source,
          snippet: file.snippet,
          lastModified: file.lastModified,
        });
      }
    }
    return results;
  }
}

export { SearchIndexer };
