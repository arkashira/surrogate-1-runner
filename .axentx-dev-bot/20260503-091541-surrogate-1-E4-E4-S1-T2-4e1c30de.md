import json
import time
import threading
from dataclasses import dataclass, asdict
from typing import Optional, List
from pathlib import Path

DB_PATH = Path(__file__).parent.parent.parent / "outbox.db"

@dataclass
class OutboxEvent:
    id: Optional[int]
    event_type: str            # "status_change"
    payload: dict              # {request_title, old_status, new_status, public_url, sla_state}
    created_at: float
    attempts: int = 0
    last_attempt_at: Optional[float] = None
    delivered: bool = False
    error: Optional[str] = None

class Outbox:
    _local = threading.local()

    @classmethod
    def _conn(cls):
        if not hasattr(cls._local, "conn"):
            cls._local.conn = sqlite3.connect(str(DB_PATH), check_same_thread=False)
            cls._local.conn.row_factory = sqlite3.Row
            cls._init_db(cls._local.conn)
        return cls._local.conn

    @staticmethod
    def _init_db(conn: sqlite3.Connection):
        conn.execute("""
            CREATE TABLE IF NOT EXISTS outbox (
                id              INTEGER PRIMARY KEY AUTOINCREMENT,
                event_type      TEXT NOT NULL,
                payload         TEXT NOT NULL,
                created_at      REAL NOT NULL,
                attempts        INTEGER DEFAULT 0,
                last_attempt_at REAL,
                delivered       INTEGER DEFAULT 0,
                error           TEXT
            )
        """)
