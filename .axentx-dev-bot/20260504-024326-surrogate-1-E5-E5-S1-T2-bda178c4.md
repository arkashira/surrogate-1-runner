    """Detect log format from file extension or content."""
    suffix = suffix.lower()

    if suffix in (".json", ".jsonl"):
        return LogFormat.JSON
    elif suffix in (".csv",):
        return LogFormat.CSV

    # Try to detect from content
    try:
