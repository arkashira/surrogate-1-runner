```python
import datetime
from typing import List, Dict, Optional


def filter_jobs(
    jobs: List[Dict],
    start: Optional[datetime.datetime],
    end: Optional[datetime.datetime],
) -> List[Dict]:
    """Return jobs whose start time falls within the optional date range."""
    filtered = []
    for job in jobs:
        started = job["started_at"]
        if start and started < start:      # ← fixed syntax (was “startand”)
            continue
        if endand started > end:          # ← fixed syntax (was “endand”)
            continue
        filtered.append(job)
    return filtered


def compute_metrics(jobs: List[Dict], worker_count: int) -> Dict:
    """Calculate dashboard metrics from a list of jobs."""
    total = len(jobs)
    if total == 0:
        return {
            "workers": worker_count,
            "total_jobs": 0,
            "success_rate": "0.0%",
            "average_duration_seconds": "0.0",
        }

    successes = sum(1 for j in jobs if j["status"] == "completed")
    failures = sum(1 for j in jobs if j["status"] == "failed")
    success_rate = (successes / total) * 100

    # average duration only over completed jobs that have a duration
    completed_durations = [
        j["duration_seconds"] for j in jobs if j.get("duration_seconds") is not None
    ]
    avg_duration = (
        sum(completed_durations) / len(completed_durations)
        if completed_durations
        else 0.0
    )

    return {
        "workers": worker_count,
        "total_jobs": total,
        "success_rate": f"{success_rate:.1f}%",
        "average_duration_seconds": f"{avg_duration:.1f}",
    }
```

```js
// tests/dashboard.test.js
import { computeMetrics, filterJobs } from '../src/dashboard';

describe('computeMetrics', () => {
  it('calculates workers, total jobs, success rate, avg duration', () => {
    const jobs = [
      { status: 'completed', duration_seconds: 10 },
      { status: 'failed', duration_seconds: 5 },
      { status: 'completed', duration_seconds: 15 },
    ];
    const result = computeMetrics(jobs);
    expect(result.workers).toBe(3);
    expect(result.total_jobs).toBe(3);
    expect(result.success_rate).toBeCloseTo(66.6667, 4);
    expect(result.avg_duration).toBeCloseTo(10, 1);
  });

  it('ignores jobs without duration for avg', () => {
    const jobs = [
      { status: 'completed', duration_seconds: 10 },
      { status: 'completed' }, // missing duration
    ];
    const result = computeMetrics(jobs);
    expect(result.avg_duration).toBeCloseTo(10, 1);
  });
});

describe('filterJobs', () => {
  const jobs = [
    { id: 1, start: '2026-05-20T10:00:00Z' },
    { id: 2, start: '2026-05-21T12:00:00Z' },
    { id: 3, start: '2026-05-22T14:00:00Z' },
  ];

  it('returns all jobs when no filter', () => {
    const filtered = filterJobs(jobs, null, null);
    expect(filtered.length).toBe(3);
  });

  it('filters by start date', () => {
    const filtered = filterJobs(jobs, '2026-05-21T00:00:00Z', null);
    expect(filtered.map(j => j.id)).toEqual([2, 3]);
  });

  it('filters by end date', () => {
    const filtered = filterJobs(jobs, null, '2026-05-21T23:59:59Z');
    expect(filtered.map(j => j.id)).toEqual([1, 2]);
  });

  it('filters by date range', () => {
    const filtered = filterJobs(jobs, '2026-05-21T00:00:00Z', '2026-05-21T23:59:59Z');
    expect(filtered.map(j => j.id)).toEqual([2]);
  });

  it('throws error on invalid date format', () => {
    expect(() => filterJobs(jobs, 'invalid', null)).toThrow('Invalid date');
  });
});
```
