// extract.test.ts
import { extractDocument } from '../services/extract';
import { FILE_SIZE_LIMIT } from '../constants';

describe('extractDocument', () => {
  test('extracts metadata and fullText from valid PDF buffer', async () => {
    const pdfBuffer = mockPdfBuffer({ title: 'T', author: 'A', year: 2020, fullText: 'hello world' });
    const result = await extractDocument(pdfBuffer, 'doc.pdf');
    expect(result.title).toBe('T');
    expect(result.author).toBe('A');
    expect(result.publicationYear).toBe(2020);
    expect(result.fullText).toContain('hello world');
  });

  test('extracts metadata and fullText from valid ePub buffer', async () => {
    const epubBuffer = mockEpubBuffer({ title: 'E', author: 'B', year: 2021, fullText: 'epub text' });
    const result = await extractDocument(epubBuffer, 'book.epub');
    expect(result.title).toBe('E');
    expect(result.author).toBe('B');
    expect(result.publicationYear).toBe(2021);
    expect(result.fullText).toContain('epub text');
  });

  test('falls back to filename parsing when metadata missing', async () => {
    const buffer = mockPdfBuffer({ title: null, author: null, year: null, fullText: 'content' });
    const result = await extractDocument(buffer, '2019 - Jane Doe - Title.pdf');
    expect(result.title).toBe('Title');
    expect(result.author).toBe('Jane Doe');
    expect(result.publicationYear).toBe(2019);
    expect(result.fullText).toBe('content');
  });

  test('rejects when buffer exceeds 100MB', async () => {
    const oversized = Buffer.alloc(FILE_SIZE_LIMIT + 1);
    await expect(extractDocument(oversized, 'big.pdf')).rejects.toThrow('FileSizeLimitExceeded');
  });

  test('rejects with clear error for invalid/corrupted buffer', async () => {
    const invalid = Buffer.from('not a pdf or epub');
    await expect(extractDocument(invalid, 'bad.pdf')).rejects.toThrow(/unsupported|corrupt|parse/i);
  });
});
