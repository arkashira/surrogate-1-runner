# File: /opt/axentx/surrogate-1/src/routes/requests.py
from flask import Blueprint, request, jsonify
from src.schemas.request import RequestSchema
import uuid
import time

requests_bp = Blueprint('requests', __name__)

# In-memory storage for demo purposes - would use DB in production
requests_storage = {}

@requests_bp.route('/requests', methods=['POST'])
def create_request():
    try:
        data = request.get_json()
        
        # Validate required fields
        required_fields = ['type', 'title', 'owner']
        for field in required_fields:
            if field not in data or not data[field]:
                return jsonify({
                    'error': f'Missing required field: {field}'
                }), 400
        
        # Create request object
        request_id = str(uuid.uuid4())
        slug = f"req-{request_id[:8]}"
        
        new_request = {
            'id': request_id,
            'slug': slug,
            'type': data['type'],
            'title': data['title'],
            'description': data.get('description', ''),
            'owner': data['owner'],
            'sla_target': data.get('sla_target'),
            'status': 'submitted',
            'created_at': int(time.time())
        }
        
        requests_storage[request_id] = new_request
        
        return jsonify({
            'id': request_id,
            'slug': slug
        }), 201
        
    except Exception as e:
        return jsonify({'error': 'Internal server error'}), 500

@requests_bp.route('/requests', methods=['GET'])
def list_requests():
    # Return minimal request fields as requested
    minimal_requests = []
    for req in requests_storage.values():
        minimal_requests.append({
            'id': req['id'],
            'slug': req['slug'],
            'type': req['type'],
            'title': req['title'],
            'owner': req['owner'],
            'status': req['status'],
            'sla_target': req.get('sla_target')
        })
    
    return jsonify(minimal_requests), 200
```
