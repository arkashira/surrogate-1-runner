from .workflow import Workflow

class ExecutionStatus(Enum):
    PENDING = "pending"
    RUNNING = "running"
    RETRYING = "retrying"
    COMPLETED = "completed"
    FAILED = "failed"
    FAILED_PERMANENT = "failed_permanent"

@dataclass
class LogEntry:
    id: str
    execution_id: str
    timestamp: str
    level: str  # 'debug', 'info', 'warn', 'error'
    message: str
    metadata: Optional[Dict[str, Any]] = None

@dataclass
class ExecutionRecord:
    id: str
    workflow_id: str
    status: ExecutionStatus
    input_params: Dict[str, Any]
    output: Optional[Dict[str, Any]] = None
    logs: List[LogEntry] = field(default_factory=list)
    retry_count: int = 0
    max_retries: int = 3

    completed_at: Optional[str] = None
    created_at: str = field(default_factory=lambda: datetime.now(timezone.utc).isoformat())
    updated_at: str = field(default_factory=lambda: datetime.now(timezone.utc).isoformat())

class WorkflowExecutor:
    """
    Orchestrates workflow execution with validation, logging, and retry logic.
    """
    
    def __init__(
        self, 
        workflow: Workflow,
        max_retries: int = 3,
        initial_delay: float = 1.0,
        max_delay: float = 30.0
    ):
        self.workflow = workflow
        self.max_retries = max_retries
        self.initial_delay = initial_delay
        self.max_delay = max_delay
        self.logger = logging.getLogger(__name__)
        self._execution_record: Optional[ExecutionRecord] = None

    def execute(self, input_params: Dict[str, Any]) -> Dict[str, Any]:
        """
        Execute workflow with full lifecycle management.
        
        Args:
            input_params: Parameters matching workflow schema
            
        Returns:
            Workflow execution result
            
        Raises:
            ValidationError: If input doesn't match schema
            WorkflowExecutionError: If execution fails after all retries
        """
        # Validate input before any execution attempt
        self._validate_input(input_params)
        
        # Initialize execution record
        execution_id = self._generate_execution_id()
        self._execution_record = ExecutionRecord(
            id=execution_id,
            workflow_id=self.workflow.name,
            status=ExecutionStatus.PENDING,
            input_params=input_params,
            max_retries=self.max_retries
        )
        
        self._log(execution_id, "info", f"Starting workflow execution: {self.workflow.name}")
        
        # Execute with retry logic
        result = self._execute_with_retry(input_params, execution_id)
        
        self._execution_record.output = result
        self._execution_record.status = ExecutionStatus.COMPLETED
        self._execution_record.completed_at = datetime.now(timezone.utc).isoformat()
        self._log(execution_id, "info", "Workflow execution completed successfully")
        
        return result

    def _validate_input(self, input_params: Dict[str, Any]) -> None:
        """Validate input against workflow schema."""
        is_valid, error_details = self.workflow.schema.validate(input_params)
        
        if not is_valid:
            self._log(
                "validation",
                "error",
                f"Input validation failed: {error_details}",
                {"input_params": input_params}
            )
