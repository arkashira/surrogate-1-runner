import { v4 as uuidv4 } from 'uuid';

export type RequestType = string; // e.g. 'incident', 'change', 'question'

export interface Team {
  id: string;
  name: string;
  defaultSlaHoursPerRequestType: Record<RequestType, number>;
  createdAt: string;
  updatedAt: string;
}

export interface CreateTeamInput {
  name: string;
  defaultSlaHoursPerRequestType?: Record<RequestType, number>;
}

export interface UpdateTeamInput {
  name?: string;
  defaultSlaHoursPerRequestType?: Record<RequestType, number>;
}

// Lightweight in-memory store (swap for DB adapter later)
export class TeamService {
  private teams: Map<string, Team> = new Map();

  create(input: CreateTeamInput): Team {
    const now = new Date().toISOString();
    const team: Team = {
      id: uuidv4(),
      name: input.name,
      defaultSlaHoursPerRequestType: input.defaultSlaHoursPerRequestType || {},
      createdAt: now,
      updatedAt: now,
    };
    this.teams.set(team.id, team);
    return team;
  }

  findById(id: string): Team | undefined {
    return this.teams.get(id);
  }

  findAll(): Team[] {
    return Array.from(this.teams.values()).sort((a, b) => a.name.localeCompare(b.name));
  }

  update(id: string, input: UpdateTeamInput): Team | null {
    const existing = this.teams.get(id);
    if (!existing) return null;
    const updated: Team = {
      ...existing,
      ...(input.name !== undefined && { name: input.name }),
      ...(input.defaultSlaHoursPerRequestType !== undefined && {
        defaultSlaHoursPerRequestType: input.defaultSlaHoursPerRequestType,
      }),
      updatedAt: new Date().toISOString(),
    };
    this.teams.set(id, updated);
    return updated;
  }

  patch(id: string, input: Partial<UpdateTeamInput>): Team | null {
    const existing = this.teams.get(id);
    if (!existing) return null;
    const merged = {
      ...existing,
      ...(input.name !== undefined && { name: input.name }),
      ...(input.defaultSlaHoursPerRequestType !== undefined && {
        defaultSlaHoursPerRequestType: {
          ...existing.defaultSlaHoursPerRequestType,
          ...input.defaultSlaHoursPerRequestType,
        },
      }),
      updatedAt: new Date().toISOString(),
    };
    this.teams.set(id, merged);
    return merged;
  }
