// src/models/team.ts
export type RequestType = 'incident' | 'service_request' | 'change' | 'problem' | 'question';

export interface TeamSettings {
  id: string;            // team id
  name: string;
  defaultSlaHoursByRequestType: Partial<Record<RequestType, number>>;
  createdAt: string;
  updatedAt: string;
}

export interface TeamSummary {
  teamId: string;
  countsByState: Record<string, number>;
  slaHealth: {
    requestType: RequestType;
    total: number;
    breached: number;
    atRisk: number;     // >75% of SLA elapsed
    ok: number;
  }[];
}

// In-memory store (replace with DB adapter in production)
const store = new Map<string, TeamSettings>();

export const TeamSettingsModel = {
  create(payload: Omit<TeamSettings, 'id' | 'createdAt' | 'updatedAt'>): TeamSettings {
    const now = new Date().toISOString();
    const item: TeamSettings = {
      ...payload,
      id: `team_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`,
      createdAt: now,
      updatedAt: now,
    };
    store.set(item.id, item);
    return item;
  },

  findById(id: string): TeamSettings | undefined {
    return store.get(id);
  },

  update(id: string, patch: Partial<Omit<TeamSettings, 'id' | 'createdAt'>>): TeamSettings | undefined {
    const existing = store.get(id);
    if (!existing) return undefined;
    const updated: TeamSettings = {
      ...existing,
      ...patch,
      updatedAt: new Date().toISOString(),
    };
    store.set(id, updated);
    return updated;
  },

  delete(id: string): boolean {
    return store.delete(id);
  },

  list(): TeamSettings[] {
    return Array.from(store.values());
  },

  getDefaultSlaHours(teamId: string, requestType: RequestType): number | undefined {
    const team = store.get(teamId);
    if (!team) return undefined;
    return team.defaultSlaHoursByRequestType[requestType];
  },
};
