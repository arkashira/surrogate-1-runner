# Audio Guard Calibration System

## Overview
This system provides a comprehensive audio calibration solution with real-time monitoring and control capabilities. The implementation combines React components with Redux state management to create an intuitive user interface for audio device calibration.

## Core Components

### 1. Calibration Interface (`CalibrationWizard`)
A React component that provides a user-friendly interface for audio calibration:

```jsx
import React, { useState } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { startCalibration, adjustGain } from './audio_actions';

const CalibrationWizard = () => {
  const dispatch = useDispatch();
  const [inputLevel, setInputLevel] = useState(-6);
  const [isCalibrating, setIsCalibrating] = useState(false);
  const [calibrationStatus, setCalibrationStatus] = useState('idle');

  const handleStartCalibration = () => {
    setIsCalibrating(true);
    setCalibrationStatus('calibrating');
    dispatch(startCalibration());
  };

  const handleAdjustGain = (newLevel) => {
    if (!isCalibrating) return;
    
    setInputLevel(newLevel);
    dispatch(adjustGain(newLevel));
    setCalibrationStatus('adjusting');
  };

  const handleStopCalibration = () => {
    setIsCalibrating(false);
    setCalibrationStatus('completed');
  };

  return (
    <div className="calibration-wizard">
      <h1>Audio Device Calibration</h1>
      
      <div className="status-indicator">
        Status: <span className={`status ${calibrationStatus}`}>{calibrationStatus}</span>
      </div>
      
      <div className="current-level">
        Current Input Level: {inputLevel} dBFS
      </div>
      
      <div className="controls">
        {!isCalibrating ? (
          <button 
            onClick={handleStartCalibration} 
            className="start-button"
          >
            Start Calibration
          </button>
        ) : (
          <>
            <button 
              onClick={() => handleAdjustGain(inputLevel + 1)} 
              className="gain-control"
              disabled={inputLevel >= 0}
            >
              Increase Gain (+1dB)
            </button>
            
            <button 
              onClick={() => handleAdjustGain(inputLevel - 1)} 
              className="gain-control"
              disabled={inputLevel <= -40}
            >
              Decrease Gain (-1dB)
            </button>
            
            <button 
              onClick={handleStopCalibration}
              className="stop-button"
            >
              Complete Calibration
            </button>
          </>
        )}
      </div>
    </div>
  );
};

export default CalibrationWizard;
```
