```typescript
export interface SLASummary {
  created_at: string;
  resolved_at: string | null;
  target_hours: number;
  target_ms: number;
  duration_ms: number | null;
  breached: boolean;
}

/**
 * Compute SLA summary for a request.
 * If request is unresolved, duration is null and breached is false.
 */
export function computeSLASummary(request: {
  created_at: string | Date;
  resolved_at: string | Date | null | undefined;
  sla_target_hours?: number | null;
}): SLASummary {
  const created = new Date(request.created_at).getTime();
  const resolved = request.resolved_at ? new Date(request.resolved_at).getTime() : null;
  const targetHours = request.sla_target_hours ?? 72; // Default SLA target is 72 hours
  const targetMs = targetHours * 60 * 60 * 1000;

  const durationMs = resolved !== null ? resolved - created : null;
  const breached = durationMs !== null && durationMs > targetMs;

  return {
    created_at: new Date(created).toISOString(),
    resolved_at: resolved ? new Date(resolved).toISOString() : null,
    target_hours: targetHours,
    target_ms: targetMs,
    duration_ms: durationMs,
    breached,
  };
}
```
