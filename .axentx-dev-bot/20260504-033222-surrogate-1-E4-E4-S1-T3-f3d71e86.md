Daily high-impact message digest generator.

This script produces a concise daily summary (≤10 items) ordered by confidence score,
with a headline summarizing overall status. It outputs both Markdown (for Slack/Teams)
and plain-text (for email) formats, completing within 30 seconds for typical 200-message loads.

    generate_digest.py --messages-path /path/to/messages.json

    confidence: float
    type: str = "normal"

    """Represents a high-impact item in the digest."""
    author: str
    timestamp: datetime
    snippet: str
    confidence: float
    source: str = "unknown"

class Digest:
    """Represents the complete daily digest."""
    headline: str
    items: List[DigestItem]
    generated_at: datetime
    message_count: int

    def to_markdown(self) -> str:
        """Render digest in Markdown format for Slack/Teams."""
        lines = [
            f"## 📊 Daily Digest - {self.generated_at.strftime('%Y-%m-%d')}",
            "",
            f"**{self.headline}**",
            "",

            "",
            "---",
            ""
        ]

                f"  - 👤 {item.author} • 🕐 {ts} • 🎯 {item.confidence:.0%}",
                ""
            ])
