const fs = require('fs');
const path = require('path');

/**
 * DocumentationService
 * 
 * Generates human-readable markdown documentation for workflow definitions.
 * Reads workflow files from a configured directory and produces documentation
 * in Markdown format suitable for GitHub or plain text viewing.
 * 
 * @example
 * const service = new DocumentationService('./workflows');
 * const allDocs = service.getAllDocs();
 * const specificDoc = service.getDocForWorkflow('userOnboarding');
 */
class DocumentationService {
  /**
   * @param {Object} options - Configuration options
   * @param {string} [options.workflowsDir] - Directory containing workflow definition files
   * @param {boolean} [options.enableCache=true] - Whether to cache loaded workflows
   */
  constructor(options = {}) {
    const {
      workflowsDir = path.join(__dirname, '..', 'workflows'),
      enableCache = true
    } = options;
    
    this.workflowsDir = workflowsDir;
    this.enableCache = enableCache;
    this._workflowCache = null;
  }

  /**
   * Validates that the workflows directory exists
   * @throws {Error} If workflows directory doesn't exist
   */
  _validateWorkflowsDir() {
    if (!fs.existsSync(this.workflowsDir)) {
      throw new Error(`Workflows directory does not exist: ${this.workflowsDir}`);
    }
    
    const stats = fs.statSync(this.workflowsDir);
    if (!stats.isDirectory()) {
      throw new Error(`Workflows path is not a directory: ${this.workflowsDir}`);
    }
  }

  /**
   * Loads all workflow definitions from the configured directory
   * @private
   * @returns {Array<Object>} Array of workflow definition objects
   */

    if (this.enableCache && this._workflowCache !== null) {
      return this._workflowCache;
    }

    this._validateWorkflowsDir();

    const files = fs.readdirSync(this.workflowsDir)
      .filter((f) => f.endsWith('.js'))
      .sort(); // Sort for consistent ordering

    if (files.length === 0) {
      console.warn(`No workflow files found in ${this.workflowsDir}`);
      return [];
    }

    const workflows = files.map((file) => {

        this._validateWorkflow(wf, file);
        return wf;
      } catch (error) {
        throw new Error(`Failed to load workflow ${file}: ${error.message}`);
      }
    });

      this._workflowCache = workflows;
    }

    return workflows;
