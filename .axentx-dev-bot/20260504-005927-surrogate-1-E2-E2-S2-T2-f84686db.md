- **AC5** – When the selector matches no nodes, `CDPElementNotFoundError` is thrown.

import { CDPSessionManager, clickElement, CDPConnectionError, CDPElementNotFoundError } from '../src/cdp';

    const mockClient = { on: jest.fn(), send: jest.fn() };
    jest.spyOn(manager, 'createClient').mockResolvedValue(mockClient);
    await manager.connect();
    expect(manager.client).toBe(mockClient);
  });

  test('connect throws CDPConnectionError on failure', async () => {
    jest.spyOn(manager, 'createClient').mockRejectedValue(new Error('socket closed'));

describe('clickElement', () => {
  let manager: CDPSessionManager;
  let mockClient: any;

  beforeEach(() => {
    manager = new CDPSessionManager('ws://localhost:9222/devtools/browser/test-id');
    mockClient = {
      send: jest.fn(),
      on: jest.fn(),
    };
    manager.client = mockClient;
  });

  test('resolves selector to coordinates', async () => {
    mockClient.send.mockImplementation((method, params) => {
      if (method === 'DOM.getDocument') return { root: { nodeId: 1 } };
      if (method === 'DOM.querySelector') return { nodeId: 2 };
      if (method === 'DOM.getBoxModel') return { model: { content: [10, 20, 110, 120] } };
      return {};
    });

    const coords = await clickElement(manager, '#btn');
    expect(coords).toEqual({ x: 60, y: 70 });

      type: 'mousePressed',
      x: 60, y: 70,
    }));

  test('throws CDPElementNotFoundError when selector not found', async () => {

      return {};
