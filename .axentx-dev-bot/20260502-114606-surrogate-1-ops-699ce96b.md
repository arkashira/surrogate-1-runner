```python
# /opt/axentx/surrogate-1/ops_manifest.py
#!/usr/bin/env python3
"""
Generate manifest for one date folder.
Usage:
  HF_TOKEN=... python3 ops_manifest.py <date> <repo> [out_path]
Example:
  python3 ops_manifest.py 2026-04-29 my-org/surrogate-data manifest.json
"""
import json
import os
import sys
from pathlib import Path

from huggingface_hub import list_repo_tree


def main() -> None:
    if len(sys.argv) < 3:
        print("Usage: ops_manifest.py <date> <repo> [out_path]", file=sys.stderr)
        sys.exit(1)

    date_folder = sys.argv[1]            # e.g. 2026-04-29
    repo_id = sys.argv[2]                # e.g. my-org/surrogate-data
    out_path = Path(sys.argv[3] if len(sys.argv) > 3 else "manifest.json")

    # Non-recursive, single folder
    tree = list_repo_tree(repo_id=repo_id, path=date_folder, recursive=False)
    files = sorted(
        f.rfilename
        for f in tree
        if f.rfilename.endswith((".jsonl", ".parquet", ".json"))
    )

    manifest = {
        "repo_id": repo_id,
        "date": date_folder,
        "files": files,
    }

    out_path.parent.mkdir(parents=True, exist_ok=True)
    with out_path.open("w") as f:
        json.dump(manifest, f, indent=2)

    print(f"Wrote {len(files)} files to {out_path}")


if __name__ == "__main__":
    main()
```
