        items = list_repo_tree(repo, path=folder, recursive=False)
        for item in items:
            if item.path.endswith(('.parquet', '.jsonl', '.csv')):
                all_files.append(item.path)
    except Exception as e:
        print(f'Warning listing {folder}: {e}')

out_path = '${OUT_DIR}/file-list-${DATE}.json'
with open(out_path, 'w') as f:
    json.dump(all_files, f, indent=2)
print(f'Listed {len(all_files)} files -> {out_path}')
"

```bash
#!/usr/bin/env bash
# bin/dataset-enrich.sh (updated)
set -euo pipefail

REPO="axentx/surrogate-1-training-pairs"
MANIFEST="${MANIFEST:-manifests/file-list-$(date -u +%Y-%m-%d).json}"
WORKDIR="${WORKDIR:-./data}"
SHARD_ID="${SHARD_ID:-0}"
TOTAL_SHARDS="${TOTAL_SHARDS:-16}"

# CDN download - zero API calls
download_via_cdn() {
    local file_path="$1"
    local output_path="$2"
    local url="https://huggingface.co/datasets/${REPO}/resolve/main/${file_path}"
    curl -fsSL --retry 3 --retry-delay 5 -o "$output_path" "$url"
}

# Schema-agnostic projection to {prompt, response}
process_file() {
    local file_path="$1"
    local tmp_file=$(mktemp)
    
    download_via_cdn "$file_path" "$tmp_file"
    
    python3 -c "
import json, sys, os, pandas as pd, pyarrow as pa, pyarrow.parquet as pq
from pathlib import Path

tmp = Path('$tmp_file')
out_dir = Path('${WORKDIR}/processed')
out_dir.mkdir(parents=True, exist_ok=True)

try:
    table = pq.read_table(tmp)
    df = table.to_pandas()
except:
    df = pd.read_json(tmp, lines=True)

# Project to prompt/response - schema-agnostic
if 'prompt' in df.columns and 'response' in df.columns:
    pairs = df[['prompt', 'response']].dropna()
else:
    text_cols = [c for c in df.columns if df[c].dtype == 'object']
    if len(text_cols) >= 2:
        pairs = df[[text_cols[0], text_cols[1]]].dropna()
        pairs.columns = ['prompt', 'response']
    else:
        pairs = pd.DataFrame()

pairs['shard'] = ${SHARD_ID}
pairs['source_file'] = os.path.basename('$file_path')

out_file = out_dir / f'{Path(\"$file_path\").stem}_shard${SHARD_ID}.parquet'
pq.write_table(pa.Table.from_pandas(pairs), str(out_file))
print(f'Processed: {len(pairs)} pairs -> {out_file}')
"
    
    rm -f "$tmp_file"
}

# Deterministic hash-based sharding (no collisions across 16 runners)
if [[ -f "$MANIFEST" ]]; then
    python3 -c "
import json
with open('$MANIFEST') as f:
    files = json.load(f)

shard_id = int(${SHARD_ID})
total_shards = int(${TOTAL_SHARDS})

assigned = [f for f in files if hash(f) % total_shards == shard_id]
for f in assigned:
    print(f)
" | while read -r file_path; do
    echo "Processing shard ${SHARD_ID}: $file_path"
    process_file "$file_path"
done
else
    echo "ERROR: Manifest $MANIFEST not found. Generate with bin/list-public-files.sh"
    exit 1
fi
```
