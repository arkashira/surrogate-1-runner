Unified data model for marketing‑strategy input.

* Uses Pydantic for validation & (de)serialization.

* Adds timestamps and a UUID for each record.

import os
import uuid
from datetime import datetime

from typing import Dict, List, Any

from pydantic import BaseModel, Field, root_validator, validator

# --------------------------------------------------------------------------- #
# 1️⃣  Core Pydantic models
# --------------------------------------------------------------------------- #

class AudienceDemographic(BaseModel):
    age_range: str = Field(..., description="Target age range (e.g., '25‑34')")
    gender: str = Field(..., description="Target gender")
    location: str = Field(..., description="Target geographic region")
    interests: List[str] = Field(default_factory=list, description="Audience interests")

class BusinessMetrics(BaseModel):
    mrr: float = Field(..., ge=0, description="Monthly Recurring Revenue")
    cac: float = Field(..., ge=0, description="Customer Acquisition Cost")
    ltv: float = Field(..., ge=0, description="Customer Lifetime Value")

    @validator("mrr", "cac", "ltv")
    def _non_negative(cls, v, field):
        if v < 0:
            raise ValueError(f"{field.name.upper()} must be non‑negative")
        return v

class MarketingStrategyInput(BaseModel):
