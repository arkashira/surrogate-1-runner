#!/usr/bin/env python3
"""
surrogate-detect - CLI tool for detecting API signature drift during rollouts.

Usage:
    surrogate-detect --service <name> --version <version> [--baseline-dir <path>] [--signatures <path>]

Exit codes:
    0 - PASS (no drift detected)
    1 - FAIL (drift detected)
    2 - ERROR (invalid baseline or arguments)
"""

import argparse
import json
import sys
import time
from pathlib import Path
from typing import Any


def load_baseline(service: str, version: str, baseline_dir: Path) -> dict[str, Any]:
    """Load baseline signatures for the given service/version."""
    baseline_file = baseline_dir / service / f"{version}.json"
    if not baseline_file.exists():
        print(f"ERROR: No baseline found for {service}@{version} at {baseline_file}", file=sys.stderr)
        sys.exit(2)
    try:
        with open(baseline_file) as f:
            return json.load(f)
    except json.JSONDecodeError as e:
        print(f"ERROR: Invalid JSON in baseline file: {e}", file=sys.stderr)
        sys.exit(2)


def load_signatures(path: Path) -> list[dict[str, Any]]:
    """Load incoming signatures from stdin or file."""
    if path == Path("-"):
        data = sys.stdin.read()
        if not data.strip():
            return []
        try:
            return json.loads(data)
        except json.JSONDecodeError as e:
            print(f"ERROR: Invalid JSON in input: {e}", file=sys.stderr)
            sys.exit(2)
    try:
        with open(path) as f:
            return json.load(f)
    except (json.JSONDecodeError, FileNotFoundError) as e:
        print(f"ERROR: Cannot load signatures: {e}", file=sys.stderr)
        sys.exit(2)


def normalize_path(path: str) -> str:
    """Normalize API path for comparison (handle path parameters)."""
    # Convert /api/users/{id} to /api/users/:id for consistent comparison
    import re
    return re.sub(r'\{([^}]+)\}', r':\1', path)


def compare_signatures(baseline: dict[str, Any], signatures: list[dict[str, Any]]) -> tuple[bool, list[dict[str, Any]]]:
    """
    Compare incoming signatures against baseline.
    
    Returns:
        (is_pass, list of drift reports)
    """
    drifts = []
    
    # Build baseline endpoint map for quick lookup (normalize paths)
    baseline_endpoints = {}
    for sig in baseline.get("signatures", []):
        key = (sig.get("method", "").upper(), normalize_path(sig.get("", "/")))
        baseline_endpoints[key] = sig
    
    # Check for removed endpoints
    for sig in signatures:
        method = sig.get("method", "").upper()
        path = normalize_path(sig.get("path", "/"))
        key = (method, path)
        
        if key not in baseline_endpoints:
            drifts.append({
                "type": "NEW_ENDPOINT",
                "severity": "WARNING",
                "method": method,
                "path": path,
                "message": f"New endpoint detected: {method} {path}"
            })
    
    # Check baseline endpoints against incoming signatures
    baseline_paths = {normalize_path(s.get("path", "/")): s for s in baseline.get("signatures", [])}
    incoming_paths = {normalize_path(s.get("path", "/")): s for s in signatures}
    
    for path, baseline_sig in baseline_paths.items():
        method = baseline_sig.get("method", "").upper()
        key = (method, path)
        
        if key not in baseline_endpoints:
            continue
            
        if path not in incoming_paths:
            drifts.append({
                "type": "REMOVED_ENDPOINT",
                "severity": "ERROR",
                "method": method,
                "path": path,
                "message": f"Endpoint removed: {method} {path}"
            })
