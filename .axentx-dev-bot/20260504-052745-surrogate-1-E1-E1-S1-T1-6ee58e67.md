import static org.springframework.test.web.servlet.result.MockMvcResultMatchers.content;

import com.axentx.surrogate.api.LLMAgentApi;
import com.axentx.surrogate.api.dto.LLMRequest;
import com.axentx.surrogate.api.dto.LLMResponse;
import com.axentx.surrogate.service.LLMAgentService;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
import org.springframework.boot.test.mock.mockito.MockBean;
import org.springframework.http.MediaType;
import org.springframework.test.web.servlet.MockMvc;
import org.springframework.test.web.servlet.request.MockMvcRequestBuilders;

@WebMvcTest(LLMAgentApi.class)
class LLMAgentApiTest {

  @Autowired
  private MockMvc mockMvc;

  @MockBean
  private LLMAgentService llmAgentService;

  private LLMRequest validRequest;
  private LLMResponse validResponse;

  @BeforeEach
  void setUp() {
    validRequest = new LLMRequest("agentId", "requestData");
    validResponse = new LLMResponse("responseData");
  }

  @Test
  void testLLMAgentRequest_Success() throws Exception {
    when(llmAgentService.processRequest(any())).thenReturn(validResponse);

    mockMvc.perform(MockMvcRequestBuilders.post("/api/v1/llm-agent/{agentId}/request", validRequest.getAgentId())
        .contentType(MediaType.APPLICATION_JSON)
        .content(asJsonString(validRequest)))
        .andExpect(status().isOk())
        .andExpect(content().json(asJsonString(validResponse)));

    verify(llmAgentService, times(1)).processRequest(validRequest);
  }

  @Test
  void testLLMAgentRequest_Disabled() throws Exception {
    when(llmAgentService.processRequest(any())).thenReturn(validResponse);

    // Disable the new LLM agent feature
    System.setProperty("llm.agent.new.enabled", "false");

    mockMvc.perform(MockMvcRequestBuilders.post("/api/v1/llm-agent/{agentId}/request", validRequest.getAgentId())
        .contentType(MediaType.APPLICATION_JSON)
        .content(asJsonString(validRequest)))
        .andExpect(status().isNotFound());

    verify(llmAgentService, never()).processRequest(validRequest);
  }
}
