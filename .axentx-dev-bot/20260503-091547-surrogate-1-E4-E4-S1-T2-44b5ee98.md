from typing import Optional

from fastapi import APIRouter, HTTPException, Depends

router = APIRouter()

# In-memory store for demo; replace with DB in production
_comments_db = {}
_requests_db = {}  # stub for request existence check

    text: str
    visible: Optional[bool] = False

class CommentResponse(BaseModel):
    id: int
    request_id: int
    text: str
    author: str
    visible: bool
    created_at: datetime
    updated_at: datetime

def get_current_user() -> str:
    # In real system, use auth; here we default to platform-engineer
    return "platform-engineer"

@router.post("/requests/{request_id}/comments", response_model=CommentResponse, status_code=201)
def create_comment(request_id: int, comment: CommentCreate, author: str = Depends(get_current_user)):
    # Ensure request exists (stub)
    if request_id not in _requests_db:
        _requests_db[request_id] = {"id": request_id, "title": f"Request {request_id}"}

    comment_id = len(_comments_db) + 1
    now = datetime.now(timezone.utc)
    record = {
        "id": comment_id,
        "request_id": request_id,
        "text": comment.text,
        "author": author,
        "visible": comment.visible,
        "created_at": now,
        "updated_at": now,
    }
    _comments_db[comment_id] = record
    return record

@router.get("/requests/{request_id}/comments", response_model=list[CommentResponse])
def list_comments(request_id: int, only_visible: bool = False):
    results = [c for c in _comments_db.values() if c["request_id"] == request_id]
    if only_visible:
        results = [c for c in results if c["visible"]]
    return results
