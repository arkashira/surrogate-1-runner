"""
GPU discovery, hot‑plug monitoring, and allocation logic.

Implemented using `nvidia-smi` for GPU enumeration. Hot‑plug events are
detected by polling every second and invoking registered callbacks within
2 seconds of a change.

Allocation is deterministic: the GPU ID itself is returned as the
allocation handle. Allocation fails if the GPU is already reserved.
"""

from __future__ import annotations

import csv
import os
import subprocess
import threading
import time
from dataclasses import dataclass
from typing import Callable, Dict, Iterable, List, Set

# --------------------------------------------------------------------------- #
# Data structures
# --------------------------------------------------------------------------- #

@dataclass(frozen=True)
class GPUInfo:
    """Represents a single GPU's static properties."""
    id: str
    total_memory_mb: int
    pcie_bw_mbps: int
    driver_version: str

@dataclass(frozen=True)
class HotplugEvent:
    """Event emitted when a GPU is added or removed."""
    event_type: str  # "added" or "removed"
    gpu: GPUInfo

# --------------------------------------------------------------------------- #
# Discovery
# --------------------------------------------------------------------------- #

def _run_nvidia_smi() -> str:
    """Run nvidia-smi and return its stdout as a string."""
    try:
        result = subprocess.run(
            ["nvidia-smi", "--query-gpu=index,memory.total,pcie.link.gen.max,driver_version", "--format=csv,noheader,nounits"],
            check=True,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            text=True,
        )
        return result.stdout
    except (subprocess.CalledProcessError, FileNotFoundError) as exc:
        raise RuntimeError("nvidia-smi not available") from exc

def _parse_nvidia_smi_output(output: str) -> List[GPUInfo]:
    """Parse the CSV output from nvidia-smi into GPUInfo objects."""
    gpus: List[GPUInfo] = []
    reader = csv.reader(output.strip().splitlines())
    for row in reader:
        if len(row) != 4:
            continue
        idx, mem, pcie_gen, drv = [item.strip() for item in row]
        # pcie_gen is the max link generation, convert to Mbps
        # Assume 8 GT/s per generation * 8 bytes * 8 bits = 512 Gbps per gen
        # 512 Gbps = 512000 Mbps
        pcie_bw = int(pcie_gen) * 512_000

    return gpus

    """Return a list of all GPUs currently present on the system."""
    output = _run_nvidia_smi()
