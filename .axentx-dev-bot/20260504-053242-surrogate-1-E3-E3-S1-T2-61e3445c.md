import os
import json
import hashlib
import base64
from datetime import datetime, timezone
from pathlib import Path
from typing import Optional, Dict, Any, List

from cryptography.fernet import Fernet, InvalidToken


class LogStorageError(Exception):
    """Base exception for LogStorage-related errors."""
    pass


class LogIntegrityError(LogStorageError):
    """Raised when log integrity verification fails."""
    pass


class LogStorage:
    """
    Secure, tamper‑proof log storage.

    - Each log entry is encrypted at rest using Fernet (AES‑128‑CBC + HMAC‑SHA256).
    - Entries are chained via a SHA‑256 hash (`chain_hash`) that incorporates the
      previous entry's hash, providing an immutable hash‑chain.
    - The storage file is append‑only; existing entries are never overwritten.
    - Logs can be exported (decrypted) for audit purposes.
    """

    def __init__(self, log_path: Path, key: Optional[bytes] = None):
        """
        Initialise the storage.

        Args:
            log_path: Path to the encrypted log file. Parent directories are created.
            key: Optional raw Fernet key (32 url‑safe base64‑encoded bytes). If omitted,
                 the environment variable ``LOG_ENCRYPTION_KEY`` is read.
        """
        self.log_path = log_path.expanduser().resolve()
        self.log_path.parent.mkdir(parents=True, exist_ok=True)

        if key is None:
            env_key = os.getenv("LOG_ENCRYPTION_KEY")
            if not env_key:
                raise LogStorageError(
                    "Encryption key not provided and LOG_ENCRYPTION_KEY env var is missing."
                )
            try:
                # Validate base64‑url format
                base64.urlsafe_b64decode(env_key)
                key = env_key.encode()
            except Exception as exc:
                raise LogStorageError("Invalid LOG_ENCRYPTION_KEY format.") from exc

        self.fernet = Fernet(key)

        # Cache the last chain hash to avoid re‑reading the whole file on each log.
        self._last_chain_hash: Optional[str] = self._load_last_chain_hash()

    # --------------------------------------------------------------------- #
    # Internal helpers
    # --------------------------------------------------------------------- #
    def _load_last_chain_hash(self) -> Optional[str]:
        """Read the last entry's ``chain_hash`` without decrypting the whole file."""
        if not self.log_path.is_file():
            return None

        # Read the file backwards to find the last non‑empty line.
        with self.log_path.open("rb") as f:
            f.seek(0, os.SEEK_END)
            if f.tell() == 0:
                return None
            # Walk backwards until a newline is found.
            offset = -1
            while -offset < f.tell():
                f.seek(offset, os.SEEK_END)
                if f.read(1) == b"\n":
                    break
                offset -= 1
            f.seek(offset + 1, os.SEEK_END)
            last_line = f.readline().strip()
        if not last_line:
            return None
        try:
            entry = json.loads(self.fernet.decrypt(last_line))
            return entry.get("chain_hash")
        except (InvalidToken, json.JSONDecodeError):
            raise LogIntegrityError("Unable to read last log entry – possible tampering.")

    def _hash_entry(self, entry: Dict[str, Any]) -> str:
        """Deterministic SHA‑256 hash of the JSON representation of an entry."""
        entry_bytes = json.dumps(entry, sort_keys=True, separators=(",", ":")).encode()
        return hashlib.sha256(entry_bytes).hexdigest()

    def _chain_hash(self, prev_hash: Optional[str], entry_hash: str) -> str:
        """Combine previous chain hash with current entry hash."""
        combined = (prev_hash or "") + entry_hash
        return hashlib.sha256(combined.encode()).hexdigest()
