"""
# Surrogate-1 Tutorial: Common Patterns

This notebook demonstrates common usage patterns for the Surrogate-1 library.
"""

# Import necessary modules
from surrogate_1 import Functor, Monad

# Example 1: Using Functor for simple transformations
print("Example 1: Functor for simple transformations")

# Define a simple Functor
class SimpleFunctor(Functor):
    def __init__(self, value):
        self.value = value

    def map(self, func):
        return SimpleFunctor(func(self.value))

# Create a SimpleFunctor instance
simple_functor = SimpleFunctor(5)

# Apply a transformation
transformed_functor = simple_functor.map(lambda x: x * 2)

print(f"Original value: {simple_functor.value}")
print(f"Transformed value: {transformed_functor.value}")

# Example 2: Using Monad for chaining operations
print("\nExample 2: Monad for chaining operations")

# Define a simple Monad
class SimpleMonad(Monad):
    def __init__(self, value):
        self.value = value

    def bind(self, func):
        return func(self.value)

# Create a SimpleMonad instance
simple_monad = SimpleMonad(10)

# Chain operations
result = simple_monad.bind(lambda x: x + 5).bind(lambda x: x * 2)

print(f"Final result: {result}")

# Example 3: Combining Functor and Monad
print("\nExample 3: Combining Functor and Monad")

# Define a combined Functor and Monad
class CombinedFunctorMonad(Functor, Monad):
    def __init__(self, value):
        self.value = value

    def map(self, func):
        return CombinedFunctorMonad(func(self.value))

    def bind(self, func):
        return func(self.value)

# Create a CombinedFunctorMonad instance
combined = CombinedFunctorMonad(15)

# Apply transformations and chain operations
transformed_combined = combined.map(lambda x: x + 10).bind(lambda x: x * 3)

print(f"Transformed and chained result: {transformed_combined}")

# surrogate-1-runner

Parallel public-dataset ingest workers for the [axentx/surrogate-1](https://github.com/axentx/surrogate-1) HuggingFace dataset.

## What this does

Every 30 minutes (or on `workflow_dispatch`), GitHub Actions launches **16 parallel runners**. Each runner takes a deterministic 1/16 slice (`slug-hash bucket = SHARD_ID`) of the public dataset list defined in `bin/dataset-enrich.sh`, streams, normalizes per-schema, dedups via the central md5 hash store, and uploads its output to a unique path on the dataset repo:
